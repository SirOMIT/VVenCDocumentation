52 // calculate scale value of bitrate and initial delay
53 int EncHRD::xCalcScale(int x)
54 {
55  if (x==0)
56  {
57  return 0;
58  }
59  uint32_t mask = 0xffffffff;
60  int scaleValue = 32;
61 
62  while ((x&mask) != 0)
63  {
64  scaleValue--;
65  mask = (mask >> 1);
66  }
67 
68  return scaleValue;
69 }

71 void EncHRD::initHRDParameters(const VVEncCfg& encCfg, const SPS& sps)
72 {
73 // if (!encCfg.m_hrdParametersPresent && !encCfg.getCpbSaturationEnabled())
74 // {
75 // return;
76 // }
77  ProfileLevelTierFeatures profileLevelTierFeatures;
78  profileLevelTierFeatures.extractPTLInformation( sps );
79 
80  bool useSubCpbParams = false; //encCfg.getNoPicPartitionFlag() == false;
81  int bitRate = encCfg.m_RCTargetBitrate;
82  int cpbSize = profileLevelTierFeatures.getCpbSizeInBits();
83 
84  CHECK(!(cpbSize != 0), "Unspecified error"); // CPB size may not be equal to zero. ToDo: have a better default and check for level constraints
85 
86  switch (encCfg.m_FrameRate)
87  {
88  case 24:
89  generalHrdParams.numUnitsInTick = 1125000; generalHrdParams.timeScale = 27000000;
90  break;
91  case 25:
92  generalHrdParams.numUnitsInTick = 1080000; generalHrdParams.timeScale = 27000000;
93  break;
94  case 30:
95  generalHrdParams.numUnitsInTick = 900900; generalHrdParams.timeScale = 27000000;
96  break;
97  case 50:
98  generalHrdParams.numUnitsInTick = 540000; generalHrdParams.timeScale = 27000000;
99  break;
100  case 60:
101  generalHrdParams.numUnitsInTick = 450450; generalHrdParams.timeScale = 27000000;
102  break;
103  default:
104  generalHrdParams.numUnitsInTick = 1001; generalHrdParams.timeScale = 60000;
105  break;
106  }
107 
108  if (encCfg.m_temporalSubsampleRatio > 1)
109  {
110  uint32_t temporalSubsampleRatio = encCfg.m_temporalSubsampleRatio;
111  if (double(generalHrdParams.numUnitsInTick) * temporalSubsampleRatio > std::numeric_limits<uint32_t>::max())
112  {
113  generalHrdParams.timeScale = generalHrdParams.timeScale / temporalSubsampleRatio;
114  }
115  else
116  {
117  generalHrdParams.numUnitsInTick = generalHrdParams.numUnitsInTick * temporalSubsampleRatio;
118  }
119  }
120  bool rateCnt = (bitRate > 0);
121 
122  generalHrdParams.generalNalHrdParamsPresent = rateCnt;
123  generalHrdParams.generalVclHrdParamsPresent = rateCnt;
124 
125  generalHrdParams.generalSamePicTimingInAllOlsFlag = true;
126  useSubCpbParams &= (generalHrdParams.generalNalHrdParamsPresent || generalHrdParams.generalVclHrdParamsPresent);
127  generalHrdParams.generalDecodingUnitHrdParamsPresent = useSubCpbParams;
128 
129  if (generalHrdParams.generalDecodingUnitHrdParamsPresent)
130  {
131  generalHrdParams.tickDivisorMinus2 = (100 - 2);
132  }
133 
134  if (xCalcScale(bitRate) <= 6)
135  {
136  generalHrdParams.bitRateScale = 0;
137  }
138  else
139  {
140  generalHrdParams.bitRateScale = xCalcScale(bitRate) - 6;
141  }
142 
143  if (xCalcScale(cpbSize) <= 4)
144  {
145  generalHrdParams.cpbSizeScale = 0;
146  }
147  else
148  {
149  generalHrdParams.cpbSizeScale = xCalcScale(cpbSize) - 4;
150  }
151 
152  generalHrdParams.cpbSizeDuScale = 6; // in units of 2^( 4 + 6 ) = 1,024 bit
153  generalHrdParams.hrdCpbCntMinus1 = 0;
154 
155 
156  // Note: parameters for all temporal layers are initialized with the same values
157  int i, j;
158  uint32_t bitrateValue, cpbSizeValue;
159  uint32_t duCpbSizeValue;
160  uint32_t duBitRateValue = 0;
161 
162  for (i = 0; i < VVENC_MAX_TLAYER; i++)
163  {
164  olsHrdParams[i].fixedPicRateGeneralFlag = true;
165  olsHrdParams[i].fixedPicRateWithinCvsFlag = true;
166  olsHrdParams[i].elementDurationInTcMinus1 = 0;
167  olsHrdParams[i].lowDelayHrdFlag = false;
168 
170  // BitRate[ i ] = ( bit_rate_value_minus1[ i ] + 1 ) * 2^( 6 + bit_rate_scale )
171  bitrateValue = bitRate / (1 << (6 + generalHrdParams.bitRateScale)); // bitRate is in bits, so it needs to be scaled down
172  // CpbSize[ i ] = ( cpb_size_value_minus1[ i ] + 1 ) * 2^( 4 + cpb_size_scale )
173  cpbSizeValue = cpbSize / (1 << (4 + generalHrdParams.cpbSizeScale)); // using bitRate results in 1 second CPB size
174 
175  // DU CPB size could be smaller (i.e. bitrateValue / number of DUs), but we don't know
176  // in how many DUs the slice segment settings will result
177  duCpbSizeValue = bitrateValue;
178  duBitRateValue = cpbSizeValue;
179 
180  for (j = 0; j < (generalHrdParams.hrdCpbCntMinus1 + 1); j++)
181  {
182  olsHrdParams[i].bitRateValueMinus1[j][0] = bitrateValue - 1;
183  olsHrdParams[i].cpbSizeValueMinus1[j][0] = cpbSizeValue - 1;
184  olsHrdParams[i].duCpbSizeValueMinus1[j][0] = duCpbSizeValue - 1;
185  olsHrdParams[i].duBitRateValueMinus1[j][0] = duBitRateValue - 1;
186  olsHrdParams[i].cbrFlag[j][0] = false;
187 
188  olsHrdParams[i].bitRateValueMinus1[j][1] = bitrateValue - 1;
189  olsHrdParams[i].cpbSizeValueMinus1[j][1] = cpbSizeValue - 1;
190  olsHrdParams[i].duCpbSizeValueMinus1[j][1] = duCpbSizeValue - 1;
191  olsHrdParams[i].duBitRateValueMinus1[j][1] = duBitRateValue - 1;
192  olsHrdParams[i].cbrFlag[j][1] = false;
193  }
194  }
195 }
