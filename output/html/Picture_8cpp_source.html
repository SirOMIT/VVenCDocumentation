74 void BlkStat::storeBlkSize( const Picture& pic )
75 {
76  const Slice& slice = *(pic.slices[ 0 ]);
77 
78  ::memset( m_uiBlkSize, 0, sizeof( m_uiBlkSize ) );
79  ::memset( m_uiNumBlk, 0, sizeof( m_uiNumBlk ) );
80 
81  if ( ! slice.isIRAP() )
82  {
83  const int refLayer = slice.depth < NUM_AMAXBT_LAYER ? slice.depth: NUM_AMAXBT_LAYER - 1;

86  m_uiBlkSize[ refLayer ] += cu->Y().area();

102 void BlkStat::setSliceMaxBT( Slice& slice )
103 {
104  if( ! slice.isIRAP() )
105  {
106  int refLayer = slice.depth < NUM_AMAXBT_LAYER ? slice.depth: NUM_AMAXBT_LAYER - 1;
107  if( m_bResetAMaxBT && slice.poc > m_uiPrevISlicePOC )
108  {
109  ::memset( m_uiBlkSize, 0, sizeof( m_uiBlkSize ) );
110  ::memset( m_uiNumBlk, 0, sizeof( m_uiNumBlk ) );
111  m_bResetAMaxBT = false;
112  }
113 
114  if( refLayer >= 0 && m_uiNumBlk[ refLayer ] != 0 )
115  {
116  slice.picHeader->splitConsOverride = true;
117  double dBlkSize = sqrt( ( double ) m_uiBlkSize[refLayer] / m_uiNumBlk[refLayer] );
118  if( dBlkSize < AMAXBT_TH32 || slice.sps->CTUSize == 32 )
119  {
120  slice.picHeader->maxBTSize[1] = ( 32 > MAX_BT_SIZE_INTER ? MAX_BT_SIZE_INTER : 32 );
121  }
122  else if( dBlkSize < AMAXBT_TH64 || slice.sps->CTUSize == 64 )
123  {
124  slice.picHeader->maxBTSize[1] = ( 64 > MAX_BT_SIZE_INTER ? MAX_BT_SIZE_INTER : 64 );
125  }
126  else
127  {
128  slice.picHeader->maxBTSize[1] = ( 128 > MAX_BT_SIZE_INTER ? MAX_BT_SIZE_INTER : 128 );
129  }
130 
131  m_uiBlkSize[ refLayer ] = 0;
132  m_uiNumBlk [ refLayer ] = 0;
133  }
134  }
135  else
136  {
137  if( m_bResetAMaxBT )
138  {
139  ::memset( m_uiBlkSize, 0, sizeof( m_uiBlkSize ) );
140  ::memset( m_uiNumBlk, 0, sizeof( m_uiNumBlk ) );
141  }
142 
143  m_uiPrevISlicePOC = slice.poc;
144  m_bResetAMaxBT = true;
145  }
146 }

292 
293  cs->picture = this;
294  cs->refCS = cs;
295  cs->slice = nullptr; // the slices for this picture have not been set at this point. update cs->slice after swapSliceObject()
296  cs->pps = &pps;
297  cs->picHeader = picHeader;
298  if ( alfAps )
299  {
300  memcpy(cs->alfAps, alfAps, sizeof(cs->alfAps));
301  }
302  cs->lmcsAps = lmcsAps;
303  cs->pcv = pps.pcv;

306 
307  sliceDataStreams.clear();

350 void Picture::extendPicBorder()
351 {
352  if ( isBorderExtended )
353  {
354  return;
355  }
356 
357  for(int comp=0; comp<getNumberValidComponents( cs->area.chromaFormat ); comp++)
358  {
359  ComponentID compID = ComponentID( comp );
360  PelBuf p = m_bufs[ PIC_RECONSTRUCTION ].get( compID );
361  Pel* piTxt = p.bufAt(0,0);
362  int xmargin = margin >> getComponentScaleX( compID, cs->area.chromaFormat );
363  int ymargin = margin >> getComponentScaleY( compID, cs->area.chromaFormat );
364 
365  Pel* pi = piTxt;
366  // do left and right margins
367  for (int y = 0; y < p.height; y++)
368  {
369  for (int x = 0; x < xmargin; x++ )
370  {
371  pi[ -xmargin + x ] = pi[0];
372  pi[ p.width + x ] = pi[p.width-1];
373  }
374  pi += p.stride;
375  }
376 
377  // pi is now the (0,height) (bottom left of image within bigger picture
378  pi -= (p.stride + xmargin);
379  // pi is now the (-marginX, height-1)
380  for (int y = 0; y < ymargin; y++ )
381  {
382  ::memcpy( pi + (y+1)*p.stride, pi, sizeof(Pel)*(p.width + (xmargin << 1)));
383  }
384 
385  // pi is still (-marginX, height-1)
386  pi -= ((p.height-1) * p.stride);
387  // pi is now (-marginX, 0)
388  for (int y = 0; y < ymargin; y++ )
389  {
390  ::memcpy( pi - (y+1)*p.stride, pi, sizeof(Pel)*(p.width + (xmargin<<1)) );
391  }
392 
393  // reference picture with horizontal wrapped boundary
394  if (cs->sps->wrapAroundEnabled)
395  {
396  p = m_bufs[ PIC_RECON_WRAP ].get( compID );
397  p.copyFrom(m_bufs[ PIC_RECONSTRUCTION ].get( compID ));
398  piTxt = p.bufAt(0,0);
399  pi = piTxt;
400  int xoffset = cs->pps->wrapAroundOffset >> getComponentScaleX( compID, cs->area.chromaFormat );
401  for (int y = 0; y < p.height; y++)
402  {
403  for (int x = 0; x < xmargin; x++ )
404  {
405  if( x < xoffset )
406  {
407  pi[ -x - 1 ] = pi[ -x - 1 + xoffset ];
408  pi[ p.width + x ] = pi[ p.width + x - xoffset ];
409  }
410  else
411  {
412  pi[ -x - 1 ] = pi[ 0 ];
413  pi[ p.width + x ] = pi[ p.width - 1 ];
414  }
415  }
416  pi += p.stride;
417  }
418  pi -= (p.stride + xmargin);
419  for (int y = 0; y < ymargin; y++ )
420  {
421  ::memcpy( pi + (y+1)*p.stride, pi, sizeof(Pel)*(p.width + (xmargin << 1)));
422  }
423  pi -= ((p.height-1) * p.stride);
424  for (int y = 0; y < ymargin; y++ )
425  {
426  ::memcpy( pi - (y+1)*p.stride, pi, sizeof(Pel)*(p.width + (xmargin<<1)) );
427  }
428  }
429  }
430 
431  isBorderExtended = true;
432 }
