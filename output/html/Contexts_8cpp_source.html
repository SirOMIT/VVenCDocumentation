140 void BinProbModel::init( int qp, int initId )
141 {
142  int slope = (initId >> 3) - 4;
143  int offset = ((initId & 7) * 18) + 1;
144  int inistate = ((slope * (qp - 16)) >> 1) + offset;
145  int state_clip = inistate < 1 ? 1 : inistate > 127 ? 127 : inistate;
146  const int p1 = (state_clip << 8);
147  m_state[0] = p1 & MASK_0;
148  m_state[1] = p1 & MASK_1;
149 }

154 CtxSet::CtxSet( std::initializer_list<CtxSet> ctxSets )
155 {
156  uint16_t minOffset = std::numeric_limits<uint16_t>::max();
157  uint16_t maxOffset = 0;
158  for( auto iter = ctxSets.begin(); iter != ctxSets.end(); iter++ )
159  {
160  minOffset = std::min<uint16_t>( minOffset, (*iter).Offset );
161  maxOffset = std::max<uint16_t>( maxOffset, (*iter).Offset+(*iter).Size );
162  }
163  Offset = minOffset;
164  Size = maxOffset - minOffset;
165 }

179 CtxSet ContextSetCfg::addCtxSet( std::initializer_list<std::initializer_list<uint8_t>> initSet2d )
180 {
181  const std::size_t startIdx = sm_InitTables[0].size();
182  const std::size_t numValues = ( *initSet2d.begin() ).size();
183  std::size_t setId = 0;
184  for( auto setIter = initSet2d.begin(); setIter != initSet2d.end() && setId < sm_InitTables.size(); setIter++, setId++ )
185  {
186  const std::initializer_list<uint8_t>& initSet = *setIter;
187  std::vector<uint8_t>& initTable = sm_InitTables[setId];
188  CHECK( initSet.size() != numValues,
189  "Number of init values do not match for all sets (" << initSet.size() << " != " << numValues << ")." );
190  initTable.resize( startIdx + numValues );
191  std::size_t elemId = startIdx;
192  for( auto elemIter = ( *setIter ).begin(); elemIter != ( *setIter ).end(); elemIter++, elemId++ )
193  {
194  initTable[elemId] = *elemIter;
195  }
196  }
197  return CtxSet( (uint16_t)startIdx, (uint16_t)numValues );
198 }
