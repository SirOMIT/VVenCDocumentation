155 int QuantRDOQ2::xGetErrScaleCoeff( const bool needsSqrt2, SizeType width, SizeType height, int qp, const int maxLog2TrDynamicRange, const int channelBitDepth )
156 {
157  const int iTransformShift = getTransformShift(channelBitDepth, Size(width, height), maxLog2TrDynamicRange);
158  double dErrScale = (double)(1 << SCALE_BITS); // Compensate for scaling of bitcount in Lagrange cost function
159  double dTransShift = (double)iTransformShift + (needsSqrt2 ? -0.5 : 0.0);
160  dErrScale = dErrScale * pow(2.0, (-2.0*dTransShift)); // Compensate for scaling through forward transform
161  const int QStep = g_quantScales[needsSqrt2 ? 1 : 0][qp];
162  double finalErrScale = dErrScale / QStep / QStep / (1 << (DISTORTION_PRECISION_ADJUSTMENT(channelBitDepth) << 1));
163  return finalErrScale;
164 }

247 {

257 
258  const CCoeffBuf& piCoef = pSrc;

260 
261  const bool useTransformSkip = tu.mtsIdx[compID]==MTS_SKIP;
262

264

267  useRDOQ &= uiWidth > 2;
268  useRDOQ &= uiHeight > 2;
269  }
270

274  {
275  if( useTransformSkip )
276  {
277  if(tu.cu->bdpcmM[toChannelType(compID)])

280  }
281  else
282  {

284  }

321 inline cost_t QuantRDOQ2::xiGetICRateCost( const uint32_t uiAbsLevel,
322  const BinFracBits& fracBitsPar,
323  const BinFracBits& fracBitsGt1,
324  const BinFracBits& fracBitsGt2,
325  const int remRegBins,
326  unsigned goRiceZero,
327  const uint16_t ui16AbsGoRice,
328  const bool useLimitedPrefixLength,
329  const int maxLog2TrDynamicRange ) const
330 {
331  cost_t iRate = xGetIEPRate();
332  if( remRegBins < 4 )
333  {
334  uint32_t symbol = ( uiAbsLevel == 0 ? goRiceZero : uiAbsLevel <= goRiceZero ? uiAbsLevel-1 : uiAbsLevel );
335  uint32_t length;
336  const int threshold = COEF_REMAIN_BIN_REDUCTION;
337  if( symbol < ( threshold << ui16AbsGoRice ) )
338  {
339  length = symbol >> ui16AbsGoRice;
340  iRate += ( length + 1 + ui16AbsGoRice ) << SCALE_BITS;
341  }
342  else if( useLimitedPrefixLength )
343  {
344  const uint32_t maximumPrefixLength = ( 32 - ( COEF_REMAIN_BIN_REDUCTION + maxLog2TrDynamicRange ) );
345 
346  uint32_t prefixLength = 0;
347  uint32_t suffix = ( symbol >> ui16AbsGoRice ) - COEF_REMAIN_BIN_REDUCTION;
348 
349  while( ( prefixLength < maximumPrefixLength ) && ( suffix > ( ( 2 << prefixLength ) - 2 ) ) )
350  {
351  prefixLength++;
352  }
353 
354  const uint32_t suffixLength = ( prefixLength == maximumPrefixLength ) ? ( maxLog2TrDynamicRange - ui16AbsGoRice ) : ( prefixLength + 1/*separator*/ );
355 
356  iRate += ( COEF_REMAIN_BIN_REDUCTION + prefixLength + suffixLength + ui16AbsGoRice ) << SCALE_BITS;
357  }
358  else
359  {
360  length = ui16AbsGoRice;
361  symbol = symbol - ( threshold << ui16AbsGoRice );
362  while( symbol >= ( 1 << length ) )
363  {
364  symbol -= ( 1 << ( length++ ) );
365  }
366  iRate += ( threshold + length + 1 - ui16AbsGoRice + length ) << SCALE_BITS;
367  }
368  }
369  else
370  {
371  const uint32_t cthres = 4;
372  if( uiAbsLevel >= cthres )
373  {
374  uint32_t symbol = ( uiAbsLevel - cthres ) >> 1;
375  uint32_t length;
376  const int threshold = COEF_REMAIN_BIN_REDUCTION;
377  if( symbol < ( threshold << ui16AbsGoRice ) )
378  {
379  length = symbol >> ui16AbsGoRice;
380  iRate += ( length + 1 + ui16AbsGoRice ) << SCALE_BITS;
381  }
382  else if( useLimitedPrefixLength )
383  {
384  const uint32_t maximumPrefixLength = ( 32 - ( COEF_REMAIN_BIN_REDUCTION + maxLog2TrDynamicRange ) );
385 
386  uint32_t prefixLength = 0;
387  uint32_t suffix = ( symbol >> ui16AbsGoRice ) - COEF_REMAIN_BIN_REDUCTION;
388 
389  while( ( prefixLength < maximumPrefixLength ) && ( suffix > ( ( 2 << prefixLength ) - 2 ) ) )
390  {
391  prefixLength++;
392  }
393 
394  const uint32_t suffixLength = ( prefixLength == maximumPrefixLength ) ? ( maxLog2TrDynamicRange - ui16AbsGoRice ) : ( prefixLength + 1/*separator*/ );
395 
396  iRate += ( COEF_REMAIN_BIN_REDUCTION + prefixLength + suffixLength + ui16AbsGoRice ) << SCALE_BITS;
397  }
398  else
399  {
400  length = ui16AbsGoRice;
401  symbol = symbol - ( threshold << ui16AbsGoRice );
402  while( symbol >= ( 1 << length ) )
403  {
404  symbol -= ( 1 << ( length++ ) );
405  }
406  iRate += ( threshold + length + 1 - ui16AbsGoRice + length ) << SCALE_BITS;
407  }
408 
409  iRate += fracBitsGt1.intBits[1];
410  iRate += fracBitsPar.intBits[( uiAbsLevel - 2 ) & 1];
411  iRate += fracBitsGt2.intBits[1];
412  }
413  else if( uiAbsLevel == 1 )
414  {
415  iRate += fracBitsGt1.intBits[0];
416  }
417  else if( uiAbsLevel == 2 )
418  {
419  iRate += fracBitsGt1.intBits[1];
420  iRate += fracBitsPar.intBits[0];
421  iRate += fracBitsGt2.intBits[0];
422  }
423  else if( uiAbsLevel == 3 )
424  {
425  iRate += fracBitsGt1.intBits[1];
426  iRate += fracBitsPar.intBits[1];
427  iRate += fracBitsGt2.intBits[0];
428  }
429  else
430  {
431  iRate = 0;
432  }
433  }
434  return xiGetICost( iRate );
435 }

569  int piAddSBH [16];

573  int iLastScanPos = -1;
574  int lastSubSetId = -1;

579  int remRegBins = ( tu.getTbAreaAfterCoefZeroOut( compID ) * ctxBinSampleRatio ) >> 4;
