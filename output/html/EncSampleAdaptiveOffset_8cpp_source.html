73 inline double xRoundIbdi2(int bitDepth, double x)
74 {
75 #if FULL_NBIT
76  return ((x) >= 0 ? ((int)((x) + 0.5)) : ((int)((x) -0.5)));
77 #else
78  if (DISTORTION_PRECISION_ADJUSTMENT(bitDepth) == 0)
79  return ((x) >= 0 ? ((int)((x) + 0.5)) : ((int)((x) -0.5)));
80  else
81  return ((x) > 0) ? (int)(((int)(x) + (1 << (DISTORTION_PRECISION_ADJUSTMENT(bitDepth) - 1)))
82  / (1 << DISTORTION_PRECISION_ADJUSTMENT(bitDepth)))
83  : ((int)(((int)(x) - (1 << (DISTORTION_PRECISION_ADJUSTMENT(bitDepth) - 1)))
84  / (1 << DISTORTION_PRECISION_ADJUSTMENT(bitDepth))));

448 inline int EncSampleAdaptiveOffset::estIterOffset(int typeIdx, double lambda, int offsetInput, int64_t count, int64_t diffSum, int shift, int bitIncrease, int64_t& bestDist, double& bestCost, int offsetTh )
449 {
450  int iterOffset, tempOffset;
451  int64_t tempDist, tempRate;
452  double tempCost, tempMinCost;
453  int offsetOutput = 0;
454  iterOffset = offsetInput;
455  // Assuming sending quantized value 0 results in zero offset and sending the value zero needs 1 bit. entropy coder can be used to measure the exact rate here.
456  tempMinCost = lambda;
457  while (iterOffset != 0)
458  {
459  // Calculate the bits required for signaling the offset
460  tempRate = (typeIdx == SAO_TYPE_BO) ? (abs((int)iterOffset)+2) : (abs((int)iterOffset)+1);
461  if (abs((int)iterOffset)==offsetTh) //inclusive
462  {
463  tempRate --;
464  }
465  // Do the dequantization before distortion calculation
466  tempOffset = iterOffset << bitIncrease;
467  tempDist = estSaoDist( count, tempOffset, diffSum, shift);
468  tempCost = ((double)tempDist + lambda * (double) tempRate);
469  if(tempCost < tempMinCost)
470  {
471  tempMinCost = tempCost;
472  offsetOutput = iterOffset;
473  bestDist = tempDist;
474  bestCost = tempCost;
475  }
476  iterOffset = (iterOffset > 0) ? (iterOffset-1):(iterOffset+1);
477  }
478  return offsetOutput;
479 }

481 void EncSampleAdaptiveOffset::deriveOffsets(ComponentID compIdx, const int channelBitDepth, int typeIdc, SAOStatData& statData, int* quantOffsets, int& typeAuxInfo)
482 {
483  int bitDepth = channelBitDepth;
484  int shift = 2 * DISTORTION_PRECISION_ADJUSTMENT(bitDepth);
485  int offsetTh = SampleAdaptiveOffset::getMaxOffsetQVal(channelBitDepth); //inclusive
486 
487  ::memset(quantOffsets, 0, sizeof(int)*MAX_NUM_SAO_CLASSES);
488 
489  //derive initial offsets
490  int numClasses = (typeIdc == SAO_TYPE_BO)?((int)NUM_SAO_BO_CLASSES):((int)NUM_SAO_EO_CLASSES);
491  for(int classIdx=0; classIdx< numClasses; classIdx++)
492  {
493  if( (typeIdc != SAO_TYPE_BO) && (classIdx==SAO_CLASS_EO_PLAIN) )
494  {
495  continue; //offset will be zero
496  }
497 
498  if(statData.count[classIdx] == 0)
499  {
500  continue; //offset will be zero
501  }
502 
503  quantOffsets[classIdx] =
504  (int) xRoundIbdi(bitDepth, (double)(statData.diff[classIdx] << DISTORTION_PRECISION_ADJUSTMENT(bitDepth))
505  / (double)(statData.count[classIdx] << m_offsetStepLog2[compIdx]));
506  quantOffsets[classIdx] = Clip3(-offsetTh, offsetTh, quantOffsets[classIdx]);
507  }
508 
509  // adjust offsets
510  switch(typeIdc)
511  {
512  case SAO_TYPE_EO_0:
513  case SAO_TYPE_EO_90:
514  case SAO_TYPE_EO_135:
515  case SAO_TYPE_EO_45:
516  {
517  int64_t classDist;
518  double classCost;
519  for(int classIdx=0; classIdx<NUM_SAO_EO_CLASSES; classIdx++)
520  {
521  if(classIdx==SAO_CLASS_EO_FULL_VALLEY && quantOffsets[classIdx] < 0)
522  {
523  quantOffsets[classIdx] =0;
524  }
525  if(classIdx==SAO_CLASS_EO_HALF_VALLEY && quantOffsets[classIdx] < 0)
526  {
527  quantOffsets[classIdx] =0;
528  }
529  if(classIdx==SAO_CLASS_EO_HALF_PEAK && quantOffsets[classIdx] > 0)
530  {
531  quantOffsets[classIdx] =0;
532  }
533  if(classIdx==SAO_CLASS_EO_FULL_PEAK && quantOffsets[classIdx] > 0)
534  {
535  quantOffsets[classIdx] =0;
536  }
537 
538  if( quantOffsets[classIdx] != 0 ) //iterative adjustment only when derived offset is not zero
539  {
540  quantOffsets[classIdx] = estIterOffset( typeIdc, m_lambda[compIdx], quantOffsets[classIdx], statData.count[classIdx], statData.diff[classIdx], shift, m_offsetStepLog2[compIdx], classDist , classCost , offsetTh );
541  }
542  }
543 
544  typeAuxInfo =0;
545  }
546  break;
547  case SAO_TYPE_BO:
548  {
549  int64_t distBOClasses[NUM_SAO_BO_CLASSES];
550  double costBOClasses[NUM_SAO_BO_CLASSES];
551  ::memset(distBOClasses, 0, sizeof(int64_t)*NUM_SAO_BO_CLASSES);
552  for(int classIdx=0; classIdx< NUM_SAO_BO_CLASSES; classIdx++)
553  {
554  costBOClasses[classIdx]= m_lambda[compIdx];
555  if( quantOffsets[classIdx] != 0 ) //iterative adjustment only when derived offset is not zero
556  {
557  quantOffsets[classIdx] = estIterOffset( typeIdc, m_lambda[compIdx], quantOffsets[classIdx], statData.count[classIdx], statData.diff[classIdx], shift, m_offsetStepLog2[compIdx], distBOClasses[classIdx], costBOClasses[classIdx], offsetTh );
558  }
559  }
560 
561  //decide the starting band index
562  double minCost = MAX_DOUBLE, cost;
563  for(int band=0; band< NUM_SAO_BO_CLASSES- 4+ 1; band++)
564  {
565  cost = costBOClasses[band ];
566  cost += costBOClasses[band+1];
567  cost += costBOClasses[band+2];
568  cost += costBOClasses[band+3];
569 
570  if(cost < minCost)
571  {
572  minCost = cost;
573  typeAuxInfo = band;
574  }
575  }
576  //clear those unused classes
577  int clearQuantOffset[NUM_SAO_BO_CLASSES];
578  ::memset(clearQuantOffset, 0, sizeof(int)*NUM_SAO_BO_CLASSES);
579  for(int i=0; i< 4; i++)
