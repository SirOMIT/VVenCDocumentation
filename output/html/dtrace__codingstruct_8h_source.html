70 inline void dtracePicComp( DTRACE_CHANNEL channel, CodingStructure& cs, const CPelUnitBuf& pelUnitBuf, ComponentID compId )
71 {
72  if( !g_trace_ctx ) return;
73  if( pelUnitBuf.chromaFormat == CHROMA_400 && compId != COMP_Y ) return;
74  const Pel* piSrc = pelUnitBuf.bufs[compId].buf;
75  uint32_t uiStride = pelUnitBuf.bufs[compId].stride;
76  uint32_t uiWidth = pelUnitBuf.bufs[compId].width;
77  uint32_t uiHeight = pelUnitBuf.bufs[compId].height;
78  uint32_t uiChromaScaleX = getComponentScaleX( compId, pelUnitBuf.chromaFormat );
79  uint32_t uiChromaScaleY = getComponentScaleY( compId, pelUnitBuf.chromaFormat );
80 
81  DTRACE ( g_trace_ctx, channel, "\n%s: poc = %d, size=%dx%d\n\n", g_trace_ctx->getChannelName(channel), cs.slice->poc, uiWidth, uiHeight );
82  DTRACE_FRAME_BLOCKWISE( g_trace_ctx, channel, piSrc, uiStride, uiWidth, uiHeight, cs.sps->CTUSize >> uiChromaScaleX, cs.sps->CTUSize >> uiChromaScaleY);
83 }

87 inline void dtraceModeCost(CodingStructure &cs, double lambda)
88 {
89  CHECK( cs.cus.size() != 1, "Only the cost for a single CU can be show with dtraceModeCost!" );
90 
91  Distortion tempDist = cs.dist;
92 
93 #if OLD_RDCOST
94  uint64_t tempBits = cs.fracBits >> SCALE_BITS;
95  uint64_t tempCost = (uint64_t)(cs.dist + (double)tempBits * lambda);
96 #else
97  uint64_t tempBits = cs.fracBits;
98  uint64_t tempCost = (uint64_t)cs.cost;
99 #endif
100 
101  if( cs.cost == MAX_DOUBLE )
102  {
103  tempCost = 0;
104  tempBits = 0;
105  tempDist = 0;
106  }
107 
108  bool isIntra = CU::isIntra( *cs.cus.front() );
109  int intraModeL = isIntra ? cs.cus.front()->intraDir[0] : 0;
110  int intraModeC = isIntra ? cs.cus.front()->intraDir[1] : 0;
111  if (isIntra && intraModeC == DM_CHROMA_IDX)
112  intraModeC = 68;
113  int imvVal = 0;
114  imvVal = cs.cus[0]->imv;
115  DTRACE( g_trace_ctx, D_MODE_COST, "ModeCost: %6lld %3d @(%4d,%4d) [%2dx%2d] %d (qp%d,pm%d,skip%d,mrg%d,fruc%d,obmc%d,ic%d,imv%d,affn%d,%d,%d) tempCS = %lld (%d,%d)\n",
116  DTRACE_GET_COUNTER( g_trace_ctx, D_MODE_COST ),
117  cs.slice->poc,
118  cs.area.Y().x, cs.area.Y().y,
119  cs.area.Y().width, cs.area.Y().height,
120  cs.cus[0]->qtDepth,
121  cs.cus[0]->qp,
122  cs.cus[0]->predMode,
123  cs.cus[0]->skip,
124  cs.cus[0]->mergeFlag,
125  0, 0,
126  imvVal,
127  0, 0,
128  intraModeL, intraModeC,
129  tempCost, tempBits, tempDist );
130 }

132 inline void dtraceBestMode(CodingStructure *&tempCS, CodingStructure *&bestCS, double lambda, bool useEDO)
133 {
134  bool bSplitCS = tempCS->cus.size() > 1 || bestCS->cus.size() > 1;
135  ChannelType chType = tempCS->cus.back()->chType;
136 
137  // if the last CU does not align with the CS, we probably are at the edge
138  bSplitCS |= tempCS->cus.back()->blocks[chType].bottomRight() != tempCS->area.blocks[chType].bottomRight();
139 
140  Distortion tempDist = tempCS->dist;
141 
142 #if OLD_RDCOST
143  uint64_t tempBits = tempCS->fracBits >> SCALE_BITS;
144  uint64_t bestBits = bestCS->fracBits >> SCALE_BITS;
145 // uint64_t tempCost = (uint64_t)(tempCS->dist + (double)tempBits * lambda + ( useEDO ? tempCS->costDbOffset : 0 ));
146 // uint64_t bestCost = (uint64_t)(bestCS->dist + (double)bestBits * lambda + ( useEDO ? bestCS->costDbOffset : 0 ));
147  uint64_t tempCost = (uint64_t)(tempCS->dist + (double)tempBits * lambda);
148  uint64_t bestCost = (uint64_t)(bestCS->dist + (double)bestBits * lambda);
149 #else
150  uint64_t tempBits = tempCS->fracBits;
151  uint64_t bestBits = bestCS->fracBits;
152  uint64_t tempCost = (uint64_t)tempCS->cost;
153  uint64_t bestCost = (uint64_t)bestCS->cost;
154 #endif
155 
156  if( tempCS->cost == MAX_DOUBLE )
157  {
158  tempCost = 0;
159  tempBits = 0;
160  tempDist = 0;
161  }
162 
163  bool isIntra = CU::isIntra( *tempCS->cus[0] );
164  int intraModeL = isIntra ? tempCS->cus[0]->intraDir[0] : 0;
165  int intraModeC = isIntra ? tempCS->cus[0]->intraDir[1] : 0;
166 
167  if(!bSplitCS)
168  {
169  DTRACE( g_trace_ctx, D_BEST_MODE, "CheckModeCost: %6lld %3d @(%4d,%4d) [%2dx%2d] %d (%d,%d,%2d,%d,%d) tempCS = %lld (%d,%d), bestCS = %lld (%d,%d): --> choose %s\n",
170  DTRACE_GET_COUNTER( g_trace_ctx, D_BEST_MODE ),
171  tempCS->slice->poc,
172  tempCS->area.Y().x, tempCS->area.Y().y,
173  tempCS->area.Y().width, tempCS->area.Y().height,
174  tempCS->cus[0]->qtDepth,
175  tempCS->cus[0]->qp,
176  tempCS->cus[0]->predMode,
177  tempCS->cus[0]->mergeFlag,
178  intraModeL, intraModeC,
179  tempCost, tempBits, tempDist,
180  bestCost, bestBits, bestCS->dist,
181  tempCS->cost < bestCS->cost ? "TEMP" : "BEST" );
182  }
183  else
184  {
185  DTRACE( g_trace_ctx, D_BEST_MODE, "CheckModeSplitCost: %6lld %3d @(%4d,%4d) [%2dx%2d] -------------------------- tempCS = %lld (%d,%d), bestCS = %lld (%d,%d): --> choose %s\n",
186  DTRACE_GET_COUNTER( g_trace_ctx, D_BEST_MODE ),
187  tempCS->slice->poc,
188  tempCS->area.Y().x, tempCS->area.Y().y,
189  tempCS->area.Y().width, tempCS->area.Y().height,
190  tempCost, tempBits, tempDist,
191  bestCost, bestBits, bestCS->dist,
192  tempCS->cost < bestCS->cost ? "TEMP STRUCTURE" : "BEST STRUCTURE");
193  }
194 }
