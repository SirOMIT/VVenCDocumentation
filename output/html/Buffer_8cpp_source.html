64 void weightCiipCore( Pel* res, const Pel* src, const int numSamples, int numIntra )
65 {
66  if( numIntra == 1 )
67  {
68  for (int n = 0; n < numSamples; n+=2)
69  {
70  res[n ] = (res[n ] + src[n ] + 1) >> 1;
71  res[n+1] = (res[n+1] + src[n+1] + 1) >> 1;
72  }
73  }
74  else
75  {
76  const Pel* scale = numIntra ? src : res;
77  const Pel* unscale = numIntra ? res : src;
78 
79  for (int n = 0; n < numSamples; n+=2)
80  {
81  res[n ] = (unscale[n ] + 3*scale[n ] + 2) >> 2;
82  res[n+1] = (unscale[n+1] + 3*scale[n+1] + 2) >> 2;
83  }
84  }

87 template< unsigned inputSize, unsigned outputSize >
88 void mipMatrixMulCore( Pel* res, const Pel* input, const uint8_t* weight, const int maxVal, const int inputOffset, bool transpose )
89 {
90  Pel buffer[ outputSize*outputSize];
91 
92  int sum = 0;
93  for( int i = 0; i < inputSize; i++ )
94  {
95  sum += input[i];
96  }
97  const int offset = (1 << (MIP_SHIFT_MATRIX - 1)) - MIP_OFFSET_MATRIX * sum + (inputOffset << MIP_SHIFT_MATRIX);
98  CHECK( inputSize != 4 * (inputSize >> 2), "Error, input size not divisible by four" );
99 
100  Pel* mat = transpose ? buffer : res;
101  unsigned posRes = 0;
102  for( unsigned n = 0; n < outputSize*outputSize; n++ )
103  {
104  int tmp0 = input[0] * weight[0];
105  int tmp1 = input[1] * weight[1];
106  int tmp2 = input[2] * weight[2];
107  int tmp3 = input[3] * weight[3];
108  if( 8 == inputSize )
109  {
110  tmp0 += input[4] * weight[4];
111  tmp1 += input[5] * weight[5];
112  tmp2 += input[6] * weight[6];
113  tmp3 += input[7] * weight[7];
114  }
115  mat[posRes++] = Clip3<int>( 0, maxVal, ((tmp0 + tmp1 + tmp2 + tmp3 + offset) >> MIP_SHIFT_MATRIX) );
116 
117  weight += inputSize;
118  }
119 
120  if( transpose )
121  {
122  for( int j = 0; j < outputSize; j++ )
123  {
124  for( int i = 0; i < outputSize; i++ )
125  {
126  res[j * outputSize + i] = buffer[i * outputSize + j];
127  }
128  }
129  }

439 template<>
440 void AreaBuf<Pel>::scaleSignal(const int scale, const bool dir, const ClpRng& clpRng)
441 {
442  Pel* dst = buf;
443  const Pel* src = buf;
444  const int maxAbsclipBD = (1<<clpRng.bd) - 1;
445 
446  if (dir) // forward
447  {
448  if (width == 1)
449  {
450  THROW("Blocks of width = 1 not supported");
451  }
452  else
453  {
454  for (unsigned y = 0; y < height; y++)
455  {
456  for (unsigned x = 0; x < width; x++)
457  {
458  int sign = src[x] >= 0 ? 1 : -1;
459  int absval = sign * src[x];
460  dst[x] = (Pel)Clip3(-maxAbsclipBD, maxAbsclipBD, sign * (((absval << CSCALE_FP_PREC) + (scale >> 1)) / scale));
461  }
462  dst += stride;
463  src += stride;
464  }
465  }
466  }
467  else // inverse
468  {
469  for (unsigned y = 0; y < height; y++)
470  {
471  for (unsigned x = 0; x < width; x++)
472  {
473  int val = Clip3<int>((-maxAbsclipBD - 1), maxAbsclipBD, (int)src[x]);
474  int sign = src[x] >= 0 ? 1 : -1;
475  int absval = sign * val;
476  val = sign * ((absval * scale + (1 << (CSCALE_FP_PREC - 1))) >> CSCALE_FP_PREC);
477  if (sizeof(Pel) == 2) // avoid overflow when storing data
478  {
479  val = Clip3<int>(-32768, 32767, val);
480  }
481  dst[x] = (Pel)val;
482  }
483  dst += stride;
484  src += stride;
485  }
486  }
