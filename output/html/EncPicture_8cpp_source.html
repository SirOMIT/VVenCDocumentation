91 void EncPicture::compressPicture( Picture& pic, EncGOP& gopEncoder )
92 {
93  ITT_TASKSTART( itt_domain_picEncoder, itt_handle_start );
94 
95  pic.encTime.startTimer();
96 
97  pic.createTempBuffers( pic.cs->pcv->maxCUSize );
98  pic.cs->createCoeffs();
99  pic.cs->createTempBuffers( true );
100  pic.cs->initStructData( MAX_INT, false, nullptr, true );
101 
102  if( pic.useScLMCS && m_pcEncCfg->m_reshapeSignalType == RESHAPE_SIGNAL_PQ && m_pcEncCfg->m_alf )
103  {
104  const double *weights = gopEncoder.getReshaper().getlumaLevelToWeightPLUT();
105  auto& vec = m_ALF.getLumaLevelWeightTable();
106  const size_t numEl = size_t( 1 ) << m_pcEncCfg->m_internalBitDepth[0];
107 
108  vec.resize( numEl );
109  std::copy( weights, weights + numEl, vec.begin() );
110 
111  m_ALF.setAlfWSSD( 1 );
112  }
113  else
114  {
115  m_ALF.setAlfWSSD( 0 );
116  }
117 
118  // compress picture
119  xInitPicEncoder ( pic );
120  if( m_pcEncCfg->m_RCTargetBitrate > 0 )
121  {
122  pic.encRCPic = new EncRCPic;
123  pic.encRCPic->create( m_pcRateCtrl->encRCSeq, m_pcRateCtrl->encRCGOP, (pic.slices[0]->isIntra() ? 0 : pic.slices[0]->TLayer + 1), pic.slices[0]->poc, pic.rcIdxInGop, m_pcRateCtrl->m_listRCPictures );
124  gopEncoder.picInitRateControl( pic.gopId, pic, pic.slices[0], this );
125  }
126 
127  // compress current slice
128  pic.cs->slice = pic.slices[0];
129  m_SliceEncoder.compressSlice( &pic );
130 
131  ITT_TASKEND( itt_domain_picEncoder, itt_handle_start );
132 }

134 void EncPicture::finalizePicture( Picture& pic )
135 {
136  CodingStructure& cs = *(pic.cs);
137  Slice* slice = pic.slices[0];
138  // ALF
139  if( slice->sps->alfEnabled && pic.encPic )
140  {
141 #ifdef TRACE_ENABLE_ITT
142  std::stringstream ss;
143  ss << "ALF_post_" << slice->poc;
144  __itt_string_handle* itt_handle_post = __itt_string_handle_create( ss.str().c_str() );
145 #endif
146  ITT_TASKSTART( itt_domain_ALF_post, itt_handle_post );
147 
148  if( m_pcEncCfg->m_ccalf )
149  {
150  m_ALF.performCCALF( pic, cs );
151  }
152  ITT_TASKEND( itt_domain_ALF_post, itt_handle_post );
153  pic.picApsMap.setApsIdStart( m_ALF.getApsIdStart() );
154 
155  cs.slice->ccAlfFilterParam = m_ALF.getCcAlfFilterParam();
156  cs.slice->ccAlfFilterControl[0] = m_ALF.getCcAlfControlIdc(COMP_Cb);
157  cs.slice->ccAlfFilterControl[1] = m_ALF.getCcAlfControlIdc(COMP_Cr);
158 
159  DTRACE( g_trace_ctx, D_CRC, "ALF" );
160  DTRACE_CRC( g_trace_ctx, D_CRC, cs, cs.getRecoBuf() );
161  DTRACE_PIC_COMP( D_REC_CB_LUMA_ALF, cs, cs.getRecoBuf(), COMP_Y );
162  DTRACE_PIC_COMP( D_REC_CB_CHROMA_ALF, cs, cs.getRecoBuf(), COMP_Cb );
163  DTRACE_PIC_COMP( D_REC_CB_CHROMA_ALF, cs, cs.getRecoBuf(), COMP_Cr );
164  }
165 
166  if( pic.writePic )
167  {
168  // write picture
169  DTRACE_UPDATE( g_trace_ctx, std::make_pair( "bsfinal", 1 ) );
170  xWriteSliceData( pic );
171  DTRACE_UPDATE( g_trace_ctx, std::make_pair( "bsfinal", 0 ) );
172  }
173 
174  // finalize
175  pic.extendPicBorder();
176  if ( m_pcEncCfg->m_useAMaxBT )
177  {
178  pic.picBlkStat.storeBlkSize( pic );
179  }
180  // cleanup
181  if( pic.encPic )
182  {
183  pic.cs->releaseIntermediateData();
184  }
185  pic.cs->destroyTempBuffers();
186  pic.cs->destroyCoeffs();
187  pic.destroyTempBuffers();
188 
189  pic.encTime.stopTimer();
190 }

228 void EncPicture::xInitSliceCheckLDC( Slice* slice ) const
229 {
230  if ( slice->sliceType == VVENC_B_SLICE )
231  {
232  bool bLowDelay = true;
233  int iCurrPOC = slice->poc;
234  int iRefIdx = 0;
235 
236  for ( iRefIdx = 0; iRefIdx < slice->numRefIdx[ REF_PIC_LIST_0 ] && bLowDelay; iRefIdx++ )
237  {
238  if ( slice->getRefPic( REF_PIC_LIST_0, iRefIdx )->getPOC() > iCurrPOC )
239  {
240  bLowDelay = false;
241  }
242  }
243  for ( iRefIdx = 0; iRefIdx < slice->numRefIdx[ REF_PIC_LIST_1 ] && bLowDelay; iRefIdx++ )
244  {
245  if ( slice->getRefPic( REF_PIC_LIST_1, iRefIdx )->getPOC() > iCurrPOC )
246  {
247  bLowDelay = false;
248  }
249  }
250 
251  slice->checkLDC = bLowDelay;
252  }
253  else
254  {
255  slice->checkLDC = true;
256  }

260 void EncPicture::skipCompressPicture( Picture& pic, ParameterSetMap<APS>& shrdApsMap )
261 {
262  CodingStructure& cs = *(pic.cs);
263  Slice* slice = pic.slices[ 0 ];
264 
265  pic.getFilteredOrigBuffer().destroy();
266 
267  if( slice->sps->saoEnabled )
268  {
269  m_SliceEncoder.saoDisabledRate( cs, pic.getSAO( 1 ) );
270  }
271 
272  if ( slice->sps->alfEnabled && ( slice->tileGroupAlfEnabled[COMP_Y] || slice->tileGroupCcAlfCbEnabled || slice->tileGroupCcAlfCrEnabled ) )
273  {
274  // IRAP AU: reset APS map
275  int layerIdx = slice->vps == nullptr ? 0 : slice->vps->generalLayerIdx[ pic.layerId ];
276  if( !layerIdx && ( slice->pendingRasInit || slice->isIDRorBLA() ) )
277  {
278  // We have to reset all APS on IRAP, but in not encoding case we have to keep the parsed APS of current slice
279  // Get active ALF APSs from picture/slice header
280  const std::vector<int> sliceApsIdsLuma = slice->tileGroupLumaApsId;
281 
282  m_ALF.setApsIdStart( ALF_CTB_MAX_NUM_APS );
283 
284  ParameterSetMap<APS>* apsMap = &pic.picApsMap;
285  apsMap->clearActive();
286 
287  for( int apsId = 0; apsId < ALF_CTB_MAX_NUM_APS; apsId++ )
288  {
289  int psId = ( apsId << NUM_APS_TYPE_LEN ) + ALF_APS;
290  APS* aps = apsMap->getPS( psId );
291  if( aps )
292  {
293  // Check if this APS is currently the active one (used in current slice)
294  bool activeAps = false;
295  bool activeApsCcAlf = false;
296 
297  // Luma
298  for( int i = 0; i < sliceApsIdsLuma.size(); i++ )
299  {
300  if( aps->apsId == sliceApsIdsLuma[i] )
301  {
302  activeAps = true;
303  break;
304  }
305  }
306  // Chroma
307  activeAps |= aps->apsId == slice->tileGroupChromaApsId;
308  // CC-ALF
309  activeApsCcAlf |= slice->tileGroupCcAlfCbEnabled && aps->apsId == slice->tileGroupCcAlfCbApsId;
310  activeApsCcAlf |= slice->tileGroupCcAlfCrEnabled && aps->apsId == slice->tileGroupCcAlfCrApsId;
311 
312  if( !activeAps && !activeApsCcAlf )
313  {
314  apsMap->clearChangedFlag( psId );
315  }
316  if( !activeAps )
317  {
318  aps->alfParam.reset();
319  }
320  if( !activeApsCcAlf )
321  {
322  aps->ccAlfParam.reset();
323  }
324  }
325  }
326  }
327 
328  // Assign tne correct APS to slice and emulate the setting of ALF start APS ID
329  int changedApsId = -1;
330  for( int apsId = ALF_CTB_MAX_NUM_APS - 1; apsId >= 0; apsId-- )
331  {
332  ParameterSetMap<APS>* apsMap = &pic.picApsMap;
333  int psId = ( apsId << NUM_APS_TYPE_LEN ) + ALF_APS;
334  APS* aps = apsMap->getPS( psId );
335  if( aps )
336  {
337  DTRACE( g_trace_ctx, D_ALF, "%d: id=%d, apsId=%d, t=%d, ch=%d, lid=%d, tid=%d, nf=%d,%d\n", slice->poc, psId, aps->apsId, aps->apsType, apsMap->getChangedFlag( psId ), aps->layerId, aps->temporalId, aps->alfParam.newFilterFlag[CH_L], aps->alfParam.newFilterFlag[CH_C] );
338  // In slice, replace the old APS (from decoder map) with the APS from encoder map due to later checks while bitstream writing
339  if( slice->alfAps[apsId] )
340  {
341  slice->alfAps[apsId] = aps;
342  }
343 
344  if( apsMap->getChangedFlag( psId ) )
345  {
346  changedApsId = apsId;
347  }
348  }
349  }
350 
351  if( changedApsId >= 0 )
352  {
353  m_ALF.setApsIdStart( changedApsId );
354  }
355 
356  DTRACE( g_trace_ctx, D_ALF, "m_apsIdStart=%d\n", m_ALF.getApsIdStart() );
357  }
358 
359  // LMCS APS
360  // Here the decoded LMCS-APS is contained in the global shared APS map
361  // We have to sync the picture LMCS-APS with the global shared LMCS-APS
362  // NOTE: the "ChangedFlag" is also synced, enabling the APS writing later
363  if( slice->sps->lumaReshapeEnable )
364  {
365  ParameterSetMap<APS>& apsMap = shrdApsMap;
366  const int lmcsApsId = slice->picHeader->lmcsApsId;
367  const int apsMapIdx = ( lmcsApsId << NUM_APS_TYPE_LEN ) + LMCS_APS;
368  APS* aps = apsMap.getPS( apsMapIdx );
369  if( aps )
370  {
371  // LMCS parameters are not completely coming from APS, some data have to come from picture/slice header
372  // Better structuring of parameters is suggested
373  if( slice->picHeader->lmcsEnabled )
374  {
375  aps->lmcsParam.sliceReshaperModelPresent = true;
376  aps->lmcsParam.sliceReshaperEnabled = slice->picHeader->lmcsEnabled;
377  aps->lmcsParam.enableChromaAdj = slice->picHeader->lmcsChromaResidualScale;
378  }
379  else
380  {
381  aps->lmcsParam.sliceReshaperModelPresent = false;
382  aps->lmcsParam.sliceReshaperEnabled = false;
383  }
384 
385  ParameterSetMap<APS>& picApsMap = pic.picApsMap;
386  APS* picAps = picApsMap.getPS( apsMapIdx );
387  if ( picAps == nullptr )
388  {
389  picAps = picApsMap.allocatePS( apsMapIdx );
390  picAps->apsType = LMCS_APS;
391  picAps->apsId = lmcsApsId;
392  }
393  picAps->lmcsParam = aps->lmcsParam;
394  picApsMap.setChangedFlag( apsMapIdx, apsMap.getChangedFlag( apsMapIdx ) );
395  apsMap.clearChangedFlag( apsMapIdx ); // clear flag in the global map, preventing the reusing of it
396  slice->picHeader->lmcsAps = picAps; // just to be sure
397  }
398  }
399 }
