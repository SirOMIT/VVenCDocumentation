196 {
197  Picture* picExtract;
198  Picture* picInsert;
199 
200  PicList::iterator iterPicExtract;
201  PicList::iterator iterPicExtract_1;
202  PicList::iterator iterPicInsert;
203 
204  for (int i = 1; i < (int)(rcListPic.size()); i++)
205  {
206  iterPicExtract = rcListPic.begin();
207  for (int j = 0; j < i; j++)
208  {
209  iterPicExtract++;
210  }
211  picExtract = *(iterPicExtract);
212 
213  iterPicInsert = rcListPic.begin();
214  while (iterPicInsert != iterPicExtract)
215  {
216  picInsert = *(iterPicInsert);
217  if (picInsert->getPOC() >= picExtract->getPOC())
218  {
219  break;
220  }
221 
222  iterPicInsert++;
223  }
224 
225  iterPicExtract_1 = iterPicExtract; iterPicExtract_1++;
226 
227  // swap iterPicExtract and iterPicInsert, iterPicExtract = curr. / iterPicInsert = insertion position
228  rcListPic.insert( iterPicInsert, iterPicExtract, iterPicExtract_1 );
229  rcListPic.erase( iterPicExtract );
230  }

234 Picture* Slice::xGetLongTermRefPic( PicList& rcListPic, int poc, bool pocHasMsb)
235 {
236  PicList::iterator iterPic = rcListPic.begin();
237  Picture* picCand = *(iterPic);
238  Picture* pcStPic = picCand;
239 
240  int pocCycle = 1 << sps->bitsForPOC;
241  if (!pocHasMsb)
242  {
243  poc = poc & (pocCycle - 1);
244  }
245 
246  while ( iterPic != rcListPic.end() )
247  {
248  picCand = *(iterPic);
249  if (picCand && picCand->poc != poc && picCand->isReferenced)
250  {
251  int picPoc = picCand->poc;
252  if (!pocHasMsb)
253  {
254  picPoc = picPoc & (pocCycle - 1);
255  }
256 
257  if (poc == picPoc)
258  {
259  if(picCand->isLongTerm)
260  {
261  return picCand;
262  }
263 
264  pcStPic = picCand;
265  break;
266  }
267  }
268 
269  iterPic++;
270  }
271 
272  return pcStPic;
273 }

287 void Slice::setSMVDParam()
288 {
289  if ( sps->SMVD && checkLDC == false && picHeader->mvdL1Zero == false )
290  {
291  int currPOC = poc;
292  int forwardPOC = poc;
293  int backwardPOC = poc;
294  int ref = 0;
295  int refIdx0 = -1;
296  int refIdx1 = -1;
297 
298  // search nearest forward POC in List 0
299  for (ref = 0; ref < numRefIdx[REF_PIC_LIST_0]; ref++)
300  {
301  const int refPoc = getRefPic(REF_PIC_LIST_0, ref)->getPOC();
302  const bool isRefLongTerm = getRefPic(REF_PIC_LIST_0, ref)->isLongTerm;
303  if (refPoc < currPOC && (refPoc > forwardPOC || refIdx0 == -1) && !isRefLongTerm)
304  {
305  forwardPOC = refPoc;
306  refIdx0 = ref;
307  }
308  }
309 
310  // search nearest backward POC in List 1
311  for (ref = 0; ref < numRefIdx[REF_PIC_LIST_1]; ref++)
312  {
313  const int refPoc = getRefPic(REF_PIC_LIST_1, ref)->getPOC();
314  const bool isRefLongTerm = getRefPic(REF_PIC_LIST_1, ref)->isLongTerm;
315  if (refPoc > currPOC && (refPoc < backwardPOC || refIdx1 == -1) && !isRefLongTerm)
316  {
317  backwardPOC = refPoc;
318  refIdx1 = ref;
319  }
320  }
321 
322  if (!(forwardPOC < currPOC && backwardPOC > currPOC))
323  {
324  forwardPOC = currPOC;
325  backwardPOC = currPOC;
326  refIdx0 = -1;
327  refIdx1 = -1;
328 
329  // search nearest backward POC in List 0
330  for (ref = 0; ref < numRefIdx[REF_PIC_LIST_0]; ref++)
331  {
332  const int refPoc = getRefPic(REF_PIC_LIST_0, ref)->getPOC();
333  const bool isRefLongTerm = getRefPic(REF_PIC_LIST_0, ref)->isLongTerm;
334  if (refPoc > currPOC && (refPoc < backwardPOC || refIdx0 == -1) && !isRefLongTerm)
335  {
336  backwardPOC = refPoc;
337  refIdx0 = ref;
338  }
339  }
340 
341  // search nearest forward POC in List 1
342  for (ref = 0; ref < numRefIdx[REF_PIC_LIST_1]; ref++)
343  {
344  const int refPoc = getRefPic(REF_PIC_LIST_1, ref)->getPOC();
345  const bool isRefLongTerm = getRefPic(REF_PIC_LIST_1, ref)->isLongTerm;
346  if (refPoc < currPOC && (refPoc > forwardPOC || refIdx1 == -1) && !isRefLongTerm)
347  {
348  forwardPOC = refPoc;
349  refIdx1 = ref;
350  }
351  }
352  }
353 
354  if (forwardPOC < currPOC && backwardPOC > currPOC)
355  {
356  biDirPred = true;
357  symRefIdx[0] = refIdx0;
358  symRefIdx[1] = refIdx1;
359  return;
360  }
361  }

364  symRefIdx[0] = -1;
365  symRefIdx[1] = -1;

385 void Slice::constructRefPicList(PicList& rcListPic, bool extBorder)
386 {
387  ::memset(isUsedAsLongTerm, 0, sizeof(isUsedAsLongTerm));
388  if (sliceType == VVENC_I_SLICE)
389  {
390  ::memset(refPicList, 0, sizeof(refPicList));
391  ::memset(numRefIdx, 0, sizeof(numRefIdx));
392  return;
393  }
394 
395  Picture* pcRefPic = NULL;
396  uint32_t numOfActiveRef = 0;
397  //construct L0
398  for (int refList = 0; refList < NUM_REF_PIC_LIST_01; refList++)
399  {
400  RefPicList eRefList = RefPicList(refList);
401  numOfActiveRef = numRefIdx[ eRefList ];
402  for (int ii = 0; ii < numOfActiveRef; ii++)
403  {
404  if (!rpl[eRefList]->isLongtermRefPic[ii])
405  {
406  int poc_ = poc + rpl[eRefList]->refPicIdentifier[ii];
407 
408  PicList::iterator iterPic = rcListPic.begin();
409  pcRefPic = *(iterPic);
410 
411  while ( iterPic != rcListPic.end() )
412  {
413  if(pcRefPic->getPOC() == poc_)
414  {
415  break;
416  }
417  iterPic++;
418  pcRefPic = *(iterPic);
419  }
420 
421  pcRefPic->isLongTerm = false;
422  }
423  else
424  {
425  int pocBits = sps->bitsForPOC;
426  int pocMask = (1 << pocBits) - 1;
427  int ltrpPoc = rpl[eRefList]->refPicIdentifier[ii] & pocMask;
428  ltrpPoc += rplLocal[eRefList].deltaPocMSBPresent[ii] ? (pocMask + 1) * rplLocal[eRefList].deltaPocMSBCycleLT[ii] : 0;
429  pcRefPic = xGetLongTermRefPic(rcListPic, ltrpPoc, rplLocal[eRefList].deltaPocMSBPresent[ii]);
430  pcRefPic->isLongTerm = true;
431  }
432  if ( extBorder )
433  {
434  pcRefPic->extendPicBorder();
435  }
436  refPicList[eRefList][ii] = pcRefPic;
437  isUsedAsLongTerm[eRefList][ii] = pcRefPic->isLongTerm;
438  }
439  }

595  }
