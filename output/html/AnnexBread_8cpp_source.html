71 static void
72 _byteStreamNALUnit(
73  InputByteStream& bs,
74  std::vector<uint8_t>& nalUnit,
75  AnnexBStats& stats)
76 {
77  /* At the beginning of the decoding process, the decoder initialises its
78  * current position in the byte stream to the beginning of the byte stream.
79  * It then extracts and discards each leading_zero_8bits syntax element (if
80  * present), moving the current position in the byte stream forward one
81  * byte at a time, until the current position in the byte stream is such
82  * that the next four bytes in the bitstream form the four-byte sequence
83  * 0x00000001.
84  */
85  while ((bs.eofBeforeNBytes(24/8) || bs.peekBytes(24/8) != 0x000001)
86  && (bs.eofBeforeNBytes(32/8) || bs.peekBytes(32/8) != 0x00000001))
87  {
88  uint8_t leading_zero_8bits = bs.readByte();
89  if(leading_zero_8bits != 0) { THROW( "Leading zero bits not zero" ); }
90  stats.m_numLeadingZero8BitsBytes++;
91  }
92 
93  /* 1. When the next four bytes in the bitstream form the four-byte sequence
94  * 0x00000001, the next byte in the byte stream (which is a zero_byte
95  * syntax element) is extracted and discarded and the current position in
96  * the byte stream is set equal to the position of the byte following this
97  * discarded byte.
98  */
99  /* NB, the previous step guarantees this will succeed -- if EOF was
100  * encountered, an exception will stop execution getting this far */
101  if (bs.peekBytes(24/8) != 0x000001)
102  {
103  uint8_t zero_byte = bs.readByte();
104  CHECK( zero_byte != 0, "Zero byte not '0'" );
105  stats.m_numZeroByteBytes++;
106  }
107 
108  /* 2. The next three-byte sequence in the byte stream (which is a
109  * start_code_prefix_one_3bytes) is extracted and discarded and the current
110  * position in the byte stream is set equal to the position of the byte
111  * following this three-byte sequence.
112  */
113  /* NB, (1) guarantees that the next three bytes are 0x00 00 01 */
114  uint32_t start_code_prefix_one_3bytes = bs.readBytes(24/8);
115  if(start_code_prefix_one_3bytes != 0x000001) { THROW( "Invalid code prefix" );}
116  stats.m_numStartCodePrefixBytes += 3;
117 
118  /* 3. NumBytesInNALunit is set equal to the number of bytes starting with
119  * the byte at the current position in the byte stream up to and including
120  * the last byte that precedes the location of any of the following
121  * conditions:
122  * a. A subsequent byte-aligned three-byte sequence equal to 0x000000, or
123  * b. A subsequent byte-aligned three-byte sequence equal to 0x000001, or
124  * c. The end of the byte stream, as determined by unspecified means.
125  */
126  /* 4. NumBytesInNALunit bytes are removed from the bitstream and the
127  * current position in the byte stream is advanced by NumBytesInNALunit
128  * bytes. This sequence of bytes is nal_unit( NumBytesInNALunit ) and is
129  * decoded using the NAL unit decoding process
130  */
131  /* NB, (unsigned)x > 2 implies n!=0 && n!=1 */
132  while (bs.eofBeforeNBytes(24/8) || bs.peekBytes(24/8) > 2)
133  {
134  nalUnit.push_back(bs.readByte());
135  }
136 
137  /* 5. When the current position in the byte stream is:
138  * - not at the end of the byte stream (as determined by unspecified means)
139  * - and the next bytes in the byte stream do not start with a three-byte
140  * sequence equal to 0x000001
141  * - and the next bytes in the byte stream do not start with a four byte
142  * sequence equal to 0x00000001,
143  * the decoder extracts and discards each trailing_zero_8bits syntax
144  * element, moving the current position in the byte stream forward one byte
145  * at a time, until the current position in the byte stream is such that:
146  * - the next bytes in the byte stream form the four-byte sequence
147  * 0x00000001 or
148  * - the end of the byte stream has been encountered (as determined by
149  * unspecified means).
150  */
151  /* NB, (3) guarantees there are at least three bytes available or none */
152  while ((bs.eofBeforeNBytes(24/8) || bs.peekBytes(24/8) != 0x000001)
153  && (bs.eofBeforeNBytes(32/8) || bs.peekBytes(32/8) != 0x00000001))
154  {
155  uint8_t trailing_zero_8bits = bs.readByte();
156  CHECK( trailing_zero_8bits != 0, "Trailing zero bits not '0'" );
157  stats.m_numTrailingZero8BitsBytes++;
158  }
159 }

168 bool
169 byteStreamNALUnit(
170  InputByteStream& bs,
171  std::vector<uint8_t>& nalUnit,
172  AnnexBStats& stats)
173 {
174  bool eof = false;
175  try
176  {
177  _byteStreamNALUnit(bs, nalUnit, stats);
178  }
179  catch (...)
180  {
181  eof = true;
182  }
183  stats.m_numBytesInNALUnit = uint32_t(nalUnit.size());
184  return eof;
185 }
