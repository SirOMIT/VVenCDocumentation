75 QpParam::QpParam(const TransformUnit& tu, const ComponentID &compID, const bool allowACTQpoffset)
76 {
77  const ChannelType chType = toChannelType( compID );
78  const SPS &sps = *tu.cu->cs->sps;
79  const int qpBdOffset = sps.qpBDOffset[chType];
80  const bool useJQP = isChroma( compID ) && abs( TU::getICTMode( tu ) ) == 2;
81  const ComponentID jCbCr = useJQP ? COMP_JOINT_CbCr : compID;
82 
83  int chromaQpOffset = 0;
84 
85  if( isChroma( compID ) )
86  {
87  const PPS &pps = *tu.cu->slice->pps;
88  chromaQpOffset = pps.chromaQpOffset [jCbCr];
89  chromaQpOffset += tu.cu->slice->sliceChromaQpDelta[jCbCr];
90  chromaQpOffset += pps.getChromaQpOffsetListEntry( tu.cu->chromaQpAdj ).u.offset[int( jCbCr ) - 1];
91  }
92 
93  int baseQp;
94  int qpy = tu.cu->qp;
95  //bool skip = tu.mtsIdx[compID] == MTS_SKIP;
96 
97  if( isLuma( compID ) )
98  {
99  baseQp = tu.cu->qp + qpBdOffset;
100  }
101  else
102  {
103  int qpi = Clip3( -qpBdOffset, MAX_QP, qpy );
104  baseQp = sps.chromaQpMappingTable.getMappedChromaQpValue( jCbCr, qpi );
105  baseQp = Clip3( -qpBdOffset, MAX_QP, baseQp + chromaQpOffset ) + qpBdOffset;
106  }
107 
108  if( allowACTQpoffset && tu.cu->colorTransform )
109  {
110  baseQp += DELTA_QP_ACT[jCbCr];
111  }
112 
113  baseQp = Clip3( 0, MAX_QP + qpBdOffset, baseQp );
114 
115  //if( !skip )
116  {
117  Qps [0] = baseQp;
118  pers[0] = baseQp / 6;
119  rems[0] = baseQp % 6;
120  }
121  //else
122  {
123  int internalMinusInputBitDepth = sps.internalMinusInputBitDepth[chType];
124  int baseQpTS = std::max( baseQp, 4 + 6 * internalMinusInputBitDepth );
125 
126  Qps [1] = baseQpTS;
127  pers[1] = baseQpTS / 6;
128  rems[1] = baseQpTS % 6;
129  }
130 }

136 static void DeQuantCore(const int maxX,const int maxY,const int scale,const TCoeffSig* const piQCoef,const size_t piQCfStride,TCoeff *const piCoef,const int rightShift,const int inputMaximum,const TCoeff transformMaximum)
137 {
138  const int inputMinimum = -(inputMaximum+1);
139  const TCoeff transformMinimum = -(transformMaximum+1);
140  if (rightShift>0)
141  {
142  const Intermediate_Int iAdd = (Intermediate_Int) 1 << (rightShift - 1);
143  for( int y = 0, n = 0; y <= maxY; y++)
144  {
145  for( int x = 0; x <= maxX; x++, n++ )
146  {
147  const TCoeff clipQCoef = TCoeff(Clip3<Intermediate_Int>(inputMinimum, inputMaximum, piQCoef[x + y * piQCfStride]));
148  Intermediate_Int iCoeffQ = (Intermediate_Int(clipQCoef) * scale + iAdd) >> rightShift;
149  piCoef[n] = TCoeff(Clip3<Intermediate_Int>(transformMinimum,transformMaximum,iCoeffQ));
150  }
151  }
152  }
153  else // rightshift <0
154  {
155  int leftShift = -rightShift;
156  for( int y = 0, n = 0; y <= maxY; y++)
157  {
158  for( int x = 0; x <= maxX; x++, n++ )
159  {
160  const TCoeff clipQCoef = TCoeff(Clip3<Intermediate_Int>(inputMinimum, inputMaximum, piQCoef[x + y * piQCfStride]));
161  const Intermediate_Int iCoeffQ = (Intermediate_Int(clipQCoef) * scale) << leftShift;
162  piCoef[n] = TCoeff(Clip3<Intermediate_Int>(transformMinimum,transformMaximum,iCoeffQ));
163  }
164  }
165  }
166 }

183 void invResDPCM( const TransformUnit& tu, const ComponentID compID, CoeffSigBuf& dstBuf )
184 {
185  const CompArea& rect = tu.blocks[compID];
186  const int wdt = rect.width;
187  const int hgt = rect.height;
188  const CCoeffSigBuf coeffs = tu.getCoeffs(compID);
189 
190  const int maxLog2TrDynamicRange = tu.cs->sps->getMaxLog2TrDynamicRange(toChannelType(compID));
191  const TCoeff inputMinimum = -(1 << maxLog2TrDynamicRange);
192  const TCoeff inputMaximum = (1 << maxLog2TrDynamicRange) - 1;
193 
194  const TCoeffSig* coef = &coeffs.buf[0];
195  TCoeffSig* dst = &dstBuf.buf[0];
196 
197  if ( tu.cu->bdpcmM[toChannelType(compID)] == 1)
198  {
199  for( int y = 0; y < hgt; y++ )
200  {
201  dst[0] = coef[0];
202  for( int x = 1; x < wdt; x++ )
203  {
204  dst[x] = Clip3(inputMinimum, inputMaximum, TCoeff( dst[x - 1] ) + TCoeff( coef[x] ));
205  }
206  coef += coeffs.stride;
207  dst += dstBuf.stride;
208  }
209  }
210  else
211  {
212  for( int x = 0; x < wdt; x++ )
213  {
214  dst[x] = coef[x];
215  }
216  for( int y = 0; y < hgt - 1; y++ )
217  {
218  for( int x = 0; x < wdt; x++ )
219  {
220  dst[dstBuf.stride + x] = Clip3(inputMinimum, inputMaximum, TCoeff( dst[x] ) + TCoeff( coef[coeffs.stride + x] ));
221  }
222  coef += coeffs.stride;
223  dst += dstBuf.stride;
224  }
225  }
226 }

228 void fwdResDPCM( TransformUnit& tu, const ComponentID compID )
229 {
230  const CompArea& rect = tu.blocks[compID];
231  const int wdt = rect.width;
232  const int hgt = rect.height;
233  CoeffSigBuf coeffs = tu.getCoeffs(compID);
234 
235  TCoeffSig* coef = &coeffs.buf[0];
236  if (tu.cu->bdpcmM[toChannelType(compID)] == 1)
237  {
238  for( int y = 0; y < hgt; y++ )
239  {
240  for( int x = wdt - 1; x > 0; x-- )
241  {
242  coef[x] -= coef[x - 1];
243  }
244  coef += coeffs.stride;
245  }
246  }
247  else
248  {
249  coef += coeffs.stride * (hgt - 1);
250  for( int y = 0; y < hgt - 1; y++ )
251  {
252  for ( int x = 0; x < wdt; x++ )
253  {
254  coef[x] -= coef[x - coeffs.stride];
255  }
256  coef -= coeffs.stride;
257  }
258  }
259 }

262 void Quant::xSignBitHidingHDQ( TCoeffSig* pQCoef, const TCoeff* pCoef, TCoeff* deltaU, const CoeffCodingContext& cctx, const int maxLog2TrDynamicRange )
263 {
264  const uint32_t width = cctx.width();
265  const uint32_t height = cctx.height();
266  const uint32_t groupSize = 1 << cctx.log2CGSize();
267 
268  const TCoeff entropyCodingMinimum = -(1 << maxLog2TrDynamicRange);
269  const TCoeff entropyCodingMaximum = (1 << maxLog2TrDynamicRange) - 1;
270 
271  int lastCG = -1;
272  int absSum = 0 ;
273  int n ;
274 
275  for( int subSet = (width*height-1) >> cctx.log2CGSize(); subSet >= 0; subSet-- )
276  {
277  int subPos = subSet << cctx.log2CGSize();
278  int firstNZPosInCG=groupSize , lastNZPosInCG=-1 ;
279  absSum = 0 ;
280 
281  for(n = groupSize-1; n >= 0; --n )
282  {
283  if( pQCoef[ cctx.blockPos( n + subPos ) ] )
284  {
285  lastNZPosInCG = n;
286  break;
287  }
288  }
289 
290  for(n = 0; n <groupSize; n++ )
291  {
292  if( pQCoef[ cctx.blockPos( n + subPos ) ] )
293  {
294  firstNZPosInCG = n;
295  break;
296  }
297  }
298 
299  for(n = firstNZPosInCG; n <=lastNZPosInCG; n++ )
300  {
301  absSum += int(pQCoef[ cctx.blockPos( n + subPos ) ]);
302  }
303 
304  if(lastNZPosInCG>=0 && lastCG==-1)
305  {
306  lastCG = 1 ;
307  }
308 
309  if( lastNZPosInCG-firstNZPosInCG>=SBH_THRESHOLD )
310  {
311  uint32_t signbit = (pQCoef[cctx.blockPos(subPos+firstNZPosInCG)]>0?0:1) ;
312  if( signbit!=(absSum&0x1) ) //compare signbit with sum_parity
313  {
314  TCoeff curCost = std::numeric_limits<TCoeff>::max();
315  TCoeff minCostInc = std::numeric_limits<TCoeff>::max();
316  int minPos =-1, finalChange=0, curChange=0;
317 
318  for( n = (lastCG==1?lastNZPosInCG:groupSize-1) ; n >= 0; --n )
319  {
320  uint32_t blkPos = cctx.blockPos( n+subPos );
321  if(pQCoef[ blkPos ] != 0 )
322  {
323  if(deltaU[blkPos]>0)
324  {
325  curCost = - deltaU[blkPos];
326  curChange=1 ;
327  }
328  else
329  {
330  //curChange =-1;
331  if(n==firstNZPosInCG && abs(pQCoef[blkPos])==1)
332  {
333  curCost = std::numeric_limits<TCoeff>::max();
334  }
335  else
336  {
337  curCost = deltaU[blkPos];
338  curChange =-1;
339  }
340  }
341  }
342  else
343  {
344  if(n<firstNZPosInCG)
345  {
346  uint32_t thisSignBit = (pCoef[blkPos]>=0?0:1);
347  if(thisSignBit != signbit )
348  {
349  curCost = std::numeric_limits<TCoeff>::max();
350  }
351  else
352  {
353  curCost = - (deltaU[blkPos]) ;
354  curChange = 1 ;
355  }
356  }
357  else
358  {
359  curCost = - (deltaU[blkPos]) ;
360  curChange = 1 ;
361  }
362  }
363 
364  if( curCost<minCostInc)
365  {
366  minCostInc = curCost ;
367  finalChange = curChange ;
368  minPos = blkPos ;
369  }
370  } //CG loop
371 
372  if(pQCoef[minPos] == entropyCodingMaximum || pQCoef[minPos] == entropyCodingMinimum)
373  {
374  finalChange = -1;
375  }
376 
377  if(pCoef[minPos]>=0)
378  {
379  pQCoef[minPos] += finalChange ;
380  }
381  else
382  {
383  pQCoef[minPos] -= finalChange ;
384  }
385  } // Hide
386  }
387  if(lastCG==1)
388  {
389  lastCG=0 ;
390  }
391  } // TU loop
392 
393  return;
394 }

396 void Quant::dequant(const TransformUnit& tu,
397  CoeffBuf& dstCoeff,
398  const ComponentID compID,
399  const QpParam& cQP)
400 {
401  const SPS *sps = tu.cs->sps;
402  const CompArea &area = tu.blocks[compID];
403  const uint32_t uiWidth = area.width;
404  const uint32_t uiHeight = area.height;
405  TCoeff *const piCoef = dstCoeff.buf;
406  const uint32_t numSamplesInBlock = uiWidth * uiHeight;
407  const int maxLog2TrDynamicRange = sps->getMaxLog2TrDynamicRange(toChannelType(compID));
408  const TCoeff transformMinimum = -(1 << maxLog2TrDynamicRange);
409  const TCoeff transformMaximum = (1 << maxLog2TrDynamicRange) - 1;
410  const bool isTransformSkip = tu.mtsIdx[compID] == MTS_SKIP;
411  const bool isLfnstApplied = tu.cu->lfnstIdx > 0 && (CU::isSepTree(*tu.cu) ? true : isLuma(compID));
412  const bool enableScalingLists = getUseScalingList(uiWidth, uiHeight, isTransformSkip, isLfnstApplied);
413  const int scalingListType = getScalingListType(tu.cu->predMode, compID);
414  const int channelBitDepth = sps->bitDepths[toChannelType(compID)];
415 
416  const TCoeffSig *coef = tu.getCoeffs( compID ).buf;
417  const ptrdiff_t piStride = tu.getCoeffs( compID ).stride;
418 
419  if( tu.cu->bdpcmM[toChannelType( compID )] )
420  {
421  CoeffSigBuf coefBuf( m_tmpBdpcm, uiWidth, uiHeight );
422  invResDPCM( tu, compID, coefBuf );
423  coef = m_tmpBdpcm;
424  }
425 
426  const TCoeffSig *const piQCoef = coef;
427  CHECK(scalingListType >= SCALING_LIST_NUM, "Invalid scaling list");
428 
429  // Represents scaling through forward transform
430  const bool bClipTransformShiftTo0 = tu.mtsIdx[compID]!=MTS_SKIP && sps->spsRExt.extendedPrecisionProcessing;
431  const int originalTransformShift = getTransformShift(channelBitDepth, area.size(), maxLog2TrDynamicRange);
432  const bool needSqrtAdjustment = TU::needsSqrt2Scale( tu, compID );
433  const int iTransformShift = (bClipTransformShiftTo0 ? std::max<int>(0, originalTransformShift) : originalTransformShift) + (needSqrtAdjustment?-1:0);
434 
435  const int QP_per = cQP.per(isTransformSkip);
436  const int QP_rem = cQP.rem(isTransformSkip);
437 
438  const int rightShift = (IQUANT_SHIFT - ((isTransformSkip ? 0 : iTransformShift) + QP_per)) + (enableScalingLists ? LOG2_SCALING_LIST_NEUTRAL_VALUE : 0);
439 
440  if(enableScalingLists)
441  {
442  //from the dequantization equation:
443  //iCoeffQ = ((Intermediate_Int(clipQCoef) * piDequantCoef[deQuantIdx]) + iAdd ) >> rightShift
444  //(sizeof(Intermediate_Int) * 8) = inputBitDepth + dequantCoefBits - rightShift
445  const uint32_t dequantCoefBits = 1 + IQUANT_SHIFT + SCALING_LIST_BITS;
446  const uint32_t targetInputBitDepth = std::min<uint32_t>((maxLog2TrDynamicRange + 1), (((sizeof(Intermediate_Int) * 8) + rightShift) - dequantCoefBits));
447 
448  const Intermediate_Int inputMinimum = -(1 << (targetInputBitDepth - 1));
449  const Intermediate_Int inputMaximum = (1 << (targetInputBitDepth - 1)) - 1;
450 
451  const uint32_t uiLog2TrWidth = Log2(uiWidth);
452  const uint32_t uiLog2TrHeight = Log2(uiHeight);
453  int* piDequantCoef = getDequantCoeff(scalingListType, QP_rem, uiLog2TrWidth, uiLog2TrHeight);
454 
455  if(rightShift > 0)
456  {
457  const Intermediate_Int iAdd = (Intermediate_Int) 1 << (rightShift - 1);
458  for( int n = 0; n < numSamplesInBlock; n++ )
459  {
460  const TCoeff clipQCoef = TCoeff(Clip3<Intermediate_Int>(inputMinimum, inputMaximum, piQCoef[n]));
461  const Intermediate_Int iCoeffQ = ((Intermediate_Int(clipQCoef) * piDequantCoef[n]) + iAdd ) >> rightShift;
462  piCoef[n] = TCoeff(Clip3<Intermediate_Int>(transformMinimum,transformMaximum,iCoeffQ));
463  }
464  }
465  else
466  {
467  const int leftShift = -rightShift;
468  for( int n = 0; n < numSamplesInBlock; n++ )
469  {
470  const TCoeff clipQCoef = TCoeff(Clip3<Intermediate_Int>(inputMinimum, inputMaximum, piQCoef[n]));
471  const Intermediate_Int iCoeffQ = (Intermediate_Int(clipQCoef) * piDequantCoef[n]) << leftShift;
472  piCoef[n] = TCoeff(Clip3<Intermediate_Int>(transformMinimum,transformMaximum,iCoeffQ));
473  }
474  }
475  }
476  else
477  {
478  const int scale = g_invQuantScales[needSqrtAdjustment?1:0][QP_rem];
479  const int scaleBits = ( IQUANT_SHIFT + 1 );
480  //from the dequantisation equation:
481  //iCoeffQ = Intermediate_Int((int64_t(clipQCoef) * scale + iAdd) >> rightShift);
482  //(sizeof(Intermediate_Int) * 8) = inputBitDepth + scaleBits - rightShift
483  const uint32_t targetInputBitDepth = std::min<uint32_t>((maxLog2TrDynamicRange + 1), (((sizeof(Intermediate_Int) * 8) + rightShift) - scaleBits));
484  const Intermediate_Int inputMaximum = (1 << (targetInputBitDepth - 1)) - 1;
485  DeQuant(uiWidth-1,uiHeight-1,scale,piQCoef,piStride,piCoef,rightShift,inputMaximum,transformMaximum);
486  }
487 }

587 void Quant::xDestroyScalingList()
588 {
589  if( !m_isScalingListOwner ) return;
590 
591  for(uint32_t sizeIdX = 0; sizeIdX < SCALING_LIST_SIZE_NUM; sizeIdX++)
592  {
593  for(uint32_t sizeIdY = 0; sizeIdY < SCALING_LIST_SIZE_NUM; sizeIdY++)
594  {
595  for(uint32_t listId = 0; listId < SCALING_LIST_NUM; listId++)
596  {
597  for(uint32_t qp = 0; qp < SCALING_LIST_REM_NUM; qp++)
598  {
599  if(m_quantCoef[sizeIdX][sizeIdY][listId][qp])
600  {

613 void Quant::quant(TransformUnit& tu, const ComponentID compID, const CCoeffBuf& pSrc, TCoeff &uiAbsSum, const QpParam& cQP, const Ctx& ctx)
614 {
615  const SPS &sps = *tu.cs->sps;
616  const CompArea& rect = tu.blocks[compID];
617  const uint32_t uiWidth = rect.width;
618  const uint32_t uiHeight = rect.height;
619  const int channelBitDepth = sps.bitDepths[toChannelType(compID)];
620 
621  const CCoeffBuf& piCoef = pSrc;
622  CoeffSigBuf piQCoef = tu.getCoeffs(compID);
623 
624  const bool useTransformSkip = tu.mtsIdx[compID] == MTS_SKIP;
625  const int maxLog2TrDynamicRange = sps.getMaxLog2TrDynamicRange(toChannelType(compID));
626 
627  {
628  CoeffCodingContext cctx(tu, compID, tu.cs->slice->signDataHidingEnabled);
629 
630  const TCoeff entropyCodingMinimum = -(1 << maxLog2TrDynamicRange);
631  const TCoeff entropyCodingMaximum = (1 << maxLog2TrDynamicRange) - 1;
632 
633  TCoeff deltaU[MAX_TB_SIZEY * MAX_TB_SIZEY];
634  int scalingListType = getScalingListType(tu.cu->predMode, compID);
635  CHECK(scalingListType >= SCALING_LIST_NUM, "Invalid scaling list");
636  const uint32_t uiLog2TrWidth = Log2(uiWidth);
637  const uint32_t uiLog2TrHeight = Log2(uiHeight);
638  int *piQuantCoeff = getQuantCoeff(scalingListType, cQP.rem(useTransformSkip), uiLog2TrWidth, uiLog2TrHeight);
639 
640  const bool isLfnstApplied = tu.cu->lfnstIdx > 0 && (CU::isSepTree(*tu.cu) ? true : isLuma(compID));
641  const bool enableScalingLists = getUseScalingList(uiWidth, uiHeight, useTransformSkip, isLfnstApplied);
642 
643  // for blocks that where width*height != 4^N, the effective scaling applied during transformation cannot be
644  // compensated by a bit-shift (the quantised result will be sqrt(2) * larger than required).
645  // The quantScale table and shift is used to compensate for this.
646  const bool needSqrtAdjustment= TU::needsSqrt2Scale( tu, compID );
647  const int defaultQuantisationCoefficient = g_quantScales[needSqrtAdjustment?1:0][cQP.rem(useTransformSkip)];
648  int iTransformShift = getTransformShift(channelBitDepth, rect.size(), maxLog2TrDynamicRange) + ( needSqrtAdjustment?-1:0);
649 
650  if (useTransformSkip && sps.spsRExt.extendedPrecisionProcessing)
651  {
652  iTransformShift = std::max<int>(0, iTransformShift);
653  }
654 
655  const int iQBits = QUANT_SHIFT + cQP.per(useTransformSkip) + (useTransformSkip ? 0 : iTransformShift);
656  // QBits will be OK for any internal bit depth as the reduction in transform shift is balanced by an increase in Qp_per due to QpBDOffset
657 
658  const int64_t iAdd = int64_t(tu.cs->slice->isIRAP() ? 171 : 85) << int64_t(iQBits - 9);
659  const int qBits8 = iQBits - 8;
660 
661  const uint32_t lfnstIdx = tu.cu->lfnstIdx;
662  const int maxNumberOfCoeffs = lfnstIdx > 0 ? ((( uiWidth == 4 && uiHeight == 4 ) || ( uiWidth == 8 && uiHeight == 8) ) ? 8 : 16) : piQCoef.area();
663  piQCoef.memset( 0 );
664  for (int uiBlockPos = 0; uiBlockPos < maxNumberOfCoeffs; uiBlockPos++ )
665  {
666  const TCoeff iLevel = piCoef.buf[uiBlockPos];
667  const TCoeff iSign = (iLevel < 0 ? -1: 1);
668 
669  const int64_t tmpLevel = (int64_t)abs(iLevel) * (enableScalingLists ? piQuantCoeff[uiBlockPos] : defaultQuantisationCoefficient);
670 
671  const TCoeff quantisedMagnitude = TCoeff((tmpLevel + iAdd ) >> iQBits);
672  deltaU[uiBlockPos] = (TCoeff)((tmpLevel - ((int64_t)quantisedMagnitude<<iQBits) )>> qBits8);
673 
674  uiAbsSum += quantisedMagnitude;
675  const TCoeff quantisedCoefficient = quantisedMagnitude * iSign;
676 
677  piQCoef.buf[uiBlockPos] = Clip3<TCoeff>( entropyCodingMinimum, entropyCodingMaximum, quantisedCoefficient );
678  } // for n
679  if (tu.cu->bdpcmM[toChannelType(compID)])
680  {
681  fwdResDPCM( tu, compID );
682  }
683  if( cctx.signHiding() )
684  {
685  if(uiAbsSum >= 2) //this prevents TUs with only one coefficient of value 1 from being tested
686  {
687  xSignBitHidingHDQ(piQCoef.buf, piCoef.buf, deltaU, cctx, maxLog2TrDynamicRange);
688  }
689  }
690  int lastScanPos = -1;
691  for( int scanPos = 0; scanPos < cctx.maxNumCoeff(); scanPos++)
692  {
693  unsigned blkPos = cctx.blockPos( scanPos );
694  if( piQCoef.buf[blkPos] )
695  {
696  lastScanPos = scanPos;
697  }
698  }
699  tu.lastPos[compID] = lastScanPos;
700  } //if RDOQ
701  //return;
702 }

704 bool Quant::xNeedRDOQ(TransformUnit& tu, const ComponentID compID, const CCoeffBuf& pSrc, const QpParam& cQP)
705 {
706  const SPS &sps = *tu.cs->sps;
707  const CompArea& rect = tu.blocks[compID];
708  const uint32_t uiWidth = rect.width;
709  const uint32_t uiHeight = rect.height;
710  const int channelBitDepth = sps.bitDepths[toChannelType(compID)];
711 
712  const CCoeffBuf piCoef = pSrc;
713 
714  const bool useTransformSkip = tu.mtsIdx[compID] == MTS_SKIP;
715  const int maxLog2TrDynamicRange = sps.getMaxLog2TrDynamicRange(toChannelType(compID));
716 
717  int scalingListType = getScalingListType(tu.cu->predMode, compID);
718  CHECK(scalingListType >= SCALING_LIST_NUM, "Invalid scaling list");
719 
720  const uint32_t uiLog2TrWidth = Log2(uiWidth);
721  const uint32_t uiLog2TrHeight = Log2(uiHeight);
722  int *piQuantCoeff = getQuantCoeff(scalingListType, cQP.rem(useTransformSkip), uiLog2TrWidth, uiLog2TrHeight);
723 
724  const bool isLfnstApplied = tu.cu->lfnstIdx > 0 && (CU::isSepTree(*tu.cu) ? true : isLuma(compID));
725  const bool enableScalingLists = getUseScalingList(uiWidth, uiHeight, (useTransformSkip != 0), isLfnstApplied);
726 
727  /* for 422 chroma blocks, the effective scaling applied during transformation is not a power of 2, hence it cannot be
728  * implemented as a bit-shift (the quantised result will be sqrt(2) * larger than required). Alternatively, adjust the
729  * uiLog2TrSize applied in iTransformShift, such that the result is 1/sqrt(2) the required result (i.e. smaller)
730  * Then a QP+3 (sqrt(2)) or QP-3 (1/sqrt(2)) method could be used to get the required result
731  */
732  const bool needSqrtAdjustment= TU::needsSqrt2Scale( tu, compID );
733  const int defaultQuantisationCoefficient = g_quantScales[needSqrtAdjustment?1:0][cQP.rem(useTransformSkip)];
734  int iTransformShift = getTransformShift(channelBitDepth, rect.size(), maxLog2TrDynamicRange) + (needSqrtAdjustment?-1:0);
735 
736  if (useTransformSkip && sps.spsRExt.extendedPrecisionProcessing)
737  {
738  iTransformShift = std::max<int>(0, iTransformShift);
739  }
740 
741 
742  const int iQBits = QUANT_SHIFT + cQP.per(useTransformSkip) + iTransformShift;
743  assert(iQBits>=0);
744  // QBits will be OK for any internal bit depth as the reduction in transform shift is balanced by an increase in Qp_per due to QpBDOffset
745 
746  // iAdd is different from the iAdd used in normal quantization
747  const int64_t iAdd = int64_t(compID == COMP_Y ? 171 : 256) << (iQBits - 9);
748 
749  for (int uiBlockPos = 0; uiBlockPos < rect.area(); uiBlockPos++)
750  {
751  const TCoeff iLevel = piCoef.buf[uiBlockPos];
752  const int64_t tmpLevel = (int64_t)abs(iLevel) * (enableScalingLists ? piQuantCoeff[uiBlockPos] : defaultQuantisationCoefficient);
753  const TCoeff quantisedMagnitude = TCoeff((tmpLevel + iAdd ) >> iQBits);
754 
755  if (quantisedMagnitude != 0)
756  {
757  return true;
758  }
759  } // for n
760  return false;
761 }
