68 void offsetBlock_core(const int channelBitDepth, const ClpRng& clpRng, int typeIdx, int* offset, int startIdx,
69  const Pel* srcBlk, Pel* resBlk, ptrdiff_t srcStride, ptrdiff_t resStride, int width, int height,
70  uint8_t availMask, std::vector<int8_t> &signLineBuf1, std::vector<int8_t> &signLineBuf2)
71 {
72  int x, y, startX, startY, endX, endY, edgeType;
73  int firstLineStartX, firstLineEndX, lastLineStartX, lastLineEndX;
74  int8_t signLeft, signRight, signDown;
75 
76  const Pel* srcLine = srcBlk;
77  Pel* resLine = resBlk;
78 
79  switch (typeIdx)
80  {
81  case SAO_TYPE_EO_0:
82  {
83  offset += 2;
84  startX = availMask&LeftAvail ? 0 : 1;
85  endX = availMask&RightAvail ? width : (width - 1);
86  for (y = 0; y < height; y++)
87  {
88  signLeft = (int8_t)sgn(srcLine[startX] - srcLine[startX - 1]);
89  for (x = startX; x < endX; x++)
90  {
91  signRight = (int8_t)sgn(srcLine[x] - srcLine[x + 1]);
92  edgeType = signRight + signLeft;
93  signLeft = -signRight;
94 
95  resLine[x] = ClipPel<int>(srcLine[x] + offset[edgeType], clpRng);
96  }
97  srcLine += srcStride;
98  resLine += resStride;
99  }
100 
101  }
102  break;
103  case SAO_TYPE_EO_90:
104  {
105  offset += 2;
106  int8_t *signUpLine = &signLineBuf1[0];
107 
108  startY = availMask&AboveAvail ? 0 : 1;
109  endY = availMask&BelowAvail ? height : height - 1;
110  if (!(availMask&AboveAvail))
111  {
112  srcLine += srcStride;
113  resLine += resStride;
114  }
115 
116  const Pel* srcLineAbove = srcLine - srcStride;
117  for (x = 0; x < width; x++)
118  {
119  signUpLine[x] = (int8_t)sgn(srcLine[x] - srcLineAbove[x]);
120  }
121 
122  const Pel* srcLineBelow;
123  for (y = startY; y < endY; y++)
124  {
125  srcLineBelow = srcLine + srcStride;
126 
127  for (x = 0; x < width; x++)
128  {
129  signDown = (int8_t)sgn(srcLine[x] - srcLineBelow[x]);
130  edgeType = signDown + signUpLine[x];
131  signUpLine[x] = -signDown;
132 
133  resLine[x] = ClipPel<int>(srcLine[x] + offset[edgeType], clpRng);
134  }
135  srcLine += srcStride;
136  resLine += resStride;
137  }
138 
139  }
140  break;
141  case SAO_TYPE_EO_135:
142  {
143  offset += 2;
144  int8_t *signUpLine, *signDownLine, *signTmpLine;
145 
146  signUpLine = &signLineBuf1[0];
147  signDownLine = &signLineBuf2[0];
148 
149  startX = availMask&LeftAvail ? 0 : 1;
150  endX = availMask&RightAvail ? width : (width - 1);
151 
152  //prepare 2nd line's upper sign
153  const Pel* srcLineBelow = srcLine + srcStride;
154  for (x = startX; x < endX + 1; x++)
155  {
156  signUpLine[x] = (int8_t)sgn(srcLineBelow[x] - srcLine[x - 1]);
157  }
158 
159  //1st line
160  const Pel* srcLineAbove = srcLine - srcStride;
161  firstLineStartX = availMask&AboveLeftAvail ? 0 : 1;
162  firstLineEndX = availMask&AboveAvail ? endX : 1;
163  for (x = firstLineStartX; x < firstLineEndX; x++)
164  {
165  edgeType = sgn(srcLine[x] - srcLineAbove[x - 1]) - signUpLine[x + 1];
166 
167  resLine[x] = ClipPel<int>(srcLine[x] + offset[edgeType], clpRng);
168  }
169  srcLine += srcStride;
170  resLine += resStride;
171 
172 
173  //middle lines
174  for (y = 1; y < height - 1; y++)
175  {
176  srcLineBelow = srcLine + srcStride;
177 
178  for (x = startX; x < endX; x++)
179  {
180  signDown = (int8_t)sgn(srcLine[x] - srcLineBelow[x + 1]);
181  edgeType = signDown + signUpLine[x];
182  resLine[x] = ClipPel<int>(srcLine[x] + offset[edgeType], clpRng);
183 
184  signDownLine[x + 1] = -signDown;
185  }
186  signDownLine[startX] = (int8_t)sgn(srcLineBelow[startX] - srcLine[startX - 1]);
187 
188  signTmpLine = signUpLine;
189  signUpLine = signDownLine;
190  signDownLine = signTmpLine;
191 
192  srcLine += srcStride;
193  resLine += resStride;
194  }
195 
196  //last line
197  srcLineBelow = srcLine + srcStride;
198  lastLineStartX = availMask&BelowAvail ? startX : (width - 1);
199  lastLineEndX = availMask&BelowRightAvail ? width : (width - 1);
200  for (x = lastLineStartX; x < lastLineEndX; x++)
201  {
202  edgeType = sgn(srcLine[x] - srcLineBelow[x + 1]) + signUpLine[x];
203  resLine[x] = ClipPel<int>(srcLine[x] + offset[edgeType], clpRng);
204 
205  }
206  }
207  break;
208  case SAO_TYPE_EO_45:
209  {
210  offset += 2;
211  int8_t *signUpLine = &signLineBuf1[1];
212 
213  startX = availMask&LeftAvail ? 0 : 1;
214  endX = availMask&RightAvail ? width : (width - 1);
215 
216  //prepare 2nd line upper sign
217  const Pel* srcLineBelow = srcLine + srcStride;
218  for (x = startX - 1; x < endX; x++)
219  {
220  signUpLine[x] = (int8_t)sgn(srcLineBelow[x] - srcLine[x + 1]);
221  }
222 
223 
224  //first line
225  const Pel* srcLineAbove = srcLine - srcStride;
226  firstLineStartX = availMask&AboveAvail ? startX : (width - 1);
227  firstLineEndX = availMask&AboveRightAvail ? width : (width - 1);
228  for (x = firstLineStartX; x < firstLineEndX; x++)
229  {
230  edgeType = sgn(srcLine[x] - srcLineAbove[x + 1]) - signUpLine[x - 1];
231  resLine[x] = ClipPel<int>(srcLine[x] + offset[edgeType], clpRng);
232  }
233  srcLine += srcStride;
234  resLine += resStride;
235 
236  //middle lines
237  for (y = 1; y < height - 1; y++)
238  {
239  srcLineBelow = srcLine + srcStride;
240 
241  for (x = startX; x < endX; x++)
242  {
243  signDown = (int8_t)sgn(srcLine[x] - srcLineBelow[x - 1]);
244  edgeType = signDown + signUpLine[x];
245  resLine[x] = ClipPel<int>(srcLine[x] + offset[edgeType], clpRng);
246  signUpLine[x - 1] = -signDown;
247  }
248  signUpLine[endX - 1] = (int8_t)sgn(srcLineBelow[endX - 1] - srcLine[endX]);
249  srcLine += srcStride;
250  resLine += resStride;
251  }
252 
253  //last line
254  srcLineBelow = srcLine + srcStride;
255  lastLineStartX = availMask&BelowLeftAvail ? 0 : 1;
256  lastLineEndX = availMask&BelowAvail ? endX : 1;
257  for (x = lastLineStartX; x < lastLineEndX; x++)
258  {
259  edgeType = sgn(srcLine[x] - srcLineBelow[x - 1]) + signUpLine[x];
260  resLine[x] = ClipPel<int>(srcLine[x] + offset[edgeType], clpRng);
261 
262  }
263  }
264  break;
265  case SAO_TYPE_BO:
266  {
267  const int shiftBits = channelBitDepth - NUM_SAO_BO_CLASSES_LOG2;
268  for (y = 0; y < height; y++)
269  {
270  for (x = 0; x < width; x++)
271  {
272  resLine[x] = ClipPel<int>(srcLine[x] + offset[srcLine[x] >> shiftBits], clpRng);
273  }
274  srcLine += srcStride;
275  resLine += resStride;
276  }
277  }
278  break;
279  default:
280  {
281  THROW("Not a supported SAO types\n");
282  }
283  }

337 void SampleAdaptiveOffset::invertQuantOffsets(ComponentID compIdx, int typeIdc, int typeAuxInfo, int* dstOffsets, int* srcOffsets)
338 {
339  int codedOffset[MAX_NUM_SAO_CLASSES];
340 
341  ::memcpy(codedOffset, srcOffsets, sizeof(int)*MAX_NUM_SAO_CLASSES);
342  ::memset(dstOffsets, 0, sizeof(int)*MAX_NUM_SAO_CLASSES);
343 
344  if(typeIdc == SAO_TYPE_START_BO)
345  {
346  for(int i=0; i< 4; i++)
347  {
348  dstOffsets[(typeAuxInfo+ i)%NUM_SAO_BO_CLASSES] = codedOffset[(typeAuxInfo+ i)%NUM_SAO_BO_CLASSES]*(1<<m_offsetStepLog2[compIdx]);
349  }
350  }
351  else //EO
352  {
353  for(int i=0; i< NUM_SAO_EO_CLASSES; i++)
354  {
355  dstOffsets[i] = codedOffset[i] *(1<<m_offsetStepLog2[compIdx]);
356  }
357  CHECK(dstOffsets[SAO_CLASS_EO_PLAIN] != 0, "EO offset is not '0'"); //keep EO plain offset as zero
358  }

362 int SampleAdaptiveOffset::getMergeList(CodingStructure& cs, int ctuRsAddr, SAOBlkParam* blkParams, SAOBlkParam* mergeList[NUM_SAO_MERGE_TYPES])
363 {
364  const PreCalcValues& pcv = *cs.pcv;
365 
366  int ctuX = ctuRsAddr % pcv.widthInCtus;
367  int ctuY = ctuRsAddr / pcv.widthInCtus;
368  const CodingUnit& cu = *cs.getCU(Position(ctuX*pcv.maxCUSize, ctuY*pcv.maxCUSize), CH_L, TREE_D);
369  int mergedCTUPos;
370  int numValidMergeCandidates = 0;
371 
372  for(int mergeType=0; mergeType< NUM_SAO_MERGE_TYPES; mergeType++)
373  {
374  SAOBlkParam* mergeCandidate = NULL;
375 
376  switch(mergeType)
377  {
378  case SAO_MERGE_ABOVE:
379  {
380  if(ctuY > 0)
381  {
382  mergedCTUPos = ctuRsAddr- pcv.widthInCtus;
383  if(cs.getCURestricted(Position(ctuX*pcv.maxCUSize, (ctuY-1)*pcv.maxCUSize), cu, cu.chType))
384  {
385  mergeCandidate = &(blkParams[mergedCTUPos]);
386  }
387  }
388  }
389  break;
390  case SAO_MERGE_LEFT:
391  {
392  if(ctuX > 0)
393  {
394  mergedCTUPos = ctuRsAddr- 1;
395  if(cs.getCURestricted(Position((ctuX-1)*pcv.maxCUSize, ctuY*pcv.maxCUSize), cu, cu.chType))
396  {
397  mergeCandidate = &(blkParams[mergedCTUPos]);
398  }
399  }
400  }
401  break;
402  default:
403  {
404  THROW("not a supported merge type");
405  }
406  }
407 
408  mergeList[mergeType]=mergeCandidate;
409  if (mergeCandidate != NULL)
410  {
411  numValidMergeCandidates++;
412  }
413  }
414 
415  return numValidMergeCandidates;
416 }

419 void SampleAdaptiveOffset::reconstructBlkSAOParam(SAOBlkParam& recParam, SAOBlkParam* mergeList[NUM_SAO_MERGE_TYPES])
420 {
421  const int numberOfComponents = m_numberOfComponents;
422  for(int compIdx = 0; compIdx < numberOfComponents; compIdx++)
423  {
424  const ComponentID component = ComponentID(compIdx);
425  SAOOffset& offsetParam = recParam[component];
426 
427  if(offsetParam.modeIdc == SAO_MODE_OFF)
428  {
429  continue;
430  }
431 
432  switch(offsetParam.modeIdc)
433  {
434  case SAO_MODE_NEW:
435  {
436  invertQuantOffsets(component, offsetParam.typeIdc, offsetParam.typeAuxInfo, offsetParam.offset, offsetParam.offset);
437  }
438  break;
439  case SAO_MODE_MERGE:
440  {
441  SAOBlkParam* mergeTarget = mergeList[offsetParam.typeIdc];
442  CHECK(mergeTarget == NULL, "Merge target does not exist");
443 
444  offsetParam = (*mergeTarget)[component];
445  }
446  break;
447  default:
448  {
449  THROW("Not a supported mode");
450  }
451  }
452  }
453 }

481 void SampleAdaptiveOffset::offsetCTU( const UnitArea& area, const CPelUnitBuf& src, PelUnitBuf& res, SAOBlkParam& saoblkParam, CodingStructure& cs)
482 {
483  const uint32_t numberOfComponents = getNumberValidComponents( area.chromaFormat );
484  bool bAllOff=true;
485  for( uint32_t compIdx = 0; compIdx < numberOfComponents; compIdx++)
486  {
487  if (saoblkParam[compIdx].modeIdc != SAO_MODE_OFF)
488  {
489  bAllOff=false;
490  }
491  }
492  if (bAllOff)
493  {
494  return;
495  }
496 
497  uint8_t availMask;
498  //block boundary availability
499  deriveLoopFilterBoundaryAvailibility(cs, area.Y(), availMask);
500 
501  const size_t lineBufferSize = area.Y().width + 1;
502  if (m_signLineBuf1.size() < lineBufferSize)
503  {
504  m_signLineBuf1.resize(lineBufferSize);
505  m_signLineBuf2.resize(lineBufferSize);
506  }
507 
508  for(int compIdx = 0; compIdx < numberOfComponents; compIdx++)
509  {
510  const ComponentID compID = ComponentID(compIdx);
511  const CompArea& compArea = area.block(compID);
512  SAOOffset& ctbOffset = saoblkParam[compIdx];
513 
514  if(ctbOffset.modeIdc != SAO_MODE_OFF)
515  {
516  int srcStride = src.get(compID).stride;
517  const Pel* srcBlk = src.get(compID).bufAt(compArea);
518  int resStride = res.get(compID).stride;
519  Pel* resBlk = res.get(compID).bufAt(compArea);
520 
521  offsetBlock( cs.sps->bitDepths[toChannelType(compID)],
522  cs.slice->clpRngs[compID],
523  ctbOffset.typeIdc, ctbOffset.offset, ctbOffset.typeAuxInfo
524  , srcBlk, resBlk, srcStride, resStride, compArea.width, compArea.height, availMask
525  , m_signLineBuf1, m_signLineBuf2 );
526  }
527  } //compIdx
528 }

530 void SampleAdaptiveOffset::SAOProcess( CodingStructure& cs, SAOBlkParam* saoBlkParams )
531 {
532  CHECK(!saoBlkParams, "No parameters present");
533 
534  xReconstructBlkSAOParams(cs, saoBlkParams);
535 
536  const uint32_t numberOfComponents = getNumberValidComponents(cs.area.chromaFormat);
537  bool bAllDisabled = true;
538  for (uint32_t compIdx = 0; compIdx < numberOfComponents; compIdx++)
539  {
540  if (m_picSAOEnabled[compIdx])
541  {
542  bAllDisabled = false;
543  }
544  }
545  if (bAllDisabled)
546  {
547  return;
548  }
549 
550  const PreCalcValues& pcv = *cs.pcv;
551  Picture& pic = *cs.picture;
552  PelUnitBuf recBuf = pic.getRecoBuf();
553  PelUnitBuf saoBuf = pic.getSaoBuf();
554  saoBuf.copyFrom( recBuf );
555 
556  int ctuRsAddr = 0;
557  for( uint32_t yPos = 0; yPos < pcv.lumaHeight; yPos += pcv.maxCUSize )
558  {
559  const uint32_t height = (yPos + pcv.maxCUSize > pcv.lumaHeight) ? (pcv.lumaHeight - yPos) : pcv.maxCUSize;
560  for( uint32_t xPos = 0; xPos < pcv.lumaWidth; xPos += pcv.maxCUSize )
561  {
562  const uint32_t width = (xPos + pcv.maxCUSize > pcv.lumaWidth) ? (pcv.lumaWidth - xPos) : pcv.maxCUSize;
563  const UnitArea area( cs.area.chromaFormat, Area(xPos , yPos, width, height) );
564 
565  offsetCTU( area, saoBuf, recBuf, cs.picture->getSAO()[ctuRsAddr], cs);
566  ctuRsAddr++;
567  }
568  }
569 
570  DTRACE_PIC_COMP(D_REC_CB_LUMA_SAO, cs, cs.getRecoBuf(), COMP_Y);
571  DTRACE_PIC_COMP(D_REC_CB_CHROMA_SAO, cs, cs.getRecoBuf(), COMP_Cb);
572  DTRACE_PIC_COMP(D_REC_CB_CHROMA_SAO, cs, cs.getRecoBuf(), COMP_Cr);
573 }

575 void SampleAdaptiveOffset::deriveLoopFilterBoundaryAvailibility(CodingStructure& cs, const Position& pos, uint8_t& availMask ) const
576 {
577  const int cuSize = cs.pcv->maxCUSize;
578  const CodingUnit* cuCurr = cs.getCU(pos, CH_L, TREE_D);
579  const CodingUnit* cuLeft = cs.getCU(pos.offset(-cuSize, 0), CH_L, TREE_D);
580  const CodingUnit* cuRight = cs.getCU(pos.offset(cuSize, 0), CH_L, TREE_D);
581  const CodingUnit* cuAbove = cs.getCU(pos.offset(0, -cuSize), CH_L, TREE_D);
582  const CodingUnit* cuBelow = cs.getCU(pos.offset(0, cuSize), CH_L, TREE_D);
583  const CodingUnit* cuAboveLeft = cs.getCU(pos.offset(-cuSize, -cuSize), CH_L, TREE_D);
584  const CodingUnit* cuAboveRight = cs.getCU(pos.offset(cuSize, -cuSize), CH_L, TREE_D);
585  const CodingUnit* cuBelowLeft = cs.getCU(pos.offset(-cuSize, cuSize), CH_L, TREE_D);
586  const CodingUnit* cuBelowRight = cs.getCU(pos.offset(cuSize, cuSize), CH_L, TREE_D);
587  availMask = 0;
588 
589  // check cross slice flags
590  if( cs.pps->loopFilterAcrossSlicesEnabled )
591  {
592  availMask |= (cuLeft != NULL) ? LeftAvail : 0;
593  availMask |= (cuAbove != NULL) ? AboveAvail : 0;
594  availMask |= (cuRight != NULL) ? RightAvail : 0;
595  availMask |= (cuBelow != NULL) ? BelowAvail : 0;
596  availMask |= (cuAboveLeft != NULL) ? AboveLeftAvail : 0;
597  availMask |= (cuBelowRight != NULL) ? BelowRightAvail : 0;
598  availMask |= (cuAboveRight != NULL) ? AboveRightAvail : 0;
599  availMask |= (cuBelowLeft != NULL) ? BelowLeftAvail : 0;
600  }
601  else
602  {
603  availMask |= ((cuLeft != NULL) && CU::isSameSlice(*cuCurr, *cuLeft) ) ? LeftAvail : 0;
604  availMask |= ((cuAbove != NULL) && CU::isSameSlice(*cuCurr, *cuAbove) ) ? AboveAvail : 0;
605  availMask |= ((cuRight != NULL) && CU::isSameSlice(*cuCurr, *cuRight)) ? RightAvail : 0;
606  availMask |= ((cuBelow != NULL) && CU::isSameSlice(*cuCurr, *cuBelow) ) ? BelowAvail : 0;
607  availMask |= ((cuAboveLeft != NULL) && CU::isSameSlice(*cuCurr, *cuAboveLeft)) ? AboveLeftAvail : 0;
