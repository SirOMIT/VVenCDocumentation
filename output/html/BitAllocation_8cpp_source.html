68  return d < 1.5e-13 ? -128 : int (floor (3.0 * log (d) / log (2.0) + 0.5));

72 {
73  if (bitDepth > 16 || avgLumaValue >= (1u << bitDepth)) return 0;

83 static double filterAndCalculateAverageActivity (const Pel* pSrc, const int iSrcStride, const int height, const int width,
84  const Pel* pSM1, const int iSM1Stride, const Pel* pSM2, const int iSM2Stride,
85  uint32_t frameRate, const uint32_t bitDepth, const bool isUHD)
86 {
87  double meanAct = 0.0;
88  uint64_t saAct = 0; // spatial absolute activity sum
89  uint64_t taAct = 0; // temporal absolute activity sum
90  const Pel* pS0 = pSrc;
91 
92  if (pSrc == nullptr || iSrcStride <= 0) return 0.0;
93  // force 1st-order delta if only prev. frame available
94  if (pSM2 == nullptr || iSM2Stride <= 0) frameRate = 24;
95 
96  // skip first row as there may be a black border frame
97  pSrc += iSrcStride;
98  // center rows
99  if (isUHD) // high-pass with downsampling
100  {
101  const int i2ndStride = iSrcStride * 2;
102  const int i3rdStride = iSrcStride * 3;
103 
104  pSrc += iSrcStride;
105  for (int y = 2; y < height - 2; y += 2)
106  {
107  for (int x = 2; x < width - 2; x += 2) // cnt cols
108  {
109  const int s = 12 * ((int) pSrc[x ] + (int) pSrc[x+1 ] + (int) pSrc[x +iSrcStride] + (int) pSrc[x+1+iSrcStride])
110  - 3 * ((int) pSrc[x-1 ] + (int) pSrc[x+2 ] + (int) pSrc[x-1+iSrcStride] + (int) pSrc[x+2+iSrcStride]
111  + (int) pSrc[x -iSrcStride] + (int) pSrc[x+1-iSrcStride] + (int) pSrc[x +i2ndStride] + (int) pSrc[x+1+i2ndStride])
112  - 2 * ((int) pSrc[x-1-iSrcStride] + (int) pSrc[x+2-iSrcStride] + (int) pSrc[x-1+i2ndStride] + (int) pSrc[x+2+i2ndStride])
113  - ((int) pSrc[x-1-i2ndStride] + (int) pSrc[x -i2ndStride] + (int) pSrc[x+1-i2ndStride] + (int) pSrc[x+2-i2ndStride]
114  + (int) pSrc[x-1+i3rdStride] + (int) pSrc[x +i3rdStride] + (int) pSrc[x+1+i3rdStride] + (int) pSrc[x+2+i3rdStride]
115  + (int) pSrc[x-2-iSrcStride] + (int) pSrc[x-2 ] + (int) pSrc[x-2+iSrcStride] + (int) pSrc[x-2+i2ndStride]
116  + (int) pSrc[x+3-iSrcStride] + (int) pSrc[x+3 ] + (int) pSrc[x+3+iSrcStride] + (int) pSrc[x+3+i2ndStride]);
117  saAct += abs (s);
118  }
119  pSrc += i2ndStride;
120  }
121 
122  meanAct = double (saAct) / double ((width - 4) * (height - 4));
123  }
124  else // HD high-pass without downsampling
125  {
126  for (int y = 1; y < height - 1; y++)
127  {
128  for (int x = 1; x < width - 1; x++) // center cols
129  {
130  const int s = 12 * (int) pSrc[x ] - 2 * ((int) pSrc[x-1] + (int) pSrc[x+1] + (int) pSrc[x -iSrcStride] + (int) pSrc[x +iSrcStride])
131  - ((int) pSrc[x-1-iSrcStride] + (int) pSrc[x+1-iSrcStride] + (int) pSrc[x-1+iSrcStride] + (int) pSrc[x+1+iSrcStride]);
132  saAct += abs (s);
133  }
134  pSrc += iSrcStride;
135  }
136 
137  meanAct = double (saAct) / double ((width - 2) * (height - 2));
138  }
139 
140  // skip first row as there may be a black border frame
141  pSrc = pS0 + iSrcStride;
142  // center rows
143  if (isUHD) // high-pass with downsampling
144  {
145  const int i2M0Stride = iSrcStride * 2;
146  const int i2M1Stride = iSM1Stride * 2;
147 
148  CHECK (pSM1 == nullptr || iSM1Stride <= 0 || iSM1Stride < width, "Pel buffer pointer pSM1 must not be null!");
149 
150  pSrc += iSrcStride;
151  pSM1 += i2M1Stride;
152  if (frameRate <= 31) // 1st-order delta
153  {
154  for (int y = 2; y < height - 2; y += 2)
155  {
156  for (int x = 2; x < width - 2; x += 2) // c cols
157  {
158  const int t = (int) pSrc[x] + (int) pSrc[x+1] + (int) pSrc[x+iSrcStride] + (int) pSrc[x+1+iSrcStride]
159  - ((int) pSM1[x] + (int) pSM1[x+1] + (int) pSM1[x+iSM1Stride] + (int) pSM1[x+1+iSM1Stride]);
160  taAct += (1 + 3 * abs (t)) >> 1;
161  }
162  pSrc += i2M0Stride;
163  pSM1 += i2M1Stride;
164  }
165  }
166  else // 2nd-order delta (diff of diffs)
167  {
168  const int i2M2Stride = iSM2Stride * 2;
169 
170  CHECK (pSM2 == nullptr || iSM2Stride <= 0 || iSM2Stride < width, "Pel buffer pointer pSM2 must not be null!");
171 
172  pSM2 += i2M2Stride;
173  for (int y = 2; y < height - 2; y += 2)
174  {
175  for (int x = 2; x < width - 2; x += 2) // c cols
176  {
177  const int t = (int) pSrc[x] + (int) pSrc[x+1] + (int) pSrc[x+iSrcStride] + (int) pSrc[x+1+iSrcStride]
178  - 2 * ((int) pSM1[x] + (int) pSM1[x+1] + (int) pSM1[x+iSM1Stride] + (int) pSM1[x+1+iSM1Stride])
179  + (int) pSM2[x] + (int) pSM2[x+1] + (int) pSM2[x+iSM2Stride] + (int) pSM2[x+1+iSM2Stride];
180  taAct += abs (t);
181  }
182  pSrc += i2M0Stride;
183  pSM1 += i2M1Stride;
184  pSM2 += i2M2Stride;
185  }
186  }
187 
188  meanAct += (2.0 * taAct) / double ((width - 4) * (height - 4));
189  }
190  else // HD high-pass without downsampling
191  {
192  CHECK (pSM1 == nullptr || iSM1Stride <= 0 || iSM1Stride < width, "Pel buffer pointer pSM1 must not be null!");
193 
194  pSM1 += iSM1Stride;
195  if (frameRate <= 31) // 1st-order delta
196  {
197  for (int y = 1; y < height - 1; y++)
198  {
199  for (int x = 1; x < width - 1; x++) // cnt cols
200  {
201  const int t = (int) pSrc[x] - (int) pSM1[x];
202 
203  taAct += (1 + 3 * abs (t)) >> 1;
204  }
205  pSrc += iSrcStride;
206  pSM1 += iSM1Stride;
207  }
208  }
209  else // 2nd-order delta (diff of diffs)
210  {
211  CHECK (pSM2 == nullptr || iSM2Stride <= 0 || iSM2Stride < width, "Pel buffer pointer pSM2 must not be null!");
212 
213  pSM2 += iSM2Stride;
214  for (int y = 1; y < height - 1; y++)
215  {
216  for (int x = 1; x < width - 1; x++) // cnt cols
217  {
218  const int t = (int) pSrc[x] - 2 * (int) pSM1[x] + (int) pSM2[x];
219 
220  taAct += abs (t);
221  }
222  pSrc += iSrcStride;
223  pSM1 += iSM1Stride;
224  pSM2 += iSM2Stride;
225  }
226  }
227 
228  meanAct += (2.0 * taAct) / double ((width - 2) * (height - 2));
229  }
230 
231  // lower limit, compensate for high-pass amplification
232  return std::max (meanAct, double (1 << (bitDepth - 6)));
233 }

235 static double getAveragePictureActivity (const uint32_t picWidth, const uint32_t picHeight,
236  const int scaledAverageGopActivity,
237  const bool tempFiltering, const uint32_t bitDepth)
238 {
239  if (scaledAverageGopActivity > 0)
240  {
241  return (double (scaledAverageGopActivity) / double (1 << (24 - bitDepth)));
242  }
243  const double hpEnerPic = (tempFiltering ? 32.0 : 16.0) * double (1 << (2 * bitDepth - 10)) * sqrt ((3840.0 * 2160.0) / double (picWidth * picHeight));
244 
245  return sqrt (hpEnerPic); // square-root of a_pic value
246 }

248 static int getGlaringColorQPOffset (Picture* const pic, const int ctuAddr, const int bitDepth, uint32_t &avgLumaValue)
249 {
250  const PreCalcValues& pcv = *pic->cs->pcv;
251  const ChromaFormat chrFmt = pic->chromaFormat;
252  const SizeType chrWidth = pcv.maxCUSize >> getChannelTypeScaleX (CH_C, chrFmt);
253  const SizeType chrHeight = pcv.maxCUSize >> getChannelTypeScaleY (CH_C, chrFmt);
254  const unsigned w = pcv.widthInCtus;
255  const int midLevel = 1 << (bitDepth - 1);
256  int chrValue = MAX_INT;
257 
258  avgLumaValue = uint32_t ((ctuAddr >= 0) ? pic->ctuAdaptedQP[ctuAddr] : pic->getOrigBuf().Y().getAvg());
259 
260  for (uint32_t comp = COMP_Cb; comp < MAX_NUM_COMP; comp++)
261  {
262  const ComponentID compID = (ComponentID) comp;
263  int avgCompValue;
264 
265  if (ctuAddr >= 0) // chroma
266  {
267  const CompArea chrArea = clipArea (CompArea (compID, chrFmt, Area ((ctuAddr % w) * chrWidth, (ctuAddr / w) * chrHeight, chrWidth, chrHeight)), pic->block (compID));
268 
269  avgCompValue = pic->getOrigBuf (chrArea).getAvg();
270  }
271  else avgCompValue = pic->getOrigBuf (pic->block (compID)).getAvg();
272 
273  if (chrValue > avgCompValue) chrValue = avgCompValue; // minimum of the DC offsets
274  }
275  CHECK (chrValue < 0, "mean chroma value cannot be negative!");
276 
277  chrValue = (int) avgLumaValue - chrValue;
278 
279  if (chrValue > midLevel) return apprI3Log2 (double (chrValue * chrValue) / double (midLevel * midLevel));
280 
281  return 0;
282 }

285 {
286  const ChromaFormat chrFmt = pic->chromaFormat;
287  const SizeType chrWidth = lumaArea.width >> getChannelTypeScaleX (CH_C, chrFmt);

289  const PosType chrPosX = lumaArea.x >> getChannelTypeScaleX (CH_C, chrFmt);
290  const PosType chrPosY = lumaArea.y >> getChannelTypeScaleY (CH_C, chrFmt);
291  const int midLevel = 1 << (bitDepth - 1);
292  int chrValue = MAX_INT;
293 
294  avgLumaValue = pic->getOrigBuf (lumaArea).getAvg();
295 
296  for (uint32_t comp = COMP_Cb; comp < MAX_NUM_COMP; comp++)
297  {
298  const ComponentID compID = (ComponentID) comp;
299  const CompArea chrArea = clipArea (CompArea (compID, chrFmt, Area (chrPosX, chrPosY, chrWidth, chrHeight)), pic->block (compID));
300 
301  int avgCompValue = pic->getOrigBuf (chrArea).getAvg();
302 
303  if (chrValue > avgCompValue) chrValue = avgCompValue; // minimum of the DC offsets
304  }
305  CHECK (chrValue < 0, "mean chroma value cannot be negative!");
306 
307  chrValue = (int) avgLumaValue - chrValue;
308

316 int BitAllocation::applyQPAdaptationChroma (const Slice* slice, const VVEncCfg* encCfg, const int sliceQP,
317  std::vector<int>& ctuPumpRedQP, int optChromaQPOffset[2], double* picVisActY /*= nullptr*/)
318 {
319  Picture* const pic = (slice != nullptr ? slice->pic : nullptr);
320  double hpEner[MAX_NUM_COMP] = {0.0, 0.0, 0.0};
321  int savedLumaQP = -1;
322  uint32_t meanLuma = MAX_UINT;
323 
324  if (pic == nullptr || encCfg == nullptr || optChromaQPOffset == nullptr) return -1;
325 
326  const bool isHDR = encCfg->m_HdrMode != vvencHDRMode::VVENC_HDR_OFF;
327  const bool isHighResolution = (encCfg->m_PadSourceWidth > 2048 || encCfg->m_PadSourceHeight > 1280);
328  const int bitDepth = slice->sps->bitDepths[CH_L];
329 
330  optChromaQPOffset[0] = optChromaQPOffset[1] = 0;
331 
332  for (uint32_t comp = 0; comp < getNumberValidComponents (pic->chromaFormat); comp++)
333  {
334  const ComponentID compID = (ComponentID) comp;
335  const CPelBuf picOrig = pic->getOrigBuf (compID);
336  const CPelBuf picPrv1 = pic->getOrigBufPrev (compID, false);
337  const CPelBuf picPrv2 = pic->getOrigBufPrev (compID, true );
338 
339  hpEner[comp] = filterAndCalculateAverageActivity (picOrig.buf, picOrig.stride, picOrig.height, picOrig.width,
340  picPrv1.buf, picPrv1.stride, picPrv2.buf, picPrv2.stride, encCfg->m_FrameRate,
341  bitDepth, isHighResolution && (isLuma (compID) || pic->chromaFormat == CHROMA_444));
342  if (isChroma (compID))
343  {
344  const int adaptChromaQPOffset = 2.0 * hpEner[comp] <= hpEner[0] ? 0 : apprI3Log2 (2.0 * hpEner[comp] / hpEner[0]);
345 
346  if (savedLumaQP < 0)
347  {
348  int averageAdaptedLumaQP = Clip3 (0, MAX_QP, sliceQP + apprI3Log2 (hpEner[0] / getAveragePictureActivity (encCfg->m_PadSourceWidth, encCfg->m_PadSourceHeight,
349  encCfg->m_RCNumPasses == 2 ? 0 : ctuPumpRedQP.back(),
350  (encCfg->m_usePerceptQPATempFiltISlice || !slice->isIntra()), bitDepth)));
351  if (isChromaEnabled (pic->chromaFormat) && (averageAdaptedLumaQP < MAX_QP))
352  {
353  averageAdaptedLumaQP += getGlaringColorQPOffset (pic, -1 /*ctuAddr*/, slice->sps->bitDepths[CH_C], meanLuma);
354 
355  if ((averageAdaptedLumaQP > MAX_QP) && !isHDR) averageAdaptedLumaQP = MAX_QP;
356  }
357  // change mean picture QP index based on picture's average luma value (Sharp)
358  if (isHDR)
359  {
360  if (meanLuma == MAX_UINT) meanLuma = pic->getOrigBuf().Y().getAvg();
361 
362  averageAdaptedLumaQP = Clip3 (0, MAX_QP, averageAdaptedLumaQP + lumaDQPOffset (meanLuma, bitDepth));
363  }
364 
365  savedLumaQP = averageAdaptedLumaQP;
366  } // savedLumaQP < 0
367 
368  GCC_WARNING_DISABLE_maybe_uninitialized // probably spurious warning, when building with -fsanitize=undefined: "error: 'encCfg.33' may be used uninitialized in this function"
369  const int lumaChromaMappingDQP = (savedLumaQP - slice->sps->chromaQpMappingTable.getMappedChromaQpValue (compID, savedLumaQP)) >> (encCfg->m_QP >= MAX_QP_PERCEPT_QPA ? 1 : 0);
370  GCC_WARNING_RESET
371 
372  optChromaQPOffset[comp-1] = std::min (3 + lumaChromaMappingDQP, adaptChromaQPOffset + lumaChromaMappingDQP);
373  } // isChroma (compID)
374  else if (picVisActY != nullptr) *picVisActY = hpEner[comp];
375  }
376 
377  return savedLumaQP;
378 }

380 int BitAllocation::applyQPAdaptationLuma (const Slice* slice, const VVEncCfg* encCfg, const int savedQP, const double lambda,
381  std::vector<int>& ctuPumpRedQP, std::vector<uint8_t>* ctuRCQPMemory,
382  const uint32_t ctuStartAddr, const uint32_t ctuBoundingAddr)
383 {
384  Picture* const pic = (slice != nullptr ? slice->pic : nullptr);
385  double hpEnerPic, hpEnerAvg = 0.0;
386  int adaptedSliceQP = -1;
387  uint32_t meanLuma = MAX_UINT;
388 
389  if (pic == nullptr || pic->cs == nullptr || encCfg == nullptr || ctuStartAddr >= ctuBoundingAddr) return -1;
390 
391  const bool isHDR = encCfg->m_HdrMode != vvencHDRMode::VVENC_HDR_OFF;
392  const bool isHighResolution = (encCfg->m_PadSourceWidth > 2048 || encCfg->m_PadSourceHeight > 1280);
393  const bool useFrameWiseQPA = (encCfg->m_QP > MAX_QP_PERCEPT_QPA);
394  const int bitDepth = slice->sps->bitDepths[CH_L];
395  const int sliceQP = (savedQP < 0 ? slice->sliceQp : savedQP);
396  const PreCalcValues& pcv = *pic->cs->pcv;
397 
398  if (!useFrameWiseQPA || (savedQP < 0)) // mean visual activity value and luma value in each CTU
399  {
400  const PosType guardSize = (isHighResolution ? 2 : 1);
401  for (uint32_t ctuTsAddr = ctuStartAddr; ctuTsAddr < ctuBoundingAddr; ctuTsAddr++)
402  {
403  const uint32_t ctuRsAddr = /*tileMap.getCtuBsToRsAddrMap*/ (ctuTsAddr);
404  const Position pos ((ctuRsAddr % pcv.widthInCtus) * pcv.maxCUSize, (ctuRsAddr / pcv.widthInCtus) * pcv.maxCUSize);
405  const CompArea ctuArea = clipArea (CompArea (COMP_Y, pic->chromaFormat, Area (pos.x, pos.y, pcv.maxCUSize, pcv.maxCUSize)), pic->Y());
406  const SizeType fltWidth = pcv.maxCUSize + guardSize * (pos.x > 0 ? 2 : 1);
407  const SizeType fltHeight = pcv.maxCUSize + guardSize * (pos.y > 0 ? 2 : 1);
408  const CompArea fltArea = clipArea (CompArea (COMP_Y, pic->chromaFormat, Area (pos.x > 0 ? pos.x - guardSize : 0, pos.y > 0 ? pos.y - guardSize : 0, fltWidth, fltHeight)), pic->Y());
409  const CPelBuf picOrig = pic->getOrigBuf (fltArea);
410  const CPelBuf picPrv1 = pic->getOrigBufPrev (fltArea, false);
411  const CPelBuf picPrv2 = pic->getOrigBufPrev (fltArea, true );
412 
413  hpEnerPic = filterAndCalculateAverageActivity (picOrig.buf, picOrig.stride, picOrig.height, picOrig.width,
414  picPrv1.buf, picPrv1.stride, picPrv2.buf, picPrv2.stride, encCfg->m_FrameRate,
415  bitDepth, isHighResolution);
416  hpEnerAvg += hpEnerPic;
417  pic->ctuQpaLambda[ctuRsAddr] = hpEnerPic; // temporary backup of CTU mean visual activity
418  pic->ctuAdaptedQP[ctuRsAddr] = pic->getOrigBuf (ctuArea).getAvg(); // and mean luma value
419  }
420 
421  hpEnerAvg /= double (ctuBoundingAddr - ctuStartAddr);
422  }
423 
424  hpEnerPic = 1.0 / getAveragePictureActivity (encCfg->m_PadSourceWidth, encCfg->m_PadSourceHeight, encCfg->m_RCNumPasses == 2 ? 0 : ctuPumpRedQP.back(),
425  (encCfg->m_usePerceptQPATempFiltISlice || !slice->isIntra()), bitDepth);
426 
427  if (useFrameWiseQPA || (sliceQP >= MAX_QP))
428  {
429  int averageAdaptedLumaQP = (savedQP < 0 ? Clip3 (0, MAX_QP, slice->sliceQp + apprI3Log2 (hpEnerAvg * hpEnerPic)) : savedQP);
430 
431  if (isChromaEnabled (pic->chromaFormat) && (averageAdaptedLumaQP < MAX_QP) && (savedQP < 0))
432  {
433  averageAdaptedLumaQP += getGlaringColorQPOffset (pic, -1 /*ctuAddr*/, slice->sps->bitDepths[CH_C], meanLuma);
434 
435  if ((averageAdaptedLumaQP > MAX_QP) && !isHDR) averageAdaptedLumaQP = MAX_QP;
436  }
437  // change mean picture QP index depending on picture's average luma value (Sharp)
438  if (isHDR && (savedQP < 0))
439  {
440  if (meanLuma == MAX_UINT) meanLuma = pic->getOrigBuf().Y().getAvg();
441 
442  averageAdaptedLumaQP = Clip3 (0, MAX_QP, averageAdaptedLumaQP + lumaDQPOffset (meanLuma, bitDepth));
443  }
444 
445  hpEnerAvg = lambda * pow (2.0, double (averageAdaptedLumaQP - slice->sliceQp) / 3.0);
446 
447  for (uint32_t ctuTsAddr = ctuStartAddr; ctuTsAddr < ctuBoundingAddr; ctuTsAddr++)
448  {
449  const uint32_t ctuRsAddr = /*tileMap.getCtuBsToRsAddrMap*/ (ctuTsAddr);
450 
451  pic->ctuQpaLambda[ctuRsAddr] = hpEnerAvg; // store adapted slice-level lambda
452  pic->ctuAdaptedQP[ctuRsAddr] = (Pel) averageAdaptedLumaQP; // store fixed QPs
453  }
454 
455  adaptedSliceQP = averageAdaptedLumaQP;
456  }
457  else // use CTU-wise QPA
458  {
459  const int nCtu = int (ctuBoundingAddr - ctuStartAddr);
460  const int dvsr = encCfg->m_IntraPeriod - encCfg->m_GOPSize;
461 
462  for (uint32_t ctuTsAddr = ctuStartAddr; ctuTsAddr < ctuBoundingAddr; ctuTsAddr++)
463  {
464  const uint32_t ctuRsAddr = /*tileMap.getCtuBsToRsAddrMap*/ (ctuTsAddr);
465  int adaptedLumaQP = Clip3 (0, MAX_QP, slice->sliceQp + apprI3Log2 (pic->ctuQpaLambda[ctuRsAddr]/*hpEner*/ * hpEnerPic));
466 
467  if ((encCfg->m_usePerceptQPATempFiltISlice == 2) && slice->isIntra() && (ctuPumpRedQP.size() > ctuRsAddr))
468  {
469  if (ctuRCQPMemory != nullptr) // save I-frame QP for second rate control pass
470  {
471  if (ctuRsAddr & 1) ctuRCQPMemory->back() |= (Clip3 (-8, 7, ctuPumpRedQP[ctuRsAddr]) + 8) << 4;
472  else /*even addr*/ ctuRCQPMemory->push_back (Clip3 (-8, 7, ctuPumpRedQP[ctuRsAddr]) + 8);
473  if (adaptedLumaQP > 0)
474  {
475  adaptedLumaQP--; // this is a first-pass tuning to stabilize rate control
476  }
477  }
478  if (ctuPumpRedQP[ctuRsAddr] < 0) adaptedLumaQP = Clip3 (0, MAX_QP, adaptedLumaQP + (ctuPumpRedQP[ctuRsAddr] * encCfg->m_GOPSize - (dvsr >> 1)) / dvsr);
479  else /*ctuPumpRedQP[addr] >= 0*/ adaptedLumaQP = Clip3 (0, MAX_QP, adaptedLumaQP + (ctuPumpRedQP[ctuRsAddr] * encCfg->m_GOPSize + (dvsr >> 1)) / dvsr);
480 
481  ctuPumpRedQP[ctuRsAddr] = 0; // reset QP memory for temporal pumping analysis
482  }
483  if ((encCfg->m_usePerceptQPATempFiltISlice == 2) && !slice->isIntra() && (slice->TLayer == 0) && (ctuRCQPMemory != nullptr) && (adaptedLumaQP < MAX_QP))
484  {
485  adaptedLumaQP++; // this is a first-pass tuning to stabilize the rate control
486  }
487  meanLuma = MAX_UINT;
488  if (isChromaEnabled (pic->chromaFormat) && (adaptedLumaQP < MAX_QP))
489  {
490  adaptedLumaQP += getGlaringColorQPOffset (pic, (int) ctuRsAddr, slice->sps->bitDepths[CH_C], meanLuma);
491 
492  if ((adaptedLumaQP > MAX_QP) && !isHDR) adaptedLumaQP = MAX_QP;
493  }
494  // change the CTU-level QP index based on CTU area's average luma value (Sharp)
495  if (isHDR)
496  {
497  if (meanLuma == MAX_UINT) meanLuma = pic->ctuAdaptedQP[ctuRsAddr];
498 
499  adaptedLumaQP = Clip3 (0, MAX_QP, adaptedLumaQP + lumaDQPOffset (meanLuma, bitDepth));
500  }
501  // reduce delta-QP variance, avoid wasting precious bit budget at low bit-rates
502  if ((3 + encCfg->m_QP > MAX_QP_PERCEPT_QPA) && (savedQP >= 0))
503  {
504  adaptedLumaQP = ((1 + MAX_QP_PERCEPT_QPA - encCfg->m_QP) * adaptedLumaQP + (3 + encCfg->m_QP - MAX_QP_PERCEPT_QPA) * sliceQP + 1) >> 2;
505  }
506 
507  hpEnerAvg = lambda * pow (2.0, double (adaptedLumaQP - slice->sliceQp) / 3.0);
508 
509  pic->ctuQpaLambda[ctuRsAddr] = hpEnerAvg; // store adapted CTU-level lambdas
510  pic->ctuAdaptedQP[ctuRsAddr] = (Pel) adaptedLumaQP; // store adapted CTU QPs
511 
512  adaptedSliceQP += adaptedLumaQP;
513 
514  // try to reduce local bitrate peaks using minimum smoothing of the adapted QPs
515  if ((pcv.widthInCtus > 1) && (encCfg->m_PadSourceWidth <= 640 && encCfg->m_PadSourceHeight <= 480))
516  {
517  const uint32_t ctuIdxX = ctuRsAddr % pcv.widthInCtus; // horizontal CTU index
518 
519  if (ctuIdxX == 0)
520  {
521  adaptedLumaQP = (ctuRsAddr > 1 ? pic->ctuAdaptedQP[ctuRsAddr - 2] : 0);
522  }
523  else // ctuIdxX > 0
524  {
525  adaptedLumaQP = (ctuIdxX > 1 ? std::min (pic->ctuAdaptedQP[ctuRsAddr - 2], pic->ctuAdaptedQP[ctuRsAddr]) : pic->ctuAdaptedQP[ctuRsAddr]);
526  }
527  if (ctuRsAddr > pcv.widthInCtus) // previous vertical
528  {
529  adaptedLumaQP = std::min (adaptedLumaQP, (int) pic->ctuAdaptedQP[ctuRsAddr - 1 - pcv.widthInCtus]);
530  }
531  if ((ctuRsAddr > 0) && (pic->ctuAdaptedQP[ctuRsAddr - 1] < (Pel) adaptedLumaQP))
532  {
533  hpEnerAvg = lambda * pow (2.0, double (adaptedLumaQP - slice->sliceQp) / 3.0);
534 
535  pic->ctuQpaLambda[ctuRsAddr - 1] = hpEnerAvg; // lambda after smoothing
536  pic->ctuAdaptedQP[ctuRsAddr - 1] = (Pel) adaptedLumaQP; // min. smoothing
537  }
538 
539  if ((ctuTsAddr == ctuBoundingAddr - 1) && (ctuRsAddr > pcv.widthInCtus)) // last CTU in the given slice
540  {
541  adaptedLumaQP = std::min (pic->ctuAdaptedQP[ctuRsAddr - 1], pic->ctuAdaptedQP[ctuRsAddr - pcv.widthInCtus]);
542 
543  if (pic->ctuAdaptedQP[ctuRsAddr] < (Pel) adaptedLumaQP)
544  {
545  hpEnerAvg = lambda * pow (2.0, double (adaptedLumaQP - slice->sliceQp) / 3.0);
546 
547  pic->ctuQpaLambda[ctuRsAddr] = hpEnerAvg;
548  pic->ctuAdaptedQP[ctuRsAddr] = (Pel) adaptedLumaQP;
549  }
550  }
551  }
552  } // end CTU-wise loop
553 
554  adaptedSliceQP = ((savedQP < 0 || ((encCfg->m_usePerceptQPATempFiltISlice == 2) && (slice->isIntra() || slice->TLayer == 0))) && (ctuBoundingAddr > ctuStartAddr)
555  ? (adaptedSliceQP < 0 ? adaptedSliceQP - ((nCtu + 1) >> 1) : adaptedSliceQP + ((nCtu + 1) >> 1)) / nCtu : sliceQP); // mean adapted luma QP
556  }
557 
558  return adaptedSliceQP;
559 }

561 int BitAllocation::applyQPAdaptationSubCtu (const Slice* slice, const VVEncCfg* encCfg, const Area& lumaArea)
562 {
563  Picture* const pic = (slice != nullptr ? slice->pic : nullptr);
564  double hpEnerPic, hpEnerSub = 0.0;
565  int adaptedSubCtuQP = -1;
566  uint32_t meanLuma = MAX_UINT;
567 
568  if (pic == nullptr || encCfg == nullptr) return -1;
569 
570  const bool isHDR = encCfg->m_HdrMode != vvencHDRMode::VVENC_HDR_OFF;
571  const bool isHighResolution = (encCfg->m_PadSourceWidth > 2048 || encCfg->m_PadSourceHeight > 1280);
572  const int bitDepth = slice->sps->bitDepths[CH_L];
573  const PosType guardSize = (isHighResolution ? 2 : 1);
574  const Position pos = lumaArea.pos();
575  const CompArea subArea = clipArea (CompArea (COMP_Y, pic->chromaFormat, Area (pos.x, pos.y, lumaArea.width, lumaArea.height)), pic->Y());
576  const SizeType fltWidth = lumaArea.width + guardSize * (pos.x > 0 ? 2 : 1);
577  const SizeType fltHeight = lumaArea.height + guardSize * (pos.y > 0 ? 2 : 1);
578  const CompArea fltArea = clipArea (CompArea (COMP_Y, pic->chromaFormat, Area (pos.x > 0 ? pos.x - guardSize : 0, pos.y > 0 ? pos.y - guardSize : 0, fltWidth, fltHeight)), pic->Y());
579  const CPelBuf picOrig = pic->getOrigBuf (fltArea);
580  const CPelBuf picPrv1 = pic->getOrigBufPrev (fltArea, false);
581  const CPelBuf picPrv2 = pic->getOrigBufPrev (fltArea, true );
582 
583  hpEnerSub = filterAndCalculateAverageActivity (picOrig.buf, picOrig.stride, picOrig.height, picOrig.width,
584  picPrv1.buf, picPrv1.stride, picPrv2.buf, picPrv2.stride, encCfg->m_FrameRate,
585  bitDepth, isHighResolution);
586  hpEnerPic = 1.0 / getAveragePictureActivity (encCfg->m_PadSourceWidth, encCfg->m_PadSourceHeight, 0,
587  (encCfg->m_usePerceptQPATempFiltISlice || !slice->isIntra()), bitDepth);
588  adaptedSubCtuQP = Clip3 (0, MAX_QP, pic->picInitialQP + apprI3Log2 (hpEnerSub * hpEnerPic));
589 
590  if (isChromaEnabled (pic->chromaFormat) && (adaptedSubCtuQP < MAX_QP))
591  {
592  adaptedSubCtuQP += getGlaringColorQPOffsetSubCtu (pic, subArea, slice->sps->bitDepths[CH_C], meanLuma);
593 
594  if ((adaptedSubCtuQP > MAX_QP) && !isHDR) adaptedSubCtuQP = MAX_QP;
595  }
596  // change the sub-CTU-level QP index based on sub-area's average luma value (Sharp)
597  if (isHDR)
598  {
599  if (meanLuma == MAX_UINT) meanLuma = pic->getOrigBuf (subArea).getAvg();
600 
601  adaptedSubCtuQP = Clip3 (0, MAX_QP, adaptedSubCtuQP + lumaDQPOffset (meanLuma, bitDepth));
602  }
603 
604  return adaptedSubCtuQP;
605 }

607 int BitAllocation::getCtuPumpingReducingQP (const Slice* slice, const CPelBuf& origY, const Distortion uiSadBestForQPA,
608  std::vector<int>& ctuPumpRedQP, const uint32_t ctuRsAddr, const int baseQP)
609 {
610  if (slice == nullptr || !slice->pps->useDQP || ctuPumpRedQP.size() <= ctuRsAddr) return 0;
611 
612  const int32_t avgOrig = origY.getAvg();
613  uint32_t sumAbsZmOrig = 0; // zero-mean
614  const Pel* src = origY.buf;
615 
616  for (SizeType y = 0; y < origY.height; y++) // sum up the zero-mean absolute values
617  {
618  for (SizeType x = 0; x < origY.width; x++)
619  {
620  sumAbsZmOrig += (uint32_t) abs (src[x] - avgOrig);
621  }
622  src += origY.stride;
623  }
624 
625  const double sumAbsRatio = double (uiSadBestForQPA * 3 /*TODO: or 4? fine-tune!*/) / double (sumAbsZmOrig == 0 ? 1 : sumAbsZmOrig);
626  const int pumpingReducQP = (int (log (Clip3 (0.25, 4.0, sumAbsRatio)) / log (2.0) + (sumAbsRatio < 1.0 ? -0.5 : 0.5))) >> (baseQP >= 38/*MAX_QP_PERCEPT_QPA*/ ? 1 : 0);
627 
628  ctuPumpRedQP[ctuRsAddr] += pumpingReducQP;
629 
630  return pumpingReducQP;
631 }
