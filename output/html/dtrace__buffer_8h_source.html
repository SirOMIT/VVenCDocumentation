62 inline unsigned calcCheckSum( const int iWidth, const int iHeight, const Pel* p, const uint32_t stride, const int bitdepth )
63 {
64  unsigned checksum = 0;
65  for( unsigned y = 0; y < iHeight; y++)
66  {
67  for( unsigned x = 0; x < iWidth; x++)
68  {
69  uint8_t xor_mask = (x & 0xff) ^ (y & 0xff) ^ (x >> 8) ^ (y >> 8);
70  checksum = (checksum + ((p[y*stride+x] & 0xff) ^ xor_mask)) & 0xffffffff;
71 
72  if(bitdepth > 8)
73  {
74  checksum = (checksum + ((p[y*stride+x]>>8) ^ xor_mask)) & 0xffffffff;
75  }
76  }
77  }
78  return checksum;
79 }

161 inline void dtraceCRC( CDTrace *trace_ctx, DTRACE_CHANNEL channel, const CodingStructure& cs, const CPelUnitBuf& pelUnitBuf, const Area* parea = NULL )
162 {
163  const Area& area = parea ? *parea : cs.area.Y();
164  if( cs.pcv->chrFormat == VVENC_CHROMA_400)
165  {
166  DTRACE( trace_ctx, channel, " CRC: %6lld %3d @(%4d,%4d) [%2dx%2d] ,Checksum(%x)\n",
167  DTRACE_GET_COUNTER( g_trace_ctx, channel ),
168  cs.slice->poc,

171  }
172  else

176  cs.slice->poc,
177  area.x, area.y, area.width, area.height,
178  calcCheckSum( pelUnitBuf.bufs[COMP_Y], cs.sps->bitDepths[CH_L]),
179  calcCheckSum( pelUnitBuf.bufs[COMP_Cb], cs.sps->bitDepths[CH_C]),

194 inline void dtraceMotField( CDTrace *trace_ctx, const CodingUnit& cu )
195 {
196  DTRACE( trace_ctx, D_MOT_FIELD, "CU %d,%d @ %d,%d\n", cu.lwidth(), cu.lheight(), cu.lx(), cu.ly() );
197  const CMotionBuf mb = cu.getMotionBuf();
198  for( uint32_t listIdx = 0; listIdx < 2; listIdx++ )
199  {
200  RefPicList eListIdx = RefPicList( listIdx );
201  for( int y = 0, i = 0; y < cu.lheight(); y += 4 )
202  {
203  for( int x = 0; x < cu.lwidth(); x += 4, i++ )
204  {
205  const MotionInfo &mi = mb.at( x >> 2, y >> 2 );
206  DTRACE( trace_ctx, D_MOT_FIELD, "%d,%d:%d ", mi.mv[eListIdx].hor, mi.mv[eListIdx].ver, mi.refIdx[eListIdx] );
207  }
208  DTRACE( trace_ctx, D_MOT_FIELD, "\n" );
209  }
210  DTRACE( trace_ctx, D_MOT_FIELD, "\n" );
