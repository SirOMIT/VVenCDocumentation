88 enum EncTestModeOpts

98 static inline void getAreaIdxNew(const Area& area, const PreCalcValues &pcv, unsigned& idx1, unsigned& idx2, unsigned& idx3, unsigned& idx4)
99 {
100  idx1 = Log2( area.width )-2;
101  idx2 = Log2( area.height )-2;
102  idx3 = (area.x & pcv.maxCUSizeMask) >> MIN_CU_LOG2;
103  idx4 = (area.y & pcv.maxCUSizeMask) >> MIN_CU_LOG2;
