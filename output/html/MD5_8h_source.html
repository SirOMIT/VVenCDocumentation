65 /*
66  * Note: this code is harmless on little-endian machines.
67  */
68 void byteReverse(uint32_t *buf, unsigned len)
69 {
70  uint32_t t;
71  do {
72  char* bytes = (char *) buf;
73  t = ((unsigned) bytes[3] << 8 | bytes[2]) << 16 |
74  ((unsigned) bytes[1] << 8 | bytes[0]);
75  *buf = t;
76  buf++;
77  } while (--len);
78 }
79 #endif

115  void update(unsigned char *buf, unsigned len)
116  {
117  uint32_t t;
118 
119  /* Update bitcount */
120 
121  t = ctx.bits[0];
122  if ((ctx.bits[0] = t + ((uint32_t) len << 3)) < t)
123  ctx.bits[1]++; /* Carry from low to high */
124  ctx.bits[1] += len >> 29;
125 
126  t = (t >> 3) & 0x3f; /* Bytes already in shsInfo->data */
127 
128  /* Handle any leading odd-sized chunks */
129 
130  if (t) {
131  unsigned char *p = ctx.in.b8 + t;
132 
133  t = 64 - t;
134  if (len < t) {
135  ::memcpy(p, buf, len);
136  return;
137  }
138  memcpy(p, buf, t);
139  byteReverse(ctx.in.b32, 16);
140  MD5Transform(ctx.buf, ctx.in.b32);
141  buf += t;
142  len -= t;
143  }
144  /* Process data in 64-byte chunks */
145 
146  while (len >= 64) {
147  memcpy(ctx.in.b8, buf, 64);
148  byteReverse(ctx.in.b32, 16);
149  MD5Transform(ctx.buf, ctx.in.b32);
150  buf += 64;
151  len -= 64;
152  }
153 
154  /* Handle any remaining bytes of data. */
155 
156  memcpy(ctx.in.b8, buf, len);
157  }

162  void finalize(unsigned char digest[MD5_DIGEST_STRING_LENGTH])
163  {
164  unsigned count;
165  unsigned char *p;
166 
167  /* Compute number of bytes mod 64 */
168  count = (ctx.bits[0] >> 3) & 0x3F;
169 
170  /* Set the first char of padding to 0x80. This is safe since there is
171  always at least one byte free */
172  p = ctx.in.b8 + count;
173  *p++ = 0x80;
174 
175  /* Bytes of padding needed to make 64 bytes */
176  count = 64 - 1 - count;
177 
178  /* Pad out to 56 mod 64 */
179  if (count < 8) {
180  /* Two lots of padding: Pad the first block to 64 bytes */
181  memset(p, 0, count);
182  byteReverse(ctx.in.b32, 16);
183  MD5Transform(ctx.buf, ctx.in.b32);
184 
185  /* Now fill the next block with 56 bytes */
186  memset(ctx.in.b8, 0, 56);
187  } else {
188  /* Pad block to 56 bytes */
189  memset(p, 0, count - 8);
190  }
191  byteReverse(ctx.in.b32, 14);
192 
193  /* Append length in bits and transform */
194  ctx.in.b32[14] = ctx.bits[0];
195  ctx.in.b32[15] = ctx.bits[1];
196 
197  MD5Transform(ctx.buf, ctx.in.b32);
198  byteReverse((uint32_t *) ctx.buf, 4);
199  memcpy(digest, ctx.buf, 16);
200 
201  memset(&ctx, 0, sizeof( ctx)); /* In case it's sensitive */
202  /* The original version of this code omitted the asterisk. In
203  effect, only the first part of ctx was wiped with zeros, not
204  the whole thing. Bug found by Derek Jones. Original line: */
205  // memset(ctx, 0, sizeof(ctx)); /* In case it's sensitive */
206  }
