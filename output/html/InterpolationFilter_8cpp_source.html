249 void InterpolationFilter::filterCopy( const ClpRng& clpRng, const Pel* src, int srcStride, Pel* dst, int dstStride, int width, int height, bool biMCForDMVR)
250 {
251  int row, col;
252 
253  if ( isFirst == isLast )
254  {
255  for (row = 0; row < height; row++)
256  {
257  for (col = 0; col < width; col++)
258  {
259  dst[col] = src[col];
260  }
261 
262  src += srcStride;
263  dst += dstStride;
264  }
265  }
266  else if ( isFirst )
267  {
268  const unsigned shift = std::max<int>(2, (IF_INTERNAL_PREC - clpRng.bd));
269 
270  if (biMCForDMVR)
271  {
272  int shift10BitOut, offset;
273  if ((clpRng.bd - IF_INTERNAL_PREC_BILINEAR) > 0)
274  {
275  shift10BitOut = (clpRng.bd - IF_INTERNAL_PREC_BILINEAR);
276  offset = (1 << (shift10BitOut - 1));
277  for (row = 0; row < height; row++)
278  {
279  for (col = 0; col < width; col++)
280  {
281  dst[col] = (src[col] + offset) >> shift10BitOut;
282  }
283  src += srcStride;
284  dst += dstStride;
285  }
286  }
287  else
288  {
289  shift10BitOut = (IF_INTERNAL_PREC_BILINEAR - clpRng.bd);
290  for (row = 0; row < height; row++)
291  {
292  for (col = 0; col < width; col++)
293  {
294  dst[col] = src[col] << shift10BitOut;
295  }
296  src += srcStride;
297  dst += dstStride;
298  }
299  }
300  }
301  else
302  {
303  for (row = 0; row < height; row++)
304  {
305  for (col = 0; col < width; col++)
306  {
307  Pel val = leftShiftU(src[col], shift);
308  dst[col] = val - (Pel)IF_INTERNAL_OFFS;
309  }
310 
311  src += srcStride;
312  dst += dstStride;
313  }
314  }
315  }
316  else
317  {
318  const unsigned shift = std::max<int>(2, (IF_INTERNAL_PREC - clpRng.bd));
319 
320  if (biMCForDMVR)
321  {
322  int shift10BitOut, offset;
323  if ((clpRng.bd - IF_INTERNAL_PREC_BILINEAR) > 0)
324  {
325  shift10BitOut = (clpRng.bd - IF_INTERNAL_PREC_BILINEAR);
326  offset = (1 << (shift10BitOut - 1));
327  for (row = 0; row < height; row++)
328  {
329  for (col = 0; col < width; col++)
330  {
331  dst[col] = (src[col] + offset) >> shift10BitOut;
332  }
333  src += srcStride;
334  dst += dstStride;
335  }
336  }
337  else
338  {
339  shift10BitOut = (IF_INTERNAL_PREC_BILINEAR - clpRng.bd);
340  for (row = 0; row < height; row++)
341  {
342  for (col = 0; col < width; col++)
343  {
344  dst[col] = src[col] << shift10BitOut;
345  }
346  src += srcStride;
347  dst += dstStride;
348  }
349  }
350  }
351  else
352  {
353  const Pel offset = ((1) << (shift - 1)) + IF_INTERNAL_OFFS;
354  for (row = 0; row < height; row++)
355  {
356  for (col = 0; col < width; col++)
357  {
358  Pel val = src[ col ];
359  val = rightShiftU((val + offset), shift);
360 
361  dst[col] = ClipPel( val, clpRng );
362  }
363 
364  src += srcStride;
365  dst += dstStride;
366  }
367  }
368  }
369 }

395 template<int N, bool isVertical, bool isFirst, bool isLast>
396 void InterpolationFilter::filter(const ClpRng& clpRng, Pel const *src, int srcStride, Pel* dst, int dstStride, int width, int height, TFilterCoeff const *coeff, bool biMCForDMVR)
397 {
398  int row, col;
399 
400  Pel c[8];
401  c[0] = coeff[0];
402  c[1] = coeff[1];
403  if ( N >= 4 )
404  {
405  c[2] = coeff[2];
406  c[3] = coeff[3];
407  }
408  if ( N >= 6 )
409  {
410  c[4] = coeff[4];
411  c[5] = coeff[5];
412  }
413  if ( N == 8 )
414  {
415  c[6] = coeff[6];
416  c[7] = coeff[7];
417  }
418 
419  int cStride = ( isVertical ) ? srcStride : 1;
420  src -= ( N/2 - 1 ) * cStride;
421 
422  int offset;
423  int headRoom = std::max<int>(2, (IF_INTERNAL_PREC - clpRng.bd));
424  int shift = IF_FILTER_PREC;
425  // with the current settings (IF_INTERNAL_PREC = 14 and IF_FILTER_PREC = 6), though headroom can be
426  // negative for bit depths greater than 14, shift will remain non-negative for bit depths of 8->20
427  CHECK(shift < 0, "Negative shift");
428 
429  if ( isLast )
430  {
431  shift += (isFirst) ? 0 : headRoom;
432  offset = 1 << (shift - 1);
433  offset += (isFirst) ? 0 : IF_INTERNAL_OFFS << IF_FILTER_PREC;
434  }
435  else
436  {
437  shift -= (isFirst) ? headRoom : 0;
438  offset = (isFirst) ? -IF_INTERNAL_OFFS << shift : 0;
439  }
440 
441  if (biMCForDMVR)
442  {
443  if( isFirst )
444  {
445  shift = IF_FILTER_PREC_BILINEAR - (IF_INTERNAL_PREC_BILINEAR - clpRng.bd);
446  offset = 1 << (shift - 1);
447  }
448  else
449  {
450  shift = 4;
451  offset = 1 << (shift - 1);
452  }
453  }
454  for (row = 0; row < height; row++)
455  {
456  for (col = 0; col < width; col++)
457  {
458  int sum;
459 
460  sum = src[ col + 0 * cStride] * c[0];
461  sum += src[ col + 1 * cStride] * c[1];
462  if ( N >= 4 )
463  {
464  sum += src[ col + 2 * cStride] * c[2];
465  sum += src[ col + 3 * cStride] * c[3];
466  }
467  if ( N >= 6 )
468  {
469  sum += src[ col + 4 * cStride] * c[4];
470  sum += src[ col + 5 * cStride] * c[5];
471  }
472  if ( N == 8 )
473  {
474  sum += src[ col + 6 * cStride] * c[6];
475  sum += src[ col + 7 * cStride] * c[7];
476  }
477 
478  Pel val = ( sum + offset ) >> shift;
479  if ( isLast )
480  {
481  val = ClipPel( val, clpRng );
482  }
483  dst[col] = val;
484  }
485 
486  src += srcStride;
487  dst += dstStride;
488  }
489 }

580 void InterpolationFilter::filterHor(const ComponentID compID, Pel const *src, int srcStride, Pel* dst, int dstStride, int width, int height, int frac, bool isLast, const ChromaFormat fmt, const ClpRng& clpRng, bool useAltHpelIf, int nFilterIdx, bool biMCForDMVR)
581 {
582  if( frac == 0 )
583  {
584  m_filterCopy[true][isLast](clpRng, src, srcStride, dst, dstStride, width, height, biMCForDMVR);
585  }
586  else if( isLuma( compID ) )
587  {
588  CHECK( frac < 0 || frac >= LUMA_INTERPOLATION_FILTER_SUB_SAMPLE_POSITIONS, "Invalid fraction" );
589  if( nFilterIdx == 1 )
590  {
591  filterHor<NTAPS_BILINEAR>(clpRng, src, srcStride, dst, dstStride, width, height, isLast, m_bilinearFilterPrec4[frac], biMCForDMVR);
592  }
593  else
594  {
595  if (frac == 8 && useAltHpelIf)
596  {
597  filterHor<NTAPS_LUMA>(clpRng, src, srcStride, dst, dstStride, width, height, isLast, m_lumaAltHpelIFilter, biMCForDMVR);
598  }
599  else if ((width == 4 && height == 4) || (width == 4 && height == (4 + NTAPS_LUMA - 1)))
600  {
601  filterHor<NTAPS_LUMA>(clpRng, src, srcStride, dst, dstStride, width, height, isLast, m_lumaFilter4x4[frac], biMCForDMVR);
602  }
603  else
604  {
605  filterHor<NTAPS_LUMA>(clpRng, src, srcStride, dst, dstStride, width, height, isLast, m_lumaFilter[frac], biMCForDMVR);
606  }
607  }
608  }
609  else
610  {
611  const uint32_t csx = getComponentScaleX( compID, fmt );
612  CHECK( frac < 0 || csx >= 2 || ( frac << ( 1 - csx ) ) >= CHROMA_INTERPOLATION_FILTER_SUB_SAMPLE_POSITIONS, "Invalid fraction" );
613  filterHor<NTAPS_CHROMA>( clpRng, src, srcStride, dst, dstStride, width, height, isLast, m_chromaFilter[frac << ( 1 - csx )], biMCForDMVR);
614  }
615 }

754 template<bool isLast, int w>
755 void InterpolationFilter::filterXxY_N4( const ClpRng& clpRng, const Pel* src, int srcStride, Pel* _dst, int dstStride, int width, int height, const TFilterCoeff *coeffH, const TFilterCoeff *coeffV )
756 {
757  int row, col;
758 
759  Pel cH[4];
760  cH[0] = coeffH[0]; cH[1] = coeffH[1];
761  cH[2] = coeffH[2]; cH[3] = coeffH[3];
762  Pel cV[4];
763  cV[0] = coeffV[0]; cV[1] = coeffV[1];
764  cV[2] = coeffV[2]; cV[3] = coeffV[3];
765 
766  int offset1st, offset2nd;
767  int headRoom = std::max<int>( 2, ( IF_INTERNAL_PREC - clpRng.bd ) );
768  int shift1st = IF_FILTER_PREC, shift2nd = IF_FILTER_PREC;
769  // with the current settings (IF_INTERNAL_PREC = 14 and IF_FILTER_PREC = 6), though headroom can be
770  // negative for bit depths greater than 14, shift will remain non-negative for bit depths of 8->20
771 
772  if( isLast )
773  {
774  shift1st -= headRoom;
775  shift2nd += headRoom;
776  offset1st = -IF_INTERNAL_OFFS << shift1st;
777  offset2nd = 1 << ( shift2nd - 1 );
778  offset2nd += IF_INTERNAL_OFFS << IF_FILTER_PREC;
779  }
780  else
781  {
782  shift1st -= headRoom;
783  offset1st = -IF_INTERNAL_OFFS << shift1st;
784  offset2nd = 0;
785  }
786 
787  src -= 1 + srcStride;
788 
789  int *tmp = ( int * ) alloca( w * height * sizeof( int ) );
790  memset( tmp, 0, w * height * sizeof( int ) );
791 
792  int** dst = ( int ** ) alloca( height * sizeof( int * ) );
793 
794  for( int i = 0; i < height; i++ ) dst[i] = &tmp[i * w];
795 
796  for( row = 0; row < ( height + 3 ); row++ )
797  {
798  for( col = 0; col < w; col++ )
799  {
800  int sum;
801 
802  sum = src[col ] * cH[0];
803  sum += src[col + 1] * cH[1];
804  sum += src[col + 2] * cH[2];
805  sum += src[col + 3] * cH[3];
806 
807  sum = ( sum + offset1st ) >> shift1st;
808 
809  if( row >= 0 && row < ( height + 0 ) ) dst[row ][col] += sum * cV[0];
810  if( row >= 1 && row < ( height + 1 ) ) dst[row - 1][col] += sum * cV[1];
811  if( row >= 2 && row < ( height + 2 ) ) dst[row - 2][col] += sum * cV[2];
812 
813  if( row >= 3 )
814  {
815  int val = ( dst[row - 3][col] + sum * cV[3] + offset2nd ) >> shift2nd;
816  if( isLast )
817  {
818  val = ClipPel( val, clpRng );
819  }
820  _dst[col] = val;
821  }
822  }
823 
824  src += srcStride;
825  if( row >= 3 ) _dst += dstStride;
826  }
827 }

925 void InterpolationFilter::xWeightedGeoBlk(const ClpRngs &clpRngs, const CodingUnit& cu, const uint32_t width,
926  const uint32_t height,
927  const ComponentID compIdx, const uint8_t splitDir,PelUnitBuf &predDst,
928  PelUnitBuf &predSrc0, PelUnitBuf &predSrc1)
929 {
930  Pel * dst = predDst.get(compIdx).buf;
931  Pel * src0 = predSrc0.get(compIdx).buf;
932  Pel * src1 = predSrc1.get(compIdx).buf;
933  int32_t strideDst = predDst.get(compIdx).stride - width;
934  int32_t strideSrc0 = predSrc0.get(compIdx).stride - width;
935  int32_t strideSrc1 = predSrc1.get(compIdx).stride - width;
936 
937  const char log2WeightBase = 3;
938  // const ClpRng clipRng = cu.cs->slice->clpRngs[compIdx] // cu.slice->clpRngs().comp[compIdx];
939  const int32_t clipbd = clpRngs[compIdx].bd;
940  const int32_t shiftWeighted = std::max<int>(2, (IF_INTERNAL_PREC - clipbd)) + log2WeightBase;
941  const int32_t offsetWeighted = (1 << (shiftWeighted - 1)) + (IF_INTERNAL_OFFS << log2WeightBase);
942  const uint32_t scaleX = getComponentScaleX(compIdx, cu.chromaFormat);
943  const uint32_t scaleY = getComponentScaleY(compIdx, cu.chromaFormat);
944 
945  int16_t angle = g_GeoParams[splitDir][0];
946  int16_t wIdx = floorLog2(cu.lwidth()) - GEO_MIN_CU_LOG2;
947  int16_t hIdx = floorLog2(cu.lheight()) - GEO_MIN_CU_LOG2;
948  int16_t stepX = 1 << scaleX;
949  int16_t stepY = 0;
950  int16_t *weight = nullptr;
951  if (g_angle2mirror[angle] == 2)
952  {
953  stepY = -(int) ((GEO_WEIGHT_MASK_SIZE << scaleY) + cu.lwidth());
954  weight =
955  &g_globalGeoWeights[g_angle2mask[angle]]
956  [(GEO_WEIGHT_MASK_SIZE - 1 - g_weightOffset[hIdx][wIdx][splitDir][1]) * GEO_WEIGHT_MASK_SIZE
957  + g_weightOffset[hIdx][wIdx][splitDir][0]];
958  }
959  else if (g_angle2mirror[angle] == 1)
960  {
961  stepX = -1 << scaleX;
962  stepY = (GEO_WEIGHT_MASK_SIZE << scaleY) + cu.lwidth();
963  weight =
964  &g_globalGeoWeights[g_angle2mask[angle]][g_weightOffset[hIdx][wIdx][splitDir][1] * GEO_WEIGHT_MASK_SIZE
965  + (GEO_WEIGHT_MASK_SIZE - 1 - g_weightOffset[hIdx][wIdx][splitDir][0])];
966  }
967  else
968  {
969  stepY = (GEO_WEIGHT_MASK_SIZE << scaleY) - cu.lwidth();
970  weight = &g_globalGeoWeights[g_angle2mask[angle]][g_weightOffset[hIdx][wIdx][splitDir][1] * GEO_WEIGHT_MASK_SIZE
971  + g_weightOffset[hIdx][wIdx][splitDir][0]];
972  }
973  for (int y = 0; y < height; y++)
974  {
975  for (int x = 0; x < width; x++)
976  {
977  *dst++ = ClipPel(rightShiftU((*weight * (*src0++) + ((8 - *weight) * (*src1++)) + offsetWeighted), shiftWeighted),clpRngs[compIdx]);
978  weight += stepX;
979  }
980  dst += strideDst;
981  src0 += strideSrc0;
982  src1 += strideSrc1;
983  weight += stepY;
984  }
985 }
