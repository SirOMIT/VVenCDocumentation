85 struct MvField

90  MvField() : refIdx( NOT_VALID ) {}

92 
93  void setMvField( Mv const & cMv, const int iRefIdx )
94  {
95  CHECK( iRefIdx == -1 && cMv != Mv(0,0), "Must not happen." );
96  mv = cMv;
97  refIdx = iRefIdx;
98  }
99 
100  bool operator==( const MvField& other ) const
101  {
102  CHECK( refIdx == -1 && mv != Mv(0,0), "Error in operator== of MvField." );
103  CHECK( other.refIdx == -1 && other.mv != Mv(0,0), "Error in operator== of MvField." );
104  return refIdx == other.refIdx && mv == other.mv;
105  }
106  bool operator!=( const MvField& other ) const
107  {
108  CHECK( refIdx == -1 && mv != Mv(0,0), "Error in operator!= of MvField." );
109  CHECK( other.refIdx == -1 && other.mv != Mv(0,0), "Error in operator!= of MvField." );
110  return refIdx != other.refIdx || mv != other.mv;
111  }

114 struct MotionInfo
115 {
116  Mv mv [NUM_REF_PIC_LIST_01];
117  int8_t refIdx[NUM_REF_PIC_LIST_01] = { NOT_VALID, NOT_VALID };
118 
119  uint16_t sliceIdx = 0;
120  bool isInter = false;
121  char interDir = 0;
122  bool isIBCmot = false;
123  Mv bv;
124 
125  bool operator==( const MotionInfo& mi ) const
126  {
127  if( isInter != mi.isInter ) return false;
128  if( isIBCmot != mi.isIBCmot ) return false;
129  if (isInter)
130  {
131  if (sliceIdx != mi.sliceIdx) return false;
132  if (interDir != mi.interDir) return false;
133 
134  if (interDir != 2)
135  {
136  if (refIdx[0] != mi.refIdx[0]) return false;
137  if (mv[0] != mi.mv[0]) return false;
138  }
139 
140  if (interDir != 1)
141  {
142  if (refIdx[1] != mi.refIdx[1]) return false;
143  if (mv[1] != mi.mv[1]) return false;
144  }
145  }
146  return true;
147  }
148 
149  bool operator!=( const MotionInfo& mi ) const
150  {
151  return !( *this == mi );
152  }
153 };
