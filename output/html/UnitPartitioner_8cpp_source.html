99 {

102 
103  for( const auto &level : m_partStack )
104  {

106  else splitSeries += static_cast< SplitSeries >( level.split ) << ( depth * SPLIT_DMULT );
107 
108  depth++;
109  }
110 
111  return splitSeries;

167 void Partitioner::setMaxMinDepth( unsigned& minDepth, unsigned& maxDepth, const CodingStructure& cs, bool refineMinMax ) const
168 {
169  unsigned stdMinDepth = 0;
170  unsigned stdMaxDepth = cs.pcv->getMaxDepth( cs.slice->sliceType, chType );
171  const Position pos = currArea().blocks[chType].pos();
172  const unsigned curSliceIdx = cs.slice->independentSliceIdx;
173  const unsigned curTileIdx = cs.pps->getTileIdx( currArea().lumaPos() );
174 
175  const CodingUnit* cuLeft = cs.getCURestricted( pos.offset( -1, 0 ), pos, curSliceIdx, curTileIdx, chType, treeType );
176  const CodingUnit* cuBelowLeft = cs.getCURestricted( pos.offset( -1, currArea().blocks[chType].height), pos, curSliceIdx, curTileIdx, chType, treeType );
177  const CodingUnit* cuAbove = cs.getCURestricted( pos.offset( 0, -1 ), pos, curSliceIdx, curTileIdx, chType, treeType );
178  const CodingUnit* cuAboveRight = cs.getCURestricted( pos.offset( currArea().blocks[chType].width, -1 ), pos, curSliceIdx, curTileIdx, chType, treeType );
179 
180  minDepth = stdMaxDepth;
181  maxDepth = stdMinDepth;
182 
183  if( cuLeft )
184  {
185  minDepth = std::min<unsigned>( minDepth, cuLeft->qtDepth );
186  maxDepth = std::max<unsigned>( maxDepth, cuLeft->qtDepth );
187  }
188  else
189  {
190  minDepth = stdMinDepth;
191  maxDepth = stdMaxDepth;
192  }
193 
194  if( cuBelowLeft )
195  {
196  minDepth = std::min<unsigned>( minDepth, cuBelowLeft->qtDepth );
197  maxDepth = std::max<unsigned>( maxDepth, cuBelowLeft->qtDepth );
198  }
199  else
200  {
201  minDepth = stdMinDepth;
202  maxDepth = stdMaxDepth;
203  }
204 
205  if( cuAbove )
206  {
207  minDepth = std::min<unsigned>( minDepth, cuAbove->qtDepth );
208  maxDepth = std::max<unsigned>( maxDepth, cuAbove->qtDepth );
209  }
210  else
211  {
212  minDepth = stdMinDepth;
213  maxDepth = stdMaxDepth;
214  }
215 
216  if( cuAboveRight )
217  {
218  minDepth = std::min<unsigned>( minDepth, cuAboveRight->qtDepth );
219  maxDepth = std::max<unsigned>( maxDepth, cuAboveRight->qtDepth );
220  }
221  else
222  {
223  minDepth = stdMinDepth;
224  maxDepth = stdMaxDepth;
225  }
226 
227  minDepth = ( minDepth >= 1 ? minDepth - 1 : 0 );
228  maxDepth = std::min<unsigned>( stdMaxDepth, maxDepth + 1 );
229 
230  if( refineMinMax )
231  {
232  int minDepthCur = stdMaxDepth;
233  int maxDepthCur = stdMinDepth;
234  int amountN = 0;
235  for (int n = 0; n < 3; n++)
236  {
237  const CodingUnit* cuNeigh = (n==0)?cs.getCURestricted(pos.offset(-1, -1), pos, curSliceIdx, curTileIdx, chType, treeType): (n==1)? cuAbove : cuLeft;
238  if (cuNeigh)
239  {
240  amountN++;
241  minDepthCur = std::min<unsigned>(minDepthCur, cuNeigh->qtDepth);
242  maxDepthCur = std::max<unsigned>(maxDepthCur, cuNeigh->qtDepth);
243  }
244  }
245  if (amountN)
246  {
247  minDepthCur = (minDepthCur >= 1 ? minDepthCur - 1 : 0);
248  maxDepthCur = std::min<unsigned>(stdMaxDepth, maxDepthCur + 1);
249  maxDepth = std::min<unsigned>(maxDepthCur, maxDepth);
250  minDepth = std::max<unsigned>(minDepthCur, minDepth);
251  }
252  }
253 }

291 void Partitioner::splitCurrArea( const PartSplit split, const CodingStructure& cs )
292 {
293  if ((split != TU_1D_HORZ_SPLIT) && (split != TU_1D_VERT_SPLIT))
294  {
295  CHECKD(!canSplit(split, cs), "Trying to apply a prohibited split!");
296  }
297 
298  bool isImplicit = isSplitImplicit( split, cs );
299  bool canQtSplit = canSplit( CU_QUAD_SPLIT, cs );
300  bool qgEnable = currQgEnable();
301  bool qgChromaEnable = currQgChromaEnable();
302 
303  const UnitArea& area = currArea();
304  m_partStack.resize_noinit( m_partStack.size() + 1 );
305  PartLevel& back = m_partStack.back();
306  back.init();
307  back.split = split;
308  back.parts = &m_partBuf[m_partBufIdx];
309  int numParts;
310 
311  CHECK( m_partBufIdx > partBufSize, "Partition buffer overflow" );
312 
313  switch( split )
314  {
315  case CU_QUAD_SPLIT:
316  numParts = PartitionerImpl::getCUSubPartitions( back.parts, area, cs, split );
317  back.modeType = modeType;
318  break;
319  case CU_HORZ_SPLIT:
320  case CU_VERT_SPLIT:
321  numParts = PartitionerImpl::getCUSubPartitions( back.parts, area, cs, split );
322  back.modeType = modeType;
323  break;
324  case CU_TRIH_SPLIT:
325  case CU_TRIV_SPLIT:
326  numParts = PartitionerImpl::getCUSubPartitions( back.parts, area, cs, split );
327  back.modeType = modeType;
328  break;
329  case TU_MAX_TR_SPLIT:
330  numParts = PartitionerImpl::getMaxTuTiling( back.parts, area, cs );
331  break;
332  case SBT_VER_HALF_POS0_SPLIT:
333  case SBT_VER_HALF_POS1_SPLIT:
334  case SBT_HOR_HALF_POS0_SPLIT:
335  case SBT_HOR_HALF_POS1_SPLIT:
336  case SBT_VER_QUAD_POS0_SPLIT:
337  case SBT_VER_QUAD_POS1_SPLIT:
338  case SBT_HOR_QUAD_POS0_SPLIT:
339  case SBT_HOR_QUAD_POS1_SPLIT:
340  numParts = PartitionerImpl::getSbtTuTiling( back.parts, area, cs, split );
341  break;
342  case TU_1D_HORZ_SPLIT:
343  case TU_1D_VERT_SPLIT:
344  {
345  numParts = PartitionerImpl::getTUIntraSubPartitions(back.parts, area, cs, split, TREE_D);
346  break;
347  }
348  default:
349  THROW( "Unknown split mode" );
350  break;
351  }
352 
353  back.numParts = numParts;
354  m_partBufIdx += numParts;
355 
356  CHECK( m_partBufIdx > partBufSize, "Partition buffer overflow" );
357 
358  currDepth++;
359  currSubdiv++;
360 #if _DEBUG
361  m_currArea = m_partStack.back().parts[0];
362 #endif
363 
364  if ((split == TU_MAX_TR_SPLIT) || (split == TU_1D_HORZ_SPLIT) || (split == TU_1D_VERT_SPLIT))
365  {
366  currTrDepth++;
367  }
368  else if( split >= SBT_VER_HALF_POS0_SPLIT && split <= SBT_HOR_QUAD_POS1_SPLIT )
369  {
370  currTrDepth++;
371  }
372  else
373  {
374  currTrDepth = 0;
375  }
376 
377  if( split == CU_HORZ_SPLIT || split == CU_VERT_SPLIT || split == CU_TRIH_SPLIT || split == CU_TRIV_SPLIT )
378  {
379  currBtDepth++;
380  if( isImplicit ) currImplicitBtDepth++;
381  currMtDepth++;
382 
383  if( split == CU_TRIH_SPLIT || split == CU_TRIV_SPLIT )
384  {
385  // first and last part of triple split are equivalent to double bt split
386  currBtDepth++;
387  currSubdiv++;
388  }
389  m_partStack.back().canQtSplit = canQtSplit;
390  }
391  else if( split == CU_QUAD_SPLIT )
392  {
393  CHECK( currBtDepth > 0, "Cannot split a non-square area other than with a binary split" );
394  CHECK( currMtDepth > 0, "Cannot split a non-square area other than with a binary split" );
395  currMtDepth = 0;
396  currBtDepth = 0;
397  currQtDepth++;
398  currSubdiv++;
399  }
400 
401  qgEnable &= (currSubdiv <= (cs.slice->isIntra() ? cs.slice->picHeader->cuQpDeltaSubdivIntra : cs.slice->picHeader->cuQpDeltaSubdivInter ));
402  qgChromaEnable &= (currSubdiv <= (cs.slice->isIntra() ? cs.slice->picHeader->cuChromaQpOffsetSubdivIntra : cs.slice->picHeader->cuChromaQpOffsetSubdivInter ));
403  m_partStack.back().qgEnable = qgEnable;
404  m_partStack.back().qgChromaEnable = qgChromaEnable;
405  if (qgEnable)
406  currQgPos = currArea().lumaPos();
407  if (qgChromaEnable)
408  currQgChromaPos = currArea().chromaPos();
409 }

411 void Partitioner::canSplit( const CodingStructure &cs, bool& canNo, bool& canQt, bool& canBh, bool& canBv, bool& canTh, bool& canTv )
412 {
413  const PartSplit implicitSplit = m_partStack.back().checkdIfImplicit ? m_partStack.back().implicitSplit : getImplicitSplit( cs );
414 
415  canNo = canQt = canBh = canTh = canBv = canTv = true;
416  bool canBtt = currMtDepth < (maxBTD + currImplicitBtDepth);
417 
418  // the minimal and maximal sizes are given in luma samples
419  const CompArea& area = currArea().Y();
420  const CompArea *areaC = (chType == CH_C) ? &(currArea().Cb()) : nullptr;
421  PartLevel& level = m_partStack.back();
422 
423  const PartSplit lastSplit = level.split;
424  const PartSplit parlSplit = lastSplit == CU_TRIH_SPLIT ? CU_HORZ_SPLIT : CU_VERT_SPLIT;
425 
426  // don't allow QT-splitting below a BT split
427  if( lastSplit != CTU_LEVEL && lastSplit != CU_QUAD_SPLIT ) canQt = false;
428  // minQtSize is in luma samples unit
429  const unsigned minQTThreshold = minQtSize >> ((area.chromaFormat == CHROMA_400) ? 0 : ((int) getChannelTypeScaleX(CH_C, area.chromaFormat) - (int) getChannelTypeScaleY(CH_C, area.chromaFormat)));
430  if( area.width <= minQTThreshold ) canQt = false;
431  if( areaC && areaC->width <= MIN_DUALTREE_CHROMA_WIDTH ) canQt = false;
432  if( treeType == TREE_C )
433  {
434  canQt = canBh = canTh = canBv = canTv = false;
435  return;
436  }
437  if( implicitSplit != CU_DONT_SPLIT )
438  {
439  canNo = canTh = canTv = false;
440 
441  canBh = implicitSplit == CU_HORZ_SPLIT;
442  canBv = implicitSplit == CU_VERT_SPLIT;
443  if (areaC && areaC->width == 4) canBv = false;
444  if( !canBh && !canBv && !canQt ) canQt = true;
445  return;
446  }
447 
448  if( ( lastSplit == CU_TRIH_SPLIT || lastSplit == CU_TRIV_SPLIT ) && currPartIdx() == 1 )
449  {
450  canBh = parlSplit != CU_HORZ_SPLIT;
451  canBv = parlSplit != CU_VERT_SPLIT;
452  }
453 
454  if( canBtt && ( area.width <= minTSize && area.height <= minTSize ) )
455  {
456  canBtt = false;
457  }
458  if( canBtt && ( area.width > maxBtSize || area.height > maxBtSize )
459  && ( ( area.width > maxTtSize || area.height > maxTtSize ) ) )
460  {
461  canBtt = false;
462  }
463 
464  if( !canBtt )
465  {
466  canBh = canTh = canBv = canTv = false;
467 
468  return;
469  }
470 
471  if( area.width > maxBtSize || area.height > maxBtSize )
472  {
473  canBh = canBv = false;
474  }
475 
476  // specific check for BT splits
477  if( area.height <= minTSize ) canBh = false;
478  if( area.width > MAX_TB_SIZEY && area.height <= MAX_TB_SIZEY ) canBh = false;
479  if( areaC && areaC->width * areaC->height <= MIN_DUALTREE_CHROMA_SIZE ) canBh = false;
480  if( area.width <= minTSize ) canBv = false;
481  if( area.width <= MAX_TB_SIZEY && area.height > MAX_TB_SIZEY ) canBv = false;
482  if (areaC && (areaC->width * areaC->height <= MIN_DUALTREE_CHROMA_SIZE || areaC->width == 4)) canBv = false;
483  if( modeType == MODE_TYPE_INTER && area.width * area.height == 32 ) canBv = canBh = false;
484  if( area.height <= 2 * minTSize || area.height > maxTtSize || area.width > maxTtSize )
485  canTh = false;
486  if( area.width > MAX_TB_SIZEY || area.height > MAX_TB_SIZEY ) canTh = false;
487  if( areaC && areaC->width * areaC->height <= MIN_DUALTREE_CHROMA_SIZE*2 ) canTh = false;
488  if( area.width <= 2 * minTSize || area.width > maxTtSize || area.height > maxTtSize )
489  canTv = false;
490  if( area.width > MAX_TB_SIZEY || area.height > MAX_TB_SIZEY ) canTv = false;
491  if (areaC && (areaC->width * areaC->height <= MIN_DUALTREE_CHROMA_SIZE * 2 || areaC->width == 8)) canTv = false;
492  if( modeType == MODE_TYPE_INTER && area.width * area.height == 64 ) canTv = canTh = false;
493 }

495 bool Partitioner::canSplit( const PartSplit split, const CodingStructure &cs )
496 {
497  const CompArea area = currArea().Y();
498  const unsigned maxTrSize = cs.sps->getMaxTbSize();
499 
500  bool canNo, canQt, canBh, canTh, canBv, canTv;
501 
502  canSplit( cs, canNo, canQt, canBh, canBv, canTh, canTv );
503  switch( split )
504  {
505  case CTU_LEVEL:
506  THROW( "Checking if top level split is possible" );
507  return true;
508  break;
509  case TU_MAX_TR_SPLIT:
510  return area.width > maxTrSize || area.height > maxTrSize;
511  break;
512  case SBT_VER_HALF_POS0_SPLIT:
513  case SBT_VER_HALF_POS1_SPLIT:
514  case SBT_HOR_HALF_POS0_SPLIT:
515  case SBT_HOR_HALF_POS1_SPLIT:
516  case SBT_VER_QUAD_POS0_SPLIT:
517  case SBT_VER_QUAD_POS1_SPLIT:
518  case SBT_HOR_QUAD_POS0_SPLIT:
519  case SBT_HOR_QUAD_POS1_SPLIT:
520  return currTrDepth == 0;
521  break;
522  case CU_QUAD_SPLIT:
523  return canQt;
524  case CU_DONT_SPLIT:
525  return canNo;
526  case CU_HORZ_SPLIT:
527  return canBh;
528  case CU_VERT_SPLIT:
529  return canBv;
530  case CU_TRIH_SPLIT:
531  return canTh;
532  case CU_TRIV_SPLIT:
533  return canTv;
534  case CU_MT_SPLIT:
535  return ( canBh || canTh || canBv || canTv );
536  case CU_BT_SPLIT:
537  return ( canBh || canBv );
538  break;
539  default:
540  THROW( "Unknown split mode" );
541  return false;
542  break;
543  }
544 
545  return true;
546 }

577 PartSplit Partitioner::getImplicitSplit( const CodingStructure &cs )
578 {
579  if( m_partStack.back().checkdIfImplicit )
580  {
581  return m_partStack.back().implicitSplit;
582  }
583 
584  PartSplit split = CU_DONT_SPLIT;
585 
586  if( split == CU_DONT_SPLIT )
587  {
588  const bool isBlInPic = cs.picture->Y().contains( currArea().Y().bottomLeft() );
589  const bool isTrInPic = cs.picture->Y().contains( currArea().Y().topRight() );
590 
591  const CompArea& area = currArea().Y();
592  const bool isBtAllowed = area.width <= maxBtSize && area.height <= maxBtSize && currMtDepth < (maxBTD + currImplicitBtDepth);
593  // minQtSize is in luma samples unit
594  const unsigned minQTThreshold = minQtSize >> ((area.chromaFormat == CHROMA_400) ? 0 : ((int) getChannelTypeScaleX(CH_C, area.chromaFormat) - (int) getChannelTypeScaleY(CH_C, area.chromaFormat)));
595  const bool isQtAllowed = area.width > minQTThreshold && currBtDepth == 0;
596 
597  if( !isBlInPic && !isTrInPic && isQtAllowed )
598  {
599  split = CU_QUAD_SPLIT;
600  }
601  else if( !isBlInPic && isBtAllowed && area.width <= MAX_TB_SIZEY )
602  {
603  split = CU_HORZ_SPLIT;
604  }
605  else if( !isTrInPic && isBtAllowed && area.height <= MAX_TB_SIZEY )

630 void Partitioner::exitCurrSplit()
631 {

633  const int currIndex = m_partStack.back().idx;
634  const int numParts = m_partStack.back().numParts;
635 
636  m_partStack.pop_back();
637  m_partBufIdx -= numParts;
638 
639  CHECK( currDepth == 0, "depth is '0', although a split was performed" );
640  currDepth--;
641  currSubdiv--;
642  if( currQgEnable() )
643  currQgPos = currArea().lumaPos();
644  if( currArea().chromaFormat != CHROMA_400 && currQgChromaEnable() )
645  currQgChromaPos = currArea().chromaPos();
646 #if _DEBUG
647  m_currArea = m_partStack.back().parts[m_partStack.back().idx];
648 #endif
649 
650  if( currSplit == CU_HORZ_SPLIT || currSplit == CU_VERT_SPLIT || currSplit == CU_TRIH_SPLIT || currSplit == CU_TRIV_SPLIT )
651  {
652  CHECK( !m_partStack.back().checkdIfImplicit, "Didn't check if the current split is implicit" );
653  CHECK( currBtDepth == 0, "BT depth is '0', athough a BT split was performed" );
654  CHECK( currMtDepth == 0, "MT depth is '0', athough a BT split was performed" );
655  currMtDepth--;
656  if( m_partStack.back().isImplicit ) currImplicitBtDepth--;
657  currBtDepth--;
658  if( ( currSplit == CU_TRIH_SPLIT || currSplit == CU_TRIV_SPLIT ) && currIndex != 1 )
659  {
660  CHECK( currBtDepth == 0, "BT depth is '0', athough a TT split was performed" );
661  currBtDepth--;
662  currSubdiv--;
663  }
664  }
665  else if( currSplit == TU_MAX_TR_SPLIT )
666  {
667  CHECK( currTrDepth == 0, "TR depth is '0', although a TU split was performed" );
668  currTrDepth--;
669  }
670  else if( currSplit >= SBT_VER_HALF_POS0_SPLIT && currSplit <= SBT_HOR_QUAD_POS1_SPLIT )
671  {
672  CHECK( currTrDepth == 0, "TR depth is '0', although a TU split was performed" );
673  currTrDepth--;
674  }
675  else if ((currSplit == TU_1D_HORZ_SPLIT) || (currSplit == TU_1D_VERT_SPLIT))
676  {
677  CHECK(currTrDepth == 0, "TR depth is '0', although a TU split was performed");
678  currTrDepth--;
679  }
680  else
681  {
682  CHECK( currTrDepth > 0, "RQT found with QTBT partitioner" );
683 
684  CHECK( currQtDepth == 0, "QT depth is '0', although a QT split was performed" );
685  currQtDepth--;
686  currSubdiv--;

746  Partitioning& sub = dst;

749  {
750  sub[i] = cuArea;
751

753  {

759  }
760

772  {

774  if (i == 1) blk.y += blk.height;
775  }
776  }
777 
778  return 2;
779  }
780  else if( splitType == CU_VERT_SPLIT )
781  {

783

785  {
786  sub[i] = cuArea;
787 
788  for( auto &blk : sub[i].blocks )

792  }
793  }
794 
795  return 2;
796  }
797  else if( splitType == CU_TRIH_SPLIT )

842 int PartitionerImpl::getTUIntraSubPartitions( Partitioning& sub, const UnitArea &tuArea, const CodingStructure &cs, const PartSplit splitType, const TreeType treeType )
843 {
844  uint32_t nPartitions;
845  uint32_t splitDimensionSize = CU::getISPSplitDim( tuArea.lumaSize().width, tuArea.lumaSize().height, splitType );
846 
847  bool isDualTree = CS::isDualITree( cs ) || treeType != TREE_D;
848 
849  if( splitType == TU_1D_HORZ_SPLIT )
850  {
851  nPartitions = tuArea.lumaSize().height >> Log2(splitDimensionSize);
852 
853  for( uint32_t i = 0; i < nPartitions; i++ )
854  {
855  sub[i] = tuArea;
856  CompArea& blkY = sub[i].blocks[COMP_Y];
857 
858  blkY.height = splitDimensionSize;
859  blkY.y = i > 0 ? sub[i - 1].blocks[COMP_Y].y + splitDimensionSize : blkY.y;
860 
861  CHECK( sub[i].lumaSize().height < 1, "the cs split causes the block to be smaller than the minimal TU size" );
862  }
863  }
864  else if( splitType == TU_1D_VERT_SPLIT )
865  {
866  nPartitions = tuArea.lumaSize().width >> Log2(splitDimensionSize);
867 
868  for( uint32_t i = 0; i < nPartitions; i++ )
869  {
870  sub[i] = tuArea;
871  CompArea& blkY = sub[i].blocks[COMP_Y];
872 
873  blkY.width = splitDimensionSize;
874  blkY.x = i > 0 ? sub[i - 1].blocks[COMP_Y].x + splitDimensionSize : blkY.x;
875  CHECK( sub[i].lumaSize().width < 1, "the split causes the block to be smaller than the minimal TU size" );
876  }
877  }
878  else
879  {
880  THROW( "Unknown TU sub-partitioning" );
881  }
882  //we only partition luma, so there is going to be only one chroma tu at the end (unless it is dual tree, in which case there won't be any chroma components)
883  uint32_t partitionsWithoutChroma = (cs.area.chromaFormat == CHROMA_400) ? 0 : (isDualTree ? nPartitions : nPartitions - 1);
884  for( uint32_t i = 0; i < partitionsWithoutChroma; i++ )
885  {
886  CompArea& blkCb = sub[i].blocks[COMP_Cb];
887  CompArea& blkCr = sub[i].blocks[COMP_Cr];
888  blkCb = CompArea();
889  blkCr = CompArea();
890  }
891 
892  return nPartitions;
893 }

922 int PartitionerImpl::getMaxTuTiling( Partitioning& dst, const UnitArea& cuArea, const CodingStructure& cs )
923 {
924  const Size area = cuArea.lumaSize();
925  const int maxTrSize = cs.sps->getMaxTbSize();

927  const int numTilesV = std::max<int>( 1, area.height / maxTrSize );
928  const int numTiles = numTilesH * numTilesV;
929  const int numLog2H = Log2( numTilesH );
930  const int* rsScanToZ = g_rsScanToZ[numLog2H];
931 
932  Partitioning& ret = dst;
933 
934  for( int i = 0; i < numTiles; i++ )
935  {
936  ret[i] = cuArea;
937 
938  const int zid = rsScanToZ[i];
939 
940  const int y = zid >> numLog2H;
941  const int x = zid & ( ( 1 << numLog2H ) - 1 );
942 
943  UnitArea& tile = ret[i];
944 
945  for( CompArea& comp : tile.blocks )
946  {
947  if( !comp.valid() ) continue;
948 
949  comp.width /= numTilesH;
950  comp.height /= numTilesV;
951 
952  comp.x += comp.width * x;
953  comp.y += comp.height * y;
954  }
955  }
956

961 {

985  heightFactor = 4;
986  yOffsetFactor = 0;
987  }
988  }
989  else
990  {

992  {
993  widthFactor = 4;
994  xOffsetFactor = 0;
995  heightFactor = 2;
996  yOffsetFactor = ( i == 0 ) ? 0 : 2;
997  }
998  else

1001  xOffsetFactor = ( i == 0 ) ? 0 : 2;

1010  {

1014  comp.y += ( comp.height * yOffsetFactor ) >> 2;
