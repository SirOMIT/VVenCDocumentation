72 void SEIEncoder::initBufferingPeriodSEI( SEIBufferingPeriod& bpSei, bool noLeadingPictures)
73 {
74  CHECK(!(m_isInitialized), "bufferingPeriodSEI already initialized");
75 
76  uint32_t uiInitialCpbRemovalDelay = (90000/2); // 0.5 sec
77  bpSei.bpNalCpbParamsPresent = true;
78  bpSei.bpVclCpbParamsPresent = true;
79  bpSei.bpMaxSubLayers = m_pcEncCfg->m_maxTempLayer;
80  bpSei.bpCpbCnt = 1;
81  for(int i=0; i < bpSei.bpMaxSubLayers; i++)
82  {
83  for(int j=0; j < bpSei.bpCpbCnt; j++)
84  {
85  bpSei.initialCpbRemovalDelay[j][i][0] = uiInitialCpbRemovalDelay;
86  bpSei.initialCpbRemovalDelay[j][i][1] = uiInitialCpbRemovalDelay;
87  bpSei.initialCpbRemovalOffset[j][i][0] = uiInitialCpbRemovalDelay;
88  bpSei.initialCpbRemovalOffset[j][i][1] = uiInitialCpbRemovalDelay;
89  }
90  }
91  // We don't set concatenation_flag here. max_initial_removal_delay_for_concatenation depends on the usage scenario.
92  // The parameters could be added to config file, but as long as the initialisation of generic buffering parameters is
93  // not controllable, it does not seem to make sense to provide settings for these.
94  bpSei.concatenationFlag = false;
95  bpSei.maxInitialRemovalDelayForConcatenation = uiInitialCpbRemovalDelay;
96  bpSei.bpDecodingUnitHrdParamsPresent = false;//m_pcEncCfg->m_noPicPartitionFlag == false;
97  bpSei.decodingUnitCpbParamsInPicTimingSeiFlag = !m_pcEncCfg->m_decodingUnitInfoSEIEnabled;
98  bpSei.initialCpbRemovalDelayLength = 16; // assuming 0.5 sec, log2( 90,000 * 0.5 ) = 16-bit
99  // Note: The following parameters require some knowledge about the GOP structure.
100  // Using getIntraPeriod() should be avoided though, because it assumes certain GOP
101  // properties, which are only valid in CTC.
102  // Still copying this setting from HM for consistency, improvements welcome
103  bool isRandomAccess = m_pcEncCfg->m_IntraPeriod > 0;
104  if( isRandomAccess && m_pcEncCfg->m_IntraPeriod < 256)
105  {
106  bpSei.cpbRemovalDelayLength = // 6 // 32 = 2^5 (plus 1)
107  bpSei.dpbOutputDelayLength = ceilLog2( m_pcEncCfg->m_IntraPeriod)+1; // 6 // 32 + 3 = 2^6
108  }
109  else
110  {
111  bpSei.cpbRemovalDelayLength = 9; // max. 2^10
112  bpSei.dpbOutputDelayLength = 9; // max. 2^10
113  }
114  bpSei.duCpbRemovalDelayIncrementLength = 7; // ceil( log2( tick_divisor_minus2 + 2 ) )
115  bpSei.dpbOutputDelayDuLength = bpSei.dpbOutputDelayLength + bpSei.duCpbRemovalDelayIncrementLength;
116  //for the concatenation, it can be set to one during splicing.
117  bpSei.concatenationFlag = 0;
118  //since the temporal layer HRDParameters is not ready, we assumed it is fixed
119  bpSei.auCpbRemovalDelayDelta = 1;
120  bool bpDeltasGOPStructure = m_pcEncCfg->m_GOPSize == 8 || m_pcEncCfg->m_GOPSize == 16; //assume GOPs specified as in CTC
121  bpSei.cpbRemovalDelayDeltasPresent = bpDeltasGOPStructure;
122  if (bpSei.cpbRemovalDelayDeltasPresent)
123  {
124  switch (m_pcEncCfg->m_GOPSize)
125  {
126  case 8:
127  {
128  if (noLeadingPictures)
129  {
130  bpSei.numCpbRemovalDelayDeltas = 5;
131  bpSei.cpbRemovalDelayDelta[0] = 1;
132  bpSei.cpbRemovalDelayDelta[1] = 2;
133  bpSei.cpbRemovalDelayDelta[2] = 3;
134  bpSei.cpbRemovalDelayDelta[3] = 6;
135  bpSei.cpbRemovalDelayDelta[4] = 7;
136  }
137  else
138  {
139  bpSei.numCpbRemovalDelayDeltas = 3;
140  bpSei.cpbRemovalDelayDelta[0] = 1;
141  bpSei.cpbRemovalDelayDelta[1] = 2;
142  bpSei.cpbRemovalDelayDelta[2] = 3;
143  }
144  }
145  break;
146  case 16:
147  {
148  if (noLeadingPictures)
149  {
150  bpSei.numCpbRemovalDelayDeltas = 9;
151  bpSei.cpbRemovalDelayDelta[0] = 1;
152  bpSei.cpbRemovalDelayDelta[1] = 2;
153  bpSei.cpbRemovalDelayDelta[2] = 3;
154  bpSei.cpbRemovalDelayDelta[3] = 4;
155  bpSei.cpbRemovalDelayDelta[4] = 6;
156  bpSei.cpbRemovalDelayDelta[5] = 7;
157  bpSei.cpbRemovalDelayDelta[6] = 9;
158  bpSei.cpbRemovalDelayDelta[7] = 14;
159  bpSei.cpbRemovalDelayDelta[8] = 15;
160  }
161  else
162  {
163  bpSei.numCpbRemovalDelayDeltas = 5;
164  bpSei.cpbRemovalDelayDelta[0] = 1;
165  bpSei.cpbRemovalDelayDelta[1] = 2;
166  bpSei.cpbRemovalDelayDelta[2] = 3;
167  bpSei.cpbRemovalDelayDelta[3] = 6;
168  bpSei.cpbRemovalDelayDelta[4] = 7;
169  }
170  }
171  break;
172  default:
173  {
174  THROW("m_cpbRemovalDelayDelta not applicable for the GOP size");
175  }
176  break;
177  }
178  }
179  bpSei.sublayerDpbOutputOffsetsPresent = true;
180  for(int i = 0; i < bpSei.bpMaxSubLayers; i++)
181  {
182  bpSei.dpbOutputTidOffset[i] = m_pcEncCfg->m_maxNumReorderPics[i] * (1<<(bpSei.bpMaxSubLayers-1-i));
183  if(bpSei.dpbOutputTidOffset[i] >= m_pcEncCfg->m_maxNumReorderPics[bpSei.bpMaxSubLayers-1])
184  {
185  bpSei.dpbOutputTidOffset[i] -= m_pcEncCfg->m_maxNumReorderPics[bpSei.bpMaxSubLayers-1];
186  }
187  else
188  {
189  bpSei.dpbOutputTidOffset[i] = 0;
190  }
191  }
192  // A commercial encoder should track the buffer state for all layers and sub-layers
193  // to ensure CPB conformance. Such tracking is required for calculating alternative
194  // CPB parameters.
195  // Unfortunately VTM does not have such tracking. Thus we cannot encode alternative
196  // CPB parameters here.
197  bpSei.altCpbParamsPresent = false;
198  bpSei.useAltCpbParamsFlag = false;
199 }

202 void SEIEncoder::initDecodedPictureHashSEI( SEIDecodedPictureHash& dphSei, const CPelUnitBuf& pic, std::string &rHashString, const BitDepths &bitDepths)
203 {
204  CHECK(!(m_isInitialized), "Unspecified error");
205 
206  dphSei.method = m_pcEncCfg->m_decodedPictureHashSEIType;
207  dphSei.singleCompFlag = m_pcEncCfg->m_internChromaFormat == 0;
208 
209  switch (m_pcEncCfg->m_decodedPictureHashSEIType)
210  {
211  case VVENC_HASHTYPE_MD5:
212  {
213  uint32_t numChar=calcMD5(pic, dphSei.pictureHash, bitDepths);
214  rHashString = hashToString(dphSei.pictureHash, numChar);
215  }
216  break;
217  case VVENC_HASHTYPE_CRC:
218  {
219  uint32_t numChar=calcCRC(pic, dphSei.pictureHash, bitDepths);
220  rHashString = hashToString(dphSei.pictureHash, numChar);
221  }
222  break;
223  case VVENC_HASHTYPE_CHECKSUM:
224  default:
225  {
226  uint32_t numChar=calcChecksum(pic, dphSei.pictureHash, bitDepths);
227  rHashString = hashToString(dphSei.pictureHash, numChar);
228  }
229  break;
230  }
231 }

233 void SEIEncoder::initPictureTimingSEI( SEIMessages& seiMessages, SEIMessages& nestedSeiMessages, SEIMessages& duInfoSeiMessages, const Slice *slice, const uint32_t numDU, const bool bpPresentInAU)
234 {
235  // Picture timing depends on buffering period. When either of those is not disabled,
236  // initialization would fail. Needs more cleanup after DU timing is integrated.
237  if (!(m_pcEncCfg->m_pictureTimingSEIEnabled && m_pcEncCfg->m_bufferingPeriodSEIEnabled))
238  {
239  return;
240  }
241 
242  const GeneralHrdParams *hrd = &slice->sps->generalHrdParams;
243 
244  // update decoding unit parameters
245  if ((m_pcEncCfg->m_pictureTimingSEIEnabled || m_pcEncCfg->m_bufferingPeriodSEIEnabled) && slice->nuhLayerId == slice->vps->layerId[0])
246  {
247  int picSptDpbOutputDuDelay = 0;
248  SEIPictureTiming *ptSei = new SEIPictureTiming();
249  const uint32_t maxNumSubLayers = slice->sps->maxTLayers;
250 
251  // DU parameters
252  if( hrd->generalDecodingUnitHrdParamsPresent )
253  {
254  ptSei->numDecodingUnitsMinus1 = numDU - 1;
255  ptSei->duCommonCpbRemovalDelayFlag = false;
256  ptSei->numNalusInDuMinus1.resize( numDU );
257  ptSei->duCpbRemovalDelayMinus1.resize( numDU * maxNumSubLayers );
258  }
259  const uint32_t cpbRemovalDelayLegth = m_pcEncHRD->bufferingPeriodSEI.cpbRemovalDelayLength;
260  ptSei->auCpbRemovalDelay[maxNumSubLayers-1] = std::min<int>(std::max<int>(1, m_totalCoded[maxNumSubLayers-1] - m_lastBPSEI[maxNumSubLayers-1]), (1<<cpbRemovalDelayLegth)); // Syntax element signalled as minus, hence the .
261  CHECK( (m_totalCoded[maxNumSubLayers-1] - m_lastBPSEI[maxNumSubLayers-1]) > (1<<cpbRemovalDelayLegth), " cpbRemovalDelayLegth too small for m_auCpbRemovalDelay[pt_max_sub_layers_minus1] at picture timing SEI " );
262  const uint32_t temporalId = slice->TLayer;
263  for( int i = temporalId ; i < maxNumSubLayers - 1 ; i ++ )
264  {
265  int indexWithinGOP = (m_totalCoded[maxNumSubLayers - 1] - m_lastBPSEI[maxNumSubLayers - 1]) % m_pcEncCfg->m_GOPSize;
266  ptSei->ptSubLayerDelaysPresent[i] = true;
267  if( ((m_rapWithLeading == true) && (indexWithinGOP == 0)) || (m_totalCoded[maxNumSubLayers - 1] == 0) || bpPresentInAU || (slice->poc + m_pcEncCfg->m_GOPSize) > m_pcEncCfg->m_framesToBeEncoded )
268  {
269  ptSei->cpbRemovalDelayDeltaEnabledFlag[i] = false;
270  }
271  else
272  {
273  ptSei->cpbRemovalDelayDeltaEnabledFlag[i] = m_pcEncHRD->bufferingPeriodSEI.cpbRemovalDelayDeltasPresent;
274  }
275  if( ptSei->cpbRemovalDelayDeltaEnabledFlag[i] )
276  {
277  if( m_rapWithLeading == false )
278  {
279  switch (m_pcEncCfg->m_GOPSize)
280  {
281  case 8:
282  {
283  if((indexWithinGOP == 1 && i == 2))
284  {
285  ptSei->cpbRemovalDelayDeltaIdx[i] = 0;
286  }
287  else if((indexWithinGOP == 2 && i == 2) || (indexWithinGOP == 6 && i == 2))
288  {
289  ptSei->cpbRemovalDelayDeltaIdx[i] = 1;
290  }
291  else if((indexWithinGOP == 1 && i == 1) || (indexWithinGOP == 3 && i == 2))
292  {
293  ptSei->cpbRemovalDelayDeltaIdx[i] = 2;
294  }
295  else if(indexWithinGOP == 2 && i == 1)
296  {
297  ptSei->cpbRemovalDelayDeltaIdx[i] = 3;
298  }
299  else if(indexWithinGOP == 1 && i == 0)
300  {
301  ptSei->cpbRemovalDelayDeltaIdx[i] = 4;
302  }
303  else
304  {
305  THROW("m_cpbRemovalDelayDeltaIdx not applicable for the sub-layer and GOP size");
306  }
307  }
308  break;
309  case 16:
310  {
311  if((indexWithinGOP == 1 && i == 3))
312  {
313  ptSei->cpbRemovalDelayDeltaIdx[i] = 0;
314  }
315  else if((indexWithinGOP == 2 && i == 3) || (indexWithinGOP == 10 && i == 3) || (indexWithinGOP == 14 && i == 3))
316  {
317  ptSei->cpbRemovalDelayDeltaIdx[i] = 1;
318  }
319  else if((indexWithinGOP == 1 && i == 2) || (indexWithinGOP == 3 && i == 3) || (indexWithinGOP == 7 && i == 3) || (indexWithinGOP == 11 && i == 3))
320  {
321  ptSei->cpbRemovalDelayDeltaIdx[i] = 2;
322  }
323  else if(indexWithinGOP == 4 && i == 3)
324  {
325  ptSei->cpbRemovalDelayDeltaIdx[i] = 3;
326  }
327  else if((indexWithinGOP == 2 && i == 2) || (indexWithinGOP == 10 && i == 2))
328  {
329  ptSei->cpbRemovalDelayDeltaIdx[i] = 4;
330  }
331  else if(indexWithinGOP == 1 && i == 1)
332  {
333  ptSei->cpbRemovalDelayDeltaIdx[i] = 5;
334  }
335  else if(indexWithinGOP == 3 && i == 2)
336  {
337  ptSei->cpbRemovalDelayDeltaIdx[i] = 6;
338  }
339  else if(indexWithinGOP == 2 && i == 1)
340  {
341  ptSei->cpbRemovalDelayDeltaIdx[i] = 7;
342  }
343  else if(indexWithinGOP == 1 && i == 0)
344  {
345  ptSei->cpbRemovalDelayDeltaIdx[i] = 8;
346  }
347  else
348  {
349  THROW("m_cpbRemovalDelayDeltaIdx not applicable for the sub-layer and GOP size");
350  }
351  }
352  break;
353  default:
354  {
355  THROW("m_cpbRemovalDelayDeltaIdx not supported for the current GOP size");
356  }
357  break;
358  }
359  }
360  else
361  {
362  switch (m_pcEncCfg->m_GOPSize)
363  {
364  case 8:
365  {
366  if((indexWithinGOP == 1 && i == 2) || (indexWithinGOP == 5 && i == 2))
367  {
368  ptSei->cpbRemovalDelayDeltaIdx[i] = 0;
369  }
370  else if(indexWithinGOP == 2 && i == 2)
371  {
372  ptSei->cpbRemovalDelayDeltaIdx[i] = 1;
373  }
374  else if(indexWithinGOP == 1 && i == 1)
375  {
376  ptSei->cpbRemovalDelayDeltaIdx[i] = 2;
377  }
378  else
379  {
380  THROW("m_cpbRemovalDelayDeltaIdx not applicable for the sub-layer and GOP size");
381  }
382  }
383  break;
384  case 16:
385  {
386  if((indexWithinGOP == 1 && i == 3) || (indexWithinGOP == 9 && i == 3) || (indexWithinGOP == 13 && i == 3))
387  {
388  ptSei->cpbRemovalDelayDeltaIdx[i] = 0;
389  }
390  else if((indexWithinGOP == 2 && i == 3) || (indexWithinGOP == 6 && i == 3) || (indexWithinGOP == 10 && i == 3))
391  {
392  ptSei->cpbRemovalDelayDeltaIdx[i] = 1;
393  }
394  else if((indexWithinGOP == 1 && i == 2) || (indexWithinGOP == 9 && i == 2) || (indexWithinGOP == 3 && i == 3))
395  {
396  ptSei->cpbRemovalDelayDeltaIdx[i] = 2;
397  }
398  else if(indexWithinGOP == 2 && i == 2)
399  {
400  ptSei->cpbRemovalDelayDeltaIdx[i] = 3;
401  }
402  else if(indexWithinGOP == 1 && i == 1)
403  {
404  ptSei->cpbRemovalDelayDeltaIdx[i] = 4;
405  }
406  else
407  {
408  THROW("m_cpbRemovalDelayDeltaIdx not applicable for the sub-layer and GOP size");
409  }
410  }
411  break;
412  default:
413  {
414  THROW("m_cpbRemovalDelayDeltaIdx not applicable for the sub-layer and GOP size");
415  }
416  break;
417  }
418  }
419  }
420  else
421  {
422  int scaledDistToBuffPeriod = (m_totalCoded[i] - m_lastBPSEI[i]) * (1<<(maxNumSubLayers - 1 - i));
423  ptSei->auCpbRemovalDelay[i] = std::min<int>(std::max<int>(1, scaledDistToBuffPeriod), (1<<cpbRemovalDelayLegth)); // Syntax element signalled as minus, hence the .
424  CHECK( scaledDistToBuffPeriod > (1<<cpbRemovalDelayLegth), " cpbRemovalDelayLegth too small for m_auCpbRemovalDelay[i] at picture timing SEI " );
425  }
426  }
427  ptSei->picDpbOutputDelay = slice->sps->numReorderPics[slice->sps->maxTLayers-1] + slice->poc - m_totalCoded[maxNumSubLayers-1];
428 // if(m_pcEncCfg->m_efficientFieldIRAPEnabled && IRAPGOPid > 0 && IRAPGOPid < m_iGopSize)
429 // {
430 // // if pictures have been swapped there is likely one more picture delay on their tid. Very rough approximation
431 // ptSei->picDpbOutputDelay ++;
432 // }
433  int factor = hrd->tickDivisorMinus2 + 2;
434  ptSei->picDpbOutputDuDelay = factor * ptSei->picDpbOutputDelay;
435  if( m_pcEncCfg->m_decodingUnitInfoSEIEnabled )
436  {
437  picSptDpbOutputDuDelay = factor * ptSei->picDpbOutputDelay;
438  }
439  if( bpPresentInAU )
440  {
441  for( int i = temporalId ; i < maxNumSubLayers ; i ++ )
442  {
443  m_lastBPSEI[i] = m_totalCoded[i];
444  }
445  if( (slice->nalUnitType == VVENC_NAL_UNIT_CODED_SLICE_IDR_W_RADL)||(slice->nalUnitType == VVENC_NAL_UNIT_CODED_SLICE_CRA) )
446  {
447  m_rapWithLeading = true;
448  }
449  }
450 
451 
452  if( m_pcEncCfg->m_pictureTimingSEIEnabled )
453  {
454  seiMessages.push_back( ptSei );
455 
456 // if (m_pcEncCfg->m_scalableNestingSEIEnabled && !m_pcEncCfg->m_samePicTimingInAllOLS)
457 // {
458 // SEIPictureTiming *pictureTimingSEIcopy = new SEIPictureTiming();
459 // *pictureTimingSEIcopy = *pictureTimingSEI;
460 // nestedSeiMessages.push_back(pictureTimingSEIcopy);
461 // }
462  }
463 
464  if( m_pcEncCfg->m_decodingUnitInfoSEIEnabled && hrd->generalDecodingUnitHrdParamsPresent )
465  {
466  for( int i = 0; i < ( ptSei->numDecodingUnitsMinus1 + 1 ); i ++ )
467  {
468  SEIDecodingUnitInfo *duInfoSEI = new SEIDecodingUnitInfo();
469  duInfoSEI->decodingUnitIdx = i;
470  for( int j = temporalId; j <= maxNumSubLayers; j++ )
471  {
472  duInfoSEI->duSptCpbRemovalDelayIncrement[j] = ptSei->duCpbRemovalDelayMinus1[i*maxNumSubLayers+j] + 1;
473  }
474  duInfoSEI->dpbOutputDuDelayPresent = false;
475  duInfoSEI->picSptDpbOutputDuDelay = picSptDpbOutputDuDelay;
476 
477  duInfoSeiMessages.push_back(duInfoSEI);
478  }
479  }
480 
481  if( !m_pcEncCfg->m_pictureTimingSEIEnabled && ptSei )
482  {
483  delete ptSei;
484  }
485  }
486 
487  // not sure if this is the final place
488  for( uint32_t i = slice->TLayer; i < slice->sps->maxTLayers; i ++ )
489  {
490  m_totalCoded[i]++;
491  }
492 }
