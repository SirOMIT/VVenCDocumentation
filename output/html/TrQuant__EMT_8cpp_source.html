370 template< int uiTrSize >
371 inline void _fastForwardMM( const TCoeff *src, TCoeff *dst, int shift, int line, int iSkipLine, int iSkipLine2, const TMatrixCoeff* tc )
372 {
373 #if !ENABLE_SIMD_TRAFO
374  const int rnd_factor = 1 << (shift - 1);
375 #endif
376  const int reducedLine = line - iSkipLine;
377  const int cutoff = uiTrSize - iSkipLine2;
378  TCoeff *pCoef;
379 
380 #if ENABLE_SIMD_TRAFO
381  if( line == 1 )
382  {
383  if( uiTrSize == 4 )
384  g_tCoeffOps.fastFwdCore4_1D( tc, src, dst, uiTrSize, line, reducedLine, cutoff, shift );
385  else
386  g_tCoeffOps.fastFwdCore8_1D( tc, src, dst, uiTrSize, line, reducedLine, cutoff, shift );
387  }
388  else
389  {
390  if( uiTrSize == 4 )
391  g_tCoeffOps.fastFwdCore4_2D( tc, src, dst, uiTrSize, line, reducedLine, cutoff, shift );
392  else
393  g_tCoeffOps.fastFwdCore8_2D( tc, src, dst, uiTrSize, line, reducedLine, cutoff, shift );
394  }
395 #else
396  for( int i = 0; i<reducedLine; i++ )
397  {
398  pCoef = dst;
399  const TMatrixCoeff* iT = tc;
400  for( int j = 0; j<cutoff; j++ )
401  {
402  int iSum = 0;
403  for( int k = 0; k<uiTrSize; k++ )
404  {
405  // dst[j * line + i] += src[i * trSize + k] * t[j * trSize + k]
406  iSum += src[k] * iT[k];
407  }
408  pCoef[i] = (iSum + rnd_factor) >> shift;

411  }
412  src += uiTrSize;
413  }
414 #endif
415 
416  if( iSkipLine )
417  {
418  pCoef = dst + reducedLine;
419  for( int j = 0; j<cutoff; j++ )
420  {
421  memset(pCoef, 0, sizeof(TCoeff) * iSkipLine);
422  pCoef += line;
423  }
424  }
425 
426  if( iSkipLine2 )
427  {
428  pCoef = dst + line*cutoff;
429  memset(pCoef, 0, sizeof(TCoeff) * line * iSkipLine2);
430  }
431 }
