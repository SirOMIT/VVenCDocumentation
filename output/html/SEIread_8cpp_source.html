134 {
135  SEIMessages seiListInCurNalu;
136  setBitstream(bs);
137  CHECK(m_pcBitstream->getNumBitsUntilByteAligned(), "Bitstream not aligned");
138 
139  do
140  {
141  xReadSEImessage(seis, nalUnitType, nuh_layer_id, temporalId, vps, sps, hrd, pDecodedMessageOutputStream);

143  /* SEI messages are an integer number of bytes, something has failed

155 void SEIReader::xReadSEImessage(SEIMessages& seis, const vvencNalUnitType nalUnitType, const uint32_t nuh_layer_id, const uint32_t temporalId, const VPS *vps, const SPS *sps, HRD &hrd, std::ostream *pDecodedMessageOutputStream)
156 {
157  DTRACE( g_trace_ctx, D_HEADER, "=========== SEI message ===========\n" );
158 
159  int payloadType = 0;
160  uint32_t val = 0;
161 
162  do
163  {
164  sei_read_code(NULL, 8, val, "payload_type");
165  payloadType += val;
166  } while (val==0xFF);
167 
168  uint32_t payloadSize = 0;
169  do
170  {
171  sei_read_code(NULL, 8, val, "payload_size");
172  payloadSize += val;
173  } while (val==0xFF);
174 
175  DTRACE( g_trace_ctx, D_HEADER, "=========== %s SEI message ===========\n", SEI::getSEIMessageString( (SEI::PayloadType)payloadType ) );
176 
177  /* extract the payload for this single SEI message.
178  * This allows greater safety in erroneous parsing of an SEI message
179  * from affecting subsequent messages.
180  * After parsing the payload, bs needs to be restored as the primary
181  * bitstream.
182  */
183  InputBitstream *bs = getBitstream();
184  setBitstream(bs->extractSubstream(payloadSize * 8));
185 
186  SEI *sei = NULL;
187  const SEIBufferingPeriod *bp = NULL;
188  const SEIPictureTiming *pt = NULL;
189 
190  if(nalUnitType == VVENC_NAL_UNIT_PREFIX_SEI)
191  {
192  switch (payloadType)
193  {
194  case SEI::USER_DATA_UNREGISTERED:
195  sei = new SEIuserDataUnregistered;
196  xParseSEIuserDataUnregistered((SEIuserDataUnregistered&) *sei, payloadSize, pDecodedMessageOutputStream);
197  break;
198  case SEI::DECODING_UNIT_INFO:
199  bp = &hrd.bufferingPeriodSEI;
200  if (!bp)
201  {
202  msg( VVENC_WARNING, "Warning: Found Decoding unit information SEI message, but no active buffering period is available. Ignoring.");
203  }
204  else
205  {
206  sei = new SEIDecodingUnitInfo;
207  xParseSEIDecodingUnitInfo((SEIDecodingUnitInfo&) *sei, payloadSize, *bp, temporalId, pDecodedMessageOutputStream);
208  }
209  break;
210  case SEI::BUFFERING_PERIOD:
211  sei = new SEIBufferingPeriod;
212  xParseSEIBufferingPeriod((SEIBufferingPeriod&) *sei, payloadSize, pDecodedMessageOutputStream);
213  hrd.bufferingPeriodSEI = *((SEIBufferingPeriod*) sei);
214  break;
215  case SEI::PICTURE_TIMING:
216  {
217  bp = &hrd.bufferingPeriodSEI;
218  if (!bp)
219  {
220  msg( VVENC_WARNING, "Warning: Found Picture timing SEI message, but no active buffering period is available. Ignoring.");
221  }
222  else
223  {
224  sei = new SEIPictureTiming;
225  xParseSEIPictureTiming((SEIPictureTiming&)*sei, payloadSize, temporalId, *bp, pDecodedMessageOutputStream);
226  hrd.pictureTimingSEI = *( (SEIPictureTiming*) sei );
227  }
228  }
229  break;
230  case SEI::SCALABLE_NESTING:
231  sei = new SEIScalableNesting;
232  xParseSEIScalableNesting((SEIScalableNesting&)*sei, nalUnitType, nuh_layer_id, payloadSize, vps, sps, pDecodedMessageOutputStream);
233  break;
234  case SEI::FRAME_FIELD_INFO:
235  sei = new SEIFrameFieldInfo;
236  pt = &hrd.pictureTimingSEI;
237  xParseSEIFrameFieldinfo((SEIFrameFieldInfo&) *sei, *pt, payloadSize, pDecodedMessageOutputStream);
238  break;
239  case SEI::DEPENDENT_RAP_INDICATION:
240  sei = new SEIDependentRAPIndication;
241  xParseSEIDependentRAPIndication((SEIDependentRAPIndication&) *sei, payloadSize, pDecodedMessageOutputStream);
242  break;
243  case SEI::FRAME_PACKING:
244  sei = new SEIFramePacking;
245  xParseSEIFramePacking((SEIFramePacking&) *sei, payloadSize, pDecodedMessageOutputStream);
246  break;
247  case SEI::PARAMETER_SETS_INCLUSION_INDICATION:
248  sei = new SEIParameterSetsInclusionIndication;
249  xParseSEIParameterSetsInclusionIndication((SEIParameterSetsInclusionIndication&)*sei, payloadSize, pDecodedMessageOutputStream);
250  break;
251  case SEI::MASTERING_DISPLAY_COLOUR_VOLUME:
252  sei = new SEIMasteringDisplayColourVolume;
253  xParseSEIMasteringDisplayColourVolume((SEIMasteringDisplayColourVolume&) *sei, payloadSize, pDecodedMessageOutputStream);
254  break;
255  case SEI::ALTERNATIVE_TRANSFER_CHARACTERISTICS:
256  sei = new SEIAlternativeTransferCharacteristics;
257  xParseSEIAlternativeTransferCharacteristics((SEIAlternativeTransferCharacteristics&) *sei, payloadSize, pDecodedMessageOutputStream);
258  break;
259  case SEI::EQUIRECTANGULAR_PROJECTION:
260  sei = new SEIEquirectangularProjection;
261  xParseSEIEquirectangularProjection((SEIEquirectangularProjection&) *sei, payloadSize, pDecodedMessageOutputStream);
262  break;
263  case SEI::SPHERE_ROTATION:
264  sei = new SEISphereRotation;
265  xParseSEISphereRotation((SEISphereRotation&) *sei, payloadSize, pDecodedMessageOutputStream);
266  break;
267  case SEI::OMNI_VIEWPORT:
268  sei = new SEIOmniViewport;
269  xParseSEIOmniViewport((SEIOmniViewport&) *sei, payloadSize, pDecodedMessageOutputStream);
270  break;
271  case SEI::REGION_WISE_PACKING:
272  sei = new SEIRegionWisePacking;
273  xParseSEIRegionWisePacking((SEIRegionWisePacking&) *sei, payloadSize, pDecodedMessageOutputStream);
274  break;
275  case SEI::GENERALIZED_CUBEMAP_PROJECTION:
276  sei = new SEIGeneralizedCubemapProjection;
277  xParseSEIGeneralizedCubemapProjection((SEIGeneralizedCubemapProjection&) *sei, payloadSize, pDecodedMessageOutputStream);
278  break;
279  case SEI::SUBPICTURE_LEVEL_INFO:
280  sei = new SEISubpicureLevelInfo;
281  xParseSEISubpictureLevelInfo((SEISubpicureLevelInfo&) *sei, payloadSize, pDecodedMessageOutputStream);
282  break;
283  case SEI::SAMPLE_ASPECT_RATIO_INFO:
284  sei = new SEISampleAspectRatioInfo;
285  xParseSEISampleAspectRatioInfo((SEISampleAspectRatioInfo&) *sei, payloadSize, pDecodedMessageOutputStream);
286  break;
287  case SEI::USER_DATA_REGISTERED_ITU_T_T35:
288  sei = new SEIUserDataRegistered;
289  xParseSEIUserDataRegistered((SEIUserDataRegistered&)*sei, payloadSize, pDecodedMessageOutputStream);
290  break;
291  case SEI::FILM_GRAIN_CHARACTERISTICS:
292  sei = new SEIFilmGrainCharacteristics;
293  xParseSEIFilmGrainCharacteristics((SEIFilmGrainCharacteristics&)*sei, payloadSize, pDecodedMessageOutputStream);
294  break;
295  case SEI::CONTENT_LIGHT_LEVEL_INFO:
296  sei = new SEIContentLightLevelInfo;
297  xParseSEIContentLightLevelInfo((SEIContentLightLevelInfo&)*sei, payloadSize, pDecodedMessageOutputStream);
298  break;
299  case SEI::AMBIENT_VIEWING_ENVIRONMENT:
300  sei = new SEIAmbientViewingEnvironment;
301  xParseSEIAmbientViewingEnvironment((SEIAmbientViewingEnvironment&)*sei, payloadSize, pDecodedMessageOutputStream);
302  break;
303  case SEI::CONTENT_COLOUR_VOLUME:
304  sei = new SEIContentColourVolume;
305  xParseSEIContentColourVolume((SEIContentColourVolume&)*sei, payloadSize, pDecodedMessageOutputStream);
306  break;
307  default:
308  for (uint32_t i = 0; i < payloadSize; i++)
309  {
310  uint32_t seiByte;
311  sei_read_code (NULL, 8, seiByte, "unknown prefix SEI payload byte");
312  }
313  msg( VVENC_WARNING, "Unknown prefix SEI message (payloadType = %d) was found!\n", payloadType);
314  if (pDecodedMessageOutputStream)
315  {
316  (*pDecodedMessageOutputStream) << "Unknown prefix SEI message (payloadType = " << payloadType << ") was found!\n";
317  }
318  break;
319  }
320  }
321  else
322  {
323  switch (payloadType)
324  {
325  case SEI::USER_DATA_UNREGISTERED:
326  sei = new SEIuserDataUnregistered;
327  xParseSEIuserDataUnregistered((SEIuserDataUnregistered&) *sei, payloadSize, pDecodedMessageOutputStream);
328  break;
329  case SEI::DECODED_PICTURE_HASH:
330  sei = new SEIDecodedPictureHash;
331  xParseSEIDecodedPictureHash((SEIDecodedPictureHash&) *sei, payloadSize, pDecodedMessageOutputStream);
332  break;
333  case SEI::SCALABLE_NESTING:
334  sei = new SEIScalableNesting;
335  xParseSEIScalableNesting((SEIScalableNesting&)*sei, nalUnitType, nuh_layer_id, payloadSize, vps, sps, pDecodedMessageOutputStream);
336  break;
337  default:
338  for (uint32_t i = 0; i < payloadSize; i++)
339  {
340  uint32_t seiByte;
341  sei_read_code( NULL, 8, seiByte, "unknown suffix SEI payload byte");
342  }
343  msg( VVENC_WARNING, "Unknown suffix SEI message (payloadType = %d) was found!\n", payloadType);
344  if (pDecodedMessageOutputStream)
345  {
346  (*pDecodedMessageOutputStream) << "Unknown suffix SEI message (payloadType = " << payloadType << ") was found!\n";
347  }
348  break;
349  }
350  }
351 
352  if (sei != NULL)
353  {
354  seis.push_back(sei);
355  }
356 
357  /* By definition the underlying bitstream terminates in a byte-aligned manner.
358  * 1. Extract all bar the last MIN(bitsremaining,nine) bits as reserved_payload_extension_data
359  * 2. Examine the final 8 bits to determine the payload_bit_equal_to_one marker
360  * 3. Extract the remainingreserved_payload_extension_data bits.
361  *
362  * If there are fewer than 9 bits available, extract them.
363  */
364  int payloadBitsRemaining = getBitstream()->getNumBitsLeft();
365  if (payloadBitsRemaining) /* more_data_in_payload() */
366  {
367  for (; payloadBitsRemaining > 9; payloadBitsRemaining--)
368  {
369  uint32_t reservedPayloadExtensionData;
370  sei_read_code ( pDecodedMessageOutputStream, 1, reservedPayloadExtensionData, "reserved_payload_extension_data");
371  }
372 
373  /* 2 */
374  int finalBits = getBitstream()->peekBits(payloadBitsRemaining);
375  int finalPayloadBits = 0;
376  for (int mask = 0xff; finalBits & (mask >> finalPayloadBits); finalPayloadBits++)
377  {
378  continue;
379  }
380 
381  /* 3 */
382  for (; payloadBitsRemaining > 9 - finalPayloadBits; payloadBitsRemaining--)
383  {
384  uint32_t reservedPayloadExtensionData;
385  sei_read_flag ( 0, reservedPayloadExtensionData, "reserved_payload_extension_data");
386  }
387 
388  uint32_t dummy;
389  sei_read_flag( 0, dummy, "payload_bit_equal_to_one"); payloadBitsRemaining--;
390  while (payloadBitsRemaining)
391  {
392  sei_read_flag( 0, dummy, "payload_bit_equal_to_zero"); payloadBitsRemaining--;
393  }
394  }
395 
396  /* restore primary bitstream for sei_message */
397  delete getBitstream();
398  setBitstream(bs);
399 }

441 void SEIReader::xParseSEIDecodedPictureHash(SEIDecodedPictureHash& sei, uint32_t payloadSize, std::ostream *pDecodedMessageOutputStream)
442 {
443  uint32_t bytesRead = 0;
444  output_sei_message_header(sei, pDecodedMessageOutputStream, payloadSize);
445 
446  uint32_t val;
447  sei_read_code( pDecodedMessageOutputStream, 8, val, "dpb_sei_hash_type");

449  sei_read_code( pDecodedMessageOutputStream, 1, val, "dph_sei_single_component_flag");

487 void SEIReader::xParseSEIScalableNesting(SEIScalableNesting& sei, const vvencNalUnitType nalUnitType, const uint32_t nuhLayerId, uint32_t payloadSize, const VPS *vps, const SPS *sps, std::ostream *decodedMessageOutputStream)
488 {
489  uint32_t symbol;
490  SEIMessages seis;
491  output_sei_message_header(sei, decodedMessageOutputStream, payloadSize);
492 
493  sei_read_flag(decodedMessageOutputStream, symbol, "sn_ols_flag"); sei.snOlsFlag = symbol;
494  sei_read_flag(decodedMessageOutputStream, symbol, "sn_subpic_flag"); sei.snSubpicFlag = symbol;
495  if (sei.snOlsFlag)
496  {
497  sei_read_uvlc(decodedMessageOutputStream, symbol, "sn_num_olss_minus1"); sei.snNumOlssMinus1 = symbol;
498  for (uint32_t i = 0; i <= sei.snNumOlssMinus1; i++)
499  {
500  sei_read_uvlc(decodedMessageOutputStream, symbol, "sn_ols_idx_delta_minus1[i]"); sei.snOlsIdxDeltaMinus1[i] = symbol;
501  }
502  for (uint32_t i = 0; i <= sei.snNumOlssMinus1; i++)
503  {
504  if (i == 0)
505  {
506  sei.snOlsIdx[i] = sei.snOlsIdxDeltaMinus1[i];
507  }
508  else
509  {
510  sei.snOlsIdx[i] = sei.snOlsIdxDeltaMinus1[i] + sei.snOlsIdxDeltaMinus1[i - 1] + 1;
511  }
512  }
513  if (vps && vps->vpsId != 0)
514  {
515  uint32_t lowestLayerId = MAX_UINT;
516  for (uint32_t olsIdxForSEI = 0; olsIdxForSEI <= sei.snNumOlssMinus1; olsIdxForSEI++)
517  {
518  int olsIdx = sei.snOlsIdx[olsIdxForSEI];
519  for (int layerIdx = 0; layerIdx < vps->numLayersInOls[olsIdx]; layerIdx++)
520  {
521  if (lowestLayerId > vps->layerIdInOls[olsIdx][layerIdx])
522  {
523  lowestLayerId = vps->layerIdInOls[olsIdx][layerIdx];
524  }
525  }
526  }
527  CHECK(lowestLayerId!= nuhLayerId, "nuh_layer_id is not equal to the lowest layer among Olss that the scalable SEI applies");
528  }
529  }
530  else
531  {
532  sei_read_flag(decodedMessageOutputStream, symbol, "sn_all_layers_flag"); sei.snAllLayersFlag = symbol;
533  if (!sei.snAllLayersFlag)
534  {
535  sei_read_uvlc(decodedMessageOutputStream, symbol, "sn_num_layers_minus1"); sei.snNumLayersMinus1 = symbol;
536  sei.snLayerId[0] = nuhLayerId;
537  for (uint32_t i = 1; i <= sei.snNumLayersMinus1; i++)
538  {
539  sei_read_code(decodedMessageOutputStream, 6, symbol, "sn_layer_id[i]"); sei.snLayerId[i] = symbol;
540  }
541  }
542  }
543  if (sei.snSubpicFlag)
544  {
545  sei_read_uvlc(decodedMessageOutputStream, symbol, "sn_num_subpics_minus1"); sei.snNumSubpics = symbol + 1;
546  sei_read_uvlc(decodedMessageOutputStream, symbol, "sn_subpic_id_len_minus1"); sei.snSubpicIdLen = symbol + 1;
547  sei.snSubpicId.resize(sei.snNumSubpics);
548  for (uint32_t i = 0; i < sei.snNumSubpics; i++)
549  {
550  sei_read_code(decodedMessageOutputStream, sei.snSubpicIdLen, symbol, "sn_subpic_id[i]"); sei.snSubpicId[i] = symbol;
551  }
552  }
553 
554  sei_read_uvlc(decodedMessageOutputStream, symbol, "sn_num_seis_minus1"); sei.snNumSEIs = symbol + 1;
