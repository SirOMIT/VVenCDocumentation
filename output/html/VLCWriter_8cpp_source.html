143 void VLCWriter::xWriteUvlc ( uint32_t uiCode )
144 {
145  uint32_t uiLength = 1;
146  uint32_t uiTemp = ++uiCode;
147 
148  CHECK( !uiTemp, "Integer overflow" );
149 
150  while( 1 != uiTemp )
151  {
152  uiTemp >>= 1;
153  uiLength += 2;
154  }
155  // Take care of cases where uiLength > 32
156  m_pcBitIf->write( 0, uiLength >> 1);
157  m_pcBitIf->write( uiCode, (uiLength+1) >> 1);
158 }

193 void HLSWriter::xCodeRefPicList( const ReferencePictureList* rpl, bool isLongTermPresent, uint32_t ltLsbBitsCount, const bool isForbiddenZeroDeltaPoc, int rplIdx )
194 {
195  uint32_t numRefPic = rpl->numberOfShorttermPictures + rpl->numberOfLongtermPictures + rpl->numberOfInterLayerPictures;
196  WRITE_UVLC(numRefPic, "num_ref_entries[ listIdx ][ rplsIdx ]");
197 
198  if (isLongTermPresent && numRefPic > 0 && rplIdx != -1)
199  {
200  WRITE_FLAG(rpl->ltrpInSliceHeader, "ltrp_in_slice_header_flag[ listIdx ][ rplsIdx ]");
201  }
202  int prevDelta = MAX_INT;
203  int deltaValue = 0;
204  bool firstSTRP = true;
205  for (int ii = 0; ii < numRefPic; ii++)
206  {
207  if( rpl->interLayerPresent )
208  {
209  WRITE_FLAG( rpl->isInterLayerRefPic[ii], "inter_layer_ref_pic_flag[ listIdx ][ rplsIdx ][ i ]" );
210 
211  if( rpl->isInterLayerRefPic[ii] )
212  {
213  CHECK( rpl->interLayerRefPicIdx[ii] < 0, "Wrong inter-layer reference index" );
214  WRITE_UVLC( rpl->interLayerRefPicIdx[ii], "ilrp_idx[ listIdx ][ rplsIdx ][ i ]" );
215  }
216  }
217 
218  if( !rpl->isInterLayerRefPic[ii] )
219  {
220  if (rpl->numberOfLongtermPictures)
221  {
222  WRITE_FLAG(!rpl->isLongtermRefPic[ii], "st_ref_pic_flag[ listIdx ][ rplsIdx ][ i ]");
223  }
224  if (!rpl->isLongtermRefPic[ii])
225  {
226  if (firstSTRP)
227  {
228  firstSTRP = false;
229  deltaValue = prevDelta = rpl->refPicIdentifier[ii];
230  }
231  else
232  {
233  deltaValue = rpl->refPicIdentifier[ii] - prevDelta;
234  prevDelta = rpl->refPicIdentifier[ii];
235  }
236  unsigned int absDeltaValue = (deltaValue < 0) ? 0 - deltaValue : deltaValue;
237  if( isForbiddenZeroDeltaPoc || ii == 0 )
238  {
239  CHECK( !absDeltaValue, "Zero delta POC is not used without WP" );
240  WRITE_UVLC( absDeltaValue - 1, "abs_delta_poc_st[ listIdx ][ rplsIdx ][ i ]" );
241  }
242  else
243  WRITE_UVLC(absDeltaValue, "abs_delta_poc_st[ listIdx ][ rplsIdx ][ i ]");
244  if (absDeltaValue > 0)
245  {
246  WRITE_FLAG((deltaValue < 0), "strp_entry_sign_flag[ listIdx ][ rplsIdx ][ i ]"); //0 means negative delta POC : 1 means positive
247  }
248  }
249  else if (!rpl->ltrpInSliceHeader)
250  {
251  WRITE_CODE(rpl->refPicIdentifier[ii], ltLsbBitsCount, "poc_lsb_lt[listIdx][rplsIdx][i]");
252  }
253  }
254  }

435 void HLSWriter::codeAlfAps( const APS* pcAPS )
436 {
437  const AlfParam& param = pcAPS->alfParam;
438 
439  WRITE_FLAG(param.newFilterFlag[CH_L], "alf_luma_new_filter");
440  const CcAlfFilterParam& paramCcAlf = pcAPS->ccAlfParam;
441  if (pcAPS->chromaPresent)
442  {
443  WRITE_FLAG(param.newFilterFlag[CH_C], "alf_chroma_new_filter");
444  WRITE_FLAG(paramCcAlf.newCcAlfFilter[COMP_Cb - 1], "alf_cc_cb_filter_signal_flag");
445  WRITE_FLAG(paramCcAlf.newCcAlfFilter[COMP_Cr - 1], "alf_cc_cr_filter_signal_flag");
446  }
447 
448  if (param.newFilterFlag[CH_L])
449  {
450  WRITE_FLAG( param.nonLinearFlag[CH_L], "alf_luma_clip" );
451 
452  WRITE_UVLC(param.numLumaFilters - 1, "alf_luma_num_filters_signalled_minus1");
453  if (param.numLumaFilters > 1)
454  {
455  const int len = ceilLog2( param.numLumaFilters);
456  for (int i = 0; i < MAX_NUM_ALF_CLASSES; i++)
457  {
458  WRITE_CODE(param.filterCoeffDeltaIdx[i], len, "alf_luma_coeff_delta_idx" );
459  }
460  }
461  alfFilter(param, false, 0);
462  }
463 
464  if (param.newFilterFlag[CH_C])
465  {
466  WRITE_FLAG(param.nonLinearFlag[CH_C], "alf_nonlinear_enable_flag_chroma");
467  if( VVENC_MAX_NUM_ALF_ALTERNATIVES_CHROMA > 1 )
468  {
469  WRITE_UVLC( param.numAlternativesChroma - 1, "alf_chroma_num_alts_minus1" );
470  }
471  for( int altIdx=0; altIdx < param.numAlternativesChroma; ++altIdx )
472  {
473  alfFilter(param, true, altIdx);
474  }
475  }
476 
477  for (int ccIdx = 0; ccIdx < 2; ccIdx++)
478  {
479  if (paramCcAlf.newCcAlfFilter[ccIdx])
480  {
481  const int filterCount = paramCcAlf.ccAlfFilterCount[ccIdx];
482  CHECK(filterCount > MAX_NUM_CC_ALF_FILTERS, "CC ALF Filter count is too large");
483  CHECK(filterCount == 0, "CC ALF Filter count is too small");
484 
485  if (MAX_NUM_CC_ALF_FILTERS > 1)
486  {
487  WRITE_UVLC(filterCount - 1, ccIdx == 0 ? "alf_cc_cb_filters_signalled_minus1" : "alf_cc_cr_filters_signalled_minus1");
488  }
489 
490  for (int filterIdx = 0; filterIdx < filterCount; filterIdx++)
491  {
492  AlfFilterShape alfShape(size_CC_ALF);
493 
494  const short *coeff = paramCcAlf.ccAlfCoeff[ccIdx][filterIdx];
495  // Filter coefficients
496  for (int i = 0; i < alfShape.numCoeff - 1; i++)
497  {
498  if (coeff[i] == 0)
499  {
500  WRITE_CODE(0, CCALF_BITS_PER_COEFF_LEVEL, ccIdx == 0 ? "alf_cc_cb_mapped_coeff_abs" : "alf_cc_cr_mapped_coeff_abs");
501  }
502  else
503  {
504  WRITE_CODE(1 + floorLog2(abs(coeff[i])), CCALF_BITS_PER_COEFF_LEVEL, ccIdx == 0 ? "alf_cc_cb_mapped_coeff_abs" : "alf_cc_cr_mapped_coeff_abs");
505  WRITE_FLAG(coeff[i] < 0 ? 1 : 0, ccIdx == 0 ? "alf_cc_cb_coeff_sign" : "alf_cc_cr_coeff_sign");
506  }
507  }
508 
509  DTRACE(g_trace_ctx, D_SYNTAX, "%s coeff filterIdx %d: ", ccIdx == 0 ? "Cb" : "Cr", filterIdx);
510  for (int i = 0; i < alfShape.numCoeff; i++)
511  {
512  DTRACE(g_trace_ctx, D_SYNTAX, "%d ", coeff[i]);
513  }
514  DTRACE(g_trace_ctx, D_SYNTAX, "\n");
515  }
516  }
517  }
