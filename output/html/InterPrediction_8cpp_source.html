92 void applyPROFCore(Pel* dst, int dstStride, const Pel* src, int srcStride, int width, int height, const Pel* gradX, const Pel* gradY, int gradStride, const int* dMvX, const int* dMvY, int dMvStride, const bool& bi, int shiftNum, Pel offset, const ClpRng& clpRng)

118 void gradFilterCore(const Pel* pSrc, int srcStride, int width, int height, int gradStride, Pel* gradX, Pel* gradY, const int bitDepth)
119 {
120  const Pel* srcTmp = pSrc + srcStride + 1;
121  Pel* gradXTmp = gradX + gradStride + 1;
122  Pel* gradYTmp = gradY + gradStride + 1;
123  int shift1 = 6;
124 
125  for (int y = 0; y < (height - 2 * BDOF_EXTEND_SIZE); y++)
126  {
127  for (int x = 0; x < (width - 2 * BDOF_EXTEND_SIZE); x++)
128  {
129  gradYTmp[x] = ( srcTmp[x + srcStride] >> shift1 ) - ( srcTmp[x - srcStride] >> shift1 );
130  gradXTmp[x] = ( srcTmp[x + 1] >> shift1 ) - ( srcTmp[x - 1] >> shift1 );
131  }
132  gradXTmp += gradStride;
133  gradYTmp += gradStride;
134  srcTmp += srcStride;
135  }
136 
137  if (PAD)
138  {
139  gradXTmp = gradX + gradStride + 1;
140  gradYTmp = gradY + gradStride + 1;
141  for (int y = 0; y < (height - 2 * BDOF_EXTEND_SIZE); y++)
142  {
143  gradXTmp[-1] = gradXTmp[0];
144  gradXTmp[width - 2 * BDOF_EXTEND_SIZE] = gradXTmp[width - 2 * BDOF_EXTEND_SIZE - 1];
145  gradXTmp += gradStride;
146 
147  gradYTmp[-1] = gradYTmp[0];
148  gradYTmp[width - 2 * BDOF_EXTEND_SIZE] = gradYTmp[width - 2 * BDOF_EXTEND_SIZE - 1];
149  gradYTmp += gradStride;
150  }
151 
152  gradXTmp = gradX + gradStride;
153  gradYTmp = gradY + gradStride;
154  ::memcpy(gradXTmp - gradStride, gradXTmp, sizeof(Pel)*(width));
155  ::memcpy(gradXTmp + (height - 2 * BDOF_EXTEND_SIZE)*gradStride, gradXTmp + (height - 2 * BDOF_EXTEND_SIZE - 1)*gradStride, sizeof(Pel)*(width));
156  ::memcpy(gradYTmp - gradStride, gradYTmp, sizeof(Pel)*(width));
157  ::memcpy(gradYTmp + (height - 2 * BDOF_EXTEND_SIZE)*gradStride, gradYTmp + (height - 2 * BDOF_EXTEND_SIZE - 1)*gradStride, sizeof(Pel)*(width));
158  }

190 template<int padSize>
191 void paddingCore(Pel *ptr, int stride, int width, int height)
192 {
193  /*left and right padding*/
194  Pel *ptrTemp1 = ptr;
195  Pel *ptrTemp2 = ptr + (width - 1);
196  ptrdiff_t offset = 0;
197  for (int i = 0; i < height; i++)
198  {
199  offset = stride * i;
200  for (int j = 1; j <= padSize; j++)
201  {
202  *(ptrTemp1 - j + offset) = *(ptrTemp1 + offset);
203  *(ptrTemp2 + j + offset) = *(ptrTemp2 + offset);
204  }
205  }
206  /*Top and Bottom padding*/
207  int numBytes = (width + padSize + padSize) * sizeof(Pel);
208  ptrTemp1 = (ptr - padSize);
209  ptrTemp2 = (ptr + (stride * (height - 1)) - padSize);
210  for (int i = 1; i <= padSize; i++)
211  {
212  memcpy(ptrTemp1 - (i * stride), (ptrTemp1), numBytes);
213  memcpy(ptrTemp2 + (i * stride), (ptrTemp2), numBytes);
214  }
215 }

1170 inline int32_t div_for_maxq7(int64_t N, int64_t D)
1171 {

1174  if (N < 0)
1175  {
1176  sign = 1;
1177  N = -N;
1178  }
1179 
1180  q = 0;
1181  D = (D << 3);
1182  if (N >= D)
1183  {
1184  N -= D;
1185  q++;
1186  }
1187  q = (q << 1);
1188 
1189  D = (D >> 1);
1190  if (N >= D)
1191  {
1192  N -= D;
1193  q++;
1194  }
1195  q = (q << 1);
1196 
1197  if (N >= (D >> 1))
1198  q++;
1199 
1200  if (sign)
1201  return (-q);
1202  return(q);
