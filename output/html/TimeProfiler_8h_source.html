119 {
120 public:
121  using rep = std::milli;
122  using clock = std::chrono::steady_clock;
123  using time_point = std::chrono::time_point<clock>;
124  using duration = std::chrono::duration<double, rep>;
125 
126 private:
127  time_point previous = clock::now();
128  STAGE m_eStage;
129  const unsigned m_numStages = P_STAGES + 1;
130  int m_iLevel;
131  int m_iExtData;
132  unsigned m_numBlkHor;
133  unsigned m_numBlkVer;
134  unsigned m_curWId;
135  unsigned m_curHId;
136 
137 public:
138  const time_point start_time = previous;
139  std::vector<duration> durations;
140 
141  TimeProfiler() : m_eStage( P_OTHER )
142  {
143  init();
144  }
145  void init()
146  {
147  durations.resize( m_numStages );
148  for( size_t i = 0; i < m_numStages; ++i ) { durations[i] = durations[i].zero(); }
149  }
150  TimeProfiler& operator()( STAGE s )
151  {
152  time_point now = clock::now();
153  durations[m_eStage] += ( now - previous );
154  previous = now;
155  m_eStage = s;
156  return *this;
157  }
158  TimeProfiler& operator+=( const TimeProfiler& other )
159  {
160  auto i1 = durations.begin();
161  auto i2 = other.durations.cbegin();
162  for( ; i1 != durations.end() && i2 != other.durations.cend(); ++i1, ++i2 )
163  {
164  *i1 += *i2;
165  }
166 
167  return *this;
168  }
169 
170  void start( STAGE s )
171  {
172  previous = clock::now();
173  m_eStage = s;
174  }
175  void stop( STAGE s )
176  {
177  time_point now = clock::now();
178  durations[m_eStage] += ( now - previous );
179  }
180  STAGE curStage() { return m_eStage; }

253 class TimeProfiler2D
254 {
255 public:
256  using clock = std::chrono::steady_clock;
257  using time_point = std::chrono::time_point<clock>;
258 
259 private:
260  time_point m_previous = clock::now();
261  const time_point m_startTime = m_previous;
262  STAGE m_stage;
263  unsigned m_numX;
264  unsigned m_numY;
265  unsigned m_numZ;
266  unsigned m_curX;
267  unsigned m_curY;
268  unsigned m_curZ;
269  std::vector<StatCounter2DSet<double>> m_counters;
270 
271 public:
272  TimeProfiler2D( unsigned numX = 1, unsigned numY = 1, unsigned numZ = 1 ) : m_stage( P_OTHER ), m_numX( numX ), m_numY( numY ), m_numZ( numZ ), m_curX( 0 ), m_curY( 0 ), m_curZ( 0 )
273  {
274  m_counters.resize( m_numZ );
275  for( int i = 0; i < m_numZ; i++ )
276  {
277  m_counters[i].init( std::vector<std::string> { stageNames, std::end( stageNames ) }, m_numX, m_numY );
278  }
279  }
280  void count( STAGE s, unsigned x, unsigned y, unsigned z )
281  {
282  time_point now = clock::now();
283  m_counters[z][m_stage][y][x] += ( now - m_previous ).count();
284  m_previous = now;
285  m_stage = s;
286  m_curX = x;
287  m_curY = y;
288  m_curZ = z;
289  }
290  void start( STAGE s )
291  {
292  m_previous = clock::now();
293  m_stage = s;
294  }
295  void stop( STAGE s )
296  {
297  count( m_stage, m_curX, m_curY, m_curZ );
298  }
299  unsigned numStages() { return m_counters[0].getNumCntTypes(); }
300  STAGE curStage() { return m_stage; }
301  unsigned curX() { return m_curX; }
302  unsigned curY() { return m_curY; }
303  unsigned curZ() { return m_curZ; }
304 
305  std::vector<StatCounter2DSet<double>>& getCountersSet() { return m_counters; }
306  const std::vector<StatCounter2DSet<double>>& getCountersSet() const { return m_counters; }
307 };
