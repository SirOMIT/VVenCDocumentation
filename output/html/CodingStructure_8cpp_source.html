145 const int CodingStructure::signalModeCons( const PartSplit split, Partitioner &partitioner, const ModeType modeTypeParent ) const
146 {
147  if (CS::isDualITree(*this) || modeTypeParent != MODE_TYPE_ALL || partitioner.currArea().chromaFormat == CHROMA_444 || partitioner.currArea().chromaFormat == CHROMA_400 )
148  return LDT_MODE_TYPE_INHERIT;
149  int minLumaArea = partitioner.currArea().lumaSize().area();
150  if (split == CU_QUAD_SPLIT || split == CU_TRIH_SPLIT || split == CU_TRIV_SPLIT) // the area is split into 3 or 4 parts
151  {
152  minLumaArea = minLumaArea >> 2;
153  }
154  else if (split == CU_VERT_SPLIT || split == CU_HORZ_SPLIT) // the area is split into 2 parts
155  {
156  minLumaArea = minLumaArea >> 1;
157  }
158  int minChromaBlock = minLumaArea >> (getChannelTypeScaleX(CH_C, partitioner.currArea().chromaFormat) + getChannelTypeScaleY(CH_C, partitioner.currArea().chromaFormat));
159  bool is2xNChroma = (partitioner.currArea().chromaSize().width == 4 && split == CU_VERT_SPLIT) || (partitioner.currArea().chromaSize().width == 8 && split == CU_TRIV_SPLIT);
160  return minChromaBlock >= 16 && !is2xNChroma ? LDT_MODE_TYPE_INHERIT : ((minLumaArea < 32) || slice->isIntra()) ? LDT_MODE_TYPE_INFER : LDT_MODE_TYPE_SIGNAL;
161 }

208 TransformUnit* CodingStructure::getTU( const Position& pos, const ChannelType effChType, const int subTuIdx )
209 {
210  const CompArea& _blk = area.blocks[effChType];
211 
212  if( !_blk.contains( pos ) )
213  {
214  if( parent ) return parent->getTU( pos, effChType );
215  else return nullptr;
216  }
217  else
218  {
219  TransformUnit* ptu = m_tuPtr[effChType][rsAddr(pos, _blk.pos(), _blk.width, unitScale[effChType])];
220  unsigned idx = ptu->idx;
221  if (!ptu && m_isTuEnc)
222  return parent->getTU(pos, effChType);
223  else
224  {
225  if (isLuma(effChType))
226  {
227  const TransformUnit& tu = *tus[idx - 1];
228  if (tu.cu->ispMode)
229  {
230  if (subTuIdx == -1)
231  {
232  unsigned extraIdx = 0;
233  while (!tus[idx - 1 + extraIdx]->blocks[getFirstComponentOfChannel(effChType)].contains(pos))
234  {
235  extraIdx++;
236  CHECK(tus[idx - 1 + extraIdx]->cu->treeType == TREE_C,
237  "tu searched by position points to a chroma tree CU");
238  CHECK(extraIdx > 3, "extraIdx > 3");
239  }
240  return tus[idx - 1 + extraIdx];
241  }
242  return tus[idx - 1 + subTuIdx];
243  }
244  }
245  return ptu;
246  }
247  }

292 CodingUnit& CodingStructure::addCU( const UnitArea& unit, const ChannelType chType, CodingUnit* cuInit )
293 {
294  CodingUnit* cu;
295 
296  if( cuInit )
297  {
298  cu = cuInit;
299  }
300  else
301  {
302  if( m_unitCacheMutex ) m_unitCacheMutex->lock();
303

305 
306  if( m_unitCacheMutex ) m_unitCacheMutex->unlock();
307 
308  cu->UnitArea::operator=( unit );
309  cu->initData();
310  cu->slice = nullptr;
311  }
312 
313  cu->next = nullptr;
314  cu->firstTU = nullptr;
315  cu->lastTU = nullptr;
316  cu->chType = chType;
317  cu->cs = this;
318 
319  CodingUnit *prevCU = m_numCUs > 0 ? cus.back() : nullptr;
320 
321  if( prevCU )
322  {
323  const int prevCuCtuRsAddr = getCtuAddr( recalcPosition( area.chromaFormat, prevCU->chType, CH_L, prevCU->blocks[prevCU->chType] ), *pcv );
324  const int currCuCtuRsAddr = getCtuAddr( recalcPosition( area.chromaFormat, chType, CH_L, cu->blocks[ chType] ), *pcv );
325 
326  if( prevCuCtuRsAddr == currCuCtuRsAddr )
327  {
328  prevCU->next = cu;
329  }
330  }
331 
332  cus.push_back( cu );
333 
334  Mv* prevCuMvd = cuInit ? cuInit->mvdL0SubPu : nullptr;
335 
336  uint32_t idx = ++m_numCUs;
337  cu->idx = idx;
338  cu->mvdL0SubPu = nullptr;
339 
340  if( isLuma( chType ) && unit.lheight() >= 8 && unit.lwidth() >= 8 && unit.Y().area() >= 128 )
341  {
342  CHECKD( m_dmvrMvCacheOffset >= m_dmvrMvCache.size(), "dmvr cache offset out of bounds" );
343 
344  int mvdArrSize = std::max<int>( 1, unit.lwidth() >> DMVR_SUBCU_SIZE_LOG2 ) * std::max<int>( 1, unit.lheight() >> DMVR_SUBCU_SIZE_LOG2 );
345  cu->mvdL0SubPu = &m_dmvrMvCache[m_dmvrMvCacheOffset];
346  m_dmvrMvCacheOffset += mvdArrSize;
347 
348  if( prevCuMvd )
349  memcpy( cu->mvdL0SubPu, prevCuMvd, sizeof( Mv ) * mvdArrSize );
350  }
351 
352  uint32_t numCh = getNumberValidChannels( area.chromaFormat );
353 
354  for( uint32_t i = 0; i < numCh; i++ )
355  {
356  if( !cu->blocks[i].valid() )
357  {
358  continue;
359  }
360 
361  const CompArea& _selfBlk = area.blocks[i];
362  const CompArea &_blk = cu-> blocks[i];
363 
364  const UnitScale& scale = unitScale[_blk.compID];
365  const Area scaledSelf = scale.scale( _selfBlk );
366  const Area scaledBlk = scale.scale( _blk );
367  CodingUnit **cuPtr = m_cuPtr[i] + rsAddr( scaledBlk.pos(), scaledSelf.pos(), scaledSelf.width );
368 
369  CHECKD( *cuPtr, "Overwriting a pre-existing value, should be '0'!" );
370 
371  g_pelBufOP.fillPtrMap( ( void** ) cuPtr, scaledSelf.width, scaledBlk.width, scaledBlk.height, ( void* ) cu );
372  }
373 
374  return *cu;
375 }

377 TransformUnit& CodingStructure::addTU( const UnitArea& unit, const ChannelType chType, CodingUnit* cu, TransformUnit* tuInit )
378 {
379  TransformUnit* tu;
380 
381  if( tuInit )
382  {
383  tu = tuInit;
384  }
385  else
386  {
387  if( m_unitCacheMutex ) m_unitCacheMutex->lock();

392 
393  tu->UnitArea::operator=( unit );

396 
397  tu->next = nullptr;
398  tu->prev = nullptr;
399  tu->cs = this;
400  tu->cu = cu;
401  tu->chType = chType;
402 
403  TransformUnit *prevTU = m_numTUs > 0 ? tus.back() : nullptr;
404 
405  if( prevTU && prevTU->cu == tu->cu )
406  {
407  prevTU->next = tu;
408  tu->prev = prevTU;
409  }
410 
411  tus.push_back( tu );
412 
413  if( tu->cu )
414  {
415  if( tu->cu->firstTU == nullptr )
416  {
417  tu->cu->firstTU = tu;
418  }
419  tu->cu->lastTU = tu;
420  }
421 
422  uint32_t idx = ++m_numTUs;
423  tu->idx = idx;
424 
425  TCoeffSig *coeffs[3] = { nullptr, nullptr, nullptr };
426 
427  uint32_t numCh = getNumberValidComponents( area.chromaFormat );
428 
429  for( uint32_t i = 0; i < numCh; i++ )
430  {
431  if( !tu->blocks[i].valid() )
432  {
433  continue;
434  }
435 
436  if (i < getNumberValidChannels(area.chromaFormat))
437  {
438  const CompArea& _selfBlk = area.blocks[i];
439  const CompArea &_blk = tu-> blocks[i];
440 
441  bool isIspTu = tu->cu != nullptr && tu->cu->ispMode && isLuma( _blk.compID );
442 
443  bool isFirstIspTu = false;
444  if( isIspTu )
445  {
446  isFirstIspTu = CU::isISPFirst( *tu->cu, _blk, getFirstComponentOfChannel( ChannelType( i ) ) );
447  }
448  if( !isIspTu || isFirstIspTu )
449  {
450  const UnitScale& scale = unitScale[_blk.compID];
451 
452  const Area scaledSelf = scale.scale( _selfBlk );
453  const Area scaledBlk = isIspTu ? scale.scale( tu->cu->blocks[i] ) : scale.scale( _blk );
454  TransformUnit **tuPtr = m_tuPtr[i] + rsAddr( scaledBlk.pos(), scaledSelf.pos(), scaledSelf.width );
455 
456  CHECKD( *tuPtr, "Overwriting a pre-existing value, should be '0'!" );
457 
458  g_pelBufOP.fillPtrMap( ( void** ) tuPtr, scaledSelf.width, scaledBlk.width, scaledBlk.height, ( void* ) tu );
459  }
460  }
461 
462  coeffs[i] = m_coeffs[i] + m_offsets[i];
463 
464  unsigned areaSize = tu->blocks[i].area();
465  m_offsets[i] += areaSize;
466 
467  if( tuInit )
468  memcpy( coeffs[i], tu->m_coeffs[i], areaSize * sizeof( TCoeffSig ) );
469  }
470 
471  tu->init( coeffs );
472 
473  return *tu;
474 }
