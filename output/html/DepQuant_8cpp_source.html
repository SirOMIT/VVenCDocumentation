160  void Rom::xInitScanArrays()
161  {
162  if( m_scansInitialized )
163  {
164  return;
165  }
166  ::memset( m_scanId2NbInfoSbbArray, 0, sizeof(m_scanId2NbInfoSbbArray) );
167  ::memset( m_scanId2NbInfoOutArray, 0, sizeof(m_scanId2NbInfoOutArray) );
168  ::memset( m_tuParameters, 0, sizeof(m_tuParameters) );
169 
170  uint32_t raster2id[ MAX_CU_SIZE * MAX_CU_SIZE ];
171  ::memset(raster2id, 0, sizeof(raster2id));
172 
173  for( int hd = 0; hd < MAX_TU_SIZE_IDX; hd++ )
174  {
175  for( int vd = 0; vd < MAX_TU_SIZE_IDX; vd++ )
176  {
177  if( (hd == 0 && vd <= 1) || (hd <= 1 && vd == 0) )
178  {
179  continue;
180  }
181  const uint32_t blockWidth = (1 << hd);
182  const uint32_t blockHeight = (1 << vd);
183  const uint32_t log2CGWidth = g_log2SbbSize[hd][vd][0];
184  const uint32_t log2CGHeight = g_log2SbbSize[hd][vd][1];
185  const uint32_t groupWidth = 1 << log2CGWidth;
186  const uint32_t groupHeight = 1 << log2CGHeight;
187  const uint32_t groupSize = groupWidth * groupHeight;
188  const SizeType blkWidthIdx = Log2( blockWidth );
189  const SizeType blkHeightIdx = Log2( blockHeight );
190  const ScanElement * scanId2RP = getScanOrder( SCAN_GROUPED_4x4, blkWidthIdx, blkHeightIdx );
191  NbInfoSbb*& sId2NbSbb = m_scanId2NbInfoSbbArray[hd][vd];
192  NbInfoOut*& sId2NbOut = m_scanId2NbInfoOutArray[hd][vd];
193  // consider only non-zero-out region
194  const uint32_t blkWidthNZOut = std::min<unsigned>( JVET_C0024_ZERO_OUT_TH, blockWidth );
195  const uint32_t blkHeightNZOut= std::min<unsigned>( JVET_C0024_ZERO_OUT_TH, blockHeight );
196  const uint32_t totalValues = blkWidthNZOut * blkHeightNZOut;
197 
198  sId2NbSbb = new NbInfoSbb[ totalValues ];
199  sId2NbOut = new NbInfoOut[ totalValues ];
200 
201  for( uint32_t scanId = 0; scanId < totalValues; scanId++ )
202  {
203  raster2id[scanId2RP[scanId].idx] = scanId;
204  }
205 
206  for( unsigned scanId = 0; scanId < totalValues; scanId++ )
207  {
208  const int posX = scanId2RP[scanId].x;
209  const int posY = scanId2RP[scanId].y;
210  const int rpos = scanId2RP[scanId].idx;
211  {
212  //===== inside subband neighbours =====
213  NbInfoSbb& nbSbb = sId2NbSbb[ scanId ];
214  const int begSbb = scanId - ( scanId & (groupSize-1) ); // first pos in current subblock
215  int cpos[5];
216 
217  cpos[0] = ( posX + 1 < blkWidthNZOut ? ( raster2id[rpos+1 ] < groupSize + begSbb ? raster2id[rpos+1 ] - begSbb : 0 ) : 0 );
218  cpos[1] = ( posX + 2 < blkWidthNZOut ? ( raster2id[rpos+2 ] < groupSize + begSbb ? raster2id[rpos+2 ] - begSbb : 0 ) : 0 );
219  cpos[2] = ( posX + 1 < blkWidthNZOut && posY + 1 < blkHeightNZOut ? ( raster2id[rpos+1+blockWidth] < groupSize + begSbb ? raster2id[rpos+1+blockWidth] - begSbb : 0 ) : 0 );
220  cpos[3] = ( posY + 1 < blkHeightNZOut ? ( raster2id[rpos+ blockWidth] < groupSize + begSbb ? raster2id[rpos+ blockWidth] - begSbb : 0 ) : 0 );
221  cpos[4] = ( posY + 2 < blkHeightNZOut ? ( raster2id[rpos+2*blockWidth] < groupSize + begSbb ? raster2id[rpos+2*blockWidth] - begSbb : 0 ) : 0 );
222 
223  for( nbSbb.num = 0; true; )
224  {
225  int nk = -1;
226  for( int k = 0; k < 5; k++ )
227  {
228  if( cpos[k] != 0 && ( nk < 0 || cpos[k] < cpos[nk] ) )
229  {
230  nk = k;
231  }
232  }
233  if( nk < 0 )
234  {
235  break;
236  }
237  nbSbb.inPos[ nbSbb.num++ ] = uint8_t( cpos[nk] );
238  cpos[nk] = 0;
239  }
240  for( int k = nbSbb.num; k < 5; k++ )
241  {
242  nbSbb.inPos[k] = 0;
243  }
244  }
245  {
246  //===== outside subband neighbours =====
247  NbInfoOut& nbOut = sId2NbOut[ scanId ];
248  const int begSbb = scanId - ( scanId & (groupSize-1) ); // first pos in current subblock
249  int cpos[5];
250 
251  cpos[0] = ( posX + 1 < blkWidthNZOut ? ( raster2id[rpos+1 ] >= groupSize + begSbb ? raster2id[rpos+1 ] : 0 ) : 0 );
252  cpos[1] = ( posX + 2 < blkWidthNZOut ? ( raster2id[rpos+2 ] >= groupSize + begSbb ? raster2id[rpos+2 ] : 0 ) : 0 );
253  cpos[2] = ( posX + 1 < blkWidthNZOut && posY + 1 < blkHeightNZOut ? ( raster2id[rpos+1+blockWidth] >= groupSize + begSbb ? raster2id[rpos+1+blockWidth] : 0 ) : 0 );
254  cpos[3] = ( posY + 1 < blkHeightNZOut ? ( raster2id[rpos+ blockWidth] >= groupSize + begSbb ? raster2id[rpos+ blockWidth] : 0 ) : 0 );
255  cpos[4] = ( posY + 2 < blkHeightNZOut ? ( raster2id[rpos+2*blockWidth] >= groupSize + begSbb ? raster2id[rpos+2*blockWidth] : 0 ) : 0 );
256 
257  for( nbOut.num = 0; true; )
258  {
259  int nk = -1;
260  for( int k = 0; k < 5; k++ )
261  {
262  if( cpos[k] != 0 && ( nk < 0 || cpos[k] < cpos[nk] ) )
263  {
264  nk = k;
265  }
266  }
267  if( nk < 0 )
268  {
269  break;
270  }
271  nbOut.outPos[ nbOut.num++ ] = uint16_t( cpos[nk] );
272  cpos[nk] = 0;
273  }
274  for( int k = nbOut.num; k < 5; k++ )
275  {
276  nbOut.outPos[k] = 0;
277  }
278  nbOut.maxDist = ( scanId == 0 ? 0 : sId2NbOut[scanId-1].maxDist );
279  for( int k = 0; k < nbOut.num; k++ )
280  {
281  if( nbOut.outPos[k] > nbOut.maxDist )
282  {
283  nbOut.maxDist = nbOut.outPos[k];
284  }
285  }
286  }
287  }
288 
289  // make it relative
290  for( unsigned scanId = 0; scanId < totalValues; scanId++ )
291  {
292  NbInfoOut& nbOut = sId2NbOut[scanId];
293  const int begSbb = scanId - ( scanId & (groupSize-1) ); // first pos in current subblock
294  for( int k = 0; k < nbOut.num; k++ )
295  {
296  CHECK(begSbb > nbOut.outPos[k], "Position must be past sub block begin");
297  nbOut.outPos[k] -= begSbb;
298  }
299  nbOut.maxDist -= scanId;
300  }
301 
302  for( int chId = 0; chId < MAX_NUM_CH; chId++ )
303  {
304  m_tuParameters[hd][vd][chId] = new TUParameters( *this, blockWidth, blockHeight, ChannelType(chId) );
305  }
306  }
307  }
308  m_scansInitialized = true;
309  }

345  TUParameters::TUParameters( const Rom& rom, const unsigned width, const unsigned height, const ChannelType chType )
346  {
347  m_chType = chType;
348  m_width = width;
349  m_height = height;
350  const uint32_t nonzeroWidth = std::min<uint32_t>(JVET_C0024_ZERO_OUT_TH, m_width);
351  const uint32_t nonzeroHeight = std::min<uint32_t>(JVET_C0024_ZERO_OUT_TH, m_height);
352  m_numCoeff = nonzeroWidth * nonzeroHeight;
353  m_log2SbbWidth = g_log2SbbSize[ Log2(m_width) ][ Log2(m_height) ][0];
354  m_log2SbbHeight = g_log2SbbSize[ Log2(m_width) ][ Log2(m_height) ][1];
355  m_log2SbbSize = m_log2SbbWidth + m_log2SbbHeight;
356  m_sbbSize = ( 1 << m_log2SbbSize );
357  m_sbbMask = m_sbbSize - 1;
358  m_widthInSbb = nonzeroWidth >> m_log2SbbWidth;
359  m_heightInSbb = nonzeroHeight >> m_log2SbbHeight;
360  m_numSbb = m_widthInSbb * m_heightInSbb;
361  SizeType hsbb = Log2( m_widthInSbb );
362  SizeType vsbb = Log2( m_heightInSbb );
363  SizeType hsId = Log2( m_width );
364  SizeType vsId = Log2( m_height );
365  m_scanSbbId2SbbPos = getScanOrder( SCAN_UNGROUPED , hsbb , vsbb );
366  m_scanId2BlkPos = getScanOrder( SCAN_GROUPED_4x4 , hsId , vsId );
367  int log2W = Log2( m_width );
368  int log2H = Log2( m_height );
369  m_scanId2NbInfoSbb = rom.getNbInfoSbb( log2W, log2H );
370  m_scanId2NbInfoOut = rom.getNbInfoOut( log2W, log2H );
371  m_scanInfo = new ScanInfo[ m_numCoeff ];
372  for( int scanIdx = 0; scanIdx < m_numCoeff; scanIdx++ )
373  {
374  xSetScanInfo( m_scanInfo[scanIdx], scanIdx );
375  }
376  }

379  void TUParameters::xSetScanInfo( ScanInfo& scanInfo, int scanIdx )
380  {
381  scanInfo.sbbSize = m_sbbSize;
382  scanInfo.numSbb = m_numSbb;
383  scanInfo.scanIdx = scanIdx;
384  scanInfo.rasterPos = m_scanId2BlkPos[scanIdx].idx;
385  scanInfo.sbbPos = m_scanSbbId2SbbPos[scanIdx >> m_log2SbbSize].idx;
386  scanInfo.insidePos = scanIdx & m_sbbMask;
387  scanInfo.spt = SCAN_ISCSBB;
388  if( scanInfo.insidePos == m_sbbMask && scanIdx > scanInfo.sbbSize && scanIdx < m_numCoeff - 1 )
389  scanInfo.spt = SCAN_SOCSBB;
390  else if( scanInfo.insidePos == 0 && scanIdx > 0 && scanIdx < m_numCoeff - m_sbbSize )
391  scanInfo.spt = SCAN_EOCSBB;
392  scanInfo.posX = m_scanId2BlkPos[scanIdx].x;
393  scanInfo.posY = m_scanId2BlkPos[scanIdx].y;
394  if( scanIdx )
395  {
396  const int nextScanIdx = scanIdx - 1;
397  const int diag = m_scanId2BlkPos[nextScanIdx].x + m_scanId2BlkPos[nextScanIdx].y;
398  if( m_chType == CH_L )
399  {
400  scanInfo.sigCtxOffsetNext = ( diag < 2 ? 8 : diag < 5 ? 4 : 0 );
401  scanInfo.gtxCtxOffsetNext = ( diag < 1 ? 16 : diag < 3 ? 11 : diag < 10 ? 6 : 1 );
402  }
403  else
404  {
405  scanInfo.sigCtxOffsetNext = ( diag < 2 ? 4 : 0 );
406  scanInfo.gtxCtxOffsetNext = ( diag < 1 ? 6 : 1 );
407  }
408  scanInfo.nextInsidePos = nextScanIdx & m_sbbMask;
409  scanInfo.nextNbInfoSbb = m_scanId2NbInfoSbb[ nextScanIdx ];
410  if( scanInfo.insidePos == 0 )
411  {
412  const int nextSbbPos = m_scanSbbId2SbbPos[nextScanIdx >> m_log2SbbSize].idx;
413  const int nextSbbPosY = nextSbbPos / m_widthInSbb;
414  const int nextSbbPosX = nextSbbPos - nextSbbPosY * m_widthInSbb;
415  scanInfo.nextSbbRight = ( nextSbbPosX < m_widthInSbb - 1 ? nextSbbPos + 1 : 0 );
416  scanInfo.nextSbbBelow = ( nextSbbPosY < m_heightInSbb - 1 ? nextSbbPos + m_widthInSbb : 0 );
417  }
418  }

472  void RateEstimator::xSetLastCoeffOffset( const FracBitsAccess& fracBitsAccess, const TUParameters& tuPars, const TransformUnit& tu, const ComponentID compID )
473  {
474  const ChannelType chType = ( compID == COMP_Y ? CH_L : CH_C );
475  int32_t cbfDeltaBits = 0;
476  if( compID == COMP_Y && !CU::isIntra(*tu.cu) && !tu.depth )
477  {
478  const BinFracBits bits = fracBitsAccess.getFracBitsArray( Ctx::QtRootCbf() );
479  cbfDeltaBits = int32_t( bits.intBits[1] ) - int32_t( bits.intBits[0] );
480  }
481  else
482  {
483  BinFracBits bits;
484  bool prevLumaCbf = false;
485  bool lastCbfIsInferred = false;
486  bool useIntraSubPartitions = tu.cu->ispMode && isLuma(chType);
487  if( useIntraSubPartitions )
488  {
489  bool rootCbfSoFar = false;
490  bool isLastSubPartition = CU::isISPLast(*tu.cu, tu.Y(), compID);
491  uint32_t nTus = tu.cu->ispMode == HOR_INTRA_SUBPARTITIONS ? tu.cu->lheight() >> Log2(tu.lheight()) : tu.cu->lwidth() >> Log2(tu.lwidth());
492  if( isLastSubPartition )
493  {
494  TransformUnit* tuPointer = tu.cu->firstTU;
495  for( int tuIdx = 0; tuIdx < nTus - 1; tuIdx++ )
496  {
497  rootCbfSoFar |= TU::getCbfAtDepth(*tuPointer, COMP_Y, tu.depth);
498  tuPointer = tuPointer->next;
499  }
500  if( !rootCbfSoFar )
501  {
502  lastCbfIsInferred = true;
503  }
504  }
505  if( !lastCbfIsInferred )
506  {
507  prevLumaCbf = TU::getPrevTuCbfAtDepth(tu, compID, tu.depth);
508  }
509  bits = fracBitsAccess.getFracBitsArray(Ctx::QtCbf[compID](DeriveCtx::CtxQtCbf(compID, prevLumaCbf, true)));
510  }
511  else
512  {
513  bits = fracBitsAccess.getFracBitsArray(Ctx::QtCbf[compID](DeriveCtx::CtxQtCbf(compID, tu.cbf[COMP_Cb])));
514  }
515  cbfDeltaBits = lastCbfIsInferred ? 0 : int32_t(bits.intBits[1]) - int32_t(bits.intBits[0]);
516  }
517 
518  static const unsigned prefixCtx[] = { 0, 0, 0, 3, 6, 10, 15, 21 };
519  uint32_t ctxBits [ LAST_SIGNIFICANT_GROUPS ];
520  for( unsigned xy = 0; xy < 2; xy++ )
521  {
522  int32_t bitOffset = ( xy ? cbfDeltaBits : 0 );
523  int32_t* lastBits = ( xy ? m_lastBitsY : m_lastBitsX );
524  const unsigned size = ( xy ? tuPars.m_height : tuPars.m_width );

654  inline int ceil_log2(uint64_t x)
655  {
656 #ifdef TARGET_SIMD_X86
657  uint64_t y = _bit_scan_reverse( x ) + ( ( ( x & ( x - 1 ) ) == 0 ) ? 0 : 1 );
658 #else
659  static const uint64_t t[6] = { 0xFFFFFFFF00000000ull, 0x00000000FFFF0000ull, 0x000000000000FF00ull, 0x00000000000000F0ull, 0x000000000000000Cull, 0x0000000000000002ull };
660  int y = (((x & (x - 1)) == 0) ? 0 : 1);
661  int j = 32;
662  for( int i = 0; i < 6; i++)
663  {
664  int k = (((x & t[i]) == 0) ? 0 : j);
665  y += k;
666  x >>= k;
667  j >>= 1;
668  }
669 #endif
670

673  void Quantizer::initQuantBlock(const TransformUnit& tu, const ComponentID compID, const QpParam& cQP, const double lambda, int gValue = -1)
674  {
675  CHECKD( lambda <= 0.0, "Lambda must be greater than 0" );
676 
677  const int qpDQ = cQP.Qp(tu.mtsIdx[compID]==MTS_SKIP) + 1;
678  const int qpPer = qpDQ / 6;
679  const int qpRem = qpDQ - 6 * qpPer;
680  const SPS& sps = *tu.cs->sps;
681  const CompArea& area = tu.blocks[ compID ];
682  const ChannelType chType = toChannelType( compID );
683  const int channelBitDepth = sps.bitDepths[ chType ];
684  const int maxLog2TrDynamicRange = sps.getMaxLog2TrDynamicRange( chType );
685  const int nomTransformShift = getTransformShift( channelBitDepth, area.size(), maxLog2TrDynamicRange );
686  const bool clipTransformShift = ( tu.mtsIdx[compID]==MTS_SKIP && sps.spsRExt.extendedPrecisionProcessing );
687  const bool needsSqrt2ScaleAdjustment = TU::needsSqrt2Scale(tu, compID);
688  const int transformShift = ( clipTransformShift ? std::max<int>( 0, nomTransformShift ) : nomTransformShift ) + (needsSqrt2ScaleAdjustment?-1:0);
689  // quant parameters
690  m_QShift = QUANT_SHIFT - 1 + qpPer + transformShift;
691  m_QAdd = -( ( 3 << m_QShift ) >> 1 );
692  Intermediate_Int invShift = IQUANT_SHIFT + 1 - qpPer - transformShift;
693  m_QScale = g_quantScales[needsSqrt2ScaleAdjustment?1:0][ qpRem ];
694  const unsigned qIdxBD = std::min<unsigned>( maxLog2TrDynamicRange + 1, 8*sizeof(Intermediate_Int) + invShift - IQUANT_SHIFT - 1 );
695  m_maxQIdx = ( 1 << (qIdxBD-1) ) - 4;
696  if( m_QShift )
697  m_thresLast = TCoeff((int64_t(m_DqThrVal) << (m_QShift-1)));
698  else
699  m_thresLast = TCoeff((int64_t(m_DqThrVal>>1) << m_QShift));
700  m_thresSSbb = TCoeff((int64_t(3) << m_QShift));
701  // distortion calculation parameters
702  const int64_t qScale = (gValue==-1) ? m_QScale : gValue;
703  const int nomDShift =
704  SCALE_BITS - 2 * (nomTransformShift + DISTORTION_PRECISION_ADJUSTMENT(channelBitDepth)) + m_QShift + (needsSqrt2ScaleAdjustment ? 1 : 0);
705  const double qScale2 = double( qScale * qScale );
706  const double nomDistFactor = ( nomDShift < 0 ? 1.0/(double(int64_t(1)<<(-nomDShift))*qScale2*lambda) : double(int64_t(1)<<nomDShift)/(qScale2*lambda) );
707  const int64_t pow2dfShift = (int64_t)( nomDistFactor * qScale2 ) + 1;
708  const int dfShift = ceil_log2( pow2dfShift );
709  m_DistShift = 62 + m_QShift - 2*maxLog2TrDynamicRange - dfShift;
710  m_DistAdd = (int64_t(1) << m_DistShift) >> 1;
711  m_DistStepAdd = (int64_t)( nomDistFactor * double(int64_t(1)<<(m_DistShift+m_QShift)) + .5 );
712  m_DistOrgFact = (int64_t)( nomDistFactor * double(int64_t(1)<<(m_DistShift+1 )) + .5 );
713  }

715  void Quantizer::dequantBlock( const TransformUnit& tu, const ComponentID compID, const QpParam& cQP, CoeffBuf& recCoeff, bool enableScalingLists, int* piDequantCoef) const
716  {
717 
718  //----- set basic parameters -----
719  const CompArea& area = tu.blocks[ compID ];
720  const int numCoeff = area.area();
721  const SizeType hsId = Log2( area.width );
722  const SizeType vsId = Log2( area.height );
723  const ScanElement *scan = getScanOrder( SCAN_GROUPED_4x4, hsId, vsId );
724  const TCoeffSig* qCoeff = tu.getCoeffs( compID ).buf;
725  TCoeff* tCoeff = recCoeff.buf;
726 
727  //----- reset coefficients and get last scan index -----
728  ::memset( tCoeff, 0, numCoeff * sizeof( TCoeff ) );
729  int lastScanIdx = tu.lastPos[compID];
730  if( lastScanIdx < 0 )
731  {
732  return;
733  }
734 
735  //----- set dequant parameters -----
736  const int qpDQ = cQP.Qp(tu.mtsIdx[compID]==MTS_SKIP) + 1;
737  const int qpPer = qpDQ / 6;
738  const int qpRem = qpDQ - 6 * qpPer;
739  const SPS& sps = *tu.cs->sps;
740  const ChannelType chType = toChannelType( compID );
741  const int channelBitDepth = sps.bitDepths[ chType ];
742  const int maxLog2TrDynamicRange = sps.getMaxLog2TrDynamicRange( chType );
743  const TCoeff minTCoeff = -( 1 << maxLog2TrDynamicRange );
744  const TCoeff maxTCoeff = ( 1 << maxLog2TrDynamicRange ) - 1;
745  const int nomTransformShift = getTransformShift( channelBitDepth, area.size(), maxLog2TrDynamicRange );
746  const bool clipTransformShift = ( tu.mtsIdx[compID]==MTS_SKIP && sps.spsRExt.extendedPrecisionProcessing );
747  const bool needsSqrt2ScaleAdjustment = TU::needsSqrt2Scale(tu, compID);
748  const int transformShift = ( clipTransformShift ? std::max<int>( 0, nomTransformShift ) : nomTransformShift ) + (needsSqrt2ScaleAdjustment?-1:0);
749  Intermediate_Int shift = IQUANT_SHIFT + 1 - qpPer - transformShift + (enableScalingLists ? LOG2_SCALING_LIST_NEUTRAL_VALUE : 0);
750  Intermediate_Int invQScale = g_invQuantScales[needsSqrt2ScaleAdjustment?1:0][ qpRem ];
751  Intermediate_Int add = (shift < 0) ? 0 : ((1 << shift) >> 1);
752  //----- dequant coefficients -----
753  for( int state = 0, scanIdx = lastScanIdx; scanIdx >= 0; scanIdx-- )
754  {
755  const unsigned rasterPos = scan[scanIdx].idx;
756  const TCoeffSig& level = qCoeff[ rasterPos ];
757  if( level )
758  {
759  if (enableScalingLists)
760  invQScale = piDequantCoef[rasterPos];//scalingfactor*levelScale
761  if (shift < 0 && (enableScalingLists || scanIdx == lastScanIdx))
762  {
763  invQScale <<= -shift;
764  }
765  Intermediate_Int qIdx = ( level << 1 ) + ( level > 0 ? -(state>>1) : (state>>1) );
766  int64_t nomTCoeff = ((int64_t)qIdx * (int64_t)invQScale + add) >> ((shift < 0) ? 0 : shift);
767  tCoeff[rasterPos] = (TCoeff)Clip3<int64_t>(minTCoeff, maxTCoeff, nomTCoeff);
768  }
769  state = ( 32040 >> ((state<<2)+((level&1)<<1)) ) & 3; // the 16-bit value "32040" represent the state transition table
770  }
771  }

773  inline void Quantizer::preQuantCoeff(const TCoeff absCoeff, PQData *pqData, int quanCoeff) const
774  {
775  // TODO: vectorize
776  int64_t scaledOrg = int64_t( absCoeff ) * quanCoeff;
777  TCoeff qIdx = std::max<TCoeff>( 1, std::min<TCoeff>( m_maxQIdx, TCoeff( ( scaledOrg + m_QAdd ) >> m_QShift ) ) );
778  int64_t scaledAdd = qIdx * m_DistStepAdd - scaledOrg * m_DistOrgFact;
779 
780  PQData& pq_a = pqData[ ( qIdx + 0 ) & 3 ];
781  PQData& pq_b = pqData[ ( qIdx + 1 ) & 3 ];
782  PQData& pq_c = pqData[ ( qIdx + 2 ) & 3 ];
783  PQData& pq_d = pqData[ ( qIdx + 3 ) & 3 ];
784 
785  pq_a.deltaDist = ( ( scaledAdd + 0 * m_DistStepAdd ) * ( qIdx + 0 ) + m_DistAdd ) >> m_DistShift;
786  pq_a.absLevel = ( qIdx + 1 ) >> 1;
787 
788  pq_b.deltaDist = ( ( scaledAdd + 1 * m_DistStepAdd ) * ( qIdx + 1 ) + m_DistAdd ) >> m_DistShift;
789  pq_b.absLevel = ( qIdx + 2 ) >> 1;
790 
791  pq_c.deltaDist = ( ( scaledAdd + 2 * m_DistStepAdd ) * ( qIdx + 2 ) + m_DistAdd ) >> m_DistShift;
792  pq_c.absLevel = ( qIdx + 3 ) >> 1;
793 
794  pq_d.deltaDist = ( ( scaledAdd + 3 * m_DistStepAdd ) * ( qIdx + 3 ) + m_DistAdd ) >> m_DistShift;
795  pq_d.absLevel = ( qIdx + 4 ) >> 1;
796  }
