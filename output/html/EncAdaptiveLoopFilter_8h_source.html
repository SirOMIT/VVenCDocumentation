66 {
67  static constexpr int MaxAlfNumClippingValues = AdaptiveLoopFilter::MaxAlfNumClippingValues;
68  using TE = double[MAX_NUM_ALF_LUMA_COEFF][MAX_NUM_ALF_LUMA_COEFF];
69  using Ty = double[MAX_NUM_ALF_LUMA_COEFF];
70  using TKE = TE**;
71  using TKy = Ty*;
72 
73  int numCoeff;
74  int numBins;
75 private:
76  int _numBinsAlloc;
77 public:
78 
79  TKy y;
80  TKE E;
81  double pixAcc;
82  bool all0;
83 
84  AlfCovariance() : numBins( -1 ), _numBinsAlloc( -1 ), y( nullptr ), E( nullptr ), all0( true ) {}
85  ~AlfCovariance() { }
86 
87  void create( int size, int num_bins )
88  {
89  if( y ) destroy();
90 
91  numCoeff = size;
92  numBins = _numBinsAlloc = num_bins;
93 
94  y = new Ty[_numBinsAlloc];
95  E = new TE*[_numBinsAlloc];
96 
97  for( int i = 0; i < _numBinsAlloc; i++ )
98  {
99  E[i] = new TE[_numBinsAlloc];
100  }
101 
102  // will be done be reset either way
103  //std::memset( y, 0, sizeof( y ) );
104  //std::memset( E, 0, sizeof( E ) );
105  }
106 
107  void destroy()
108  {
109  delete[] y;
110  y = nullptr;
111 
112  if( E )
113  {
114  for( int i = 0; i < _numBinsAlloc; i++ )
115  {
116  delete[] E[i];
117  E[i] = nullptr;
118  }
119 
120  delete[] E;
121  E = nullptr;
122  }
123  }
124 
125  void reset()
126  {
127  pixAcc = 0;
128  all0 = true;
129 
130  for( int i = 0; i < _numBinsAlloc; i++ )
131  {
132  for( int j = 0; j < _numBinsAlloc; j++ )
133  {
134  std::memset( E[i][j], 0, sizeof( TE ) );
135  }
136 
137  std::memset( y[i], 0, sizeof( Ty ) );
138  }
139  }
140 
141  const AlfCovariance& operator=( const AlfCovariance& src )
142  {
143  if( _numBinsAlloc < src.numBins )
144  {
145  destroy();
146  create( src.numCoeff, src.numBins );
147  }
148 
149  numCoeff = src.numCoeff;
150  numBins = src.numBins;
151 
152  for( int i = 0; i < numBins; i++ )
153  {
154  for( int j = 0; j < numBins; j++ )
155  {
156  std::memcpy( E[i][j], src.E[i][j], sizeof( TE ) );
157  }
158 
159  std::memcpy( y[i], src.y[i], sizeof( Ty ) );
160  }
161 
162  pixAcc = src.pixAcc;
163  all0 = src.all0;
164 
165  return *this;
166  }

168 #if ENABLE_TRACING
169  void trace()

171  DTRACE( g_trace_ctx, D_ALF, "E\n");
172  for( int b0 = 0; b0 < numBins; b0++ )

184  DTRACE( g_trace_ctx, D_ALF, "\n" );

187  }
188  DTRACE( g_trace_ctx, D_ALF, "y\n");
189  for( int b = 0; b < numBins; b++ )
190  {
191  for( int j = 0; j < numCoeff; j++ )
192  {
193  DTRACE( g_trace_ctx, D_ALF, "%0.2f ", y[b][j] );
194  }
195  DTRACE( g_trace_ctx, D_ALF, "\n" );

201  void add( const AlfCovariance& lhs, const AlfCovariance& rhs )
202  {
203  if( _numBinsAlloc < lhs.numBins )
204  {
205  destroy();
206  create( lhs.numCoeff, lhs.numBins );
207  }
208 
209  numCoeff = lhs.numCoeff;
210  numBins = lhs.numBins;
211  all0 = lhs.all0 & rhs.all0;
212 
213  for( int b0 = 0; b0 < numBins; b0++ )
214  {
215  for( int b1 = 0; b1 < numBins; b1++ )
216  {
217  for( int j = 0; j < numCoeff; j++ )
218  {
219  for( int i = 0; i < numCoeff; i++ )
220  {
221  E[b0][b1][j][i] = lhs.E[b0][b1][j][i] + rhs.E[b0][b1][j][i];
222  }
223  }
224  }
225  }
226 
227  for( int b = 0; b < numBins; b++ )
228  {
229  for( int j = 0; j < numCoeff; j++ )
230  {
231  y[b][j] = lhs.y[b][j] + rhs.y[b][j];
232  }
233  }
234 
235  pixAcc = lhs.pixAcc + rhs.pixAcc;
236  all0 = lhs.all0 && rhs.all0;
237  }
238 
239  const AlfCovariance& operator+= ( const AlfCovariance& src )
240  {
241  for( int b0 = 0; b0 < numBins; b0++ )
242  {
243  for( int b1 = 0; b1 < numBins; b1++ )
244  {
245  for( int j = 0; j < numCoeff; j++ )
246  {
247  for( int i = 0; i < numCoeff; i++ )
248  {
249  E[b0][b1][j][i] += src.E[b0][b1][j][i];
250  }
251  }
252  }
253  }
254 
255  for( int b = 0; b < numBins; b++ )
256  {
257  for( int j = 0; j < numCoeff; j++ )
258  {
259  y[b][j] += src.y[b][j];
260  }
261  }
262 
263  pixAcc += src.pixAcc;
264  all0 &= src.all0;
265 
266  return *this;
267  }
268 
269  const AlfCovariance& operator-= ( const AlfCovariance& src )
270  {
271  for( int b0 = 0; b0 < numBins; b0++ )
272  {
273  for( int b1 = 0; b1 < numBins; b1++ )
274  {
275  for( int j = 0; j < numCoeff; j++ )
276  {
277  for( int i = 0; i < numCoeff; i++ )
278  {
279  E[b0][b1][j][i] -= src.E[b0][b1][j][i];
280  }
281  }
282  }
283  }
284 
285  for( int b = 0; b < numBins; b++ )
286  {
287  for( int j = 0; j < numCoeff; j++ )
288  {
289  y[b][j] -= src.y[b][j];
290  }
291  }
292 
293  pixAcc -= src.pixAcc;
294  all0 &= src.all0;
295 
296  return *this;
297  }
