80  Array2D() : m_width(0), m_height(0), v() { }

82 
83  int w() const { return m_width; }
84  int h() const { return m_height; }
85 
86  void allocate(int width, int height, const T& value=T())
87  {
88  m_width=width;
89  m_height=height;
90  v.resize(std::size_t(m_width*m_height), value);
91  }
92 
93  T& get(int x, int y)
94  {
95  assert(x<m_width && y<m_height);
96  return v[y*m_width+x];
97  }
98 
99  const T& get(int x, int y) const
100  {
101  assert(x<m_width && y<m_height);
102  return v[y*m_width+x];
