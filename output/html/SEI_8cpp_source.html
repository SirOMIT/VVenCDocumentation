74 {
75  SEIMessages result;
76 
77  for (SEIMessages::const_iterator it=seiList.begin(); it!=seiList.end(); it++)
78  {

87 SEIMessages extractSeisByType(SEIMessages &seiList, SEI::PayloadType seiType)
88 {
89  SEIMessages result;
90 
91  SEIMessages::iterator it=seiList.begin();
92  while ( it!=seiList.end() )
93  {
94  if ((*it)->payloadType() == seiType)
95  {
96  result.push_back(*it);
97  it = seiList.erase(it);
98  }
99  else
100  {
101  it++;
102  }
103  }
104  return result;
105 }
