69 static void convertPayloadToRBSP(std::vector<uint8_t>& nalUnitBuf, InputBitstream *bitstream, bool isVclNalUnit)
70 {
71  uint32_t zeroCount = 0;
72  std::vector<uint8_t>::iterator it_read, it_write;
73 
74  uint32_t pos = 0;
75  bitstream->clearEmulationPreventionByteLocation();
76  for (it_read = it_write = nalUnitBuf.begin(); it_read != nalUnitBuf.end(); it_read++, it_write++, pos++)
77  {
78  CHECK(zeroCount >= 2 && *it_read < 0x03, "Zero count is '2' and read value is small than '3'");
79  if (zeroCount == 2 && *it_read == 0x03)
80  {
81  bitstream->pushEmulationPreventionByteLocation( pos );
82  pos++;
83  it_read++;
84  zeroCount = 0;
85  if (it_read == nalUnitBuf.end())
86  {
87  break;
88  }
89  CHECK(*it_read > 0x03, "Read a value bigger than '3'");
90  }
91  zeroCount = (*it_read == 0x00) ? zeroCount+1 : 0;
92  *it_write = *it_read;
93  }
94  CHECK(zeroCount != 0, "Zero count not '0'");
95 
96  if (isVclNalUnit)
97  {
98  // Remove cabac_zero_word from payload if present
99  int n = 0;
100 
101  while (it_write[-1] == 0x00)
102  {
103  it_write--;
104  n++;
105  }
106 
107  if (n > 0)
108  {
109  msg( VVENC_NOTICE, "\nDetected %d instances of cabac_zero_word\n", n/2);
110  }
111  }
112 
113  nalUnitBuf.resize(it_write - nalUnitBuf.begin());
114 }

132 void readNalUnitHeader(InputNALUnit& nalu)
133 {
134  InputBitstream& bs = nalu.getBitstream();
135 
136  nalu.m_forbiddenZeroBit = bs.read(1); // forbidden zero bit
137  nalu.m_nuhReservedZeroBit = bs.read(1); // nuh_reserved_zero_bit
138  nalu.m_nuhLayerId = bs.read(6); // nuh_layer_id
139  CHECK(nalu.m_nuhLayerId > 55, "The value of nuh_layer_id shall be in the range of 0 to 55, inclusive");
140  nalu.m_nalUnitType = (vvencNalUnitType) bs.read(5); // nal_unit_type
141  nalu.m_temporalId = bs.read(3) - 1; // nuh_temporal_id_plus1
142 
143 #if ENABLE_TRACING
144  xTraceNalUnitHeader(nalu);
145 #endif
146 
147  // only check these rules for base layer
148  if (nalu.m_nuhLayerId == 0 && nalu.m_temporalId == 0)
149  {
150  CHECK(nalu.m_nalUnitType == VVENC_NAL_UNIT_CODED_SLICE_STSA
151  , "When NAL unit type is equal to STSA_NUT, TemporalId shall not be equal to 0");
152  }

158 void read(InputNALUnit& nalu)
159 {
160  InputBitstream &bitstream = nalu.getBitstream();
161  std::vector<uint8_t>& nalUnitBuf=bitstream.getFifo();
162  // perform anti-emulation prevention
163  convertPayloadToRBSP(nalUnitBuf, &bitstream, (nalUnitBuf[0] & 64) == 0);
164  bitstream.resetToStart();
165  readNalUnitHeader(nalu);
166 }
