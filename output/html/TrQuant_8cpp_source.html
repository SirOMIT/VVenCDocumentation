99 template<int signedMode> std::pair<int64_t,int64_t> fwdTransformCbCr( const PelBuf& resCb, const PelBuf& resCr, PelBuf& resC1, PelBuf& resC2 )
100 {
101  const Pel* cb = resCb.buf;
102  const Pel* cr = resCr.buf;
103  Pel* c1 = resC1.buf;
104  Pel* c2 = resC2.buf;
105  int64_t d1 = 0;
106  int64_t d2 = 0;
107  for( SizeType y = 0; y < resCb.height; y++, cb += resCb.stride, cr += resCr.stride, c1 += resC1.stride, c2 += resC2.stride )
108  {
109  for( SizeType x = 0; x < resCb.width; x++ )
110  {
111  int cbx = cb[x], crx = cr[x];
112  if ( signedMode == 1 )
113  {
114  c1[x] = Pel( ( 4*cbx + 2*crx ) / 5 );
115  d1 += square( cbx - c1[x] ) + square( crx - (c1[x]>>1) );
116  }
117  else if ( signedMode == -1 )
118  {
119  c1[x] = Pel( ( 4*cbx - 2*crx ) / 5 );
120  d1 += square( cbx - c1[x] ) + square( crx - (-c1[x]>>1) );
121  }
122  else if ( signedMode == 2 )
123  {
124  c1[x] = Pel( ( cbx + crx ) / 2 );
125  d1 += square( cbx - c1[x] ) + square( crx - c1[x] );
126  }
127  else if ( signedMode == -2 )
128  {
129  c1[x] = Pel( ( cbx - crx ) / 2 );
130  d1 += square( cbx - c1[x] ) + square( crx + c1[x] );
131  }
132  else if ( signedMode == 3 )
133  {
134  c2[x] = Pel( ( 4*crx + 2*cbx ) / 5 );
135  d1 += square( cbx - (c2[x]>>1) ) + square( crx - c2[x] );
136  }
137  else if ( signedMode == -3 )
138  {
139  c2[x] = Pel( ( 4*crx - 2*cbx ) / 5 );
140  d1 += square( cbx - (-c2[x]>>1) ) + square( crx - c2[x] );
141  }
142  else
143  {
144  d1 += square( cbx );
145  d2 += square( crx );
146  }
147  }
148  }
149  return std::make_pair(d1,d2);
150 }

152 template<int signedMode> void invTransformCbCr( PelBuf& resCb, PelBuf& resCr )
153 {
154  Pel* cb = resCb.buf;
155  Pel* cr = resCr.buf;
156  for( SizeType y = 0; y < resCb.height; y++, cb += resCb.stride, cr += resCr.stride )
157  {
158  for( SizeType x = 0; x < resCb.width; x++ )
159  {
160  if ( signedMode == 1 ) { cr[x] = cb[x] >> 1; }
161  else if ( signedMode == -1 ) { cr[x] = -cb[x] >> 1; }
162  else if ( signedMode == 2 ) { cr[x] = cb[x]; }
163  else if ( signedMode == -2 ) { cr[x] = -cb[x]; }
164  else if ( signedMode == 3 ) { cb[x] = cr[x] >> 1; }
165  else if ( signedMode == -3 ) { cb[x] = -cr[x] >> 1; }
166  }
167  }

309 std::vector<int> TrQuant::selectICTCandidates( const TransformUnit& tu, CompStorage* resCb, CompStorage* resCr )
310 {
311  CHECK( !resCb[0].valid() || !resCr[0].valid(), "standard components are not valid" );
312 
313  if( !CU::isIntra( *tu.cu ) )
314  {
315  int cbfMask = 3;
316  resCb[cbfMask].create( tu.blocks[COMP_Cb] );
317  resCr[cbfMask].create( tu.blocks[COMP_Cr] );
318  fwdTransformICT( tu, resCb[0], resCr[0], resCb[cbfMask], resCr[cbfMask], cbfMask );
319  std::vector<int> cbfMasksToTest;
320  cbfMasksToTest.push_back( cbfMask );
321  return cbfMasksToTest;
322  }
323 
324  std::pair<int64_t,int64_t> pairDist[4];
325  for( int cbfMask = 0; cbfMask < 4; cbfMask++ )
326  {
327  if( cbfMask )
328  {
329  if (tu.cu->lfnstIdx)
330  {
331  if (resCb[cbfMask].valid())
332  {
333  resCb[cbfMask].destroy();
334  }
335  if (resCr[cbfMask].valid())
336  {
337  resCr[cbfMask].destroy();
338  }
339  }
340  CHECK( resCb[cbfMask].valid() || resCr[cbfMask].valid(), "target components for cbfMask=" << cbfMask << " are already present" );
341  resCb[cbfMask].create( tu.blocks[COMP_Cb] );
342  resCr[cbfMask].create( tu.blocks[COMP_Cr] );
343  }
344  pairDist[cbfMask] = fwdTransformICT( tu, resCb[0], resCr[0], resCb[cbfMask], resCr[cbfMask], cbfMask );
345  }
346 
347  std::vector<int> cbfMasksToTest;
348  int64_t minDist1 = std::min<int64_t>( pairDist[0].first, pairDist[0].second );
349  int64_t minDist2 = std::numeric_limits<int64_t>::max();
350  int cbfMask1 = 0;
351  int cbfMask2 = 0;
352  for( int cbfMask : { 1, 2, 3 } )
353  {
354  if( pairDist[cbfMask].first < minDist1 )
355  {
356  cbfMask2 = cbfMask1; minDist2 = minDist1;
357  cbfMask1 = cbfMask; minDist1 = pairDist[cbfMask1].first;
358  }
359  else if( pairDist[cbfMask].first < minDist2 )
360  {
361  cbfMask2 = cbfMask; minDist2 = pairDist[cbfMask2].first;
362  }
363  }
364  if( cbfMask1 )
365  {
366  cbfMasksToTest.push_back( cbfMask1 );
367  }
368  if( cbfMask2 && ( ( minDist2 < (9*minDist1)/8 ) || ( !cbfMask1 && minDist2 < (3*minDist1)/2 ) ) )
369  {
370  cbfMasksToTest.push_back( cbfMask2 );
371  }
372 
373  return cbfMasksToTest;
374 }

445 void TrQuant::xT( const TransformUnit& tu, const ComponentID compID, const CPelBuf& resi, CoeffBuf& dstCoeff, const int width, const int height )
446 {
447  PROFILER_SCOPE_AND_STAGE( 1, g_timeProfiler, P_TRAFO );
448 
449  const unsigned maxLog2TrDynamicRange = tu.cs->sps->getMaxLog2TrDynamicRange( toChannelType( compID ) );
450  const unsigned bitDepth = tu.cs->sps->bitDepths[ toChannelType( compID ) ];
451  const int TRANSFORM_MATRIX_SHIFT = g_transformMatrixShift[TRANSFORM_FORWARD];
452  const uint32_t transformWidthIndex = Log2(width ) - 1; // nLog2WidthMinus1, since transform start from 2-point
453  const uint32_t transformHeightIndex = Log2(height) - 1; // nLog2HeightMinus1, since transform start from 2-point
454 
455  int trTypeHor = DCT2;
456  int trTypeVer = DCT2;
457 
458  xSetTrTypes( tu, compID, width, height, trTypeHor, trTypeVer );
459 
460  int skipWidth = ( trTypeHor != DCT2 && width == 32 ) ? 16 : width > JVET_C0024_ZERO_OUT_TH ? width - JVET_C0024_ZERO_OUT_TH : 0;
461  int skipHeight = ( trTypeVer != DCT2 && height == 32 ) ? 16 : height > JVET_C0024_ZERO_OUT_TH ? height - JVET_C0024_ZERO_OUT_TH : 0;
462 
463  if( tu.cu->lfnstIdx )
464  {
465  if ((width == 4 && height > 4) || (width > 4 && height == 4))
466  {
467  skipWidth = width - 4;
468  skipHeight = height - 4;
469  }
470  else if ((width >= 8 && height >= 8))
471  {
472  skipWidth = width - 8;
473  skipHeight = height - 8;
474  }
475  }
476 
477  TCoeff* block = m_blk;
478  TCoeff* tmp = m_tmp;
479 
480  const Pel* resiBuf = resi.buf;
481  const int resiStride = resi.stride;
482 
483 #if ENABLE_SIMD_TRAFO
484  if( width & 3 )
485 #endif
486  {
487  for( int y = 0; y < height; y++ )
488  {
489  for( int x = 0; x < width; x++ )
490  {
491  block[( y * width ) + x] = resiBuf[( y * resiStride ) + x];
492  }
493  }
494  }
495 #if ENABLE_SIMD_TRAFO
496  else if( width & 7 )
497  {
498  g_tCoeffOps.cpyCoeff4( resiBuf, resiStride, block, width, height );
499  }
500  else
501  {
502  g_tCoeffOps.cpyCoeff8( resiBuf, resiStride, block, width, height );
503  }
504 #endif //ENABLE_SIMD_TRAFO
505 
506  if (width > 1 && height > 1)
507  {
508  const int shift_1st = ((Log2(width )) + bitDepth + TRANSFORM_MATRIX_SHIFT) - maxLog2TrDynamicRange + COM16_C806_TRANS_PREC;
509  const int shift_2nd = (Log2(height)) + TRANSFORM_MATRIX_SHIFT + COM16_C806_TRANS_PREC;
510  CHECK( shift_1st < 0, "Negative shift" );
511  CHECK( shift_2nd < 0, "Negative shift" );
512  fastFwdTrans[trTypeHor][transformWidthIndex](block, tmp, shift_1st, height, 0, skipWidth);
513  fastFwdTrans[trTypeVer][transformHeightIndex](tmp, dstCoeff.buf, shift_2nd, width, skipWidth, skipHeight);
514  }
515  else if (height == 1) // 1-D horizontal transform
516  {
517  const int shift = ((Log2(width )) + bitDepth + TRANSFORM_MATRIX_SHIFT) - maxLog2TrDynamicRange + COM16_C806_TRANS_PREC;
518  CHECK( shift < 0, "Negative shift" );
519  fastFwdTrans[trTypeHor][transformWidthIndex](block, dstCoeff.buf, shift, 1, 0, skipWidth);
520  }
521  else // if (iWidth == 1) //1-D vertical transform
522  {
523  int shift = ((floorLog2(height)) + bitDepth + TRANSFORM_MATRIX_SHIFT) - maxLog2TrDynamicRange + COM16_C806_TRANS_PREC;
524  CHECK(shift < 0, "Negative shift");
525  CHECKD((transformHeightIndex < 0), "There is a problem with the height.");
526  fastFwdTrans[trTypeVer][transformHeightIndex](block, dstCoeff.buf, shift, 1, 0, skipHeight);
527  }
528 }
