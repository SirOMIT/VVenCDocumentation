90 void clipMv(Mv& rcMv, const Position& pos, const struct Size& size, const PreCalcValues& pcv, const PPS& pps, bool m_clipMvInSubPic)
91 {
92  if (pcv.wrapArround)
93  {
94  return;
95  }
96  int iMvShift = MV_FRACTIONAL_BITS_INTERNAL;
97  int iOffset = 8;
98  int iHorMax = (pcv.lumaWidth + iOffset - (int)pos.x - 1) << iMvShift;
99  int iHorMin = (-(int)pcv.maxCUSize - iOffset - (int)pos.x + 1) << iMvShift;
100 
101  int iVerMax = (pcv.lumaHeight + iOffset - (int)pos.y - 1) << iMvShift;
102  int iVerMin = (-(int)pcv.maxCUSize - iOffset - (int)pos.y + 1) << iMvShift;
103  const SubPic& curSubPic = pps.getSubPicFromPos(pos);
104  if (curSubPic.treatedAsPic && m_clipMvInSubPic)
105  {
106  iHorMax = ((curSubPic.subPicRight + 1) + iOffset - (int)pos.x - 1) << iMvShift;
107  iHorMin = (-(int)pcv.maxCUSize - iOffset - ((int)pos.x - curSubPic.subPicLeft) + 1) << iMvShift;
108 
109  iVerMax = ((curSubPic.subPicBottom + 1) + iOffset - (int)pos.y - 1) << iMvShift;
110  iVerMin = (-(int)pcv.maxCUSize - iOffset - ((int)pos.y - curSubPic.subPicTop) + 1) << iMvShift;
111  }
112  rcMv.hor = (std::min(iHorMax, std::max(iHorMin, rcMv.hor)));
113  rcMv.ver = (std::min(iVerMax, std::max(iVerMin, rcMv.ver)));
114 }

116 bool wrapClipMv( Mv& rcMv, const Position& pos, const struct Size& size, const CodingStructure& cs )
117 {
118  bool wrapRef = true;
119  int iMvShift = MV_FRACTIONAL_BITS_INTERNAL;
120  int iOffset = 8;
121  int iHorMax = ( cs.pcv->lumaWidth + cs.pcv->maxCUSize - size.width + iOffset - ( int ) pos.x - 1 ) << iMvShift;
122  int iHorMin = ( -( int ) cs.pcv->maxCUSize - iOffset - ( int ) pos.x + 1 ) << iMvShift;
123  int iVerMax = ( cs.pcv->lumaHeight + iOffset - ( int ) pos.y - 1 ) << iMvShift;
124  int iVerMin = ( -( int ) cs.pcv->maxCUSize - iOffset - ( int ) pos.y + 1 ) << iMvShift;
125  int mvX = rcMv.hor;
126 
127  if(mvX > iHorMax)
128  {
129  mvX -= ( cs.pps->wrapAroundOffset << iMvShift );
130  mvX = std::min( iHorMax, std::max( iHorMin, mvX ) );
131  wrapRef = false;
132  }
133  if(mvX < iHorMin)
134  {
135  mvX += ( cs.pps->wrapAroundOffset << iMvShift );
136  mvX = std::min( iHorMax, std::max( iHorMin, mvX ) );
137  wrapRef = false;
138  }
139 
140  rcMv.hor = ( mvX );
141  rcMv.ver = ( std::min( iVerMax, std::max( iVerMin, rcMv.ver ) ) );
142  return wrapRef;
143 }
