238

653  }
654  else
655  {
656  return partitioner.canSplit( CU_QUAD_SPLIT, cs );
657  }
658  }
659 
660  const Slice& slice = *cs.slice;
661  const uint32_t width = partitioner.currArea().lumaSize().width;
662  const CodingStructure *bestCS = cuECtx.bestCS;
663  const CodingUnit *bestCU = cuECtx.bestCU;
664 
665  if( cuECtx.minDepth > partitioner.currQtDepth && partitioner.canSplit( CU_QUAD_SPLIT, cs ) )
666  {
667  // enforce QT

671  {
672  // don't check this QT depth
673  return false;
674  }
675 
676  CHECK( ! isModeSplit( encTestmode ), "wrong method" );
677 
679  // skip-history rule - don't split further if at least for three past levels
680  // in the split tree it was found that skip is the best mode
682  int skipScore = 0;
683 
684  if ((!slice.isIntra() || slice.sps->IBC) && cuECtx.isBestNoSplitSkip )
685  {

688  if( ( m_ComprCUCtxList.end() - i )->isBestNoSplitSkip )
689  {

691  }
692  else
693  {
694  break;
695  }
696  }
697  }
698 
699  const PartSplit split = getPartSplit( encTestmode );
700  if( !partitioner.canSplit( split, cs ) || skipScore >= 2 )
701  {
702  if( split == CU_HORZ_SPLIT ) cuECtx.didHorzSplit = false;
703  if( split == CU_VERT_SPLIT ) cuECtx.didVertSplit = false;
704  if( split == CU_QUAD_SPLIT ) cuECtx.didQuadSplit = false;
705  return false;
706  }
707 
708  if( m_pcEncCfg->m_contentBasedFastQtbt )
709  {
710  const CompArea& currArea = partitioner.currArea().Y();
711  int cuHeight = currArea.height;
712  int cuWidth = currArea.width;
713 
714  const bool condIntraInter = m_pcEncCfg->m_IntraPeriod == 1 ? ( partitioner.currBtDepth == 0 ) : ( cuHeight > 32 && cuWidth > 32 );
715 
716  if( cuWidth == cuHeight && condIntraInter && split != CU_QUAD_SPLIT )
717  {
718  const double th1 = m_pcEncCfg->m_IntraPeriod == 1 ? 1.2 : 1.0;
719  const double th2 = m_pcEncCfg->m_IntraPeriod == 1 ? (1.2 / sqrt( 2 )) : (1.0 / sqrt( 2 ));
720 
721  if( cuECtx.grad_horVal > th1 * cuECtx.grad_verVal && cuECtx.grad_horVal > th2 * cuECtx.grad_dowVal && cuECtx.grad_horVal > th2 * cuECtx.grad_dupVal && ( split == CU_HORZ_SPLIT || split == CU_TRIH_SPLIT ) )
722  {
723  return false;
724  }
725  if( th2 * cuECtx.grad_dupVal < cuECtx.grad_verVal && th2 * cuECtx.grad_dowVal < cuECtx.grad_verVal && th1 * cuECtx.grad_horVal < cuECtx.grad_verVal && ( split == CU_VERT_SPLIT || split == CU_TRIV_SPLIT ) )
726  {
727  return false;
728  }
729  }
730 
731  if( m_pcEncCfg->m_IntraPeriod == 1 && cuWidth <= 32 && cuHeight <= 32 && bestCS && bestCS->tus.size() == 1 && bestCU && bestCU->depth == partitioner.currDepth && partitioner.currBtDepth > 1 && isLuma( partitioner.chType ) )
732  {
733  if( !bestCU->rootCbf )
734  {
735  return false;
736  }
737  }
738  }
739

744  {
745  return false;
746  }
747  }
748  if( bestCU && bestCU->skip && bestCU->mtDepth >= m_skipThresholdE0023FastEnc && !isModeSplit( lastTestmode ) )
749  {
750  return false;
751  }
752 
753  bool resetFeature = false;
754 
755  switch( split )
756  {
757  case CU_QUAD_SPLIT:
758  {
759  if( !cuECtx.qtBeforeBt && bestCU )
760  {
761  unsigned maxBTD = cs.pcv->getMaxMTTDepth( slice, partitioner.chType );
762  const CodingUnit *cuBR = bestCS->cus.back();
763  unsigned height = partitioner.currArea().lumaSize().height;
764 
765  if(((bestCU->btDepth == 0 && maxBTD >= ((slice.isIntra() && !slice.sps->IBC) ? 3 : 2))

770  return false;
771  }

781  int limitBLsize = stopSplit ? 2048 : 1024;

783  {
784  return false;
785  }
786  }
787  }
788  break;
789  case CU_HORZ_SPLIT:
790  case CU_VERT_SPLIT:
791  resetFeature = true;
792  break;
793  case CU_TRIH_SPLIT:
794  if( cuECtx.didHorzSplit && bestCU && bestCU->btDepth == partitioner.currBtDepth && !bestCU->rootCbf )
795  {
796  return false;
797  }
798

802  if( /*maxBTD == 1 && */cuECtx.didHorzSplit && cuECtx.didVertSplit && cuECtx.bestCostHorzSplit > cuECtx.bestCostVertSplit )
803  {
804  return false;
805  }
806  }
807 
808  if( !cuECtx.doTriHorzSplit )
809  {
810  return false;
811  }
812  break;
813  case CU_TRIV_SPLIT:
814  if( cuECtx.didVertSplit && bestCU && bestCU->btDepth == partitioner.currBtDepth && !bestCU->rootCbf )
815  {
816  return false;
817  }
818 
819  if( m_pcEncCfg->m_qtbttSpeedUp > 1 )
820  {
821  //unsigned maxBTD = cs.pcv->getMaxMTTDepth( slice, partitioner.chType );
822  if( /*maxBTD == 1 && */cuECtx.didHorzSplit && cuECtx.didVertSplit && cuECtx.bestCostHorzSplit < cuECtx.bestCostVertSplit )
823  {
824  return false;
825  }
826  }
827 
828  if( !cuECtx.doTriVertSplit )
829  {
830  return false;
831  }
832  break;

835  return false;
836  break;
837  }
838 
839  switch( split )
840  {
841  case CU_HORZ_SPLIT:
842  case CU_TRIH_SPLIT:
843  if( cuECtx.qtBeforeBt && cuECtx.didQuadSplit )
844  {
845  if( cuECtx.maxQtSubDepth > partitioner.currQtDepth + 1 )
846  {
847  if( resetFeature ) cuECtx.didHorzSplit = false;
848  return false;
849  }
850  }
851  break;
852  case CU_VERT_SPLIT:
853  case CU_TRIV_SPLIT:
854  if( cuECtx.qtBeforeBt && cuECtx.didQuadSplit )
855  {
856  if( cuECtx.maxQtSubDepth > partitioner.currQtDepth + 1 )
857  {
858  if( resetFeature ) cuECtx.didVertSplit = false;
859  return false;
860  }
861  }
862  break;
863  default:
864  break;
865  }
866 
867  if( split == CU_QUAD_SPLIT )
868  {
869  cuECtx.didQuadSplit = m_pcEncCfg->m_qtbttSpeedUp <= 1 || !!cuECtx.doMoreSplits;
870  }
871 
872  return m_pcEncCfg->m_qtbttSpeedUp <= 1 || !!cuECtx.doMoreSplits;
