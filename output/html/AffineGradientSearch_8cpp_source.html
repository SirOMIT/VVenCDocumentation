146  void AffineGradientSearch::xEqualCoeffComputer(Pel* pResidue, int residueStride, int **ppDerivate, int derivateBufStride, int64_t(*pEqualCoeff)[7], int width, int height, bool b6Param)
147  {
148  int affineParamNum = b6Param ? 6 : 4;
149 
150  for (int j = 0; j != height; j++)
151  {
152  int cy = ((j >> 2) << 2) + 2;
153  for (int k = 0; k != width; k++)
154  {
155  int iC[6];
156 
157  int idx = j * derivateBufStride + k;
158  int cx = ((k >> 2) << 2) + 2;
159  if (!b6Param)
160  {
161  iC[0] = ppDerivate[0][idx];
162  iC[1] = cx * ppDerivate[0][idx] + cy * ppDerivate[1][idx];
163  iC[2] = ppDerivate[1][idx];
164  iC[3] = cy * ppDerivate[0][idx] - cx * ppDerivate[1][idx];
165  }
166  else
167  {
168  iC[0] = ppDerivate[0][idx];
169  iC[1] = cx * ppDerivate[0][idx];
170  iC[2] = ppDerivate[1][idx];
171  iC[3] = cx * ppDerivate[1][idx];
172  iC[4] = cy * ppDerivate[0][idx];
173  iC[5] = cy * ppDerivate[1][idx];
174  }
175  for (int col = 0; col < affineParamNum; col++)
176  {
177  for (int row = 0; row < affineParamNum; row++)
178  {
179  pEqualCoeff[col + 1][row] += (int64_t)iC[col] * iC[row];
180  }
181  pEqualCoeff[col + 1][affineParamNum] += ((int64_t)iC[col] * pResidue[idx]) << 3;
182  }
183  }
184  }
185  }
