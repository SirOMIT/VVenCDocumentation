111 unsigned BinDecoderBase::decodeBinEP()
112 {
113  m_Value += m_Value;
114  if( ++m_bitsNeeded >= 0 )
115  {
116  m_Value += m_Bitstream->readByte();
117  m_bitsNeeded = -8;
118  }
119 
120  unsigned bin = 0;
121  unsigned SR = m_Range << 7;
122  if( m_Value >= SR )
123  {
124  m_Value -= SR;
125  bin = 1;
126  }
127  DTRACE( g_trace_ctx, D_CABAC, "%d" " " "%d" " EP=%d \n", DTRACE_GET_COUNTER( g_trace_ctx, D_CABAC ), m_Range, bin );
128  return bin;
129 }

132 unsigned BinDecoderBase::decodeBinsEP( unsigned numBins )
133 {
134 #if ENABLE_TRACING
135  int numBinsOrig = numBins;
136 #endif
137 
138  if( m_Range == 256 )
139  {
140  return decodeAlignedBinsEP( numBins );
141  }
142  unsigned remBins = numBins;
143  unsigned bins = 0;
144  while( remBins > 8 )
145  {
146  m_Value = ( m_Value << 8 ) + ( m_Bitstream->readByte() << ( 8 + m_bitsNeeded ) );
147  unsigned SR = m_Range << 15;
148  for( int i = 0; i < 8; i++ )
149  {
150  bins += bins;
151  SR >>= 1;
152  if( m_Value >= SR )
153  {
154  bins ++;
155  m_Value -= SR;
156  }
157  }
158  remBins -= 8;
159  }
160  m_bitsNeeded += remBins;
161  m_Value <<= remBins;
162  if( m_bitsNeeded >= 0 )
163  {
164  m_Value += m_Bitstream->readByte() << m_bitsNeeded;
165  m_bitsNeeded -= 8;
166  }
167  unsigned SR = m_Range << ( remBins + 7 );
168  for ( int i = 0; i < remBins; i++ )
169  {
170  bins += bins;
171  SR >>= 1;
172  if( m_Value >= SR )
173  {
174  bins ++;
175  m_Value -= SR;
176  }
177  }
178 #if ENABLE_TRACING
179  for( int i = 0; i < numBinsOrig; i++ )
180  {
181  DTRACE( g_trace_ctx, D_CABAC, "%d" " " "%d" " EP=%d \n", DTRACE_GET_COUNTER( g_trace_ctx, D_CABAC ), m_Range, ( bins >> ( numBinsOrig - 1 - i ) ) & 1 );
182  }
183 #endif
184  return bins;
185 }

187 unsigned BinDecoderBase::decodeRemAbsEP(unsigned goRicePar, unsigned cutoff, int maxLog2TrDynamicRange)
188 {
189  unsigned prefix = 0;
190  {
191  const unsigned maxPrefix = 32 - maxLog2TrDynamicRange;
192  unsigned codeWord = 0;
193  do
194  {
195  prefix++;
196  codeWord = decodeBinEP();
197  } while (codeWord && prefix < maxPrefix);
198  prefix -= 1 - codeWord;
199  }
200 
201  unsigned length = goRicePar, offset;
202  if (prefix < cutoff)
203  {
204  offset = prefix << goRicePar;
205  }
206  else
207  {
208  offset = (((1 << (prefix - cutoff)) + cutoff - 1) << goRicePar);
209  {
210  length += (prefix == (32 - maxLog2TrDynamicRange) ? maxLog2TrDynamicRange - goRicePar : prefix - cutoff);
211  }
212  }
213  return offset + decodeBinsEP(length);
214 }

248 unsigned BinDecoderBase::decodeAlignedBinsEP( unsigned numBins )
249 {
250 #if ENABLE_TRACING
251  int numBinsOrig = numBins;
252 #endif
253  unsigned remBins = numBins;
254  unsigned bins = 0;
255  while( remBins > 0 )
256  {
257  // The MSB of m_Value is known to be 0 because range is 256. Therefore:
258  // > The comparison against the symbol range of 128 is simply a test on the next-most-significant bit
259  // > "Subtracting" the symbol range if the decoded bin is 1 simply involves clearing that bit.
260  // As a result, the required bins are simply the <binsToRead> next-most-significant bits of m_Value
261  // (m_Value is stored MSB-aligned in a 16-bit buffer - hence the shift of 15)
262  //
263  // m_Value = |0|V|V|V|V|V|V|V|V|B|B|B|B|B|B|B| (V = usable bit, B = potential buffered bit (buffer refills when m_bitsNeeded >= 0))
264  //
265  unsigned binsToRead = std::min<unsigned>( remBins, 8 ); //read bytes if able to take advantage of the system's byte-read function
266  unsigned binMask = ( 1 << binsToRead ) - 1;
267  unsigned newBins = ( m_Value >> (15 - binsToRead) ) & binMask;
268  bins = ( bins << binsToRead) | newBins;
269  m_Value = ( m_Value << binsToRead) & 0x7FFF;
270  remBins -= binsToRead;
271  m_bitsNeeded += binsToRead;
272  if( m_bitsNeeded >= 0 )
273  {
274  m_Value |= m_Bitstream->readByte() << m_bitsNeeded;
275  m_bitsNeeded -= 8;
276  }
277  }
278 #if ENABLE_TRACING
279  for( int i = 0; i < numBinsOrig; i++ )
280  {
281  DTRACE( g_trace_ctx, D_CABAC, "%d" " " "%d" " " "EP=%d \n", DTRACE_GET_COUNTER( g_trace_ctx, D_CABAC ), m_Range, ( bins >> ( numBinsOrig - 1 - i ) ) & 1 );
282  }
283 #endif
284  return bins;
285 }

296 unsigned BinDecoder::decodeBin( unsigned ctxId )
297 {
298  BinProbModel& rcProbModel = m_Ctx[ctxId];
299  unsigned bin = rcProbModel.mps();
300  uint32_t LPS = rcProbModel.getLPS( m_Range );
301 
302  // DTRACE( g_trace_ctx, D_CABAC, "%d" " %d " "%d" " " "[%d:%d]" " " "%2d(MPS=%d)" " " , DTRACE_GET_COUNTER( g_trace_ctx, D_CABAC ), ctxId, m_Range, m_Range-LPS, LPS, ( unsigned int )( rcProbModel.state() ), m_Value < ( ( m_Range - LPS ) << 7 ) );
303  DTRACE( g_trace_ctx, D_CABAC, " %d " "%d" " " "[%d:%d]" " " "%2d(MPS=%d)" " ", DTRACE_GET_COUNTER( g_trace_ctx, D_CABAC ), m_Range, m_Range - LPS, LPS, (unsigned int)( rcProbModel.state() ), m_Value < ( ( m_Range - LPS ) << 7 ) );
304 
305  m_Range -= LPS;
306  uint32_t SR = m_Range << 7;
307  if( m_Value < SR )
308  {
309  // MPS path
310  if( m_Range < 256 )
311  {
312  int numBits = rcProbModel.getRenormBitsRange( m_Range );
313  m_Range <<= numBits;
314  m_Value <<= numBits;
315  m_bitsNeeded += numBits;
316  if( m_bitsNeeded >= 0 )
317  {
318  m_Value += m_Bitstream->readByte() << m_bitsNeeded;
319  m_bitsNeeded -= 8;
320  }
321  }
322  }
323  else
324  {
325  bin = 1 - bin;
326  // LPS path
327  int numBits = rcProbModel.getRenormBitsLPS( LPS );
328  m_Value -= SR;
329  m_Value = m_Value << numBits;
330  m_Range = LPS << numBits;
331  m_bitsNeeded += numBits;
332  if( m_bitsNeeded >= 0 )
333  {
334  m_Value += m_Bitstream->readByte() << m_bitsNeeded;
335  m_bitsNeeded -= 8;
336  }
337  }
338  rcProbModel.update( bin );
339  //DTRACE_DECR_COUNTER( g_trace_ctx, D_CABAC );
340  DTRACE_WITHOUT_COUNT( g_trace_ctx, D_CABAC, " - " "%d" "\n", bin );
341  return bin;
342 }
