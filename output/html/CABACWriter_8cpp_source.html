80 SliceType xGetCtxInitId( const Slice& slice, const BinEncIf& binEncoder, Ctx& ctxTest )
81 {
82  const CtxStore& ctxStoreTest = static_cast<const CtxStore&>( ctxTest );
83  const CtxStore& ctxStoreRef = static_cast<const CtxStore&>( binEncoder.getCtx() );
84  int qp = slice.sliceQp;
85  if( !slice.isIntra() )
86  {
87  SliceType aSliceTypeChoices[] = { VVENC_B_SLICE, VVENC_P_SLICE };
88  uint64_t bestCost = std::numeric_limits<uint64_t>::max();
89  SliceType bestSliceType = aSliceTypeChoices[0];
90  for (uint32_t idx=0; idx<2; idx++)
91  {
92  uint64_t curCost = 0;
93  SliceType curSliceType = aSliceTypeChoices[idx];
94  ctxTest.init( qp, (int)curSliceType );
95  for( int k = 0; k < Ctx::NumberOfContexts; k++ )
96  {
97  if( binEncoder.getNumBins(k) > 0 )
98  {
99  curCost += uint64_t( binEncoder.getNumBins(k) ) * ctxStoreRef[k].estFracExcessBits( ctxStoreTest[k] );
100  }
101  }
102  if (curCost < bestCost)
103  {
104  bestSliceType = curSliceType;
105  bestCost = curCost;
106  }
107  }
108  return bestSliceType;
109  }
110  else
111  {
112  return VVENC_I_SLICE;
113  }
114 }

123 unsigned estBits( BinEncIf& binEnc, const std::vector<bool>& bins, const Ctx& ctx, const int ctxId, const uint8_t winSize )
124 {
125  binEnc.initCtxAndWinSize( ctxId, ctx, winSize );

159 void CABACWriter::coding_tree_unit( CodingStructure& cs, const UnitArea& area, int (&qps)[2], unsigned ctuRsAddr, bool skipSao /* = false */, bool skipAlf /* = false */ )
160 {
161  CUCtx cuCtx( qps[CH_L] );
162  Partitioner *partitioner = &m_partitioner[0];
163 
164  partitioner->initCtu( area, CH_L, *cs.slice );
165 
166  if( !skipSao )
167  {
168  sao( *cs.slice, ctuRsAddr );
169  }
170 
171  if (!skipAlf)
172  {
173  for (int compIdx = 0; compIdx < MAX_NUM_COMP; compIdx++)
174  {
175  codeAlfCtuEnabledFlag(cs, ctuRsAddr, compIdx, NULL);
176  if (isLuma(ComponentID(compIdx)))
177  {
178  codeAlfCtuFilterIndex(cs, ctuRsAddr, cs.slice->tileGroupAlfEnabled[COMP_Y]);
179  }
180  if (isChroma(ComponentID(compIdx)))
181  {
182  const uint8_t* ctbAlfFlag = cs.slice->tileGroupAlfEnabled[compIdx] ? cs.slice->pic->m_alfCtuEnabled[ compIdx ].data() : nullptr;
183  if( ctbAlfFlag && ctbAlfFlag[ctuRsAddr] )
184  {
185  codeAlfCtuAlternative( cs, ctuRsAddr, compIdx );
186  }
187  }
188  }
189  }
190 
191  if ( !skipAlf )
192  {
193  for ( int compIdx = 1; compIdx < getNumberValidComponents( cs.pcv->chrFormat ); compIdx++ )
194  {
195  if (cs.slice->ccAlfFilterParam.ccAlfFilterEnabled[compIdx - 1])
196  {
197  const int filterCount = cs.slice->ccAlfFilterParam.ccAlfFilterCount[compIdx - 1];
198 
199  const int ry = ctuRsAddr / cs.pcv->widthInCtus;
200  const int rx = ctuRsAddr % cs.pcv->widthInCtus;
201  const Position lumaPos(rx * cs.pcv->maxCUSize, ry * cs.pcv->maxCUSize);
202 
203  codeCcAlfFilterControlIdc(cs.slice->ccAlfFilterControl[compIdx - 1][ctuRsAddr], cs, ComponentID(compIdx),
204  ctuRsAddr, cs.slice->ccAlfFilterControl[compIdx - 1], lumaPos, filterCount);
205  }
206  }
207  }
208 
209  if ( CS::isDualITree(cs) && cs.pcv->chrFormat != CHROMA_400 && cs.pcv->maxCUSize > 64 )
210  {
211  CUCtx chromaCuCtx(qps[CH_C]);
212  Partitioner *chromaPartitioner = &m_partitioner[1];
213  chromaPartitioner->initCtu(area, CH_C, *cs.slice);
214  coding_tree(cs, *partitioner, cuCtx, chromaPartitioner, &chromaCuCtx);
215  qps[CH_L] = cuCtx.qp;
216  qps[CH_C] = chromaCuCtx.qp;
217  }
218  else
219  {
220  coding_tree( cs, *partitioner, cuCtx );
221  qps[CH_L] = cuCtx.qp;
222  if( CS::isDualITree( cs ) && cs.pcv->chrFormat != CHROMA_400 )
223  {
224  CUCtx cuCtxChroma( qps[CH_C] );
225  partitioner->initCtu( area, CH_C, *cs.slice );
226  coding_tree( cs, *partitioner, cuCtxChroma );
227  qps[CH_C] = cuCtxChroma.qp;
228  }
229  }
230 }

242 {

250  const PreCalcValues& pcv = *cs.pcv;
251  const SAOBlkParam& sao_ctu_pars = cs.picture->getSAO()[ctuRsAddr];
252  bool slice_sao_luma_flag = ( slice.saoEnabled[ CH_L ] );
253  bool slice_sao_chroma_flag = ( slice.saoEnabled[ CH_C ] && sps.chromaFormatIdc != CHROMA_400 );
254  if( !slice_sao_luma_flag && !slice_sao_chroma_flag )
255  {
256  return;
257  }
258

260  int frame_width_in_ctus = pcv.widthInCtus;
261  int ry = ctuRsAddr / frame_width_in_ctus;
262  int rx = ctuRsAddr - ry * frame_width_in_ctus;
263  const Position pos ( rx * cs.pcv->maxCUSize, ry * cs.pcv->maxCUSize );
264  const unsigned curSliceIdx = slice.independentSliceIdx;
265  const unsigned curTileIdx = 0;//cs.picture->brickMap->getBrickIdxRsMap( pos );
266  bool leftMergeAvail = cs.getCURestricted( pos.offset( -(int)pcv.maxCUSize, 0 ), pos, curSliceIdx, curTileIdx, CH_L, TREE_D ) ? true : false;
267  bool aboveMergeAvail = cs.getCURestricted( pos.offset( 0, -(int)pcv.maxCUSize ), pos, curSliceIdx, curTileIdx, CH_L, TREE_D ) ? true : false;
268  sao_block_pars( sao_ctu_pars, sps.bitDepths, sliceEnabled, leftMergeAvail, aboveMergeAvail, false );

273 {

303 void CABACWriter::sao_offset_pars( const SAOOffset& ctbPars, ComponentID compID, bool sliceEnabled, int bitDepth )
304 {
305  if( !sliceEnabled )
306  {
307  CHECK( ctbPars.modeIdc != SAO_MODE_OFF, "Sao must be off, if it is disabled on slice level" );
308  return;
309  }
310  const bool isFirstCompOfChType = ( getFirstComponentOfChannel( toChannelType(compID) ) == compID );
311 
312  if( isFirstCompOfChType )
313  {
314  // sao_type_idx_luma / sao_type_idx_chroma
315  if( ctbPars.modeIdc == SAO_MODE_OFF )
316  {
317  m_BinEncoder.encodeBin ( 0, Ctx::SaoTypeIdx() );
318  }
319  else if( ctbPars.typeIdc == SAO_TYPE_BO )
320  {
321  m_BinEncoder.encodeBin ( 1, Ctx::SaoTypeIdx() );
322  m_BinEncoder.encodeBinEP( 0 );
323  }
324  else
325  {
326  CHECK(!( ctbPars.typeIdc < SAO_TYPE_START_BO ), "Unspecified error");
327  m_BinEncoder.encodeBin ( 1, Ctx::SaoTypeIdx() );
328  m_BinEncoder.encodeBinEP( 1 );
329  }
330  }
331 
332  if( ctbPars.modeIdc == SAO_MODE_NEW )
333  {
334  const int maxOffsetQVal = SampleAdaptiveOffset::getMaxOffsetQVal( bitDepth );
335  int numClasses = ( ctbPars.typeIdc == SAO_TYPE_BO ? 4 : NUM_SAO_EO_CLASSES );
336  int k = 0;
337  int offset[4];
338  for( int i = 0; i < numClasses; i++ )
339  {
340  if( ctbPars.typeIdc != SAO_TYPE_BO && i == SAO_CLASS_EO_PLAIN )
341  {
342  continue;
343  }
344  int classIdx = ( ctbPars.typeIdc == SAO_TYPE_BO ? ( ctbPars.typeAuxInfo + i ) % NUM_SAO_BO_CLASSES : i );
345  offset[k++] = ctbPars.offset[classIdx];
346  }
347 
348  // sao_offset_abs
349  for( int i = 0; i < 4; i++ )
350  {
351  unsigned absOffset = ( offset[i] < 0 ? -offset[i] : offset[i] );
352  unary_max_eqprob( absOffset, maxOffsetQVal );
353  }
354 
355  // band offset mode
356  if( ctbPars.typeIdc == SAO_TYPE_BO )
357  {
358  // sao_offset_sign
359  for( int i = 0; i < 4; i++ )
360  {
361  if( offset[i] )
362  {
363  m_BinEncoder.encodeBinEP( (offset[i] < 0) );
364  }
365  }
366  // sao_band_position
367  m_BinEncoder.encodeBinsEP( ctbPars.typeAuxInfo, NUM_SAO_BO_CLASSES_LOG2 );
368  }
369  // edge offset mode
370  else
371  {
372  if( isFirstCompOfChType )
373  {
374  // sao_eo_class_luma / sao_eo_class_chroma
375  CHECK( ctbPars.typeIdc - SAO_TYPE_START_EO < 0, "sao edge offset class is outside valid range" );
376  m_BinEncoder.encodeBinsEP( ctbPars.typeIdc - SAO_TYPE_START_EO, NUM_SAO_EO_TYPES_LOG2 );
377  }
378  }
379  }
380 }

391 void CABACWriter::coding_tree(const CodingStructure& cs, Partitioner& partitioner, CUCtx& cuCtx, Partitioner* pPartitionerChroma, CUCtx* pCuCtxChroma)
392 {
393  const PPS &pps = *cs.pps;
394  const UnitArea& currArea = partitioner.currArea();
395  const CodingUnit &cu = *cs.getCU( currArea.blocks[partitioner.chType], partitioner.chType, partitioner.treeType );
396 
397  // Reset delta QP coding flag and ChromaQPAdjustemt coding flag
398  //Note: do not reset qg at chroma CU
399  if( pps.useDQP && partitioner.currQgEnable() && !isChroma( partitioner.chType ) )
400  {
401  cuCtx.qgStart = true;
402  cuCtx.isDQPCoded = false;
403  }
404  if( cs.slice->chromaQpAdjEnabled && partitioner.currQgChromaEnable() )
405  {
406  cuCtx.isChromaQpAdjCoded = false;
407  }
408  // Reset delta QP coding flag and ChromaQPAdjustemt coding flag
409  if (CS::isDualITree(cs) && pPartitionerChroma != nullptr)
410  {
411  if (pps.useDQP && pPartitionerChroma->currQgEnable())
412  {
413  pCuCtxChroma->qgStart = true;
414  pCuCtxChroma->isDQPCoded = false;
415  }
416  if (cs.slice->chromaQpAdjEnabled && pPartitionerChroma->currQgChromaEnable())
417  {
418  pCuCtxChroma->isChromaQpAdjCoded = false;
419  }
420  }
421 
422  determineNeighborCus( cs, partitioner.currArea(), partitioner.chType, partitioner.treeType );
423 
424  const PartSplit splitMode = CU::getSplitAtDepth( cu, partitioner.currDepth );
425 
426  split_cu_mode( splitMode, cs, partitioner );
427 
428  CHECK( !partitioner.canSplit( splitMode, cs ), "The chosen split mode is invalid!" );
429 
430  if( splitMode != CU_DONT_SPLIT )
431  {
432  if (CS::isDualITree(cs) && pPartitionerChroma != nullptr && (partitioner.currArea().lwidth() >= 64 || partitioner.currArea().lheight() >= 64))
433  {
434  partitioner.splitCurrArea(CU_QUAD_SPLIT, cs);
435  pPartitionerChroma->splitCurrArea(CU_QUAD_SPLIT, cs);
436  bool beContinue = true;
437  bool lumaContinue = true;
438  bool chromaContinue = true;
439 
440  while (beContinue)
441  {
442  if (partitioner.currArea().lwidth() > 64 || partitioner.currArea().lheight() > 64)
443  {
444  if (cs.picture->blocks[partitioner.chType].contains(partitioner.currArea().blocks[partitioner.chType].pos()))
445  {
446  coding_tree(cs, partitioner, cuCtx, pPartitionerChroma, pCuCtxChroma);
447  }
448  lumaContinue = partitioner.nextPart(cs);
449  chromaContinue = pPartitionerChroma->nextPart(cs);
450  CHECK(lumaContinue != chromaContinue, "luma chroma partition should be matched");
451  beContinue = lumaContinue;
452  }
453  else
454  {
455  //dual tree coding under 64x64 block
456  if (cs.picture->blocks[partitioner.chType].contains(partitioner.currArea().blocks[partitioner.chType].pos()))
457  {
458  coding_tree(cs, partitioner, cuCtx);
459  }
460  lumaContinue = partitioner.nextPart(cs);
461  if (cs.picture->blocks[pPartitionerChroma->chType].contains(pPartitionerChroma->currArea().blocks[pPartitionerChroma->chType].pos()))
462  {
463  coding_tree(cs, *pPartitionerChroma, *pCuCtxChroma);
464  }
465  chromaContinue = pPartitionerChroma->nextPart(cs);
466  CHECK(lumaContinue != chromaContinue, "luma chroma partition should be matched");
467  beContinue = lumaContinue;
468  }
469  }
470  partitioner.exitCurrSplit();
471  pPartitionerChroma->exitCurrSplit();
472 
473  }
474  else
475  {
476  const ModeType modeTypeParent = partitioner.modeType;
477  const ModeType modeTypeChild = CU::getModeTypeAtDepth( cu, partitioner.currDepth );
478  mode_constraint( splitMode, cs, partitioner, modeTypeChild );
479  partitioner.modeType = modeTypeChild;
480 
481  bool chromaNotSplit = modeTypeParent == MODE_TYPE_ALL && modeTypeChild == MODE_TYPE_INTRA ? true : false;
482  CHECK( chromaNotSplit && partitioner.chType != CH_L, "chType must be luma" );
483  if( partitioner.treeType == TREE_D )
484  {
485  partitioner.treeType = chromaNotSplit ? TREE_L : TREE_D;
486  }
487  partitioner.splitCurrArea( splitMode, cs );
488 
489  do
490  {
491  if( cs.picture->blocks[partitioner.chType].contains( partitioner.currArea().blocks[partitioner.chType].pos() ) )
492  {
493  coding_tree( cs, partitioner, cuCtx );
494  }
495  } while( partitioner.nextPart( cs ) );
496 
497  partitioner.exitCurrSplit();
498  if( chromaNotSplit )
499  {
500  if (isChromaEnabled(cs.pcv->chrFormat))
501  {
502  CHECK( partitioner.chType != CH_L, "must be luma status" );
503  partitioner.chType = CH_C;
504  partitioner.treeType = TREE_C;
505 
506  if( cs.picture->blocks[partitioner.chType].contains( partitioner.currArea().blocks[partitioner.chType].pos() ) )
507  {
508  coding_tree( cs, partitioner, cuCtx );
509  }
510  }
511  //recover
512  partitioner.chType = CH_L;
513  partitioner.treeType = TREE_D;
514  }
515  partitioner.modeType = modeTypeParent;
516  }
517  return;
518  }
519 
520  // Predict QP on start of quantization group
521  if( cuCtx.qgStart )
522  {
523  cuCtx.qgStart = false;
524  cuCtx.qp = CU::predictQP( cu, cuCtx.qp );
525  }
526  CHECK( cu.treeType != partitioner.treeType, "treeType mismatch" );
527 
528 
529  // coding unit
530  coding_unit( cu, partitioner, cuCtx );
531 
532  if( cu.chType == CH_C )
533  {
534  DTRACE_COND( (isEncoding()), g_trace_ctx, D_QP, "[chroma CU]x=%d, y=%d, w=%d, h=%d, qp=%d\n", cu.Cb().x, cu.Cb().y, cu.Cb().width, cu.Cb().height, cu.qp );
535  }
536  else
537  {
538  DTRACE_COND( ( isEncoding() ), g_trace_ctx, D_QP, "x=%d, y=%d, w=%d, h=%d, qp=%d\n", cu.Y().x, cu.Y().y, cu.Y().width, cu.Y().height, cu.qp );
539  }
540  DTRACE_BLOCK_REC_COND( ( !isEncoding() ), cs.picture->getRecoBuf( cu ), cu, cu.predMode );
541 }

567 void CABACWriter::split_cu_mode( const PartSplit split, const CodingStructure& cs, Partitioner& partitioner )
568 {
569  bool canNo, canQt, canBh, canBv, canTh, canTv;
570  partitioner.canSplit( cs, canNo, canQt, canBh, canBv, canTh, canTv );

892 void CABACWriter::xWriteTruncBinCode(uint32_t symbol, uint32_t maxSymbol)
893 {
894  int thresh;
895  if (maxSymbol > 256)
896  {
897  int threshVal = 1 << 8;
898  thresh = 8;
899  while (threshVal <= maxSymbol)
900  {
901  thresh++;
902  threshVal <<= 1;
903  }
904  thresh--;
905  }
906  else
907  {
908  thresh = g_tbMax[maxSymbol];
909  }
910 
911  int val = 1 << thresh;
912  assert(val <= maxSymbol);
913  assert((val << 1) > maxSymbol);
914  assert(symbol < maxSymbol);
915  int b = maxSymbol - val;
916  assert(b < val);
917  if (symbol < val - b)
918  {
919  m_BinEncoder.encodeBinsEP(symbol, thresh);
920  }
921  else
922  {
923  symbol += val - b;
924  assert(symbol < (val << 1));
925  assert((symbol >> 1) >= val - b);
926  m_BinEncoder.encodeBinsEP(symbol, thresh + 1);
927  }
