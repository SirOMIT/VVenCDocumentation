71 #define CALC_EQUAL_COEFF_8PXLS(x1,x2,y1,y2,tmp0,tmp1,tmp2,tmp3,inter0,inter1,inter2,inter3,loadLocation) \
72 { \
73 inter0 = _mm_mul_epi32(x1, y1); \
74 inter1 = _mm_mul_epi32(tmp0, tmp2); \
75 inter2 = _mm_mul_epi32(x2, y2); \
76 inter3 = _mm_mul_epi32(tmp1, tmp3); \
77 inter2 = _mm_add_epi64(inter0, inter2); \
78 inter3 = _mm_add_epi64(inter1, inter3); \
79 inter0 = _mm_loadl_epi64(loadLocation); \
80 inter3 = _mm_add_epi64(inter2, inter3); \
81 inter1 = _mm_srli_si128(inter3, 8); \
82 inter3 = _mm_add_epi64(inter1, inter3); \
83 inter3 = _mm_add_epi64(inter0, inter3); \
84 }

86  template<X86_VEXT vext>
87  static void simdHorizontalSobelFilter(Pel* const pPred, const int predStride, int *const pDerivate, const int derivateBufStride, const int width, const int height)
88  {
89  __m128i mmPred[4];
90  __m128i mm2xPred[2];
91  __m128i mmIntermediates[4];
92  __m128i mmDerivate[2];
93 
94  assert(!(height % 2));
95  assert(!(width % 4));
96 
97  /* Derivates of the rows and columns at the boundary are done at the end of this function */
98  /* The value of col and row indicate the columns and rows for which the derivates have already been computed */
99  for (int col = 1; (col + 2) < width; col += 2)
100  {
101  mmPred[0] = _mm_loadl_epi64(reinterpret_cast<const __m128i *>(&pPred[0 * predStride + col - 1]));
102  mmPred[1] = _mm_loadl_epi64(reinterpret_cast<const __m128i *>(&pPred[1 * predStride + col - 1]));
103 
104  mmPred[0] = _mm_cvtepi16_epi32(mmPred[0]);
105  mmPred[1] = _mm_cvtepi16_epi32(mmPred[1]);
106 
107  for (int row = 1; row < (height - 1); row += 2)
108  {
109  mmPred[2] = _mm_loadl_epi64(reinterpret_cast<const __m128i *>(&pPred[(row + 1) * predStride + col - 1]));
110  mmPred[3] = _mm_loadl_epi64(reinterpret_cast<const __m128i *>(&pPred[(row + 2) * predStride + col - 1]));
111 
112  mmPred[2] = _mm_cvtepi16_epi32(mmPred[2]);
113  mmPred[3] = _mm_cvtepi16_epi32(mmPred[3]);
114 
115  mm2xPred[0] = _mm_slli_epi32(mmPred[1], 1);
116  mm2xPred[1] = _mm_slli_epi32(mmPred[2], 1);
117 
118  mmIntermediates[0] = _mm_add_epi32(mm2xPred[0], mmPred[0]);
119  mmIntermediates[2] = _mm_add_epi32(mm2xPred[1], mmPred[1]);
120 
121  mmIntermediates[0] = _mm_add_epi32(mmIntermediates[0], mmPred[2]);
122  mmIntermediates[2] = _mm_add_epi32(mmIntermediates[2], mmPred[3]);
123 
124  mmPred[0] = mmPred[2];
125  mmPred[1] = mmPred[3];
126 
127  mmIntermediates[1] = _mm_srli_si128(mmIntermediates[0], 8);
128  mmIntermediates[3] = _mm_srli_si128(mmIntermediates[2], 8);
129 
130  mmDerivate[0] = _mm_sub_epi32(mmIntermediates[1], mmIntermediates[0]);
131  mmDerivate[1] = _mm_sub_epi32(mmIntermediates[3], mmIntermediates[2]);
132 
133  _mm_storel_epi64(reinterpret_cast<__m128i *> (&pDerivate[col + (row + 0) * derivateBufStride]), mmDerivate[0]);
134  _mm_storel_epi64(reinterpret_cast<__m128i *> (&pDerivate[col + (row + 1) * derivateBufStride]), mmDerivate[1]);
135  }
136  }
137 
138  for (int j = 1; j < (height - 1); j++)
139  {
140  pDerivate[j * derivateBufStride] = pDerivate[j * derivateBufStride + 1];
141  pDerivate[j * derivateBufStride + (width - 1)] = pDerivate[j * derivateBufStride + (width - 2)];
142  }
143 
144  memcpy(pDerivate, pDerivate + derivateBufStride, width * sizeof(pDerivate[0]));
145  memcpy(pDerivate + (height - 1) * derivateBufStride, pDerivate + (height - 2) * derivateBufStride, width * sizeof(pDerivate[0])
146  );
147  }

150  static void simdVerticalSobelFilter(Pel* const pPred, const int predStride, int *const pDerivate, const int derivateBufStride, const int width, const int height)
151  {
152  __m128i mmPred[4];
153  __m128i mmIntermediates[6];
154  __m128i mmDerivate[2];
155 
156  assert(!(height % 2));
157  assert(!(width % 4));
158 
159  /* Derivates of the rows and columns at the boundary are done at the end of this function */
160  /* The value of col and row indicate the columns and rows for which the derivates have already been computed */
161  for (int col = 1; col < (width - 1); col += 2)
162  {
163  mmPred[0] = _mm_loadl_epi64(reinterpret_cast<const __m128i *>(&pPred[0 * predStride + col - 1]));
164  mmPred[1] = _mm_loadl_epi64(reinterpret_cast<const __m128i *>(&pPred[1 * predStride + col - 1]));
165 
166  mmPred[0] = _mm_cvtepi16_epi32(mmPred[0]);
167  mmPred[1] = _mm_cvtepi16_epi32(mmPred[1]);
168 
169  for (int row = 1; row < (height - 1); row += 2)
170  {
171  mmPred[2] = _mm_loadl_epi64(reinterpret_cast<const __m128i *>(&pPred[(row + 1) * predStride + col - 1]));
172  mmPred[3] = _mm_loadl_epi64(reinterpret_cast<const __m128i *>(&pPred[(row + 2) * predStride + col - 1]));
173 
174  mmPred[2] = _mm_cvtepi16_epi32(mmPred[2]);
175  mmPred[3] = _mm_cvtepi16_epi32(mmPred[3]);
176 
177  mmIntermediates[0] = _mm_sub_epi32(mmPred[2], mmPred[0]);
178  mmIntermediates[3] = _mm_sub_epi32(mmPred[3], mmPred[1]);
179 
180  mmPred[0] = mmPred[2];
181  mmPred[1] = mmPred[3];
182 
183  mmIntermediates[1] = _mm_srli_si128(mmIntermediates[0], 4);
184  mmIntermediates[4] = _mm_srli_si128(mmIntermediates[3], 4);
185  mmIntermediates[2] = _mm_srli_si128(mmIntermediates[0], 8);
186  mmIntermediates[5] = _mm_srli_si128(mmIntermediates[3], 8);
187 
188  mmIntermediates[1] = _mm_slli_epi32(mmIntermediates[1], 1);
189  mmIntermediates[4] = _mm_slli_epi32(mmIntermediates[4], 1);
190 
191  mmIntermediates[0] = _mm_add_epi32(mmIntermediates[0], mmIntermediates[2]);
192  mmIntermediates[3] = _mm_add_epi32(mmIntermediates[3], mmIntermediates[5]);
193 
194  mmDerivate[0] = _mm_add_epi32(mmIntermediates[0], mmIntermediates[1]);
195  mmDerivate[1] = _mm_add_epi32(mmIntermediates[3], mmIntermediates[4]);
196 
197  _mm_storel_epi64(reinterpret_cast<__m128i *> (&pDerivate[col + (row + 0) * derivateBufStride]), mmDerivate[0]);
198  _mm_storel_epi64(reinterpret_cast<__m128i *> (&pDerivate[col + (row + 1) * derivateBufStride]), mmDerivate[1]);
199  }
200  }
201 
202  for (int j = 1; j < (height - 1); j++)
203  {
204  pDerivate[j * derivateBufStride] = pDerivate[j * derivateBufStride + 1];
205  pDerivate[j * derivateBufStride + (width - 1)] = pDerivate[j * derivateBufStride + (width - 2)];
206  }
207 
208  memcpy(pDerivate, pDerivate + derivateBufStride, width * sizeof(pDerivate[0]));
209  memcpy(pDerivate + (height - 1) * derivateBufStride, pDerivate + (height - 2) * derivateBufStride, width * sizeof(pDerivate[0]));

212  template<X86_VEXT vext>
213  static void simdEqualCoeffComputer(Pel* pResidue, int residueStride, int **ppDerivate, int derivateBufStride, int64_t(*pEqualCoeff)[7], int width, int height, bool b6Param)
214  {
215  __m128i mmFour;
216  __m128i mmTmp[4];
217  __m128i mmIntermediate[4];
218  __m128i mmIndxK, mmIndxJ;
219  __m128i mmResidue[2];
220  __m128i mmC[12];
221 
222  // Add directly to indexes to get new index
223  mmFour = _mm_set1_epi32(4);
224  mmIndxJ = _mm_set1_epi32(-2);
225 
226 
227  int n = b6Param ? 6 : 4;
228  int idx1 = 0, idx2 = 0;
229  idx1 = -2 * derivateBufStride - 4;
230  idx2 = -derivateBufStride - 4;
231 
232  for (int j = 0; j < height; j += 2)
233  {
234  if (!(j & 3))
235  mmIndxJ = _mm_add_epi32(mmIndxJ, mmFour);
236  mmIndxK = _mm_set1_epi32(-2);
237  idx1 += (derivateBufStride << 1);
238  idx2 += (derivateBufStride << 1);
239 
240  for (int k = 0; k < width; k += 4)
241  {
242  idx1 += 4;
243  idx2 += 4;
244  mmIndxK = _mm_add_epi32(mmIndxK, mmFour);
245 
246  if (b6Param)
247  {
248  // mmC[0-5] for iC[0-5] of 1st row of pixels
249  mmC[0] = _mm_loadu_si128((const __m128i*)&ppDerivate[0][idx1]);
250  mmC[2] = _mm_loadu_si128((const __m128i*)&ppDerivate[1][idx1]);
251  mmC[1] = _mm_mullo_epi32(mmIndxK, mmC[0]);
252  mmC[3] = _mm_mullo_epi32(mmIndxK, mmC[2]);
253  mmC[4] = _mm_mullo_epi32(mmIndxJ, mmC[0]);
254  mmC[5] = _mm_mullo_epi32(mmIndxJ, mmC[2]);
255 
256  // mmC[6-11] for iC[0-5] of 2nd row of pixels
257  mmC[6] = _mm_loadu_si128((const __m128i*)&ppDerivate[0][idx2]);
258  mmC[8] = _mm_loadu_si128((const __m128i*)&ppDerivate[1][idx2]);
259  mmC[7] = _mm_mullo_epi32(mmIndxK, mmC[6]);
260  mmC[9] = _mm_mullo_epi32(mmIndxK, mmC[8]);
261  mmC[10] = _mm_mullo_epi32(mmIndxJ, mmC[6]);
262  mmC[11] = _mm_mullo_epi32(mmIndxJ, mmC[8]);
263  }
264  else
265  {
266  // mmC[0-3] for iC[0-3] of 1st row of pixels
267  mmC[0] = _mm_loadu_si128((const __m128i*)&ppDerivate[0][idx1]);
268  mmC[2] = _mm_loadu_si128((const __m128i*)&ppDerivate[1][idx1]);
269  mmC[1] = _mm_mullo_epi32(mmIndxK, mmC[0]);
270  mmC[3] = _mm_mullo_epi32(mmIndxJ, mmC[0]);
271  mmTmp[0] = _mm_mullo_epi32(mmIndxJ, mmC[2]);
272  mmTmp[1] = _mm_mullo_epi32(mmIndxK, mmC[2]);
273  mmC[1] = _mm_add_epi32(mmC[1], mmTmp[0]);
274  mmC[3] = _mm_sub_epi32(mmC[3], mmTmp[1]);
275 
276  // mmC[4-7] for iC[0-3] of 1st row of pixels
277  mmC[4] = _mm_loadu_si128((const __m128i*)&ppDerivate[0][idx2]);
278  mmC[6] = _mm_loadu_si128((const __m128i*)&ppDerivate[1][idx2]);
279  mmC[5] = _mm_mullo_epi32(mmIndxK, mmC[4]);
280  mmC[7] = _mm_mullo_epi32(mmIndxJ, mmC[4]);
281  mmTmp[2] = _mm_mullo_epi32(mmIndxJ, mmC[6]);
282  mmTmp[3] = _mm_mullo_epi32(mmIndxK, mmC[6]);
283  mmC[5] = _mm_add_epi32(mmC[5], mmTmp[2]);
284  mmC[7] = _mm_sub_epi32(mmC[7], mmTmp[3]);
285  }
286 
287  // Residue
288  mmResidue[0] = _mm_loadl_epi64((const __m128i*)&pResidue[idx1]);
289  mmResidue[1] = _mm_loadl_epi64((const __m128i*)&pResidue[idx2]);
290  mmResidue[0] = _mm_cvtepi16_epi32(mmResidue[0]);
291  mmResidue[1] = _mm_cvtepi16_epi32(mmResidue[1]);
292  mmResidue[0] = _mm_slli_epi32(mmResidue[0], 3);
293  mmResidue[1] = _mm_slli_epi32(mmResidue[1], 3);
294 
295  // Calculation of coefficient matrix
296  for (int col = 0; col < n; col++)
297  {
298  mmTmp[0] = _mm_srli_si128(mmC[0 + col], 4);
299  mmTmp[1] = _mm_srli_si128(mmC[n + col], 4);
300  CALC_EQUAL_COEFF_8PXLS(mmC[0 + col], mmC[n + col], mmC[0 + col], mmC[n + col], mmTmp[0], mmTmp[1], mmTmp[0], mmTmp[1], mmIntermediate[0], mmIntermediate[1], mmIntermediate[2], mmIntermediate[3], (const __m128i*)&pEqualCoeff[col + 1][col]);
301  _mm_storel_epi64((__m128i*)&pEqualCoeff[col + 1][col], mmIntermediate[3]);
302 
303  for (int row = col + 1; row < n; row++)
304  {
305  mmTmp[2] = _mm_srli_si128(mmC[0 + row], 4);
306  mmTmp[3] = _mm_srli_si128(mmC[n + row], 4);
307  CALC_EQUAL_COEFF_8PXLS(mmC[0 + col], mmC[n + col], mmC[0 + row], mmC[n + row], mmTmp[0], mmTmp[1], mmTmp[2], mmTmp[3], mmIntermediate[0], mmIntermediate[1], mmIntermediate[2], mmIntermediate[3], (const __m128i*)&pEqualCoeff[col + 1][row]);
308  _mm_storel_epi64((__m128i*)&pEqualCoeff[col + 1][row], mmIntermediate[3]);
309  _mm_storel_epi64((__m128i*)&pEqualCoeff[row + 1][col], mmIntermediate[3]);
310  }
311 
312  mmTmp[2] = _mm_srli_si128(mmResidue[0], 4);
313  mmTmp[3] = _mm_srli_si128(mmResidue[1], 4);
314  CALC_EQUAL_COEFF_8PXLS(mmC[0 + col], mmC[n + col], mmResidue[0], mmResidue[1], mmTmp[0], mmTmp[1], mmTmp[2], mmTmp[3], mmIntermediate[0], mmIntermediate[1], mmIntermediate[2], mmIntermediate[3], (const __m128i*)&pEqualCoeff[col + 1][n]);
315  _mm_storel_epi64((__m128i*)&pEqualCoeff[col + 1][n], mmIntermediate[3]);
316  }
317  }
318 
319  idx1 -= (width);
320  idx2 -= (width);
321  }
322  }
