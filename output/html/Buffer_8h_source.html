269 template<typename T>
270 T AreaBuf <T> ::getAvg() const
271 {
272  const T* src = buf;
273  int64_t acc = 0;
274 
275 #define AVG_INC src += stride
276 #define AVG_OP(ADDR) acc += src[ADDR]
277  SIZE_AWARE_PER_EL_OP(AVG_OP, AVG_INC);
278 #undef AVG_INC
279 #undef AVG_OP
280 
281  return T ((acc + (area() >> 1)) / area());

419  const int srcStride = other.stride;

422  const int dstStride = stride;
423 
424 #if ENABLE_SIMD_OPT_BCW

426  {
427  if(!(width & 7))
428  g_pelBufOP.removeHighFreq8(dst, dstStride, src, srcStride, width, height);

432  CHECK(true, "Not supported");
433  }
434  else
435  {
436 #endif

439  src += srcStride; \

561 {

564  int h = height;
565  int w = width;
566 
567  CHECK( w > s, "Size of buffer too small to extend" );
568

571  {
572  for( int y = 0; y < marginY; y++ )
573  {
574  for( int x = 0; x < marginX; x++ )
575  {

579  }

581

588  {

590  {
