67 void CABACReader::initCtxModels( Slice& slice )
68 {
69  SliceType sliceType = slice.sliceType;
70  int qp = slice.sliceQp;
71  if( slice.pps->cabacInitPresent && slice.cabacInitFlag )

84  }

133 void CABACReader::coding_tree_unit( CodingStructure& cs, const UnitArea& area, int (&qps)[2], unsigned ctuRsAddr )
134 {
135  CUCtx cuCtx( qps[CH_L] );
136  Partitioner* partitioner = &m_partitioner[0];
137 
138  partitioner->initCtu( area, CH_L, *cs.slice );
139  partitioner->treeType = TREE_D;
140  partitioner->modeType = MODE_TYPE_ALL;
141 
142 
143  sao( cs, ctuRsAddr );
144  if (cs.sps->alfEnabled && (cs.slice->tileGroupAlfEnabled[COMP_Y]))
145  {
146  const PreCalcValues& pcv = *cs.pcv;
147  int frame_width_in_ctus = pcv.widthInCtus;
148  int ry = ctuRsAddr / frame_width_in_ctus;
149  int rx = ctuRsAddr - ry * frame_width_in_ctus;
150  const Position pos( rx * cs.pcv->maxCUSize, ry * cs.pcv->maxCUSize );
151  const uint32_t curSliceIdx = cs.slice->independentSliceIdx;
152  const uint32_t curTileIdx = cs.pps->getTileIdx( pos );
153  bool leftAvail = cs.getCURestricted( pos.offset( -(int)pcv.maxCUSize, 0 ), pos, curSliceIdx, curTileIdx, CH_L, TREE_D ) ? true : false;
154  bool aboveAvail = cs.getCURestricted( pos.offset( 0, -(int)pcv.maxCUSize ), pos, curSliceIdx, curTileIdx, CH_L, TREE_D ) ? true : false;
155 
156  int leftCTUAddr = leftAvail ? ctuRsAddr - 1 : -1;
157  int aboveCTUAddr = aboveAvail ? ctuRsAddr - frame_width_in_ctus : -1;
158 
159  for( int compIdx = 0; compIdx < MAX_NUM_COMP; compIdx++ )
160  {
161  if (cs.slice->tileGroupAlfEnabled[compIdx])
162  {
163  uint8_t* ctbAlfFlag = cs.slice->pic->m_alfCtuEnabled[ compIdx ].data();
164  int ctx = 0;
165  ctx += leftCTUAddr > -1 ? ( ctbAlfFlag[leftCTUAddr] ? 1 : 0 ) : 0;
166  ctx += aboveCTUAddr > -1 ? ( ctbAlfFlag[aboveCTUAddr] ? 1 : 0 ) : 0;
167 
168  ctbAlfFlag[ctuRsAddr] = m_BinDecoder.decodeBin( Ctx::ctbAlfFlag( compIdx * 3 + ctx ) );
169 
170  if (isLuma((ComponentID)compIdx) && ctbAlfFlag[ctuRsAddr])
171  {
172  readAlfCtuFilterIndex(cs, ctuRsAddr);
173  }
174  if( isChroma( (ComponentID)compIdx ) )
175  {
176  int apsIdx = cs.slice->tileGroupChromaApsId;
177  CHECK(cs.slice->alfAps[apsIdx] == nullptr, "APS not initialized");
178  const AlfParam& alfParam = cs.slice->alfAps[apsIdx]->alfParam;
179  const int numAlts = alfParam.numAlternativesChroma;
180  uint8_t* ctbAlfAlternative = cs.slice->pic->m_alfCtuAlternative[compIdx].data();
181  ctbAlfAlternative[ctuRsAddr] = 0;
182  if( ctbAlfFlag[ctuRsAddr] )
183  {
184  uint8_t decoded = 0;
185  while( decoded < numAlts-1 && m_BinDecoder.decodeBin( Ctx::ctbAlfAlternative( compIdx-1 ) ) )
186  ++ decoded;
187  ctbAlfAlternative[ctuRsAddr] = decoded;
188  }
189  }
190  }
191  }
192  }
193 
194  if (cs.sps->ccalfEnabled)
195  {
196  for ( int compIdx = 1; compIdx < getNumberValidComponents( cs.pcv->chrFormat ); compIdx++ )
197  {
198  if (cs.slice->ccAlfFilterParam.ccAlfFilterEnabled[compIdx - 1])
199  {
200  const int filterCount = cs.slice->ccAlfFilterParam.ccAlfFilterCount[compIdx - 1];
201 
202  const int ry = ctuRsAddr / cs.pcv->widthInCtus;
203  const int rx = ctuRsAddr % cs.pcv->widthInCtus;
204  const Position lumaPos(rx * cs.pcv->maxCUSize, ry * cs.pcv->maxCUSize);
205 
206  ccAlfFilterControlIdc(cs, ComponentID(compIdx), ctuRsAddr, cs.slice->ccAlfFilterControl[compIdx - 1], lumaPos,
207  filterCount);
208  }
209  }
210  }
211 
212  if ( CS::isDualITree(cs) && cs.pcv->chrFormat != CHROMA_400 && cs.pcv->maxCUSize > 64 )
213  {
214  Partitioner* chromaPartitioner = &m_partitioner[1];
215  chromaPartitioner->initCtu(area, CH_C, *cs.slice);
216  CUCtx cuCtxChroma(qps[CH_C]);
217  coding_tree(cs, *partitioner, cuCtx, chromaPartitioner, &cuCtxChroma);
218  qps[CH_L] = cuCtx.qp;
219  qps[CH_C] = cuCtxChroma.qp;
220  }
221  else
222  {
223  coding_tree(cs, *partitioner, cuCtx);
224  qps[CH_L] = cuCtx.qp;
225  if( CS::isDualITree( cs ) && cs.pcv->chrFormat != CHROMA_400 )
226  {
227  CUCtx cuCtxChroma( qps[CH_C] );
228  partitioner->initCtu( area, CH_C, *cs.slice );
229  coding_tree( cs, *partitioner, cuCtxChroma );
230  qps[CH_C] = cuCtxChroma.qp;
231  }
232  }
233 
234  DTRACE_COND( ctuRsAddr == 0, g_trace_ctx, D_QP_PER_CTU, "\n%4d %2d", cs.picture->poc, cs.slice->sliceQp );
235  DTRACE ( g_trace_ctx, D_QP_PER_CTU, " %3d", qps[CH_L] - cs.slice->sliceQp );
236 }

238 void CABACReader::readAlfCtuFilterIndex(CodingStructure& cs, unsigned ctuRsAddr)
239 {
240  short* alfCtbFilterSetIndex = cs.slice->pic->m_alfCtbFilterIndex.data();
241  unsigned numAps = cs.slice->tileGroupNumAps;
242  unsigned numAvailableFiltSets = numAps + NUM_FIXED_FILTER_SETS;
243  uint32_t filtIndex = 0;
244  if (numAvailableFiltSets > NUM_FIXED_FILTER_SETS)
245  {
246  unsigned usePrevFilt = m_BinDecoder.decodeBin(Ctx::AlfUseTemporalFilt());
247  if (usePrevFilt)
248  {
249  if (numAps > 1)
250  {
251  xReadTruncBinCode(filtIndex, numAvailableFiltSets - NUM_FIXED_FILTER_SETS);
252  }
253  filtIndex += (unsigned)(NUM_FIXED_FILTER_SETS);
254  }
255  else
256  {
257  xReadTruncBinCode(filtIndex, NUM_FIXED_FILTER_SETS);
258  }
259  }
260  else
261  {
262  xReadTruncBinCode(filtIndex, NUM_FIXED_FILTER_SETS);
263  }
264  alfCtbFilterSetIndex[ctuRsAddr] = filtIndex;
265 }

267 void CABACReader::ccAlfFilterControlIdc(CodingStructure &cs, const ComponentID compID, const int curIdx,
268  uint8_t *filterControlIdc, Position lumaPos, int filterCount)
269 {
270  Position leftLumaPos = lumaPos.offset(-(int)cs.pcv->maxCUSize, 0);
271  Position aboveLumaPos = lumaPos.offset(0, -(int)cs.pcv->maxCUSize);
272  const uint32_t curSliceIdx = cs.slice->independentSliceIdx;
273  const uint32_t curTileIdx = cs.pps->getTileIdx( lumaPos );
274  bool leftAvail = cs.getCURestricted( leftLumaPos, lumaPos, curSliceIdx, curTileIdx, CH_L, TREE_D ) ? true : false;
275  bool aboveAvail = cs.getCURestricted( aboveLumaPos, lumaPos, curSliceIdx, curTileIdx, CH_L, TREE_D ) ? true : false;
276  int ctxt = 0;
277 
278  if (leftAvail)
279  {
280  ctxt += ( filterControlIdc[curIdx - 1] ) ? 1 : 0;
281  }
282  if (aboveAvail)
283  {
284  ctxt += ( filterControlIdc[curIdx - cs.pcv->widthInCtus] ) ? 1 : 0;
285  }
286  ctxt += ( compID == COMP_Cr ) ? 3 : 0;
287 
288  int idcVal = m_BinDecoder.decodeBin( Ctx::CcAlfFilterControlFlag( ctxt ) );
289  if ( idcVal )
290  {
291  while ( ( idcVal != filterCount ) && m_BinDecoder.decodeBinEP() )
292  {
293  idcVal++;
294  }
295  }
296  filterControlIdc[curIdx] = idcVal;
297 
298  DTRACE(g_trace_ctx, D_SYNTAX, "ccAlfFilterControlIdc() compID=%d pos=(%d,%d) ctxt=%d, filterCount=%d, idcVal=%d\n",
299  compID, lumaPos.x, lumaPos.y, ctxt, filterCount, idcVal);
300 }

308 void CABACReader::sao( CodingStructure& cs, unsigned ctuRsAddr )
309 {
310  const SPS& sps = *cs.sps;
311 
312  if( !sps.saoEnabled )
313  {
314  return;
315  }
316 
317  const Slice& slice = *cs.slice;
318  SAOBlkParam& sao_ctu_pars = cs.picture->getSAO()[ctuRsAddr];
319  bool slice_sao_luma_flag = ( slice.saoEnabled[ CH_L ] );
320  bool slice_sao_chroma_flag = ( slice.saoEnabled[ CH_C ] && sps.chromaFormatIdc != CHROMA_400 );
321  sao_ctu_pars[ COMP_Y ].modeIdc = SAO_MODE_OFF;
322  sao_ctu_pars[ COMP_Cb ].modeIdc = SAO_MODE_OFF;
323  sao_ctu_pars[ COMP_Cr ].modeIdc = SAO_MODE_OFF;
324  if( !slice_sao_luma_flag && !slice_sao_chroma_flag )
325  {
326  return;
327  }
328 
329  // merge
330  int frame_width_in_ctus = cs.pcv->widthInCtus;
331  int ry = ctuRsAddr / frame_width_in_ctus;
332  int rx = ctuRsAddr - ry * frame_width_in_ctus;
333  int sao_merge_type = -1;
334  const Position pos( rx * cs.pcv->maxCUSize, ry * cs.pcv->maxCUSize );
335  const unsigned curSliceIdx = cs.slice->independentSliceIdx;
336 
337  const unsigned curTileIdx = cs.pps->getTileIdx( pos );
338  if( cs.getCURestricted( pos.offset(-(int)cs.pcv->maxCUSize, 0), pos, curSliceIdx, curTileIdx, CH_L, TREE_D ) )
339  {
340  // sao_merge_left_flag
341  sao_merge_type += int( m_BinDecoder.decodeBin( Ctx::SaoMergeFlag() ) );
342  }
343 
344  if( sao_merge_type < 0 && cs.getCURestricted( pos.offset(0, -(int)cs.pcv->maxCUSize), pos, curSliceIdx, curTileIdx, CH_L, TREE_D ) )
345  {
346  // sao_merge_above_flag
347  sao_merge_type += int( m_BinDecoder.decodeBin( Ctx::SaoMergeFlag() ) ) << 1;
348  }
349  if( sao_merge_type >= 0 )
350  {
351  if( slice_sao_luma_flag || slice_sao_chroma_flag )
352  {
353  sao_ctu_pars[ COMP_Y ].modeIdc = SAO_MODE_MERGE;
354  sao_ctu_pars[ COMP_Y ].typeIdc = sao_merge_type;
355  }
356  if( slice_sao_chroma_flag )
357  {
358  sao_ctu_pars[ COMP_Cb ].modeIdc = SAO_MODE_MERGE;
359  sao_ctu_pars[ COMP_Cr ].modeIdc = SAO_MODE_MERGE;
360  sao_ctu_pars[ COMP_Cb ].typeIdc = sao_merge_type;
361  sao_ctu_pars[ COMP_Cr ].typeIdc = sao_merge_type;
362  }
363  return;
364  }
365 
366  // explicit parameters
367  ComponentID firstComp = ( slice_sao_luma_flag ? COMP_Y : COMP_Cb );
368  ComponentID lastComp = ( slice_sao_chroma_flag ? COMP_Cr : COMP_Y );
369  for( ComponentID compID = firstComp; compID <= lastComp; compID = ComponentID( compID + 1 ) )
370  {
371  SAOOffset& sao_pars = sao_ctu_pars[ compID ];
372 
373  // sao_type_idx_luma / sao_type_idx_chroma
374  if( compID != COMP_Cr )
375  {
376  if( m_BinDecoder.decodeBin( Ctx::SaoTypeIdx() ) )
377  {
378  if( m_BinDecoder.decodeBinEP( ) )
379  {
380  // edge offset
381  sao_pars.modeIdc = SAO_MODE_NEW;
382  sao_pars.typeIdc = SAO_TYPE_START_EO;
383  }
384  else
385  {
386  // band offset
387  sao_pars.modeIdc = SAO_MODE_NEW;
388  sao_pars.typeIdc = SAO_TYPE_START_BO;
389  }
390  }
391  }
392  else //Cr, follow Cb SAO type
393  {
394  sao_pars.modeIdc = sao_ctu_pars[ COMP_Cb ].modeIdc;
395  sao_pars.typeIdc = sao_ctu_pars[ COMP_Cb ].typeIdc;
396  }
397  if( sao_pars.modeIdc == SAO_MODE_OFF )
398  {
399  continue;
400  }
401 
402  // sao_offset_abs
403  int offset[4];
404  const int maxOffsetQVal = SampleAdaptiveOffset::getMaxOffsetQVal( sps.bitDepths[ toChannelType(compID) ] );
405  offset [0] = (int)unary_max_eqprob( maxOffsetQVal );
406  offset [1] = (int)unary_max_eqprob( maxOffsetQVal );
407  offset [2] = (int)unary_max_eqprob( maxOffsetQVal );
408  offset [3] = (int)unary_max_eqprob( maxOffsetQVal );
409 
410  // band offset mode
411  if( sao_pars.typeIdc == SAO_TYPE_START_BO )
412  {
413  // sao_offset_sign
414  for( int k = 0; k < 4; k++ )
415  {
416  if( offset[k] && m_BinDecoder.decodeBinEP( ) )
417  {
418  offset[k] = -offset[k];
419  }
420  }
421  // sao_band_position
422  sao_pars.typeAuxInfo = m_BinDecoder.decodeBinsEP( NUM_SAO_BO_CLASSES_LOG2 );
423  for( int k = 0; k < 4; k++ )
424  {
425  sao_pars.offset[ ( sao_pars.typeAuxInfo + k ) % MAX_NUM_SAO_CLASSES ] = offset[k];
426  }
427  continue;
428  }
429 
430  // edge offset mode
431  sao_pars.typeAuxInfo = 0;
432  if( compID != COMP_Cr )
433  {
434  // sao_eo_class_luma / sao_eo_class_chroma
435  sao_pars.typeIdc += m_BinDecoder.decodeBinsEP( NUM_SAO_EO_TYPES_LOG2 );
436  }
437  else
438  {
439  sao_pars.typeIdc = sao_ctu_pars[ COMP_Cb ].typeIdc;
440  }
441  sao_pars.offset[ SAO_CLASS_EO_FULL_VALLEY ] = offset[0];
442  sao_pars.offset[ SAO_CLASS_EO_HALF_VALLEY ] = offset[1];
443  sao_pars.offset[ SAO_CLASS_EO_PLAIN ] = 0;
444  sao_pars.offset[ SAO_CLASS_EO_HALF_PEAK ] = -offset[2];
445  sao_pars.offset[ SAO_CLASS_EO_FULL_PEAK ] = -offset[3];
446  }
447 }

457 void CABACReader::coding_tree( CodingStructure& cs, Partitioner& partitioner, CUCtx& cuCtx, Partitioner* pPartitionerChroma, CUCtx* pCuCtxChroma)
458 {
459  const PPS &pps = *cs.pps;
460  const UnitArea& currArea = partitioner.currArea();
461 
462  // Reset delta QP coding flag and ChromaQPAdjustemt coding flag
463  //Note: do not reset qg at chroma CU
464  if( pps.useDQP && partitioner.currQgEnable() && !isChroma(partitioner.chType) )
465  {
466  cuCtx.qgStart = true;
467  cuCtx.isDQPCoded = false;
468  }
469  if( cs.slice->chromaQpAdjEnabled && partitioner.currQgChromaEnable() )
470  {
471  cuCtx.isChromaQpAdjCoded = false;
472  cs.chromaQpAdj = 0;
473  }
474 
475  // Reset delta QP coding flag and ChromaQPAdjustemt coding flag
476  if (CS::isDualITree(cs) && pPartitionerChroma != nullptr)
477  {
478  if (pps.useDQP && pPartitionerChroma->currQgEnable())
479  {
480  pCuCtxChroma->qgStart = true;
481  pCuCtxChroma->isDQPCoded = false;
482  }
483  if (cs.slice->chromaQpAdjEnabled && pPartitionerChroma->currQgChromaEnable())
484  {
485  pCuCtxChroma->isChromaQpAdjCoded = false;
486  cs.chromaQpAdj = 0;
487  }
488  }
489 
490  determineNeighborCus( cs, partitioner.currArea(), partitioner.chType, partitioner.treeType );
491 
492  const PartSplit splitMode = split_cu_mode( cs, partitioner );
493 
494  CHECK( !partitioner.canSplit( splitMode, cs ), "Got an invalid split!" );
495 
496  if( splitMode != CU_DONT_SPLIT )
497  {
498  if (CS::isDualITree(cs) && pPartitionerChroma != nullptr && (partitioner.currArea().lwidth() >= 64 || partitioner.currArea().lheight() >= 64))
499  {
500  partitioner.splitCurrArea(CU_QUAD_SPLIT, cs);
501  pPartitionerChroma->splitCurrArea(CU_QUAD_SPLIT, cs);
502  bool beContinue = true;
503  bool lumaContinue = true;
504  bool chromaContinue = true;
505 
506  while (beContinue)
507  {
508  if (partitioner.currArea().lwidth() > 64 || partitioner.currArea().lheight() > 64)
509  {
510  if (cs.area.blocks[partitioner.chType].contains(partitioner.currArea().blocks[partitioner.chType].pos()))
511  {
512  coding_tree(cs, partitioner, cuCtx, pPartitionerChroma, pCuCtxChroma);
513  }
514  lumaContinue = partitioner.nextPart(cs);
515  chromaContinue = pPartitionerChroma->nextPart(cs);
516  CHECK(lumaContinue != chromaContinue, "luma chroma partition should be matched");
517  beContinue = lumaContinue;
518  }
519  else
520  {
521  //dual tree coding under 64x64 block
522  if (cs.area.blocks[partitioner.chType].contains(partitioner.currArea().blocks[partitioner.chType].pos()))
523  {
524  coding_tree(cs, partitioner, cuCtx);
525  }
526  lumaContinue = partitioner.nextPart(cs);
527  if (cs.area.blocks[pPartitionerChroma->chType].contains(pPartitionerChroma->currArea().blocks[pPartitionerChroma->chType].pos()))
528  {
529  coding_tree(cs, *pPartitionerChroma, *pCuCtxChroma);
530  }
531  chromaContinue = pPartitionerChroma->nextPart(cs);
532  CHECK(lumaContinue != chromaContinue, "luma chroma partition should be matched");
533  beContinue = lumaContinue;
534  }
535  }
536  partitioner.exitCurrSplit();
537  pPartitionerChroma->exitCurrSplit();
538 
539  //cat the chroma CUs together
540  CodingUnit* currentCu = cs.getCU(partitioner.currArea().lumaPos(), CH_L, TREE_L);
541  CodingUnit* nextCu = nullptr;
542  CodingUnit* tempLastLumaCu = nullptr;
543  CodingUnit* tempLastChromaCu = nullptr;
544  ChannelType currentChType = currentCu->chType;
545  while (currentCu->next != nullptr)
546  {
547  nextCu = currentCu->next;
548  if (currentChType != nextCu->chType && currentChType == CH_L)
549  {
550  tempLastLumaCu = currentCu;
551  if (tempLastChromaCu != nullptr) //swap
552  {
553  tempLastChromaCu->next = nextCu;
554  }
555  }
556  else if (currentChType != nextCu->chType && currentChType == CH_C)
557  {
558  tempLastChromaCu = currentCu;
559  if (tempLastLumaCu != nullptr) //swap
560  {
561  tempLastLumaCu->next = nextCu;
562  }
563  }
564  currentCu = nextCu;
