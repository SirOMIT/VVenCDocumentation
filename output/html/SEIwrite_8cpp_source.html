60 void SEIWriter::xWriteSEIpayloadData(OutputBitstream &bs, const SEI& sei, HRD &hrd, const uint32_t temporalId)
61 {
62  const SEIBufferingPeriod *bp = NULL;
63  switch (sei.payloadType())
64  {
65  case SEI::USER_DATA_UNREGISTERED:
66  xWriteSEIuserDataUnregistered(*static_cast<const SEIuserDataUnregistered*>(&sei));
67  break;
68  case SEI::DECODING_UNIT_INFO:
69  bp = &hrd.bufferingPeriodSEI;
70  CHECK (bp == nullptr, "Buffering Period need to be initialized in HRD to allow writing of Decoding Unit Information SEI");
71  xWriteSEIDecodingUnitInfo(*static_cast<const SEIDecodingUnitInfo*>(& sei), *bp, temporalId);
72  break;
73  case SEI::SCALABLE_NESTING:
74  xWriteSEIScalableNesting(bs, *static_cast<const SEIScalableNesting*>(&sei));
75  break;
76  case SEI::DECODED_PICTURE_HASH:
77  xWriteSEIDecodedPictureHash(*static_cast<const SEIDecodedPictureHash*>(&sei));
78  break;
79  case SEI::BUFFERING_PERIOD:
80  xWriteSEIBufferingPeriod(*static_cast<const SEIBufferingPeriod*>(&sei));
81  hrd.bufferingPeriodSEI = *(static_cast<const SEIBufferingPeriod*>(&sei));
82  break;
83  case SEI::PICTURE_TIMING:
84  {
85  bp = &hrd.bufferingPeriodSEI;
86  CHECK (bp == nullptr, "Buffering Period need to be initialized in HRD to allow writing of Picture Timing SEI");
87  xWriteSEIPictureTiming(*static_cast<const SEIPictureTiming*>(&sei), *bp, temporalId);
88  }
89  break;
90  case SEI::FRAME_FIELD_INFO:
91  xWriteSEIFrameFieldInfo(*static_cast<const SEIFrameFieldInfo*>(&sei));
92  break;
93  case SEI::DEPENDENT_RAP_INDICATION:
94  xWriteSEIDependentRAPIndication(*static_cast<const SEIDependentRAPIndication*>(&sei));
95  break;
96  case SEI::FRAME_PACKING:
97  xWriteSEIFramePacking(*static_cast<const SEIFramePacking*>(&sei));
98  break;
99  case SEI::PARAMETER_SETS_INCLUSION_INDICATION:
100  xWriteSEIParameterSetsInclusionIndication(*static_cast<const SEIParameterSetsInclusionIndication*>(&sei));
101  break;
102  case SEI::MASTERING_DISPLAY_COLOUR_VOLUME:
103  xWriteSEIMasteringDisplayColourVolume(*static_cast<const SEIMasteringDisplayColourVolume*>(&sei));
104  break;
105  case SEI::ALTERNATIVE_TRANSFER_CHARACTERISTICS:
106  xWriteSEIAlternativeTransferCharacteristics(*static_cast<const SEIAlternativeTransferCharacteristics*>(&sei));
107  break;
108  case SEI::EQUIRECTANGULAR_PROJECTION:
109  xWriteSEIEquirectangularProjection(*static_cast<const SEIEquirectangularProjection*>(&sei));
110  break;
111  case SEI::SPHERE_ROTATION:
112  xWriteSEISphereRotation(*static_cast<const SEISphereRotation*>(&sei));
113  break;
114  case SEI::OMNI_VIEWPORT:
115  xWriteSEIOmniViewport(*static_cast<const SEIOmniViewport*>(&sei));
116  break;
117  case SEI::REGION_WISE_PACKING:
118  xWriteSEIRegionWisePacking(*static_cast<const SEIRegionWisePacking*>(&sei));
119  break;
120  case SEI::GENERALIZED_CUBEMAP_PROJECTION:
121  xWriteSEIGeneralizedCubemapProjection(*static_cast<const SEIGeneralizedCubemapProjection*>(&sei));
122  break;
123  case SEI::USER_DATA_REGISTERED_ITU_T_T35:
124  xWriteSEIUserDataRegistered(*static_cast<const SEIUserDataRegistered*>(&sei));
125  break;
126  case SEI::FILM_GRAIN_CHARACTERISTICS:
127  xWriteSEIFilmGrainCharacteristics(*static_cast<const SEIFilmGrainCharacteristics*>(&sei));
128  break;
129  case SEI::CONTENT_LIGHT_LEVEL_INFO:
130  xWriteSEIContentLightLevelInfo(*static_cast<const SEIContentLightLevelInfo*>(&sei));
131  break;
132  case SEI::AMBIENT_VIEWING_ENVIRONMENT:
133  xWriteSEIAmbientViewingEnvironment(*static_cast<const SEIAmbientViewingEnvironment*>(&sei));
134  break;
135  case SEI::CONTENT_COLOUR_VOLUME:
136  xWriteSEIContentColourVolume(*static_cast<const SEIContentColourVolume*>(&sei));
137  break;
138  case SEI::SUBPICTURE_LEVEL_INFO:
139  xWriteSEISubpictureLevelInfo(*static_cast<const SEISubpicureLevelInfo*>(&sei));
140  break;
141  case SEI::SAMPLE_ASPECT_RATIO_INFO:
142  xWriteSEISampleAspectRatioInfo(*static_cast<const SEISampleAspectRatioInfo*>(&sei));
143  break;
144  default:
145  THROW("Trying to write unhandled SEI message");
146  break;
147  }
148  xWriteByteAlign();
149 }

154 void SEIWriter::writeSEImessages(OutputBitstream& bs, const SEIMessages &seiList, HRD &hrd, bool isNested, const uint32_t temporalId)
155 {
156 #if ENABLE_TRACING
157  if (g_HLSTraceEnable)
158  DTRACE( g_trace_ctx, D_HEADER, "=========== SEI message ===========\n" );
159 #endif
160  OutputBitstream bs_count;
161 
162  for (SEIMessages::const_iterator sei=seiList.begin(); sei!=seiList.end(); sei++)
163  {
164  // calculate how large the payload data is
165  // TODO: this would be far nicer if it used vectored buffers
166  bs_count.clear();
167  setBitstream(&bs_count);
168 
169 #if ENABLE_TRACING
170  bool traceEnable = g_HLSTraceEnable;
171  g_HLSTraceEnable = false;
172 #endif
173  xWriteSEIpayloadData(bs_count, **sei, hrd, temporalId);
174 #if ENABLE_TRACING
175  g_HLSTraceEnable = traceEnable;
176 #endif
177  uint32_t payload_data_num_bits = bs_count.getNumberOfWrittenBits();
178  CHECK(0 != payload_data_num_bits % 8, "Invalid number of payload data bits");
179 
180  setBitstream(&bs);
181  uint32_t payloadType = (*sei)->payloadType();
182  for (; payloadType >= 0xff; payloadType -= 0xff)
183  {
184  WRITE_CODE(0xff, 8, "payload_type");
185  }
186  WRITE_CODE(payloadType, 8, "payload_type");
187 
188  uint32_t payloadSize = payload_data_num_bits/8;
189  for (; payloadSize >= 0xff; payloadSize -= 0xff)
190  {
191  WRITE_CODE(0xff, 8, "payload_size");
192  }
193  WRITE_CODE(payloadSize, 8, "payload_size");
194 
195  /* payloadData */
196 #if ENABLE_TRACING
197  if (g_HLSTraceEnable)
198  DTRACE( g_trace_ctx, D_HEADER, "=========== %s SEI message ===========\n", SEI::getSEIMessageString( (SEI::PayloadType)payloadType ) );
199 #endif
200 
201  xWriteSEIpayloadData(bs, **sei, hrd, temporalId);
202  }
203  if (!isNested)
204  {
205  xWriteRbspTrailingBits();
206  }

230 void SEIWriter::xWriteSEIDecodedPictureHash(const SEIDecodedPictureHash& sei)
231 {
232  const char *traceString="\0";
233  switch (sei.method)
234  {
235  case VVENC_HASHTYPE_MD5: traceString="picture_md5"; break;
236  case VVENC_HASHTYPE_CRC: traceString="picture_crc"; break;
237  case VVENC_HASHTYPE_CHECKSUM: traceString="picture_checksum"; break;
238  default: THROW("Unknown hash type"); break;
239  }
240 
241  if (traceString != 0) //use of this variable is needed to avoid a compiler error with G++ 4.6.1
242  {
243  WRITE_CODE(sei.method, 8, "dph_sei_hash_type");
244  WRITE_CODE(sei.singleCompFlag, 1, "dph_sei_single_component_flag");
245  WRITE_CODE(0, 7, "dph_sei_reserved_zero_7bits");
246  for(uint32_t i=0; i<uint32_t(sei.pictureHash.hash.size()); i++)
247  {
248  WRITE_CODE(sei.pictureHash.hash[i], 8, traceString);
249  }
250  }

279 void SEIWriter::xWriteSEIBufferingPeriod(const SEIBufferingPeriod& sei)
280 {
281  WRITE_FLAG( sei.bpNalCpbParamsPresent, "bp_nal_hrd_parameters_present_flag");
282  WRITE_FLAG( sei.bpVclCpbParamsPresent, "bp_vcl_hrd_parameters_present_flag");
283  CHECK(!sei.bpNalCpbParamsPresent && !sei.bpVclCpbParamsPresent, "bp_nal_hrd_parameters_present_flag and/or bp_vcl_hrd_parameters_present_flag must be true");
284  CHECK (sei.initialCpbRemovalDelayLength < 1, "sei.initialCpbRemovalDelayLength must be > 0");
285  WRITE_CODE( sei.initialCpbRemovalDelayLength - 1, 5, "initial_cpb_removal_delay_length_minus1" );
286  CHECK (sei.cpbRemovalDelayLength < 1, "sei.cpbRemovalDelayLength must be > 0");
287  WRITE_CODE( sei.cpbRemovalDelayLength - 1, 5, "cpb_removal_delay_length_minus1" );
288  CHECK (sei.dpbOutputDelayLength < 1, "sei.dpbOutputDelayLength must be > 0");
289  WRITE_CODE( sei.dpbOutputDelayLength - 1, 5, "dpb_output_delay_length_minus1" );
290  WRITE_FLAG( sei.bpDecodingUnitHrdParamsPresent, "bp_decoding_unit_hrd_params_present_flag" );
291  if( sei.bpDecodingUnitHrdParamsPresent )
292  {
293  CHECK (sei.duCpbRemovalDelayIncrementLength < 1, "sei.duCpbRemovalDelayIncrementLength must be > 0");
294  WRITE_CODE( sei.duCpbRemovalDelayIncrementLength - 1, 5, "du_cpb_removal_delay_increment_length_minus1" );
295  CHECK (sei.dpbOutputDelayDuLength < 1, "sei.dpbOutputDelayDuLength must be > 0");
296  WRITE_CODE( sei.dpbOutputDelayDuLength - 1, 5, "dpb_output_delay_du_length_minus1" );
297  WRITE_FLAG( sei.decodingUnitCpbParamsInPicTimingSeiFlag, "decoding_unit_cpb_params_in_pic_timing_sei_flag" );
298  WRITE_FLAG(sei.decodingUnitDpbDuParamsInPicTimingSeiFlag, "decoding_unit_dpb_du_params_in_pic_timing_sei_flag");
299  }
300 
301  WRITE_FLAG( sei.concatenationFlag, "concatenation_flag");
302  WRITE_FLAG( sei.additionalConcatenationInfoPresent, "additional_concatenation_info_present_flag");
303  if (sei.additionalConcatenationInfoPresent)
304  {
305  WRITE_CODE( sei.maxInitialRemovalDelayForConcatenation, sei.initialCpbRemovalDelayLength, "max_initial_removal_delay_for_concatenation" );
306  }
307 
308  CHECK (sei.auCpbRemovalDelayDelta < 1, "sei.auCpbRemovalDelayDelta must be > 0");
309  WRITE_CODE( sei.auCpbRemovalDelayDelta - 1, sei.cpbRemovalDelayLength, "au_cpb_removal_delay_delta_minus1" );
310 
311  CHECK(sei.bpMaxSubLayers < 1, "bp_max_sub_layers_minus1 must be > 0");
312  WRITE_CODE(sei.bpMaxSubLayers - 1, 3, "bp_max_sub_layers_minus1");
313  if (sei.bpMaxSubLayers - 1 > 0)
314  {
315  WRITE_FLAG(sei.cpbRemovalDelayDeltasPresent, "cpb_removal_delay_deltas_present_flag");
316  }
317 
318  if (sei.cpbRemovalDelayDeltasPresent)
319  {
320  CHECK (sei.numCpbRemovalDelayDeltas < 1, "m_numCpbRemovalDelayDeltas must be > 0");
321  WRITE_UVLC( sei.numCpbRemovalDelayDeltas - 1, "num_cpb_removal_delay_deltas_minus1" );
322  for( int i = 0; i < sei.numCpbRemovalDelayDeltas; i ++ )
323  {
324  WRITE_CODE( sei.cpbRemovalDelayDelta[i], sei.cpbRemovalDelayLength, "cpb_removal_delay_delta[i]" );
325  }
326  }
327  CHECK (sei.bpCpbCnt < 1, "sei.bpCpbCnt must be > 0");
328  WRITE_UVLC( sei.bpCpbCnt - 1, "bp_cpb_cnt_minus1");
329  if (sei.bpMaxSubLayers - 1 > 0)
330  {
331  WRITE_FLAG(sei.sublayerInitialCpbRemovalDelayPresent, "bp_sublayer_initial_cpb_removal_delay_present_flag");
332  }
333  for (int i = (sei.sublayerInitialCpbRemovalDelayPresent ? 0 : sei.bpMaxSubLayers - 1); i < sei.bpMaxSubLayers; i++)
334  {
335  for( int nalOrVcl = 0; nalOrVcl < 2; nalOrVcl ++ )
336  {
337  if( ( ( nalOrVcl == 0 ) && ( sei.bpNalCpbParamsPresent ) ) ||
338  ( ( nalOrVcl == 1 ) && ( sei.bpVclCpbParamsPresent ) ) )
339  {
340  for( int j = 0; j < sei.bpCpbCnt; j ++ )
341  {
342  WRITE_CODE( sei.initialCpbRemovalDelay[j][i][nalOrVcl], sei.initialCpbRemovalDelayLength, "initial_cpb_removal_delay[j][i][nalOrVcl]" );
343  WRITE_CODE( sei.initialCpbRemovalOffset[j][i][nalOrVcl], sei.initialCpbRemovalDelayLength, "initial_cpb_removal_delay_offset[j][i][nalOrVcl]" );
344  }
345  }
346  }
347  }
348  if (sei.bpMaxSubLayers-1 > 0)
349  {
350  WRITE_FLAG(sei.sublayerDpbOutputOffsetsPresent, "bp_sublayer_dpb_output_offsets_present_flag");
351  }
352 
353  if(sei.sublayerDpbOutputOffsetsPresent)
354  {
355  for(int i = 0; i < sei.bpMaxSubLayers - 1; i++)
356  {
357  WRITE_UVLC( sei.dpbOutputTidOffset[i], "dpb_output_tid_offset[i]" );

364  }

445  {
446  if( sei.ptSubLayerDelaysPresent[i] )
447  WRITE_CODE( sei.duCommonCpbRemovalDelayMinus1[i], bp.duCpbRemovalDelayIncrementLength, "du_common_cpb_removal_delay_increment_minus1[i]" );
448  }
449  }
450  for( int i = 0; i <= sei.numDecodingUnitsMinus1; i ++ )
451  {

454  {

470 }
