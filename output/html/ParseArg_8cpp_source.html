93  void Options::addOption(OptionBase *opt)
94  {
95  Names* names = new Names();
96  names->opt = opt;
97  std::string& opt_string = opt->opt_string;
98 
99  size_t opt_start = 0;
100  for (size_t opt_end = 0; opt_end != std::string::npos;)
101  {
102  opt_end = opt_string.find_first_of(',', opt_start);
103  bool force_short = 0;
104  if (opt_string[opt_start] == '-')
105  {
106  opt_start++;
107  force_short = 1;
108  }
109  std::string opt_name = opt_string.substr(opt_start, opt_end - opt_start);
110  if (force_short || opt_name.size() == 1)
111  {
112  names->opt_short.push_back(opt_name);
113  opt_short_map[opt_name].push_back(names);
114  }
115  else
116  {
117  names->opt_long.push_back(opt_name);
118 
119  std::string optLongLower = opt_name;
120  std::transform( optLongLower.begin(), optLongLower.end(), optLongLower.begin(), ::tolower );
121  opt_long_map[optLongLower].push_back(names);
122  }
123  opt_start += opt_end + 1;
124  }
125 
126  if( !subSections_list.empty() )
127  {
128  if( curSubSection.empty() ){ curSubSection = subSections_list.back(); }
129  sub_section_namelist_map[curSubSection].push_back(opt_string);
130  }
131 
132  opt_list.push_back(names);
133  }

169  /* format help text for a single option:
170  * using the formatting: "-x, --long",
171  * if a short/long option isn't specified, it is not printed
172  */
173  static void doHelpOpt(std::ostream& out, const Options::Names& entry, unsigned pad_short = 0)
174  {
175  pad_short = std::min(pad_short, 8u);
176 
177  if (!entry.opt_short.empty())
178  {
179  unsigned pad = std::max((int)pad_short - (int)entry.opt_short.front().size(), 0);
180  out << "-" << entry.opt_short.front();
181  if (!entry.opt_long.empty())
182  {
183  out << ", ";
184  }
185  out << &(spaces[40 - pad]);
186  }
187  else
188  {
189  out << " ";
190  out << &(spaces[40 - pad_short]);
191  }
192 
193  if (!entry.opt_long.empty())
194  {
195  out << "--" << entry.opt_long.front();
196  }
197  out << " [" << entry.opt->getDefault() << "] ";
198  }

200  static void doPrintHelpEntry( std::ostream& out, const Options::Names& entry, unsigned desc_width, unsigned opt_width, unsigned pad_short = 0 )
201  {
202  std::ostringstream line(std::ios_base::out);
203  line << " ";
204  doHelpOpt(line, entry, pad_short);
205 
206  const std::string& opt_desc = entry.opt->opt_desc;
207  if (opt_desc.empty())
208  {
209  /* no help text: output option, skip further processing */
210  out << line.str() << std::endl;
211  return;
212  }
213  size_t currlength = size_t(line.tellp());
214  if (currlength > opt_width)
215  {
216  /* if option text is too long (and would collide with the
217  * help text, split onto next line */
218  line << std::endl;
219  currlength = 0;
220  }
221  /* split up the help text, taking into account new lines,
222  * (add opt_width of padding to each new line) */
223  for (size_t newline_pos = 0, cur_pos = 0; cur_pos != std::string::npos; currlength = 0)
224  {
225  /* print any required padding space for vertical alignment */
226  line << &(spaces[40 - opt_width + currlength]);
227  newline_pos = opt_desc.find_first_of('\n', newline_pos);
228  if (newline_pos != std::string::npos)
229  {
230  /* newline found, print substring (newline needn't be stripped) */
231  newline_pos++;
232  line << opt_desc.substr(cur_pos, newline_pos - cur_pos);
233  cur_pos = newline_pos;
234  continue;
235  }
236  if (cur_pos + desc_width > opt_desc.size())
237  {
238  /* no need to wrap text, remainder is less than avaliable width */
239  line << opt_desc.substr(cur_pos);
240  break;
241  }
242  /* find a suitable point to split text (avoid spliting in middle of word) */
243  size_t split_pos = opt_desc.find_last_of(' ', cur_pos + desc_width);
244  if (split_pos != std::string::npos)
245  {
246  /* eat up multiple space characters */
247  split_pos = opt_desc.find_last_not_of(' ', split_pos) + 1;
248  }
249 
250  /* bad split if no suitable space to split at. fall back to width */
251  bool bad_split = split_pos == std::string::npos || split_pos <= cur_pos;
252  if (bad_split)
253  {
254  split_pos = cur_pos + desc_width;
255  }
256  line << opt_desc.substr(cur_pos, split_pos - cur_pos);
257 
258  /* eat up any space for the start of the next line */
259  if (!bad_split)
260  {
261  split_pos = opt_desc.find_first_not_of(' ', split_pos);
262  }
263  cur_pos = newline_pos = split_pos;
264 
265  if (cur_pos >= opt_desc.size())
266  {
267  break;
268  }
269  line << std::endl;
270  }
271 
272  out << line.str() << std::endl;
273  }

276  static void printFormattedConfigEntry( std::ostream& out, const Options::Names& entry, unsigned desc_width, unsigned opt_width, unsigned opt_value_width )
277  {
278  // prints a config entry. format:
279  // option : vaue # help [default]
280 
281  std::ostringstream line(std::ios_base::out);
282 
283  if (!entry.opt_long.empty())
284  {
285  line << entry.opt_long.front();
286 
287  for( size_t s = 0 ; s < (opt_width - entry.opt_long.front().length()) ; ++s )
288  {
289  line << ' ' ;
290  }
291 
292  line << ": " << entry.opt->getValue();
293  }
294  else
295  {
296  return;
297  }
298 
299  const std::string& opt_desc = entry.opt->opt_desc;
300  if (opt_desc.empty())
301  {
302  /* no help text: output option, skip further processing */
303  out << line.str() << std::endl;
304  return;
305  }
306 
307  size_t currlength = size_t(line.tellp());
308  if (currlength > opt_value_width)
309  {
310  /* if option text is too long (and would collide with the
311  * help text, split onto next line */
312  line << std::endl;
313  currlength = 0;
314  }
315  /* split up the help text, taking into account new lines,
316  * (add opt_value_width of padding to each new line) */
317  for (size_t newline_pos = 0, cur_pos = 0; cur_pos != std::string::npos; currlength = 0)
318  {
319  /* print any required padding space for vertical alignment */
320 
321  for( size_t num_sp = 0 ; num_sp < opt_value_width - currlength ; ++num_sp )
322  {
323  line << ' ' ;
324  }
325  newline_pos = opt_desc.find_first_of('\n', newline_pos);
326  if (newline_pos != std::string::npos)
327  {
328  /* newline found, print substring (newline needn't be stripped) */
329  newline_pos++;
330  line << " # ";
331  line << opt_desc.substr(cur_pos, newline_pos - cur_pos);
332  cur_pos = newline_pos;
333  continue;
334  }
335  if (cur_pos + desc_width > opt_desc.size())
336  {
337  /* no need to wrap text, remainder is less than avaliable width */
338  line << " # ";
339  line << opt_desc.substr(cur_pos);
340  line << " [" << entry.opt->getDefault( ) << "] ";
341  break;
342  }
343  /* find a suitable point to split text (avoid spliting in middle of word) */
344  size_t split_pos = opt_desc.find_last_of(' ', cur_pos + desc_width);
345  if (split_pos != std::string::npos)
346  {
347  /* eat up multiple space characters */
348  split_pos = opt_desc.find_last_not_of(' ', split_pos) + 1;
349  }
350 
351  /* bad split if no suitable space to split at. fall back to width */
352  bool bad_split = split_pos == std::string::npos || split_pos <= cur_pos;
353  if (bad_split)
354  {
355  split_pos = cur_pos + desc_width;
356  }
357  line << " # ";
358  line << opt_desc.substr(cur_pos, split_pos - cur_pos);
359

362  {
363  split_pos = opt_desc.find_first_not_of(' ', split_pos);
364  }

377  /* format the help text */
378  void doHelp(std::ostream& out, Options& opts, unsigned columns)
379  {
380  const unsigned pad_short = 3;
381  /* first pass: work out the longest option name */
382  unsigned max_width = 0;
383  for(Options::NamesPtrList::iterator it = opts.opt_list.begin(); it != opts.opt_list.end(); it++)
384  {
385  std::ostringstream line(std::ios_base::out);
386  doHelpOpt(line, **it, pad_short);
387  max_width = std::max(max_width, (unsigned) line.tellp());
388  }
389 
390  unsigned opt_width = std::min(max_width+2, 32u + pad_short) + 2;
391  unsigned desc_width = columns - opt_width;
392 
393  /* second pass: write out formatted option and help text.
394  * - align start of help text to start at opt_width
395  * - if the option text is longer than opt_width, place the help
396  * text at opt_width on the next line.
397  */
398  if( opts.subSections_list.empty())
399  {
400  for (const auto& opt: opts.opt_list)
401  {
402  doPrintHelpEntry( out, *opt, desc_width, opt_width, pad_short );
403  }
404  return;
405  }
406 
407  for (const auto& section: opts.subSections_list)
408  {
409  if( section != "__$PLACEHOLDER$__") // print sub section name (if not dummy section)
410  {
411  out << std::endl << "#======== " << section << " ================" << std::endl;
412  }
413 
414  Options::SubSectionNamesListMap::const_iterator itSectionMap = opts.sub_section_namelist_map.find(section); // get list of options in subsection
415  if (itSectionMap != opts.sub_section_namelist_map.end())
416  {
417  for( auto & s : itSectionMap->second ) // iterate over options in subsections and find/print entry in opts list
418  {
419  for(Options::NamesPtrList::const_iterator itopt = opts.opt_list.begin(); itopt != opts.opt_list.end(); itopt++)
420  {
421  if( (*itopt)->opt->opt_string == s ) // names are equal
422  {
423  doPrintHelpEntry( out, **itopt, desc_width, opt_width, pad_short );
424  break;
425  }
426  }
427  }
428  }
429  }

432  /* prints a formated configuration of Options into a ostream */
433  void saveConfig(std::ostream& out, Options& opts, std::list<std::string> ignoreParamLst, unsigned columns )
434  {
435  /* first pass: work out the longest option name */
436  unsigned max_width_optname = 0;
437  for(Options::NamesPtrList::iterator it = opts.opt_list.begin(); it != opts.opt_list.end(); it++)
438  {
439  if (!(**it).opt_long.empty())
440  {
441  unsigned w = (unsigned)(**it).opt_long.front().size();
442  max_width_optname = std::max(max_width_optname, w);
443  }
444  }
445 
446  /* second pass: work out the longest value*/
447  unsigned max_width_value = 0;
448  for(Options::NamesPtrList::iterator it = opts.opt_list.begin(); it != opts.opt_list.end(); it++)
449  {
450  std::string value = (**it).opt->getValue();
451  max_width_value = std::max(max_width_value, (unsigned) value.length() );
452  }
453 
454  unsigned max_width_opt_value = max_width_optname + max_width_value +3; // max size of option + " : " + value
455  unsigned desc_width = columns - max_width_opt_value; // max. size for description
456 
457  /* 3rd pass: write out formatted option + current value + help text + default value.
458  * format is:
459  * option : vaue # help [default]
460  * - if the option text is longer than opt_width, place the help
461  * text at opt_width on the next line.
462  */
463  if( opts.subSections_list.empty())
464  {
465  for (const auto& opt: opts.opt_list)
466  {
467  bool ignore = ignoreParamLst.end() != std::find(ignoreParamLst.begin(), ignoreParamLst.end(), opt->opt->opt_string);
468  if( !ignore)
469  {
470  printFormattedConfigEntry( out, *opt, desc_width, max_width_optname, max_width_opt_value );
471  }
472  }
473  return;
474  }
475 
476  // iterate over all subsections and print all entries for each subsection
477  for (const auto& section: opts.subSections_list)
478  {
479  if( section != "__$PLACEHOLDER$__") // print sub section name (if not dummy section)
480  {
481  out << std::endl << "#======== " << section << " ================" << std::endl;
482  }
483 
484  Options::SubSectionNamesListMap::const_iterator itSectionMap = opts.sub_section_namelist_map.find(section); // get list of options in subsection
485  if (itSectionMap != opts.sub_section_namelist_map.end())
486  {
487  for( auto & s : itSectionMap->second ) // iterate over options in subsections and find/print entry in opts list
488  {
489  for(Options::NamesPtrList::const_iterator itopt = opts.opt_list.begin(); itopt != opts.opt_list.end(); itopt++)
490  {
491  if( (*itopt)->opt->opt_string == s ) // names are equal
492  {
493  bool ignore = ignoreParamLst.end() != std::find(ignoreParamLst.begin(), ignoreParamLst.end(), (*itopt)->opt->opt_string);
494  if( !ignore)
495  {
496  printFormattedConfigEntry( out, **itopt, desc_width, max_width_optname, max_width_opt_value );
497  }
498 
499  break;
500  }
501  }
502  }
503  }
504  }

527  bool OptionWriter::storePair(bool allow_long, bool allow_short, const std::string& name, const std::string& value)
528  {
529  bool found = false;
530  Options::NamesMap::iterator opt_it;
531  if (allow_long)
532  {
533  std::string optLongLower = name;
534  std::transform( optLongLower.begin(), optLongLower.end(), optLongLower.begin(), ::tolower );
535 
536  opt_it = opts.opt_long_map.find(optLongLower);
537  if (opt_it != opts.opt_long_map.end())
538  {
539  found = true;
540  }
541  }
542 
543  /* check for the short list */
544  if (allow_short && !(found && allow_long))
545  {
546  opt_it = opts.opt_short_map.find(name);
547  if (opt_it != opts.opt_short_map.end())
548  {
549  found = true;
550  }
551  }
552 
553  if (!found)
554  {
555  error_reporter.error(where())
556  << "Unknown option `" << name << "' (value:`" << value << "')\n";
557  return false;
558  }
559 
560  setOptions((*opt_it).second, value, error_reporter);
561  return true;
562  }

579  unsigned ArgvParser::parseGNU(unsigned argc, const char* argv[])
580  {
581  /* gnu style long options can take the forms:
582  * --option=arg
583  * --option arg
584  */
585  std::string arg(argv[0]);
586  size_t arg_opt_start = arg.find_first_not_of('-');
587  size_t arg_opt_sep = arg.find_first_of('=');
588  std::string option = arg.substr(arg_opt_start, arg_opt_sep - arg_opt_start);
589 
590  std::transform( option.begin(), option.end(), option.begin(), ::tolower ); // compare option always in lower case
591 
592  unsigned extra_argc_consumed = 0;
593  if (arg_opt_sep == std::string::npos)
594  {
595  // check if we have an argument
596  if( argc > 1)
597  {
598  std::string val(argv[1]);
599  size_t val_sep = val.find_first_of('-');
600  //check if either have no - or the parameter is a number
601  if( 0 != val_sep || std::regex_match( val, std::regex( ( "((\\+|-)?[[:digit:]]+)(\\.(([[:digit:]]+)?))?" ) ) ) )
602  {
603  extra_argc_consumed++;
604  /* argument occurs after option_sep */
605  storePair(true, false, option, val);
606  return extra_argc_consumed;
607  }
608  else if( val_sep == 0 && val.size() == 1 )
609  {
610  extra_argc_consumed++;
611  /* argument occurs after option_sep */
612  storePair(true, false, option, val);
613  return extra_argc_consumed;
614  }
615  }
616 
617  /* no argument found => argument in argv[1] (maybe) */
618  /* xxx, need to handle case where option isn't required */
619  if(!storePair(true, false, option, ""))
620  {
621  return 0;
622  }
623  }
624  else
625  {
626  /* argument occurs after option_sep */
627  std::string val = arg.substr(arg_opt_sep + 1);
628  storePair(true, false, option, val);
629  }
630 
631  return extra_argc_consumed;
632  }

634  unsigned ArgvParser::parseSHORT(unsigned argc, const char* argv[])
635  {
636  /* short options can take the forms:
637  * --option arg
638  * -option arg
639  */
640  std::string arg(argv[0]);
641  size_t arg_opt_start = arg.find_first_not_of('-');
642  std::string option = arg.substr(arg_opt_start);
643  /* lookup option */
644 
645  /* argument in argv[1] */
646  /* xxx, need to handle case where option isn't required */
647  if (argc == 1)
648  {
649  storePair(false, true, option, std::string(""));
650  return 1;
651  }
652 
653  std::string argNext = argv[1];
654  if( !argNext.empty() && argNext[0] == '-' )
655  {
656  // check if bool switch and check if next param is not an option
657  if( argNext.size() > 1 )
658  {
659  if( argNext[1] == '-' ) // is long option --
660  {
661  storePair(false, true, option, std::string(""));
662  return 0;
663  }
664 
665  // check if argv is an digit number
666  if( !std::isdigit(argNext[1]) )
667  {
668  storePair(false, true, option, std::string(""));
669  return 0;
670  }
671  }
672  }
673 
674  storePair(false, true, option, std::string(argv[1]));
675 
676  return 1;
677  }

679  std::list<const char*>
680  scanArgv(Options& opts, unsigned argc, const char* argv[], ErrorReporter& error_reporter)
681  {
682  ArgvParser avp(opts, error_reporter);
683 
684  /* a list for anything that didn't get handled as an option */
685  std::list<const char*> non_option_arguments;

693  }

700  }

707  }
708

716  break;
717  }

720  i += avp.parseGNU(argc - i, &argv[i]);

747  void CfgStreamParser::scanLine(std::string& line)
748  {
749  /* strip any leading whitespace */
750  size_t start = line.find_first_not_of(" \t\n\r");
751  if (start == std::string::npos)
752  {
753  /* blank line */
754  return;
755  }
756  if (line[start] == '#')
757  {
758  /* comment line */
759  return;
760  }
761  /* look for first whitespace or ':' after the option end */
762  size_t option_end = line.find_first_of(": \t\n\r",start);
763  std::string option = line.substr(start, option_end - start);
764 
765  /* look for ':', eat up any whitespace first */
766  start = line.find_first_not_of(" \t\n\r", option_end);
767  if (start == std::string::npos)
768  {
769  /* error: badly formatted line */
770  error_reporter.warn(where()) << "line formatting error\n";
771  return;
772  }
773  if (line[start] != ':')
774  {
775  /* error: badly formatted line */
776  error_reporter.warn(where()) << "line formatting error\n";
777  return;
778  }
779 
780  /* look for start of value std::string -- eat up any leading whitespace */
781  start = line.find_first_not_of(" \t\n\r", ++start);
782  if (start == std::string::npos)
783  {
784  /* error: badly formatted line */
785  error_reporter.warn(where()) << "line formatting error\n";
786  return;
787  }
788 
789  /* extract the value part, which may contain embedded spaces
790  * by searching for a word at a time, until we hit a comment or end of line */
791  size_t value_end = start;
792  do
793  {
794  if (line[value_end] == '#')
795  {
796  /* rest of line is a comment */
797  value_end--;
798  break;
799  }
800  value_end = line.find_first_of(" \t\n\r", value_end);
801  /* consume any white space, incase there is another word.
802  * any trailing whitespace will be removed shortly */
803  value_end = line.find_first_not_of(" \t\n\r", value_end);
804  } while (value_end != std::string::npos);
805  /* strip any trailing space from value*/
806  value_end = line.find_last_not_of(" \t\n\r", value_end);
807 
808  std::string value;
809  if (value_end >= start)
810  {
811  value = line.substr(start, value_end +1 - start);
812  }
813  else
814  {
815  /* error: no value */
816  error_reporter.warn(where()) << "no value found for option " << option << "\n";
817  return;
818  }
819 
820  // reset empty strings
821  if( value == "\"\"" || value == "''")
822  {
823  value.clear();
824  }
825 
826  /* store the value in option */
827  storePair(true, false, option, value);
828  }
