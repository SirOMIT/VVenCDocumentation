67 void Channel::update( std::map< CType, int > state )
68 {
69 
70  for( std::list<Rule>::iterator rules_iter = rule_list.begin();
71  rules_iter != rule_list.end();
72  ++rules_iter ) {
73  /* iterate over conditions, get the state of the condition type
74  * and check if contion is met:
75  * if not -> go to next rule
76  * yes -> go to next condition
77  * if all conditions are met: set channel active and return */
78  bool probe = true;
79 
80  for( Rule::iterator cond_iter = rules_iter->begin();
81  cond_iter != rules_iter->end();
82  ++cond_iter ) {
83  int sVal = state[cond_iter->type];
84  if( !cond_iter->eval( cond_iter->rval, sVal ) ) {
85  probe = false;
86  break;
87  }
88  }
89  if( probe ) {
90  _active = true;
91  return;
92  }
93  }
94 
95  _active = false;
96 }

200 int CDTrace::addRule( std::string rulestring )
201 {
202  vstring chans_conds = split( rulestring, ':' );
203  vstring channels = split( chans_conds[0], ',' );
204  vstring conditions = chans_conds.size()>1 ? split( chans_conds[1], ',' ) : vstring();
205 
206  /* parse the rules first */
207  std::vector<Condition> rule;
208  for( vstring::iterator ci = conditions.begin(); ci != conditions.end(); ++ci ) {
209  /* find one of "==", "!=", "<=", ">=" */
210  const char *ops_[] = { "==", "!=", "<=", ">=" };
211  vstring operators( ops_,&ops_[sizeof( ops_ )/sizeof( ops_[0] )] );
212  vstring::iterator oi = operators.begin();
213  std::size_t pos = std::string::npos;
214  do {
215  if( ( pos = ci->find( *oi ) ) != std::string::npos ) break;
216  } while( ++oi != operators.end() );
217 
218  /* No operator found, malformed rules string -> abort */
219  if( pos == std::string::npos ) return -2;
220 
221  CType ctype( *ci,0,pos );
222  int value = std::atoi( ci->substr( pos+2, ci->length()-( pos+2 ) ).c_str() );
223  //if( condition_types.find( ctype ) == condition_types.end() ) return 0;
224 
225  /* partially apply the condition value to the associated
226  * condtion function and append it to the rule */
227  bool ( *cfunc )( int,int );
228  if( "==" == *oi ) cfunc = _cf_eq;
229  else if( "!=" == *oi ) cfunc = _cf_neq;
230  else if( "<=" == *oi ) cfunc = _cf_le;
231  else if( ">=" == *oi ) cfunc = _cf_ge;
232  else return 0; // this is already taken care of
233 
234  rule.push_back( Condition( ctype, cfunc, value ) );
235  }
236 
237  /* add the rule to each channel */
238  for( vstring::iterator chan_iter = channels.begin(); chan_iter != channels.end(); ++chan_iter ) {
239  std::map< Key, int>::iterator ichan = deserializationTable.find(*chan_iter);
240  if( ichan != deserializationTable.end() )
241  chanRules[ichan->second].add( rule );
242  else
243  return -3;
244  }
245 
246  //return (int)channels.size();
247  return 0;
248 }
