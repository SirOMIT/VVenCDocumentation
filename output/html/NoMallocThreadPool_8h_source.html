130 struct BlockingBarrier
131 {
132  void unlock()
133  {
134  std::unique_lock<std::mutex> l( m_lock );
135  m_intBarrier.unlock();
136  if( !m_intBarrier.isBlocked() )
137  {
138  m_cond.notify_all();
139  }
140  }
141 
142  void lock()
143  {
144  std::unique_lock<std::mutex> l( m_lock );
145  m_intBarrier.lock();
146  }
147 
148  bool isBlocked() const
149  {
150  return m_intBarrier.isBlocked();
151  }
152 
153  void wait() const
154  {
155  BlockingBarrier* nonconst = const_cast<BlockingBarrier*>(this);
156 
157  std::unique_lock<std::mutex> l( nonconst->m_lock );
158  nonconst->m_cond.wait( l, [=] { return !m_intBarrier.isBlocked(); } );
159  }
160 
161  BlockingBarrier() = default;
162  ~BlockingBarrier() { std::unique_lock<std::mutex> l( m_lock ); } // ensure all threads have unlocked the mutex, when we start destruction
163

179 struct WaitCounter
180 {
181  int operator++()
182  {
183  std::unique_lock<std::mutex> l( m_lock );
184  done.lock();
185  return ++m_count;
186  }
187 
188  int operator--()
189  {
190  std::unique_lock<std::mutex> l( m_lock );
191  const unsigned int new_count = --m_count;
192  if( new_count == 0 )
193  {
194  m_cond.notify_all();
195  done.unlock();
196  }
197  l.unlock(); // unlock mutex after done-barrier to prevent race between barrier and counter
198  return new_count;
199  }
200 
201  bool isBlocked() const
202  {
203  return 0 != m_count;
204  }
205 
206  void wait() const
207  {
208  WaitCounter* nonconst = const_cast<WaitCounter*>(this);
209 
210  std::unique_lock<std::mutex> l( nonconst->m_lock );
211  nonconst->m_cond.wait( l, [=] { return m_count == 0; } );
212  }
213 
214  WaitCounter() = default;
215  ~WaitCounter() { std::unique_lock<std::mutex> l( m_lock ); } // ensure all threads have unlocked the mutex, when we start destruction
216 
217  WaitCounter( const WaitCounter & ) = delete;
218  WaitCounter( WaitCounter && ) = delete;
219 
220  WaitCounter &operator=( const WaitCounter & ) = delete;
221  WaitCounter &operator=( WaitCounter && ) = delete;
222 
223  Barrier done{ false };
224 
225 private:
226  std::condition_variable m_cond;
227  std::mutex m_lock;
228  unsigned int m_count = 0;

263  class ChunkedTaskQueue
264  {
265  constexpr static int ChunkSize = 128;

267  class Chunk
268  {
269  std::array<Slot, ChunkSize> m_slots;
270  std::atomic<Chunk*> m_next{ nullptr };
271  Chunk& m_firstChunk;
272 
273  Chunk( Chunk* firstPtr ) : m_firstChunk{ *firstPtr } {}
274 
275  friend class ChunkedTaskQueue;
276  };
277 
278  public:
279  class Iterator : public std::iterator<std::forward_iterator_tag, Slot>
280  {
281  Slot* m_slot = nullptr;
282  Chunk* m_chunk = nullptr;
283 
284  public:
285  Iterator() = default;
286  Iterator( Slot* slot, Chunk* chunk ) : m_slot( slot ), m_chunk( chunk ) {}
287 
288  Iterator& operator++()
289  {
290  CHECKD( m_slot == nullptr, "incrementing invalid iterator" );
291  CHECKD( m_chunk == nullptr, "incrementing invalid iterator" );
292 
293  if( m_slot != &m_chunk->m_slots.back() )
294  {
295  ++m_slot;
296  }
297  else
298  {
299  m_chunk = m_chunk->m_next;
300  if( m_chunk )
301  {
302  m_slot = &m_chunk->m_slots.front();
303  }
304  else
305  {
306  m_slot = nullptr;
307  }
308  }
309  return *this;
310  }
311 
312  // increment iterator and wrap around, if end is reached
313  Iterator& incWrap()
314  {
315  CHECKD( m_slot == nullptr, "incrementing invalid iterator" );
316  CHECKD( m_chunk == nullptr, "incrementing invalid iterator" );
317 
318  if( m_slot != &m_chunk->m_slots.back() )
319  {
320  ++m_slot;
321  }
322  else
323  {
324  if( (Chunk*)m_chunk->m_next )
325  {
326  m_chunk = m_chunk->m_next;
327  }
328  else
329  {
330  m_chunk = &m_chunk->m_firstChunk;
331  }
332  m_slot = &m_chunk->m_slots.front();
333  }
334  return *this;
335  }
336 
337  bool operator==( const Iterator& rhs ) const { return m_slot == rhs.m_slot; } // don't need to compare m_chunk, because m_slot is a pointer
338  bool operator!=( const Iterator& rhs ) const { return m_slot != rhs.m_slot; } // don't need to compare m_chunk, because m_slot is a pointer
339 
340  Slot& operator*() { return *m_slot; }
341 
342  bool isValid() const { return m_slot != nullptr && m_chunk != nullptr; }
343  };
344 
345  ChunkedTaskQueue() = default;
346  ~ChunkedTaskQueue()
347  {
348  Chunk* next = m_firstChunk.m_next;
349  while( next )
350  {
351  Chunk* curr = next;
352  next = curr->m_next;
353  delete curr;
354  }
355  }

357  ChunkedTaskQueue( const ChunkedTaskQueue& ) = delete;
358  ChunkedTaskQueue( ChunkedTaskQueue&& ) = delete;
359 
360  // grow the queue by adding a chunk and return an iterator to the first new task-slot
361  Iterator grow()
362  {
363  std::unique_lock<std::mutex> l( m_resizeMutex ); // prevent concurrent growth of the queue. Read access while growing is no problem
364 // std::cerr << __PRETTY_FUNCTION__ << std::endl;
365 
366  m_lastChunk->m_next = new Chunk( &m_firstChunk );
367  m_lastChunk = m_lastChunk->m_next;
368 
369  return Iterator{ &m_lastChunk->m_slots.front(), m_lastChunk };
370  }
371 
372  Iterator begin() { return Iterator{ &m_firstChunk.m_slots.front(), &m_firstChunk }; }
373  Iterator end() { return Iterator{ nullptr, nullptr }; }
374 
375  private:

387  template<class TParam>
388  bool addBarrierTask( bool ( *func )( int, TParam* ),
389  TParam* param,
390  WaitCounter* counter = nullptr,
391  Barrier* done = nullptr,
392  const CBarrierVec&& barriers = {},
393  bool ( *readyCheck )( int, TParam* ) = nullptr )
394  {
395  if( m_threads.empty() )
396  {
397  // if singlethreaded, execute all pending tasks
398  if( m_nextFillSlot != m_tasks.begin() )
399  {
400  processTasksOnMainThread();
401  }
402 
403  // when no barriers block this task, execute it directly
404  if( std::none_of( barriers.begin(), barriers.end(), []( const Barrier* b ) { return b && b->isBlocked(); } )
405  && ( !readyCheck || readyCheck( 0, param ) ) )
406  {
407  if( func( 0, param ) )
408  {
409  if( done != nullptr )
410  {
411  done->unlock();
412  }
413  return true;
414  }
415  }
416  }
417 
418  while( true )
419  {
420 #if ADD_TASK_THREAD_SAFE
421  std::unique_lock<std::mutex> l(m_nextFillSlotMutex);
422 #endif
423  CHECKD( !m_nextFillSlot.isValid(), "Next fill slot iterator should always be valid" );
424  const auto startIt = m_nextFillSlot;
425 
426 #if ADD_TASK_THREAD_SAFE
427  l.unlock();
428 #endif
429 
430  bool first = true;
431  for( auto it = startIt; it != startIt || first; it.incWrap() )
432  {
433 #if ENABLE_VALGRIND_CODE
434  MutexLock lock( m_extraMutex );
435 #endif
436 
437  first = false;
438 
439  auto& t = *it;
440  auto expected = FREE;
441  if( t.state.load( std::memory_order_relaxed ) == FREE && t.state.compare_exchange_strong( expected, PREPARING ) )
442  {
443  if( counter )
444  {
445  counter->operator++();
446  }
447 
448  t.func = (TaskFunc)func;
449  t.readyCheck = (TaskFunc)readyCheck;
450  t.param = param;
451  t.done = done;
452  t.counter = counter;
453  t.barriers = std::move( barriers );
454  t.state = WAITING;
455 
456 #if ADD_TASK_THREAD_SAFE
457  l.lock();
458 #endif
459  m_nextFillSlot.incWrap();
460  return true;
461  }
462  }
463 
464 #if ADD_TASK_THREAD_SAFE
465  l.lock();
466 #endif
467  m_nextFillSlot = m_tasks.grow();
468  }
469  return false;
470  }
