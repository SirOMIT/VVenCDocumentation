381 bool VVEncAppCfg::parseCfg( int argc, char* argv[] )
382 {
383  bool do_help = false;
384  bool do_full_help = false;
385  int warnUnknowParameter = 0;
386 
387  //
388  // link custom formated configuration parameters with istream reader
389  //
390  IStreamToEnum<vvencMsgLevel> toMsgLevel ( &m_verbosity, &MsgLevelToEnumMap );
391  IStreamToFunc<vvencPresetMode> toPreset ( setPresets, this, &PresetToEnumMap,vvencPresetMode::VVENC_MEDIUM);
392  IStreamToRefVec<int> toSourceSize ( { &m_SourceWidth, &m_SourceHeight }, true, 'x' );
393 
394  IStreamToEnum<vvencProfile> toProfile ( &m_profile, &ProfileToEnumMap );
395  IStreamToEnum<vvencTier> toLevelTier ( &m_levelTier, &TierToEnumMap );
396  IStreamToEnum<vvencLevel> toLevel ( &m_level, &LevelToEnumMap );
397  IStreamToEnum<vvencSegmentMode> toSegment ( &m_SegmentMode, &SegmentToEnumMap );
398  IStreamToEnum<vvencHDRMode> toHDRMode ( &m_HdrMode, &HdrModeToIntMap );
399 
400  IStreamToFunc<BitDepthAndColorSpace> toInputFormatBitdepth( setInputBitDepthAndColorSpace, this, &BitColorSpaceToIntMap, YUV420_8);
401  IStreamToEnum<vvencDecodingRefreshType> toDecRefreshType ( &m_DecodingRefreshType, &DecodingRefreshTypeToEnumMap );
402 
403  IStreamToEnum<int> toAud ( &m_AccessUnitDelimiter, &FlagToIntMap );
404  IStreamToEnum<int> toHrd ( &m_hrdParametersPresent, &FlagToIntMap );
405  IStreamToEnum<int> toVui ( &m_vuiParametersPresent, &FlagToIntMap );
406  IStreamToEnum<bool> toQPA ( &m_usePerceptQPA, &QPAToIntMap );
407  IStreamToArr<char> toTraceRule ( &m_traceRule[0], VVENC_MAX_STRING_LEN );
408  IStreamToArr<char> toTraceFile ( &m_traceFile[0], VVENC_MAX_STRING_LEN );
409  IStreamToEnum<vvencHashType> toHashType ( &m_decodedPictureHashSEIType, &HashTypeToEnumMap );
410  //
411  // setup configuration parameters
412  //
413 
414  std::string ignoreParams;
415  po::Options opts;
416  opts.setSubSection("General Options");
417  opts.addOptions()
418  ("help", do_help, "this help text")
419  ("fullhelp", do_full_help, "show full text")
420  ("verbosity,v", toMsgLevel, "Specifies the level of the verboseness (0: silent, 1: error, 2: warning, 3: info, 4: notice, 5: verbose, 6: debug) ")
421  ("version", m_showVersion, "show version ")
422  ;
423

426  ("input,i", m_inputFileName, "original YUV input file name or '-' for reading from stdin")

428  ("format,c", toInputFormatBitdepth, "set input format (yuv420, yuv420_10)")
429 
430  ("framerate,r", m_FrameRate, "temporal rate (framerate) e.g. 25,29,30,50,59,60 ")
431  ("tickspersec", m_TicksPerSecond, "Ticks Per Second for dts generation, (1..27000000)")
432 
433  ("frames,f", m_framesToBeEncoded, "max. frames to encode [all]")
434  ("frameskip", m_FrameSkip, "Number of frames to skip at start of input YUV [off]")
435  ("segment", toSegment, "when encoding multiple separate segments, specify segment position to enable segment concatenation (first, mid, last) [off]\n"
436  "first: first segment \n"
437  "mid : all segments between first and last segment\n"
438  "last : last segment")
439  ;
440  opts.setSubSection("Output Options");
441  opts.addOptions()
442  ("output,o", m_bitstreamFileName, "Bitstream output file name")
443  ;
444  opts.setSubSection("Encoder Options");
445  opts.addOptions()
446  ("preset", toPreset, "select preset for specific encoding setting (faster, fast, medium, slow, slower)")
447 
448  ("bitrate,b", m_RCTargetBitrate, "bitrate for rate control (0: constant-QP encoding without rate control, otherwise bits/second)" )
449  ("passes,p", m_RCNumPasses, "number of rate control passes (1,2) " )

451  ("qpa", toQPA, "Enable perceptually motivated QP adaptation, XPSNR based (0:off, 1:on)", true)

457  ("refreshsec,-rs", m_IntraPeriodSec, "Intra period/refresh in seconds")
458  ("intraperiod,-ip", m_IntraPeriod, "Intra period in frames (0: use intra period in seconds (refreshsec), else: n*gopsize)")
459  ;
460 
461  opts.setSubSection("Profile, Level, Tier");
462  opts.addOptions()
463  ("profile", toProfile, "select profile (main10, main10_stillpic)")
464  ("level", toLevel, "Level limit (1.0, 2.0,2.1, 3.0,3.1, 4.0,4.1, 5.0,5.1,5.2, 6.0,6.1,6.2,6.3, 15.5)")
465  ("tier", toLevelTier, "Tier to use for interpretation of level (main or high)")

469  opts.addOptions()
470  ("hdr", toHDRMode, "set HDR mode (+SEI messages) + BT.709 or BT.2020 color space. "

475  std::ostringstream easyOpts;
476  po::doHelp( easyOpts, opts );
477 
478  opts.setSubSection("Encoder Options");

480  ("internal-bitdepth", m_internalBitDepth[0], "internal bitdepth (8,10)")
481  ("accessunitdelimiter,-aud", toAud, "Emit Access Unit Delimiter NALUs (auto(-1),off(0),on(1); default: auto - only if needed by dependent options)", true)

485  ;
486 
487  if ( vvenc_is_tracing_enabled() )
488  {
489  opts.setSubSection( "Tracing" );
490  opts.addOptions()
491  ("tracechannellist", m_listTracingChannels, "List all available tracing channels")
492  ("tracerule", toTraceRule, "Tracing rule (ex: \"D_CABAC:poc==8\" or \"D_REC_CB_LUMA:poc==8\")")
493  ("tracefile", toTraceFile, "Tracing file")
494  ;
495  }
496 
497  po::setDefaults( opts );
498  std::ostringstream fullOpts;
499  po::doHelp( fullOpts, opts );
500 
501  //
502  // parse command line parameters and read configuration files
503  //
504 
505  po::setDefaults( opts );
506  po::ErrorReporter err;
507  const list<const char*>& argv_unhandled = po::scanArgv( opts, argc, (const char**) argv, err );
508  for( auto& a : argv_unhandled )
509  {
510  cout << "Unhandled argument ignored: `" << a << "'\n";
511  }
512 
513  if ( argc == 1 || do_help )
514  {
515  cout << easyOpts.str();
516  return false;
517  }
518  else if ( do_full_help )
519  {
520  cout << fullOpts.str();
521  return false;
522  }
523 
524  if ( err.is_errored && ! warnUnknowParameter )
525  {
526  /* error report has already been printed on stderr */
527  return false;
528  }
529 
530  //
531  // check own parameters
532  //
533 
534  if( m_showVersion )
535  {
536  return true;
537  }
538 
539  if( m_bitstreamFileName.empty() )
540  {
541  cout << "error: A bitstream file name must be specified (--output=bit.266)" << std::endl;
542  return false;
543  }
544 
545  // this has to be set outside
546  if ( m_internChromaFormat < 0 || m_internChromaFormat >= VVENC_NUM_CHROMA_FORMAT )
547  {
548  m_internChromaFormat = m_inputFileChromaFormat;
549  }
550
