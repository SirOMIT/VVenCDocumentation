102 static inline int64_t xgetbv (int ctr) {
103 #if (defined (_MSC_FULL_VER) && _MSC_FULL_VER >= 160040000) || (defined (__INTEL_COMPILER) && __INTEL_COMPILER >= 1200) // Microsoft or Intel compiler supporting _xgetbv intrinsic
104 
105  return _xgetbv(ctr); // intrinsic function for XGETBV
106 
107 #elif defined(__GNUC__) // use inline assembly, Gnu/AT&T syntax
108 
109  uint32_t a, d;
110 #if GCC_VERSION_AT_LEAST(4,4) || CLANG_VERSION_AT_LEAST(3,3)
111  __asm("xgetbv" : "=a"(a),"=d"(d) : "c"(ctr) : );
112 #else
113  __asm(".byte 0x0f, 0x01, 0xd0" : "=a"(a),"=d"(d) : "c"(ctr) : );
114 #endif
115  return a | (uint64_t(d) << 32);
116 
117 #else // #elif defined (_MSC_FULL_VER) || (defined (__INTEL_COMPILER)...) // other compiler. try inline assembly with masm/intel/MS syntax
118 
119  uint32_t a, d;
120  __asm {
121  mov ecx, ctr
122  _emit 0x0f
123  _emit 0x01
124  _emit 0xd0 ; // xgetbv
125  mov a, eax
126  mov d, edx
127  }
128  return a | (uint64_t(d) << 32);
129 
130 #endif
131 }

157 NO_USE_SIMD
158 X86_VEXT _get_x86_extensions()
159 {
160  int regs[4] = {0, 0, 0, 0};
161  X86_VEXT ext;
162  ext = SCALAR;
163 
164  do_cpuid( regs, 0 );
165  if( regs[0] == 0 ) return ext;
166  do_cpuid( regs, 1 );
167  if (!(regs[2] & BIT_HAS_SSE41)) return ext;
168  ext = SSE41;
169  if (!(regs[2] & BIT_HAS_SSE42)) return ext;
170  ext = SSE42;
171 #if !SIMD_UP_TO_SSE42
172  do_cpuidex( regs, 1, 1 );
173  if (!((regs[2] & BIT_HAS_AVX) == BIT_HAS_AVX )) return ext; // first check if the cpu supports avx
174  if ((xgetbv(0) & 6) != 6) return ext; // then see if the os uses YMM state management via XSAVE etc...
175  ext = AVX;
176 // #ifdef USE_AVX2
177  do_cpuidex( regs, 7, 0 );
178  if (!(regs[1] & BIT_HAS_AVX2)) return ext;
179  ext = AVX2;
180 // #endif
181 #ifdef USE_AVX512
182  if ((xgetbv(0) & 0xE0) != 0xE0) return ext; // see if OPMASK state and ZMM are availabe and enabled
183  do_cpuidex( regs, 7, 0 );
184  if (!(regs[1] & BIT_HAS_AVX512F )) return ext;
185  if (!(regs[1] & BIT_HAS_AVX512DQ)) return ext;
186  if (!(regs[1] & BIT_HAS_AVX512BW)) return ext;
187  ext = AVX512;
188 #endif
189 #endif
190 
191  return ext;
192 }

201 {
202  //static std::atomic<bool> b_detection_finished(false);

206  {

213  {
