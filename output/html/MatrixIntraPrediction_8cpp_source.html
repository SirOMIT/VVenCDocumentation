88 void MatrixIntraPrediction::prepareInputForPred(const CPelBuf &pSrc, const Area& block, const int bitDepth)
89 {
90  // Step 1: Save block size and calculate dependent values
91  initPredBlockParams(block);
92 
93  m_refSamplesTop = pSrc.bufAt(1, 0);
94  m_refSamplesLeft = pSrc.bufAt(1, 1);
95 
96  // Step 3: Compute the reduced boundary via Haar-downsampling (input for the prediction)
97  const int inputSize = 2 * m_reducedBdrySize;
98 
99  Pel* const topReduced = m_reducedBoundary;
100  boundaryDownsampling1D( topReduced, m_refSamplesTop, block.width, m_reducedBdrySize );
101 
102  Pel* const leftReduced = m_reducedBoundary + m_reducedBdrySize;
103  boundaryDownsampling1D( leftReduced, m_refSamplesLeft, block.height, m_reducedBdrySize );
104 
105  Pel* const leftReducedTransposed = m_reducedBoundaryTransp;
106  Pel* const topReducedTransposed = m_reducedBoundaryTransp + m_reducedBdrySize;
107  for( int x = 0; x < m_reducedBdrySize; x++ )
108  {
109  topReducedTransposed[x] = topReduced[x];
110  }
111  for( int y = 0; y < m_reducedBdrySize; y++ )
112  {
113  leftReducedTransposed[y] = leftReduced[y];
114  }
115 
116  // Step 4: Rebase the reduced boundary
117  m_inputOffset = m_reducedBoundary[0];
118  m_inputOffsetTransp = m_reducedBoundaryTransp[0];
119 
120  const bool hasFirstCol = (m_sizeId < 2);
121  m_reducedBoundary [0] = hasFirstCol ? ((1 << (bitDepth - 1)) - m_inputOffset ) : 0; // first column of matrix not needed for large blocks
122  m_reducedBoundaryTransp[0] = hasFirstCol ? ((1 << (bitDepth - 1)) - m_inputOffsetTransp) : 0;
123  for (int i = 1; i < inputSize; i++)
124  {
125  m_reducedBoundary [i] -= m_inputOffset;
126  m_reducedBoundaryTransp[i] -= m_inputOffsetTransp;
127  }
128 }

130 void MatrixIntraPrediction::predBlock(Pel* const result, const int modeIdx, const bool transpose, const int bitDepth)
131 {
132  ALIGN_DATA( MEMORY_ALIGN_DEF_SIZE, Pel bufReducedPred[MIP_MAX_REDUCED_OUTPUT_SAMPLES] );
133 
134  const bool needUpsampling = ( m_upsmpFactorHor > 1 ) || ( m_upsmpFactorVer > 1 );
135  Pel* const reducedPred = needUpsampling ? bufReducedPred : result;
136  const Pel* const reducedBoundary = transpose ? m_reducedBoundaryTransp : m_reducedBoundary;
137 
138  {
139  const int outputSize = m_reducedPredSize;
140  const int inputSize = 2 * m_reducedBdrySize;
141  const int offset = transpose ? m_inputOffsetTransp : m_inputOffset;
142  const int maxVal = ( 1 << bitDepth ) - 1;
143 
144  if( outputSize == 8)
145  {
146  g_pelBufOP.mipMatrixMul_8_8( reducedPred, reducedBoundary, &mipMatrix16x16[modeIdx][0][0], maxVal, offset, transpose );
147  }
148  else
149  {
150  if( inputSize == 4)
151  {
152  g_pelBufOP.mipMatrixMul_4_4( reducedPred, reducedBoundary, &mipMatrix4x4[modeIdx][0][0], maxVal, offset, transpose );
153  }
154  else
155  {
156  g_pelBufOP.mipMatrixMul_8_4( reducedPred, reducedBoundary, &mipMatrix8x8[modeIdx][0][0], maxVal, offset, transpose );
157  }
158  }
159  }
160 
161  // Reduced prediction is transposed if ( transpose && needUpsampling ).
162  if( needUpsampling )
163  {
164  const Pel* verSrc = reducedPred;
165  SizeType verSrcStep = m_blockSize.width;
166 
167  if( m_upsmpFactorHor > 1 )
168  {
169  Pel* const horDst = result + (m_upsmpFactorVer - 1) * m_blockSize.width;
170  verSrc = horDst;
171  verSrcStep *= m_upsmpFactorVer;
172 
173  if( m_reducedPredSize == 4)
174  {
175  if( m_upsmpFactorHor == 2 )
176  predictionUpsampling1DHor<4,1>( horDst, reducedPred, &m_refSamplesLeft[0], verSrcStep, m_upsmpFactorVer );
177  else if( m_upsmpFactorHor == 4 )
178  predictionUpsampling1DHor<4,2>( horDst, reducedPred, &m_refSamplesLeft[0], verSrcStep, m_upsmpFactorVer );
179  else
180  predictionUpsampling1DHor<4,3>( horDst, reducedPred, &m_refSamplesLeft[0], verSrcStep, m_upsmpFactorVer );
181  }
182  else
183  {
184  if( m_upsmpFactorHor == 2 )
185  predictionUpsampling1DHor<8,1>( horDst, reducedPred, &m_refSamplesLeft[0], verSrcStep, m_upsmpFactorVer );
186  else if( m_upsmpFactorHor == 4 )
187  predictionUpsampling1DHor<8,2>( horDst, reducedPred, &m_refSamplesLeft[0], verSrcStep, m_upsmpFactorVer );
188  else
189  predictionUpsampling1DHor<8,3>( horDst, reducedPred, &m_refSamplesLeft[0], verSrcStep, m_upsmpFactorVer );
190  }
191  }
192 
193  if( m_upsmpFactorVer > 1 )
194  {
195  if( m_reducedPredSize == 4)
196  {
197  if( m_upsmpFactorVer == 2 )
198  predictionUpsampling1DVer<4,1>( result, verSrc, &m_refSamplesTop[0], m_blockSize.width, verSrcStep );
199  else if( m_upsmpFactorVer == 4 )
200  predictionUpsampling1DVer<4,2>( result, verSrc, &m_refSamplesTop[0], m_blockSize.width, verSrcStep );
201  else
202  predictionUpsampling1DVer<4,3>( result, verSrc, &m_refSamplesTop[0], m_blockSize.width, verSrcStep );
203  }
204  else
205  {
206  if( m_upsmpFactorVer == 2 )
207  predictionUpsampling1DVer<8,1>( result, verSrc, &m_refSamplesTop[0], m_blockSize.width, verSrcStep );
208  else if( m_upsmpFactorVer == 4 )
209  predictionUpsampling1DVer<8,2>( result, verSrc, &m_refSamplesTop[0], m_blockSize.width, verSrcStep );
210  else
211  predictionUpsampling1DVer<8,3>( result, verSrc, &m_refSamplesTop[0], m_blockSize.width, verSrcStep );
212  }
213  }
214  }

217 void MatrixIntraPrediction::initPredBlockParams(const Size& block)
218 {
219  m_blockSize = block;
220  // init size index
221  m_sizeId = getMipSizeId( m_blockSize );
222 
223  // init reduced boundary size
224  m_reducedBdrySize = (m_sizeId == 0) ? 2 : 4;
225 
226  // init reduced prediction size
227  m_reducedPredSize = ( m_sizeId < 2 ) ? 4 : 8;
228 
229  // init upsampling factors
230  m_upsmpFactorHor = m_blockSize.width / m_reducedPredSize;
231  m_upsmpFactorVer = m_blockSize.height / m_reducedPredSize;
232 
233  CHECKD( (m_upsmpFactorHor < 1) || ((m_upsmpFactorHor & (m_upsmpFactorHor - 1)) != 0), "Need power of two horizontal upsampling factor." );
234  CHECKD( (m_upsmpFactorVer < 1) || ((m_upsmpFactorVer & (m_upsmpFactorVer - 1)) != 0), "Need power of two vertical upsampling factor." );
235 }

237 void MatrixIntraPrediction::boundaryDownsampling1D(Pel* reducedDst, const Pel* const fullSrc, const SizeType srcLen, const SizeType dstLen)
238 {
239  if (dstLen < srcLen)
240  {
241  // Create reduced boundary by downsampling
242  const SizeType downsmpFactor = srcLen / dstLen;
243  const int log2DownsmpFactor = floorLog2(downsmpFactor);
244  const int roundingOffset = (1 << (log2DownsmpFactor - 1));
245 
246  SizeType srcIdx = 0;
247  for( SizeType dstIdx = 0; dstIdx < dstLen; dstIdx++ )
248  {
249  int sum = 0;
250  for( int k = 0; k < downsmpFactor; k++ )
251  {
252  sum += fullSrc[srcIdx++];
253  }
254  reducedDst[dstIdx] = (sum + roundingOffset) >> log2DownsmpFactor;
255  }
256  }
257  else
258  {
259  // Copy boundary if no downsampling is needed
260  for (SizeType i = 0; i < dstLen; ++i)
261  {
262  reducedDst[i] = fullSrc[i];
263  }
264  }
265 }

267 template< SizeType predPredSize, unsigned log2UpsmpFactor>
268 void MatrixIntraPrediction::predictionUpsampling1DHor(Pel* const dst, const Pel* const src, const Pel* const bndry, const SizeType dstStride, const SizeType bndryStep )
269 {
270  const int roundingOffset = 1 << (log2UpsmpFactor - 1);
271  const SizeType upsmpFactor = 1 << log2UpsmpFactor;
272 
273  Pel* dstLine = dst;
274  const Pel* srcLine = src;
275  const Pel* bndryLine = bndry + bndryStep - 1;
276 
277  for( SizeType idxOrthDim = 0; idxOrthDim < predPredSize; idxOrthDim++ )
278  {
279  const Pel* before = bndryLine;
280  const Pel* behind = srcLine;
281  Pel* currDst = dstLine;
282  for( SizeType idxUpsmpDim = 0; idxUpsmpDim < predPredSize; idxUpsmpDim++ )
283  {
284  int scaledVal = ( *before ) << log2UpsmpFactor;
285  for( SizeType pos = 0; pos < upsmpFactor; pos++)
286  {
287  scaledVal -= *before;
288  scaledVal += *behind;
289  *currDst = (scaledVal + roundingOffset) >> log2UpsmpFactor;
290  currDst ++;
291  }
292  before = behind;
293  behind ++;
294  }
295 
296  srcLine += predPredSize;
297  dstLine += dstStride;
298  bndryLine += bndryStep;
299  }
300 }

302 template< SizeType inHeight, unsigned log2UpsmpFactor>
303 void MatrixIntraPrediction::predictionUpsampling1DVer(Pel* const dst, const Pel* const src, const Pel* const bndry, const SizeType outWidth, const SizeType srcStep )
304 {
305  const int roundingOffset = 1 << (log2UpsmpFactor - 1);
306  const SizeType upsmpFactor = 1 << log2UpsmpFactor;
307 
308  Pel* dstLine = dst;
309  const Pel* srcLine = src;
310  const Pel* bndryLine = bndry;
311 
312  for( SizeType idxOrthDim = 0; idxOrthDim < outWidth; idxOrthDim++ )
313  {
314  const Pel* before = bndryLine;
315  const Pel* behind = srcLine;
316  Pel* currDst = dstLine;
317  for( SizeType idxUpsmpDim = 0; idxUpsmpDim < inHeight; idxUpsmpDim++ )
318  {
319  int scaledVal = ( *before ) << log2UpsmpFactor;
320  for( SizeType pos = 0; pos < upsmpFactor; pos++)
321  {
322  scaledVal -= *before;
323  scaledVal += *behind;
324  *currDst = (scaledVal + roundingOffset) >> log2UpsmpFactor;
325  currDst += outWidth;
326  }
327  before = behind;
328  behind += srcStep;
329  }
330 
331  srcLine ++;
332  dstLine ++;
333  bndryLine ++;
334  }
