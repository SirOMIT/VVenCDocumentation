96 void DecCu::decompressCtu( CodingStructure& cs, const UnitArea& ctuArea )
97 {
98  const int maxNumChannelType = cs.pcv->chrFormat != CHROMA_400 && CS::isDualITree( cs ) ? 2 : 1;
99  if (cs.resetIBCBuffer)
100  {
101  m_pcInterPred->resetIBCBuffer(cs.pcv->chrFormat, cs.slice->sps->CTUSize);
102  cs.resetIBCBuffer = false;
103  }
104  m_pcIntraPred->reset();
105 
106  for( int ch = 0; ch < maxNumChannelType; ch++ )
107  {
108  const ChannelType chType = ChannelType( ch );
109 
110  for( auto &currCU : cs.traverseCUs( CS::getArea( cs, ctuArea, chType, TREE_D ), chType ) )
111  {
112  if (currCU.Y().valid())
113  {
114  const int vSize = cs.slice->sps->CTUSize > 64 ? 64 : cs.slice->sps->CTUSize;
115  if ((currCU.Y().x % vSize) == 0 && (currCU.Y().y % vSize) == 0)
116  {
117  for (int x = currCU.Y().x; x < currCU.Y().x + currCU.Y().width; x += vSize)
118  {
119  for (int y = currCU.Y().y; y < currCU.Y().y + currCU.Y().height; y += vSize)
120  {
121  m_pcInterPred->resetVPDUforIBC(cs.pcv->chrFormat, cs.slice->sps->CTUSize, vSize, x + g_IBCBufferSize / cs.slice->sps->CTUSize / 2, y);
122  }
123  }
124  }
125  }
126  if (currCU.predMode != MODE_INTRA && currCU.predMode != MODE_PLT && currCU.Y().valid())
127  {
128  xDeriveCUMV(currCU);
129  }
130 
131  switch( currCU.predMode )
132  {
133  case MODE_INTER:
134  case MODE_IBC:
135  xReconInter( currCU );
136  break;
137  case MODE_PLT:
138  case MODE_INTRA:
139  xReconIntraQT( currCU );
140  break;
141  default:
142  THROW( "Invalid prediction mode" );
143  break;
144  }
145  m_pcInterPred->xFillIBCBuffer(currCU);
146  LoopFilter::calcFilterStrengths(currCU);
147 
148  DTRACE_BLOCK_REC( cs.picture->getRecoBuf( currCU ), currCU, currCU.predMode );
149  }
150  }
151 }

157 void DecCu::xIntraRecBlk( TransformUnit& tu, const ComponentID compID )
158 {
159  if( !tu.blocks[ compID ].valid() )
160  {
161  return;
162  }
163 
164  CodingStructure &cs = *tu.cs;
165  const CompArea& area = tu.blocks[compID];
166  const ChannelType chType = toChannelType( compID );
167  PelBuf piPred = tu.cu->ispMode && isLuma( compID ) ? cs.getPredBuf( area ) : m_PredBuffer.getCompactBuf( area );
168  const CodingUnit& cu = *tu.cu;
169  const uint32_t uiChFinalMode = CU::getFinalIntraMode( cu, chType );
170  PelBuf pReco = cs.getRecoBuf( area );
171 
172  //===== init availability pattern =====
173  CompArea areaPredReg(COMP_Y, tu.chromaFormat, area);
174  bool predRegDiffFromTB = isLuma( compID ) && CU::isPredRegDiffFromTB( *tu.cu );
175  bool firstTBInPredReg = isLuma( compID ) && CU::isFirstTBInPredReg ( *tu.cu, area );
176 
177  if( tu.cu->ispMode && isLuma( compID ) )
178  {
179  if( predRegDiffFromTB )
180  {
181  if( firstTBInPredReg )
182  {
183  CU::adjustPredArea( areaPredReg );
184  m_pcIntraPred->initIntraPatternChTypeISP( *tu.cu, areaPredReg, pReco );
185  }
186  }
187  else
188  {
189  m_pcIntraPred->initIntraPatternChTypeISP( *tu.cu, area, pReco );
190  }
191  }
192  else
193  {
194  m_pcIntraPred->initIntraPatternChType( *tu.cu, area );
195  }
196 
197  //===== get prediction signal =====
198  if( compID != COMP_Y && CU::isLMCMode( uiChFinalMode ) )
199  {
200  const CodingUnit& cu = *tu.cu;
201  m_pcIntraPred->loadLMLumaRecPels( cu, area );
202  m_pcIntraPred->predIntraChromaLM( compID, piPred, cu, area, uiChFinalMode );
203  }
204  else if( CU::isMIP( cu, chType ) )
205  {
206  m_pcIntraPred->initIntraMip( cu );
207  m_pcIntraPred->predIntraMip( piPred, cu );
208  }
209  else
210  {
211  if (predRegDiffFromTB)
212  {
213  if (firstTBInPredReg)
214  {
215  PelBuf piPredReg = cs.getPredBuf(areaPredReg);
216  m_pcIntraPred->predIntraAng(compID, piPredReg, cu);
217  }
218  }
219  else
220  {
221  m_pcIntraPred->predIntraAng(compID, piPred, cu);
222  }
223  }
224  //===== inverse transform =====
225  const Slice& slice = *cs.slice;
226  ReshapeData& reshapeData = cs.picture->reshapeData;
227  bool lmcsflag = slice.lmcsEnabled && (slice.isIntra() || (!slice.isIntra() && reshapeData.getCTUFlag()));
228  if (lmcsflag && slice.picHeader->lmcsChromaResidualScale && (compID != COMP_Y) && (tu.cbf[COMP_Cb] || tu.cbf[COMP_Cr]))
229  {
230  const Area area = tu.Y().valid() ? tu.Y() : Area(recalcPosition(tu.chromaFormat, tu.chType, CH_L, tu.blocks[tu.chType].pos()), recalcSize(tu.chromaFormat, tu.chType, CH_L, tu.blocks[tu.chType].size()));
231  const CompArea& areaY = CompArea(COMP_Y, tu.chromaFormat, area);
232  tu.chromaAdj = reshapeData.calculateChromaAdjVpduNei(tu, areaY, TREE_D);
233  }
234  //===== inverse transform =====
235  PelBuf piResi = cs.getResiBuf( area );
236 
237  const QpParam cQP( tu, compID );
238 
239  if( tu.jointCbCr && isChroma(compID) )
240  {
241  if( compID == COMP_Cb )
242  {
243  PelBuf resiCr = cs.getResiBuf( tu.blocks[ COMP_Cr ] );
244  if( tu.jointCbCr >> 1 )
245  {
246  m_pcTrQuant->invTransformNxN( tu, COMP_Cb, piResi, cQP );
247  }
248  else
249  {
250  const QpParam qpCr( tu, COMP_Cr );
251  m_pcTrQuant->invTransformNxN( tu, COMP_Cr, resiCr, qpCr );
252  }
253  m_pcTrQuant->invTransformICT( tu, piResi, resiCr );
254  }
255  }
256  else
257  if( TU::getCbf( tu, compID ) )
258  {
259  m_pcTrQuant->invTransformNxN( tu, compID, piResi, cQP );
260  }
261  else
262  {
263  piResi.fill( 0 );
264  }
265 
266  //===== reconstruction =====
267  lmcsflag &= (tu.blocks[compID].width*tu.blocks[compID].height > 4) && slice.picHeader->lmcsChromaResidualScale;
268  if (lmcsflag && (TU::getCbf(tu, compID) || tu.jointCbCr) && isChroma(compID) )
269  {
270  piResi.scaleSignal(tu.chromaAdj, 0, tu.cu->cs->slice->clpRngs[compID]);
271  }
272 
273 
274  piPred.reconstruct( piPred, piResi, tu.cu->cs->slice->clpRngs[ compID ] );
275  pReco.copyFrom( piPred );
276 
277  if( cs.pcv->isEncoder )
278  {
279  cs.picture->getRecoBuf( area ).copyFrom( pReco );
280  }
281 }

327 void DecCu::xReconInter(CodingUnit &cu)
328 {
329  CodingStructure &cs = *cu.cs;
330  // inter prediction
331 
332  PelUnitBuf predBuf = m_PredBuffer.getCompactBuf( cu );
333  const ReshapeData& reshapeData = cs.picture->reshapeData;
334  if (cu.geo)
335  {
336  m_pcInterPred->motionCompensationGeo(cu, predBuf, m_geoMrgCtx);
337  CU::spanGeoMotionInfo(cu, m_geoMrgCtx, cu.geoSplitDir, cu.geoMergeIdx0, cu.geoMergeIdx1);
338  }
339  else
340  {
341  CHECK(CU::isIBC(cu) && cu.ciip, "IBC and Ciip cannot be used together");
342  CHECK(CU::isIBC(cu) && cu.affine, "IBC and Affine cannot be used together");
343  CHECK(CU::isIBC(cu) && cu.geo, "IBC and geo cannot be used together");
344  CHECK(CU::isIBC(cu) && cu.mmvdMergeFlag, "IBC and MMVD cannot be used together");
345  const bool luma = cu.Y().valid();
346  const bool chroma = isChromaEnabled(cu.chromaFormat) && cu.Cb().valid();
347  if (luma && (chroma || !isChromaEnabled(cu.chromaFormat)))
348  {
349  cu.mvRefine = true;
350  m_pcInterPred->motionCompensation(cu, predBuf);
351  cu.mvRefine = false;
352  }
353  else
354  {
355  cu.mcControl = luma ? 0 : 4;
356  cu.mcControl |= chroma ? 0 : 2;
357  m_pcInterPred->motionCompensationIBC(cu, predBuf);
358  cu.mcControl = 0;
359  }
360  if (cu.Y().valid())
361  {
362  bool isIbcSmallBlk = CU::isIBC(cu) && (cu.lwidth() * cu.lheight() <= 16);
363  //CU::saveMotionInHMVP(cu, isIbcSmallBlk);
364  if (!cu.affine && !cu.geo && !isIbcSmallBlk)
365  {
366  const MotionInfo& mi = cu.getMotionInfo();
367  HPMVInfo hMi(mi, (mi.interDir == 3) ? cu.BcwIdx : BCW_DEFAULT, cu.imv == IMV_HPEL);
368  cs.addMiToLut(CU::isIBC(cu) ? cu.cs->motionLut.lutIbc : cu.cs->motionLut.lut, hMi);
369  }
370  }
371 
372  if (cu.ciip)
373  {
374  PelBuf ciipBuf = m_TmpBuffer.getCompactBuf( cu.Y() );
375 
376  m_pcIntraPred->initIntraPatternChType(cu, cu.Y());
377  m_pcIntraPred->predIntraAng(COMP_Y, ciipBuf, cu);
378 
379  if( cs.picHeader->lmcsEnabled && reshapeData.getCTUFlag() )
380  {
381  predBuf.Y().rspSignal( reshapeData.getFwdLUT());
382  }
383  const int numCiipIntra = m_pcIntraPred->getNumIntraCiip( cu );
384  predBuf.Y().weightCiip( ciipBuf, numCiipIntra);
385 
386  if (isChromaEnabled(cu.chromaFormat) && cu.chromaSize().width > 2)
387  {
388  PelBuf ciipBufC = m_TmpBuffer.getCompactBuf( cu.Cb() );
389  m_pcIntraPred->initIntraPatternChType(cu, cu.Cb());
390  m_pcIntraPred->predIntraAng(COMP_Cb, ciipBufC, cu);
391  predBuf.Cb().weightCiip( ciipBufC, numCiipIntra);
392 
393  m_pcIntraPred->initIntraPatternChType(cu, cu.Cr());
394  m_pcIntraPred->predIntraAng(COMP_Cr, ciipBufC, cu);
395  predBuf.Cr().weightCiip( ciipBufC, numCiipIntra);
396  }
397  }
398  }
399 
400  if (cs.slice->lmcsEnabled && reshapeData.getCTUFlag() && !cu.ciip && !CU::isIBC(cu) )
401  {
402  predBuf.Y().rspSignal(reshapeData.getFwdLUT());
403  }
404 
405  // inter recon
406  xDecodeInterTexture(cu);
407 
408  bool LumaOnly = ((predBuf.bufs.size() > 1) && (predBuf.bufs[1].stride != 0 && predBuf.bufs[2].stride != 0) ) ? false : true;
409  if (cu.rootCbf)
410  {
411  if (LumaOnly)
412  {
413  cs.getResiBuf(cu.Y()).reconstruct(predBuf.Y(), cs.getResiBuf(cu.Y()), cs.slice->clpRngs.comp[COMP_Y]);
414  cs.getRecoBuf(cu.Y()).copyFrom(cs.getResiBuf(cu.Y()));
415  }
416  else
417  {
418  cs.getResiBuf(cu).reconstruct(predBuf, cs.getResiBuf(cu), cs.slice->clpRngs);
419  cs.getRecoBuf(cu).copyFrom(cs.getResiBuf(cu));
420  }
421  }
422  else
423  {
424  if (LumaOnly)
425  {
426  cs.getRecoBuf(cu.Y()).copyClip(predBuf.Y(), cs.slice->clpRngs.comp[COMP_Y]);
427  }
428  else
429  {
430  cs.getRecoBuf(cu).copyClip(predBuf, cs.slice->clpRngs);
431  }
432  }
433 }

435 void DecCu::xDecodeInterTU( TransformUnit& currTU, const ComponentID compID )
436 {
437  if( !currTU.blocks[compID].valid() ) return;
438 
439  const CompArea& area = currTU.blocks[compID];
440 
441  CodingStructure& cs = *currTU.cs;
442 
443  //===== inverse transform =====
444  PelBuf resiBuf = cs.getResiBuf(area);
445 
446  const QpParam cQP(currTU, compID);
447 
448  if( currTU.jointCbCr && isChroma(compID) )
449  {
450  if( compID == COMP_Cb )
451  {
452  PelBuf resiCr = cs.getResiBuf( currTU.blocks[ COMP_Cr ] );
453  if( currTU.jointCbCr >> 1 )
454  {
455  m_pcTrQuant->invTransformNxN( currTU, COMP_Cb, resiBuf, cQP );
456  }
457  else
458  {
459  const QpParam qpCr( currTU, COMP_Cr );
460  m_pcTrQuant->invTransformNxN( currTU, COMP_Cr, resiCr, qpCr );
461  }
462  m_pcTrQuant->invTransformICT( currTU, resiBuf, resiCr );
463  }
464  }
465  else
466  if( TU::getCbf( currTU, compID ) )
467  {
468  m_pcTrQuant->invTransformNxN( currTU, compID, resiBuf, cQP );
469  }
470  else
471  {
472  resiBuf.fill( 0 );
473  }
474 
475  const Slice& slice = *currTU.cu->slice;
476  const ReshapeData& reshapeData = cs.picture->reshapeData;
477  if( slice.lmcsEnabled && reshapeData.getCTUFlag() && isChroma(compID) && (TU::getCbf(currTU, compID) || currTU.jointCbCr)
478  && slice.picHeader->lmcsChromaResidualScale && currTU.blocks[compID].width * currTU.blocks[compID].height > 4)
479  {
480  resiBuf.scaleSignal(currTU.chromaAdj, 0, slice.clpRngs[compID]);
481  }

510 void DecCu::xDeriveCUMV( CodingUnit &cu )
511 {
512  MergeCtx mrgCtx;
513 
514  if( cu.mergeFlag )
515  {
516  if (cu.mmvdMergeFlag || cu.mmvdSkip)
517  {
518  CHECK(cu.ciip == true, "invalid MHIntra");
519  if (cu.cs->sps->SbtMvp)
520  {
521  Size bufSize = g_miScaling.scale(cu.lumaSize());
522  mrgCtx.subPuMvpMiBuf = MotionBuf(m_subPuMiBuf, bufSize);
523  }
524  int fPosBaseIdx = cu.mmvdMergeIdx / MMVD_MAX_REFINE_NUM;
525  CU::getInterMergeCandidates(cu, mrgCtx, 1, fPosBaseIdx + 1);
526  CU::getInterMMVDMergeCandidates(cu, mrgCtx, cu.mmvdMergeIdx);
527  mrgCtx.setMmvdMergeCandiInfo(cu, cu.mmvdMergeIdx);
528  CU::spanMotionInfo(cu, mrgCtx);
529  }
530  else
531  {
532  if (cu.geo)
533  {
534  CU::getGeoMergeCandidates(cu, m_geoMrgCtx);
535  }
536  else
537  {
538  if (cu.affine)
539  {
540  AffineMergeCtx affineMergeCtx;
541  if (cu.cs->sps->SbtMvp)
542  {
543  Size bufSize = g_miScaling.scale(cu.lumaSize());
544  mrgCtx.subPuMvpMiBuf = MotionBuf(m_subPuMiBuf, bufSize);
545  affineMergeCtx.mrgCtx = &mrgCtx;
546  }
547  CU::getAffineMergeCand(cu, affineMergeCtx, cu.mergeIdx);
548  cu.interDir = affineMergeCtx.interDirNeighbours[cu.mergeIdx];
549  cu.affineType = affineMergeCtx.affineType[cu.mergeIdx];
550  cu.BcwIdx = affineMergeCtx.BcwIdx[cu.mergeIdx];
551  cu.mergeType = affineMergeCtx.mergeType[cu.mergeIdx];
552 
553  if (cu.mergeType == MRG_TYPE_SUBPU_ATMVP)
554  {
555  cu.refIdx[0] = affineMergeCtx.mvFieldNeighbours[(cu.mergeIdx << 1) + 0][0].refIdx;
556  cu.refIdx[1] = affineMergeCtx.mvFieldNeighbours[(cu.mergeIdx << 1) + 1][0].refIdx;
557  }
558  else
559  {
560  for (int i = 0; i < 2; ++i)
561  {
562  if (cu.cs->slice->numRefIdx[RefPicList(i)] > 0)
563  {
564  MvField *mvField = affineMergeCtx.mvFieldNeighbours[(cu.mergeIdx << 1) + i];
565  cu.mvpIdx[i] = 0;
566  cu.mvpNum[i] = 0;
567  cu.mvd[i][0] = Mv();
568  CU::setAllAffineMvField(cu, mvField, RefPicList(i));
569  }
570  }
571  }
572  }
573  else
574  {
575  if (CU::isIBC(cu))
576  {
577  CU::getIBCMergeCandidates(cu, mrgCtx, cu.mergeIdx);
578  }
579  else
580  {
581  CU::getInterMergeCandidates(cu, mrgCtx, 0, cu.mergeIdx);
582  }
583  mrgCtx.setMergeInfo(cu, cu.mergeIdx);
584  }
585 
586  CU::spanMotionInfo(cu, mrgCtx);
587  }
588  }
589  }
590  else
591  {
592  if (cu.affine)
593  {
594  for (uint32_t uiRefListIdx = 0; uiRefListIdx < 2; uiRefListIdx++)
595  {
596  RefPicList eRefList = RefPicList(uiRefListIdx);
597  if (cu.cs->slice->numRefIdx[eRefList] > 0 && (cu.interDir & (1 << uiRefListIdx)))
598  {
599  AffineAMVPInfo affineAMVPInfo;
600  CU::fillAffineMvpCand(cu, eRefList, cu.refIdx[eRefList], affineAMVPInfo);
601 
602  const unsigned mvp_idx = cu.mvpIdx[eRefList];
603 
604  cu.mvpNum[eRefList] = affineAMVPInfo.numCand;
605 
606  // Mv mv[3];
607  CHECK(cu.refIdx[eRefList] < 0, "Unexpected negative refIdx.");
608  if (!cu.cs->pcv->isEncoder)
609  {
610  cu.mvd[eRefList][0].changeAffinePrecAmvr2Internal(cu.imv);
