112 int EncApp::encode()
113 {
114  vvenc_config& vvencCfg = m_cEncAppCfg;
115  if( m_cEncAppCfg.m_decode )
116  {
117  return vvenc_decode_bitstream( m_cEncAppCfg.m_bitstreamFileName.c_str(), m_cEncAppCfg.m_traceFile, m_cEncAppCfg.m_traceRule );
118  }
119 
120  // initialize encoder lib
121  m_encCtx = vvenc_encoder_create();
122  if( nullptr == m_encCtx )
123  {
124  return -1;
125  }
126 
127  int iRet = vvenc_encoder_open( m_encCtx, &vvencCfg );
128  if( 0 != iRet )
129  {
130  msgApp( VVENC_ERROR, vvenc_get_last_error( m_encCtx ) );
131  vvenc_encoder_close( m_encCtx );
132  return iRet;
133  }
134 
135  if ( ! m_cEncAppCfg.m_reconFileName.empty() )
136  {
137  vvenc_encoder_set_RecYUVBufferCallback( m_encCtx, &this->m_yuvReconFile, outputYuv );
138  }
139 
140  vvenc_get_config( m_encCtx, &vvencCfg ); // get the adapted config, because changes are needed for the yuv reader (m_MSBExtendedBitDepth)
141 
142  msgApp( VVENC_INFO, "%s", m_cEncAppCfg.getConfigAsString( vvencCfg.m_verbosity ).c_str());
143 
144  if( !strcmp( m_cEncAppCfg.m_inputFileName.c_str(), "-" ) )
145  {
146  if( m_cEncAppCfg.m_RCNumPasses > 1 )
147  {
148  msgApp( VVENC_ERROR, " 2 pass rate control and reading from stdin is not supported yet\n" );
149  vvenc_encoder_close( m_encCtx );
150  return -1;
151  }
152  else
153  {
154  msgApp( VVENC_INFO, " trying to read from stdin\n" );
155  }
156  }
157 
158  if( ! openFileIO() )
159  {
160  vvenc_encoder_close( m_encCtx );
161  return -1;
162  }
163 
164  printChromaFormat();
165 
166  // create buffer for input YUV pic
167  vvencYUVBuffer yuvInBuf;
168  vvenc_YUVBuffer_default( &yuvInBuf );
169  vvenc_YUVBuffer_alloc_buffer( &yuvInBuf, vvencCfg.m_internChromaFormat, vvencCfg.m_SourceWidth, vvencCfg.m_SourceHeight );
170 
171  // create sufficient memory for output data
172  vvencAccessUnit au;
173  vvenc_accessUnit_default( &au );
174  vvenc_accessUnit_alloc_payload( &au, vvencCfg.m_SourceWidth * vvencCfg.m_SourceHeight );
175 
176  // main loop
177  int tempRate = vvencCfg.m_FrameRate;
178  int tempScale = 1;
179  switch( vvencCfg.m_FrameRate )
180  {
181  case 23: tempRate = 24000; tempScale = 1001; break;
182  case 29: tempRate = 30000; tempScale = 1001; break;
183  case 59: tempRate = 60000; tempScale = 1001; break;
184  default: break;
185  }
186 
187  int framesRcvd = 0;
188  for( int pass = 0; pass < vvencCfg.m_RCNumPasses; pass++ )
189  {
190  // open input YUV
191  if( m_yuvInputFile.open( m_cEncAppCfg.m_inputFileName, false, vvencCfg.m_inputBitDepth[0], vvencCfg.m_MSBExtendedBitDepth[0], vvencCfg.m_internalBitDepth[0],
192  m_cEncAppCfg.m_inputFileChromaFormat, vvencCfg.m_internChromaFormat, m_cEncAppCfg.m_bClipInputVideoToRec709Range, false ))
193  {
194  msgApp( VVENC_ERROR, "%s", m_yuvInputFile.getLastError().c_str() );
195  vvenc_encoder_close( m_encCtx );
196  vvenc_YUVBuffer_free_buffer( &yuvInBuf );
197  vvenc_accessUnit_free_payload( &au );
198  closeFileIO();
199  return -1;
200  }
201 
202  const int skipFrames = vvencCfg.m_FrameSkip - vvencCfg.m_vvencMCTF.MCTFNumLeadFrames;
203  if( skipFrames > 0 )
204  {
205  m_yuvInputFile.skipYuvFrames( skipFrames, vvencCfg.m_SourceWidth, vvencCfg.m_SourceHeight );
206  }
207 
208  // initialize encoder pass
209  vvenc_init_pass( m_encCtx, pass );
210 
211  // loop over input YUV data
212  bool inputDone = false;
213  bool encDone = false;
214  framesRcvd = 0;
215  while( ! inputDone || ! encDone )
216  {
217  // check for more input pictures
218  inputDone = ( vvencCfg.m_framesToBeEncoded > 0
219  && framesRcvd >= ( vvencCfg.m_framesToBeEncoded + vvencCfg.m_vvencMCTF.MCTFNumLeadFrames + vvencCfg.m_vvencMCTF.MCTFNumTrailFrames ) )
220  || m_yuvInputFile.isEof();
221 
222  // read input YUV
223  if( ! inputDone )
224  {
225  if( 0 != m_yuvInputFile.readYuvBuf( yuvInBuf, inputDone ) )
226  {
227  msgApp( VVENC_ERROR, "read file failed: %s\n", m_yuvInputFile.getLastError().c_str() );
228  vvenc_encoder_close( m_encCtx );
229  vvenc_YUVBuffer_free_buffer( &yuvInBuf );
230  vvenc_accessUnit_free_payload( &au );
231  closeFileIO();
232  return -1;
233  }
234  if( !inputDone )
235  {
236  if( vvencCfg.m_FrameRate > 0 )
237  {
238  yuvInBuf.cts = framesRcvd * vvencCfg.m_TicksPerSecond * tempScale / tempRate;
239  yuvInBuf.ctsValid = true;
240  }
241  framesRcvd += 1;
242  }
243  }
244 
245  // encode picture
246  vvencYUVBuffer* inputPacket = nullptr;
247  if( !inputDone )
248  {
249  inputPacket = &yuvInBuf;
250  }
251 
252  iRet = vvenc_encode( m_encCtx, inputPacket, &au, &encDone );
253  if( 0 != iRet )
254  {
255  msgApp( VVENC_ERROR, "encoding failed: err code %d - %s\n", iRet, vvenc_get_last_error(m_encCtx) );
256  encDone = true;
257  inputDone = true;
258  }
259 
260  // write out encoded access units
261  if( au.payloadUsedSize )
262  {
263  outputAU( au );
264  }
265 
266  // temporally skip frames
267  if( ! inputDone && vvencCfg.m_temporalSubsampleRatio > 1 )
268  {
269  m_yuvInputFile.skipYuvFrames( vvencCfg.m_temporalSubsampleRatio - 1, vvencCfg.m_SourceWidth, vvencCfg.m_SourceHeight );
270  }
271  }
272 
273  // close input YUV
274  m_yuvInputFile.close();
275  }
276 
277  printRateSummary( framesRcvd - ( vvencCfg.m_vvencMCTF.MCTFNumLeadFrames + vvencCfg.m_vvencMCTF.MCTFNumTrailFrames ) );
278 
279  // cleanup encoder lib
280  vvenc_encoder_close( m_encCtx );
281 
282  vvenc_YUVBuffer_free_buffer( &yuvInBuf );
283  vvenc_accessUnit_free_payload( &au );
284 
285  closeFileIO();
286  return iRet;
287 }

315 bool EncApp::openFileIO()
316 {
317  // output YUV
318  if( ! m_cEncAppCfg.m_reconFileName.empty() )
319  {
320  if( m_yuvReconFile.open( m_cEncAppCfg.m_reconFileName, true, m_cEncAppCfg.m_outputBitDepth[0], m_cEncAppCfg.m_outputBitDepth[0], m_cEncAppCfg.m_internalBitDepth[0],
321  m_cEncAppCfg.m_internChromaFormat, m_cEncAppCfg.m_internChromaFormat, m_cEncAppCfg.m_bClipOutputVideoToRec709Range, m_cEncAppCfg.m_packedYUVMode ))
322  {
323  msgApp( VVENC_ERROR, "%s", m_yuvReconFile.getLastError().c_str() );
324  return false;
325  }
326  }
327 
328  // output bitstream
329  m_bitstream.open( m_cEncAppCfg.m_bitstreamFileName.c_str(), fstream::binary | fstream::out );
330  if( ! m_bitstream )
331  {
332  msgApp( VVENC_ERROR, "Failed to open bitstream file %s for writing\n", m_cEncAppCfg.m_bitstreamFileName.c_str() );
333  return false;
334  }
335 
336  return true;
337 }

347 void EncApp::printRateSummary( int framesRcvd )
348 {
349  vvenc_print_summary( m_encCtx );
350 
351  double time = (double) framesRcvd / m_cEncAppCfg.m_FrameRate * m_cEncAppCfg.m_temporalSubsampleRatio;
352  msgApp( VVENC_DETAILS,"Bytes written to file: %u (%.3f kbps)\n", m_totalBytes, 0.008 * m_totalBytes / time );
353  if( m_cEncAppCfg.m_summaryVerboseness > 0 )
354  {
355  msgApp( VVENC_DETAILS, "Bytes for SPS/PPS/APS/Slice (Incl. Annex B): %u (%.3f kbps)\n", m_essentialBytes, 0.008 * m_essentialBytes / time );
356  }

359 void EncApp::printChromaFormat()
360 {
361  if( m_cEncAppCfg.m_verbosity >= VVENC_DETAILS )
362  {
363  std::stringstream ssOut;
364  ssOut << std::setw(43) << "Input ChromaFormat = ";
365  switch( m_cEncAppCfg.m_inputFileChromaFormat )

368  case VVENC_CHROMA_420: ssOut << " 420"; break;
369  case VVENC_CHROMA_422: ssOut << " 422"; break;
370  case VVENC_CHROMA_444: ssOut << " 444"; break;

373  }
374  ssOut << std::endl;
375 
376  ssOut << std::setw(43) << "Output (intern) ChromaFormat = ";
377  switch( m_cEncAppCfg.m_internChromaFormat )
378  {
379  case VVENC_CHROMA_400: ssOut << " 400"; break;
380  case VVENC_CHROMA_420: ssOut << " 420"; break;

382  case VVENC_CHROMA_444: ssOut << " 444"; break;
383  default: msgApp( VVENC_ERROR, "invalid chroma format" );
384  return;
