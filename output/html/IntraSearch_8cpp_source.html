169 void IntraSearch::xEstimateLumaRdModeList(int& numModesForFullRD,
170  static_vector<ModeInfo, FAST_UDI_MAX_RDMODE_NUM>& RdModeList,
171  static_vector<ModeInfo, FAST_UDI_MAX_RDMODE_NUM>& HadModeList,
172  static_vector<double, FAST_UDI_MAX_RDMODE_NUM>& CandCostList,
173  static_vector<double, FAST_UDI_MAX_RDMODE_NUM>& CandHadList, CodingUnit& cu, bool testMip )
174 {
175  PROFILER_SCOPE_AND_STAGE_EXT( 0, g_timeProfiler, P_INTRA_EST_RD_CAND_LUMA, cu.cs, CH_L );
176  const uint16_t intra_ctx_size = Ctx::IntraLumaMpmFlag.size() + Ctx::IntraLumaPlanarFlag.size() + Ctx::MultiRefLineIdx.size() + Ctx::ISPMode.size() + Ctx::MipFlag.size();
177  const TempCtx ctxStartIntraCtx(m_CtxCache, SubCtx(CtxSet(Ctx::IntraLumaMpmFlag(), intra_ctx_size), m_CABACEstimator->getCtx()));
178  const double sqrtLambdaForFirstPass = m_pcRdCost->getMotionLambda() * FRAC_BITS_SCALE;
179  const int numModesAvailable = NUM_LUMA_MODE; // total number of Intra modes
180 
181  CHECK(numModesForFullRD >= numModesAvailable, "Too many modes for full RD search");
182 
183  const SPS& sps = *cu.cs->sps;
184  const bool fastMip = sps.MIP && m_pcEncCfg->m_useFastMIP;
185 
186  // this should always be true
187  CHECK( !cu.Y().valid(), "CU is not valid" );
188 
189  const CompArea& area = cu.Y();
190 
191  const UnitArea localUnitArea(area.chromaFormat, Area(0, 0, area.width, area.height));
192  if( testMip)
193  {
194  numModesForFullRD += fastMip? std::max(numModesForFullRD, floorLog2(std::min(cu.lwidth(), cu.lheight())) - m_pcEncCfg->m_useFastMIP) : numModesForFullRD;
195  m_SortedPelUnitBufs->prepare( localUnitArea, numModesForFullRD + 1 );
196  }
197  else
198  {
199  m_SortedPelUnitBufs->prepare( localUnitArea, numModesForFullRD );
200  }
201 
202  CPelBuf piOrg = cu.cs->getOrgBuf(COMP_Y);
203  PelBuf piPred = m_SortedPelUnitBufs->getTestBuf(COMP_Y);
204 
205  const ReshapeData& reshapeData = cu.cs->picture->reshapeData;
206  if (cu.cs->picHeader->lmcsEnabled && reshapeData.getCTUFlag())
207  {
208  piOrg = cu.cs->getRspOrgBuf();
209  }
210  DistParam distParam = m_pcRdCost->setDistParam( piOrg, piPred, sps.bitDepths[ CH_L ], DF_HAD_2SAD); // Use HAD (SATD) cost
211 
212  const int numHadCand = (testMip ? 2 : 1) * 3;
213 
214  //*** Derive (regular) candidates using Hadamard
215  cu.mipFlag = false;
216  cu.multiRefIdx = 0;
217 
218  //===== init pattern for luma prediction =====
219  initIntraPatternChType(cu, cu.Y(), true);
220 
221  bool satdChecked[NUM_INTRA_MODE] = { false };
222 
223  unsigned mpmLst[NUM_MOST_PROBABLE_MODES];
224  CU::getIntraMPMs(cu, mpmLst);
225 
226  const int decMsk = ( 1 << m_pcEncCfg->m_IntraEstDecBit ) - 1;
227 
228  std::vector<ModeInfo> parentCandList;
229  parentCandList.reserve( ( numModesAvailable >> m_pcEncCfg->m_IntraEstDecBit ) + 2 );
230 
231  for( unsigned mode = 0; mode < numModesAvailable; mode++ )
232  {
233  // Skip checking extended Angular modes in the first round of SATD
234  if( mode > DC_IDX && ( mode & decMsk ) )
235  {
236  continue;
237  }
238 
239  parentCandList.push_back( ModeInfo( false, false, 0, NOT_INTRA_SUBPARTITIONS, mode ) );
240  }
241 
242  for( int decDst = 1 << m_pcEncCfg->m_IntraEstDecBit; decDst > 0; decDst >>= 1 )
243  {
244  for( unsigned idx = 0; idx < parentCandList.size(); idx++ )
245  {
246  int modeParent = parentCandList[idx].modeId;
247 
248  int off = decDst & decMsk;
249  int inc = decDst << 1;
250 
251 #if 1 // INTRA_AS_IN_VTM
252  if( off != 0 && ( modeParent <= ( DC_IDX + 1 ) || modeParent >= ( NUM_LUMA_MODE - 1 ) ) )
253  {
254  continue;
255  }
256 
257 #endif
258  for( int mode = modeParent - off; mode < modeParent + off + 1; mode += inc )
259  {
260  if( satdChecked[mode] || mode < 0 || mode >= NUM_LUMA_MODE )
261  {
262  continue;
263  }
264 
265  cu.intraDir[0] = mode;
266 
267  initPredIntraParams( cu, cu.Y(), sps );
268  distParam.cur.buf = piPred.buf = m_SortedPelUnitBufs->getTestBuf().Y().buf;
269  predIntraAng( COMP_Y, piPred, cu );
270 
271  // Use the min between SAD and HAD as the cost criterion
272  // SAD is scaled by 2 to align with the scaling of HAD
273  Distortion minSadHad = distParam.distFunc( distParam );
274 
275  uint64_t fracModeBits = xFracModeBitsIntraLuma( cu, mpmLst );
276 
277  //restore ctx
278  m_CABACEstimator->getCtx() = SubCtx( CtxSet( Ctx::IntraLumaMpmFlag(), intra_ctx_size ), ctxStartIntraCtx );
279 
280  double cost = ( double ) minSadHad + ( double ) fracModeBits * sqrtLambdaForFirstPass;
281  DTRACE( g_trace_ctx, D_INTRA_COST, "IntraHAD: %u, %llu, %f (%d)\n", minSadHad, fracModeBits, cost, mode );
282 
283  int insertPos = -1;
284  updateCandList( ModeInfo( false, false, 0, NOT_INTRA_SUBPARTITIONS, mode ), cost, RdModeList, CandCostList, numModesForFullRD, &insertPos );
285  updateCandList( ModeInfo( false, false, 0, NOT_INTRA_SUBPARTITIONS, mode ), ( double ) minSadHad, HadModeList, CandHadList, numHadCand );
286  m_SortedPelUnitBufs->insert( insertPos, ( int ) RdModeList.size() );
287 
288  satdChecked[mode] = true;
289  }
290  }
291 
292  parentCandList.resize( RdModeList.size() );
293  std::copy( RdModeList.cbegin(), RdModeList.cend(), parentCandList.begin() );
294  }
295 
296  const bool isFirstLineOfCtu = (((cu.block(COMP_Y).y)&((cu.cs->sps)->CTUSize - 1)) == 0);
297  if( m_pcEncCfg->m_MRL && ! isFirstLineOfCtu )
298  {
299  cu.multiRefIdx = 1;
300  unsigned multiRefMPM [NUM_MOST_PROBABLE_MODES];
301  CU::getIntraMPMs(cu, multiRefMPM);
302 
303  for (int mRefNum = 1; mRefNum < MRL_NUM_REF_LINES; mRefNum++)
304  {
305  int multiRefIdx = MULTI_REF_LINE_IDX[mRefNum];
306 
307  cu.multiRefIdx = multiRefIdx;
308  initIntraPatternChType(cu, cu.Y(), true);
309 
310  for (int x = 1; x < NUM_MOST_PROBABLE_MODES; x++)
311  {
312  cu.intraDir[0] = multiRefMPM[x];
313  initPredIntraParams(cu, cu.Y(), sps);
314  distParam.cur.buf = piPred.buf = m_SortedPelUnitBufs->getTestBuf().Y().buf;
315  predIntraAng(COMP_Y, piPred, cu);
316 
317  // Use the min between SAD and SATD as the cost criterion
318  // SAD is scaled by 2 to align with the scaling of HAD
319  Distortion minSadHad = distParam.distFunc(distParam);
320 
321  // NB xFracModeBitsIntra will not affect the mode for chroma that may have already been pre-estimated.
322  uint64_t fracModeBits = xFracModeBitsIntraLuma( cu, mpmLst );
323 
324  //restore ctx
325  m_CABACEstimator->getCtx() = SubCtx(CtxSet(Ctx::IntraLumaMpmFlag(), intra_ctx_size), ctxStartIntraCtx);
326 
327  double cost = (double) minSadHad + (double) fracModeBits * sqrtLambdaForFirstPass;
328 // DTRACE(g_trace_ctx, D_INTRA_COST, "IntraMRL: %u, %llu, %f (%d)\n", minSadHad, fracModeBits, cost, cu.intraDir[0]);
329 
330  int insertPos = -1;
331  updateCandList( ModeInfo( false, false, multiRefIdx, NOT_INTRA_SUBPARTITIONS, cu.intraDir[0] ), cost, RdModeList, CandCostList, numModesForFullRD, &insertPos );
332  updateCandList( ModeInfo( false, false, multiRefIdx, NOT_INTRA_SUBPARTITIONS, cu.intraDir[0] ), (double)minSadHad, HadModeList, CandHadList, numHadCand );
333  m_SortedPelUnitBufs->insert(insertPos, (int)RdModeList.size());
334  }
335  }
336  cu.multiRefIdx = 0;
337  }
338 
339  if (testMip)
340  {
341  cu.mipFlag = true;
342  cu.multiRefIdx = 0;
343 
344  double mipHadCost[MAX_NUM_MIP_MODE] = { MAX_DOUBLE };
345 
346  initIntraPatternChType(cu, cu.Y());
347  initIntraMip( cu );
348 
349  const int transpOff = getNumModesMip( cu.Y() );
350  const int numModesFull = (transpOff << 1);
351  for( uint32_t uiModeFull = 0; uiModeFull < numModesFull; uiModeFull++ )
352  {
353  const bool isTransposed = (uiModeFull >= transpOff ? true : false);
354  const uint32_t uiMode = (isTransposed ? uiModeFull - transpOff : uiModeFull);
355 
356  cu.mipTransposedFlag = isTransposed;
357  cu.intraDir[CH_L] = uiMode;
358  distParam.cur.buf = piPred.buf = m_SortedPelUnitBufs->getTestBuf().Y().buf;
359  predIntraMip(piPred, cu);
360 
361  // Use the min between SAD and HAD as the cost criterion
362  // SAD is scaled by 2 to align with the scaling of HAD
363  Distortion minSadHad = distParam.distFunc(distParam);
364 
365  uint64_t fracModeBits = xFracModeBitsIntraLuma( cu, mpmLst );
366 
367  //restore ctx
368  m_CABACEstimator->getCtx() = SubCtx(CtxSet(Ctx::IntraLumaMpmFlag(), intra_ctx_size), ctxStartIntraCtx);
369 
370  double cost = double(minSadHad) + double(fracModeBits) * sqrtLambdaForFirstPass;
371  mipHadCost[uiModeFull] = cost;
372  DTRACE(g_trace_ctx, D_INTRA_COST, "IntraMIP: %u, %llu, %f (%d)\n", minSadHad, fracModeBits, cost, uiModeFull);
373 
374  int insertPos = -1;
375  updateCandList( ModeInfo( true, isTransposed, 0, NOT_INTRA_SUBPARTITIONS, cu.intraDir[0] ), cost, RdModeList, CandCostList, numModesForFullRD+1, &insertPos );
376  updateCandList( ModeInfo( true, isTransposed, 0, NOT_INTRA_SUBPARTITIONS, cu.intraDir[0] ), 0.8*(double)minSadHad, HadModeList, CandHadList, numHadCand );
377  m_SortedPelUnitBufs->insert(insertPos, (int)RdModeList.size());
378  }
379 
380  const double thresholdHadCost = 1.0 + 1.4 / sqrt((double)(cu.lwidth()*cu.lheight()));
381  xReduceHadCandList(RdModeList, CandCostList, *m_SortedPelUnitBufs, numModesForFullRD, thresholdHadCost, mipHadCost, cu, fastMip);
382  }
383 
384  if( m_pcEncCfg->m_bFastUDIUseMPMEnabled )
385  {
386  const int numMPMs = NUM_MOST_PROBABLE_MODES;
387  unsigned intraMpms[numMPMs];
388 
389  cu.multiRefIdx = 0;
390 
391  const int numCand = CU::getIntraMPMs( cu, intraMpms );
392  ModeInfo mostProbableMode(false, false, 0, NOT_INTRA_SUBPARTITIONS, 0);
393 
394  for( int j = 0; j < numCand; j++ )
395  {
396  bool mostProbableModeIncluded = false;
397  mostProbableMode.modeId = intraMpms[j];
398 
399  for( int i = 0; i < numModesForFullRD; i++ )
400  {
401  mostProbableModeIncluded |= ( mostProbableMode == RdModeList[i] );
402  }
403  if( !mostProbableModeIncluded )
404  {
405  numModesForFullRD++;
406  RdModeList.push_back( mostProbableMode );
407  CandCostList.push_back(0);
408  }
409  }
410  }

413 bool IntraSearch::estIntraPredLumaQT(CodingUnit &cu, Partitioner &partitioner, double bestCost)
414 {
415  CodingStructure &cs = *cu.cs;
416  const int width = partitioner.currArea().lwidth();
417  const int height = partitioner.currArea().lheight();
418 
419  //===== loop over partitions =====
420 
421  const TempCtx ctxStart ( m_CtxCache, m_CABACEstimator->getCtx() );
422 
423  // variables for saving fast intra modes scan results across multiple LFNST passes
424  double costInterCU = xFindInterCUCost( cu );
425 
426  bool validReturn = false;
427 
428  //===== determine set of modes to be tested (using prediction signal only) =====
429  int numModesAvailable = NUM_LUMA_MODE; // total number of Intra modes
430  static_vector<ModeInfo, FAST_UDI_MAX_RDMODE_NUM> RdModeList;
431  static_vector<ModeInfo, FAST_UDI_MAX_RDMODE_NUM> HadModeList;
432  static_vector<double, FAST_UDI_MAX_RDMODE_NUM> CandCostList;
433  static_vector<double, FAST_UDI_MAX_RDMODE_NUM> CandHadList;
434 
435  int numModesForFullRD = g_aucIntraModeNumFast_UseMPM_2D[Log2(width) - MIN_CU_LOG2][Log2(height) - MIN_CU_LOG2];
436  if (m_pcEncCfg->m_numIntraModesFullRD > 0)
437  numModesForFullRD=m_pcEncCfg->m_numIntraModesFullRD;
438 
439 #if INTRA_FULL_SEARCH
440  numModesForFullRD = numModesAvailable;
441 #endif
442  const SPS& sps = *cu.cs->sps;
443  const bool mipAllowed = sps.MIP && cu.lwidth() <= sps.getMaxTbSize() && cu.lheight() <= sps.getMaxTbSize() && ((cu.lfnstIdx == 0) || allowLfnstWithMip(cu.lumaSize()));
444  const int SizeThr = 8>>std::max(0,m_pcEncCfg->m_useFastMIP-2);
445  const bool testMip = mipAllowed && (cu.lwidth() <= (SizeThr * cu.lheight()) && cu.lheight() <= (SizeThr * cu.lwidth())) && (cu.lwidth() <= MIP_MAX_WIDTH && cu.lheight() <= MIP_MAX_HEIGHT);
446  bool testISP = sps.ISP && CU::canUseISP(width, height, cu.cs->sps->getMaxTbSize());
447  if (testISP)
448  {
449  int numTotalPartsHor = (int)width >> floorLog2(CU::getISPSplitDim(width, height, TU_1D_VERT_SPLIT));
450  int numTotalPartsVer = (int)height >> floorLog2(CU::getISPSplitDim(width, height, TU_1D_HORZ_SPLIT));
451  m_ispTestedModes[0].init(numTotalPartsHor, numTotalPartsVer, 0);
452  // the total number of subpartitions is modified to take into account the cases where LFNST cannot be combined with
453  // ISP due to size restrictions
454  numTotalPartsHor = sps.LFNST && CU::canUseLfnstWithISP(cu.Y(), HOR_INTRA_SUBPARTITIONS) ? numTotalPartsHor : 0;
455  numTotalPartsVer = sps.LFNST && CU::canUseLfnstWithISP(cu.Y(), VER_INTRA_SUBPARTITIONS) ? numTotalPartsVer : 0;
456  for (int j = 1; j < NUM_LFNST_NUM_PER_SET; j++)
457  {
458  m_ispTestedModes[j].init(numTotalPartsHor, numTotalPartsVer, 0);
459  }
460  testISP = m_ispTestedModes[0].numTotalParts[0];
461  }
462  else
463  {
464  m_ispTestedModes[0].init(0, 0, 0);
465  }
466 
467  xEstimateLumaRdModeList(numModesForFullRD, RdModeList, HadModeList, CandCostList, CandHadList, cu, testMip);
468 
469  CHECK( (size_t)numModesForFullRD != RdModeList.size(), "Inconsistent state!" );
470 
471  // after this point, don't use numModesForFullRD
472  if( m_pcEncCfg->m_usePbIntraFast && !cs.slice->isIntra() && RdModeList.size() < numModesAvailable && !cs.slice->disableSATDForRd )
473  {
474  double pbintraRatio = PBINTRA_RATIO;
475  int maxSize = -1;
476  ModeInfo bestMipMode;
477  int bestMipIdx = -1;
478  for( int idx = 0; idx < RdModeList.size(); idx++ )
479  {
480  if( RdModeList[idx].mipFlg )
481  {
482  bestMipMode = RdModeList[idx];
483  bestMipIdx = idx;
484  break;
485  }
486  }
487  const int numHadCand = 3;
488  for (int k = numHadCand - 1; k >= 0; k--)
489  {
490  if (CandHadList.size() < (k + 1) || CandHadList[k] > cs.interHad * pbintraRatio) { maxSize = k; }
491  }
492  if (maxSize > 0)
493  {
494  RdModeList.resize(std::min<size_t>(RdModeList.size(), maxSize));
495  if( bestMipIdx >= 0 )
496  {
497  if( RdModeList.size() <= bestMipIdx )
498  {
499  RdModeList.push_back(bestMipMode);
500  m_SortedPelUnitBufs->swap( maxSize, bestMipIdx );
501  }
502  }
503  }
504  if (maxSize == 0)
505  {
506  cs.dist = MAX_DISTORTION;
507  cs.interHad = 0;
508  return false;
509  }
510  }
511 
512  //===== check modes (using r-d costs) =====
513  ModeInfo bestPUMode;
514 
515  CodingStructure *csTemp = m_pTempCS;
516  CodingStructure *csBest = m_pBestCS;
517 
518  csTemp->slice = csBest->slice = cs.slice;
519  csTemp->picture = csBest->picture = cs.picture;
520  csTemp->compactResize( cu );
521  csBest->compactResize( cu );
522  csTemp->initStructData();
523  csBest->initStructData();
524 
525  int bestLfnstIdx = 0;
526  const bool useBDPCM = cs.picture->useScBDPCM;
527  int NumBDPCMCand = (useBDPCM && sps.BDPCM && CU::bdpcmAllowed(cu, ComponentID(partitioner.chType))) ? 2 : 0;
528  int bestbdpcmMode = 0;
529  int bestISP = 0;
530  int bestMrl = 0;
531  bool bestMip = 0;
