68  inline Position recalcPosition(const ChromaFormat _cf, const ComponentID srcCId, const ComponentID dstCId, const Position& pos)
69 {
70  if( toChannelType( srcCId ) == toChannelType( dstCId ) )
71  {
72  return pos;
73  }
74  else if (isLuma(srcCId) && isChroma(dstCId))
75  {
76  return Position(pos.x >> getComponentScaleX(dstCId, _cf), pos.y >> getComponentScaleY(dstCId, _cf));
77  }
78  else
79  {
80  return Position(pos.x << getComponentScaleX(srcCId, _cf), pos.y << getComponentScaleY(srcCId, _cf));
81  }
82 }
