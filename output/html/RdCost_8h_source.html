173  {
174  CHECKD( iVal == std::numeric_limits<int>::min(), "Wrong value" );
175 
176 #if ENABLE_SIMD_OPT && defined( TARGET_SIMD_X86 )
177  // the proper Log2 is not restricted to 0...MAX_CU_SIZE
178  return 1 + ( Log2( iVal <= 0 ? ( unsigned( -iVal ) << 1 ) + 1 : unsigned( iVal << 1 ) ) << 1 );
179 #else
180  unsigned uiLength2 = 1, uiTemp2 = ( iVal <= 0 ) ? ( unsigned( -iVal ) << 1 ) + 1 : unsigned( iVal << 1 );
181 
182  while( uiTemp2 > MAX_CU_SIZE )
183  {
184  uiLength2 += ( MAX_CU_DEPTH << 1 );
185  uiTemp2 >>= MAX_CU_DEPTH;
186  }
187 
188  return uiLength2 + ( Log2(uiTemp2) << 1 );
189 #endif
190  }
