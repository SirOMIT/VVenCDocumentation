126 void OutputBitstream::write ( uint32_t uiBits, uint32_t uiNumberOfBits )
127 {
128  CHECK( uiNumberOfBits > 32, "Number of bits is exceeds '32'" );
129  CHECK( uiNumberOfBits != 32 && (uiBits & (~0 << uiNumberOfBits)) != 0, "Unsupported parameters" );
130 
131  /* any modulo 8 remainder of num_total_bits cannot be written this time,
132  * and will be held until next time. */
133  uint32_t num_total_bits = uiNumberOfBits + m_num_held_bits;
134  uint32_t next_num_held_bits = num_total_bits % 8;
135 
136  /* form a byte aligned word (write_bits), by concatenating any held bits
137  * with the new bits, discarding the bits that will form the next_held_bits.
138  * eg: H = held bits, V = n new bits /---- next_held_bits
139  * len(H)=7, len(V)=1: ... ---- HHHH HHHV . 0000 0000, next_num_held_bits=0
140  * len(H)=7, len(V)=2: ... ---- HHHH HHHV . V000 0000, next_num_held_bits=1
141  * if total_bits < 8, the value of v_ is not used */
142  uint8_t next_held_bits = uiBits << (8 - next_num_held_bits);
143 
144  if (!(num_total_bits >> 3))
145  {
146  /* insufficient bits accumulated to write out, append new_held_bits to
147  * current held_bits */
148  /* NB, this requires that v only contains 0 in bit positions {31..n} */
149  m_held_bits |= next_held_bits;
150  m_num_held_bits = next_num_held_bits;
151  return;
152  }
153 
154  /* topword serves to justify held_bits to align with the msb of uiBits */
155  uint32_t topword = (uiNumberOfBits - next_num_held_bits) & ~((1 << 3) -1);
156  uint32_t write_bits = (m_held_bits << topword) | (uiBits >> next_num_held_bits);
157  switch (num_total_bits >> 3)
158  {
159  case 4: m_fifo.push_back(write_bits >> 24);
160  case 3: m_fifo.push_back(write_bits >> 16);
161  case 2: m_fifo.push_back(write_bits >> 8);
162  case 1: m_fifo.push_back(write_bits);
163  }
164 
165  DTRACE( g_trace_ctx, D_BITSTREAM, " %5d: %0X\n", (int)(m_fifo.size() - 1), m_fifo.back() );
166  m_held_bits = next_held_bits;
167  m_num_held_bits = next_num_held_bits;
168 }

194 {
195  uint32_t uiNumBits = pcSubstream->getNumberOfWrittenBits();
196 
197  const std::vector<uint8_t>& rbsp = pcSubstream->getFIFO();
198  for (std::vector<uint8_t>::const_iterator it = rbsp.begin(); it != rbsp.end();)

214 int OutputBitstream::countStartCodeEmulations()
215 {
216  uint32_t cnt = 0;
217  std::vector<uint8_t>& rbsp = getFIFO();
218  for (std::vector<uint8_t>::iterator it = rbsp.begin(); it != rbsp.end();)
219  {
220  std::vector<uint8_t>::iterator found = it;
221  do
222  {
223  // find the next emulated 00 00 {00,01,02,03}
224  // NB, end()-1, prevents finding a trailing two byte sequence
225  found = search_n(found, rbsp.end()-1, 2, 0);
226  found++;
227  // if not found, found == end, otherwise found = second zero byte
228  if (found == rbsp.end())
229  {
230  break;
231  }
232  if (*(++found) <= 3)
233  {
234  break;
235  }
236  } while (true);
237  it = found;
238  if (found != rbsp.end())
239  {
240  cnt++;
241  }
242  }
243  return cnt;
244 }

270 void InputBitstream::read (uint32_t uiNumberOfBits, uint32_t& ruiBits)
271 {
272  CHECK( uiNumberOfBits > 32, "Too many bits read" );
273 
274  m_numBitsRead += uiNumberOfBits;
275 
276  /* NB, bits are extracted from the MSB of each byte. */
277  uint32_t retval = 0;
278  if (uiNumberOfBits <= m_num_held_bits)
279  {
280  /* n=1, len(H)=7: -VHH HHHH, shift_down=6, mask=0xfe
281  * n=3, len(H)=7: -VVV HHHH, shift_down=4, mask=0xf8
282  */
283  retval = m_held_bits >> (m_num_held_bits - uiNumberOfBits);
284  retval &= ~(0xff << uiNumberOfBits);
285  m_num_held_bits -= uiNumberOfBits;
286  ruiBits = retval;
287  return;
288  }
289 
290  /* all num_held_bits will go into retval
291  * => need to mask leftover bits from previous extractions
292  * => align retval with top of extracted word */
293  /* n=5, len(H)=3: ---- -VVV, mask=0x07, shift_up=5-3=2,
294  * n=9, len(H)=3: ---- -VVV, mask=0x07, shift_up=9-3=6 */
295  uiNumberOfBits -= m_num_held_bits;
296  retval = m_held_bits & ~(0xff << m_num_held_bits);
297  retval <<= uiNumberOfBits;
298 
299  /* number of whole bytes that need to be loaded to form retval */
300  /* n=32, len(H)=0, load 4bytes, shift_down=0
301  * n=32, len(H)=1, load 4bytes, shift_down=1
302  * n=31, len(H)=1, load 4bytes, shift_down=1+1
303  * n=8, len(H)=0, load 1byte, shift_down=0
304  * n=8, len(H)=3, load 1byte, shift_down=3
305  * n=5, len(H)=1, load 1byte, shift_down=1+3
306  */
307  uint32_t aligned_word = 0;
308  uint32_t num_bytes_to_load = (uiNumberOfBits - 1) >> 3;
309  CHECK(m_fifo_idx + num_bytes_to_load >= m_fifo.size(), "Exceeded FIFO size");
310 
311  switch (num_bytes_to_load)
312  {
313  case 3: aligned_word = m_fifo[m_fifo_idx++] << 24;
314  case 2: aligned_word |= m_fifo[m_fifo_idx++] << 16;
315  case 1: aligned_word |= m_fifo[m_fifo_idx++] << 8;
316  case 0: aligned_word |= m_fifo[m_fifo_idx++];
317  }
318 
319  /* resolve remainder bits */
320  uint32_t next_num_held_bits = (32 - uiNumberOfBits) % 8;
321 
322  /* copy required part of aligned_word into retval */
323  retval |= aligned_word >> next_num_held_bits;
324 
325  /* store held bits */
326  m_num_held_bits = next_num_held_bits;
327  m_held_bits = aligned_word;
328 
329  ruiBits = retval;
330 }

362 InputBitstream *InputBitstream::extractSubstream( uint32_t uiNumBits )
363 {
364  uint32_t uiNumBytes = uiNumBits/8;
365  InputBitstream *pResult = new InputBitstream;
366 
367  std::vector<uint8_t> &buf = pResult->getFifo();
368  buf.reserve((uiNumBits+7)>>3);
369 
370  if (m_num_held_bits == 0)
371  {
372  std::size_t currentOutputBufferSize=buf.size();
373  const uint32_t uiNumBytesToReadFromFifo = std::min<uint32_t>(uiNumBytes, (uint32_t)m_fifo.size() - m_fifo_idx);
374  buf.resize(currentOutputBufferSize+uiNumBytes);
375  memcpy(&(buf[currentOutputBufferSize]), &(m_fifo[m_fifo_idx]), uiNumBytesToReadFromFifo); m_fifo_idx+=uiNumBytesToReadFromFifo;
376  if (uiNumBytesToReadFromFifo != uiNumBytes)
377  {
378  memset(&(buf[currentOutputBufferSize+uiNumBytesToReadFromFifo]), 0, uiNumBytes - uiNumBytesToReadFromFifo);
379  }
380  }
381  else
382  {
383  for (uint32_t ui = 0; ui < uiNumBytes; ui++)
384  {
385  uint32_t uiByte;
386  read(8, uiByte);
387  buf.push_back(uiByte);
388  }
389  }
390  if (uiNumBits&0x7)
391  {
392  uint32_t uiByte = 0;
393  read(uiNumBits&0x7, uiByte);
394  uiByte <<= 8-(uiNumBits&0x7);
395  buf.push_back(uiByte);
396  }
397  return pResult;
398 }

400 uint32_t InputBitstream::readByteAlignment()
401 {
402  uint32_t code = 0;
403  read( 1, code );
404  CHECK(code != 1, "Code is not '1'");
405 
406  uint32_t numBits = getNumBitsUntilByteAligned();
407  if(numBits)
408  {
409  CHECK(numBits > getNumBitsLeft(), "More bits available than left");
410  read( numBits, code );
411  CHECK(code != 0, "Code not '0'");
412  }
413  return numBits+1;
