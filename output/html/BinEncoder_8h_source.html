57 class BinStore
58 {
59 public:
60  BinStore () : m_inUse(false), m_allocated(false) {}
61  ~BinStore() {}
62 
63  void reset ()
64  {
65  if( m_inUse )
66  {
67  for( unsigned n = 0; n < Ctx::NumberOfContexts; n++ )
68  {
69  m_binBuffer[n].clear();
70  }
71  }
72  }
73  void addBin ( unsigned bin, unsigned ctxId )
74  {
75  if( m_inUse )
76  {
77  std::vector<bool>& binBuffer = m_binBuffer[ctxId];
78  if( binBuffer.size() < m_maxNumBins )
79  {
80  binBuffer.push_back( bin == 1 );
81  }
82  }
83  }
84 
85  void setUse ( bool useStore ) { m_inUse = useStore; if(m_inUse){xCheckAlloc();} }
86  bool inUse () const { return m_inUse; }
87  const std::vector<bool>& getBinVector( unsigned ctxId ) const { return m_binBuffer[ctxId]; }
88 
89 private:
90  void xCheckAlloc()
91  {
92  if( !m_allocated )
93  {
94  m_binBuffer.resize( Ctx::NumberOfContexts );
95  for( unsigned n = 0; n < Ctx::NumberOfContexts; n++ )
96  {
97  m_binBuffer[n].reserve( m_maxNumBins );
98  }
99  m_allocated = true;
100  }
101  }
102 
103 private:
104  static const std::size_t m_maxNumBins = 100000;
105  bool m_inUse;
106  bool m_allocated;
107  std::vector< std::vector<bool> > m_binBuffer;
108 };
