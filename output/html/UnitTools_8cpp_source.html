81 void CS::setRefinedMotionField(CodingStructure &cs)
82 {
83  MotionBuf mb = cs.getMotionBuf();
84  MotionInfo* orgPtr = mb.buf;
85 
86  for( CodingUnit *ptrCU : cs.cus )
87  {
88  CodingUnit& cu = *ptrCU;
89 
90  if( isLuma( cu.chType ) && CU::checkDMVRCondition( cu ) )
91  {
92  const int dy = std::min<int>( cu.lumaSize().height, DMVR_SUBCU_SIZE );
93  const int dx = std::min<int>( cu.lumaSize().width, DMVR_SUBCU_SIZE );
94 
95  static const unsigned scale = 4 * std::max<int>(1, 4 * AMVP_DECIMATION_FACTOR / 4);
96  static const unsigned mask = scale - 1;
97 
98  const Position puPos = cu.lumaPos();
99  const Mv mv0 = cu.mv[0][0];
100  const Mv mv1 = cu.mv[1][0];
101 
102  for( int y = puPos.y, num = 0; y < ( puPos.y + cu.lumaSize().height ); y = y + dy )
103  {
104  for( int x = puPos.x; x < ( puPos.x + cu.lumaSize().width ); x = x + dx, num++ )
105  {
106  const Mv subPuMv0 = mv0 + cu.mvdL0SubPu[num];
107  const Mv subPuMv1 = mv1 - cu.mvdL0SubPu[num];
108 
109  int y2 = ( ( y - 1 ) & ~mask ) + scale;
110 
111  for( ; y2 < y + dy; y2 += scale )
112  {
113  int x2 = ( ( x - 1 ) & ~mask ) + scale;
114 
115  for( ; x2 < x + dx; x2 += scale )
116  {
117  const Position mbPos = g_miScaling.scale( Position{ x2, y2 } );
118  mb.buf = orgPtr + rsAddr( mbPos, mb.stride );
119 
120  MotionInfo& mi = *mb.buf;
121 
122  mi.mv[0] = subPuMv0;
123  mi.mv[1] = subPuMv1;
124  }
125  }
126  }
127  }
128  }
129  }

133 bool CU::checkCCLMAllowed(const CodingUnit& cu)
134 {
135  bool allowCCLM = false;
136 
137  if( !CS::isDualITree( *cu.cs ) ) //single tree I slice or non-I slice (Note: judging chType is no longer equivalent to checking dual-tree I slice since the local dual-tree is introduced)
138  {
139  allowCCLM = true;
140  }
141  else if( cu.slice->sps->CTUSize <= 32 ) //dual tree, CTUsize < 64
142  {
143  allowCCLM = true;
144  }
145  else //dual tree, CTU size 64 or 128
146  {
147  int depthFor64x64Node = cu.slice->sps->CTUSize == 128 ? 1 : 0;
148  const PartSplit cuSplitTypeDepth1 = CU::getSplitAtDepth( cu, depthFor64x64Node );
149  const PartSplit cuSplitTypeDepth2 = CU::getSplitAtDepth( cu, depthFor64x64Node + 1 );
150 
151  //allow CCLM if 64x64 chroma tree node uses QT split or HBT+VBT split combination
152  if( cuSplitTypeDepth1 == CU_QUAD_SPLIT || (cuSplitTypeDepth1 == CU_HORZ_SPLIT && cuSplitTypeDepth2 == CU_VERT_SPLIT) )
153  {
154  if( cu.chromaFormat == CHROMA_420 )
155  {
156  CHECK( !(cu.blocks[COMP_Cb].width <= 16 && cu.blocks[COMP_Cb].height <= 16), "chroma cu size shall be <= 16x16 for YUV420 format" );
157  }
158  allowCCLM = true;
159  }
160  //allow CCLM if 64x64 chroma tree node uses NS (No Split) and becomes a chroma CU containing 32x32 chroma blocks
161  else if( cuSplitTypeDepth1 == CU_DONT_SPLIT )
162  {
163  if( cu.chromaFormat == CHROMA_420 )
164  {
165  CHECK( !(cu.blocks[COMP_Cb].width == 32 && cu.blocks[COMP_Cb].height == 32), "chroma cu size shall be 32x32 for YUV420 format" );
166  }
167  allowCCLM = true;
168  }
169  //allow CCLM if 64x32 chroma tree node uses NS and becomes a chroma CU containing 32x16 chroma blocks
170  else if( cuSplitTypeDepth1 == CU_HORZ_SPLIT && cuSplitTypeDepth2 == CU_DONT_SPLIT )
171  {
172  if( cu.chromaFormat == CHROMA_420 )
173  {
174  CHECK( !(cu.blocks[COMP_Cb].width == 32 && cu.blocks[COMP_Cb].height == 16), "chroma cu size shall be 32x16 for YUV420 format" );
175  }
176  allowCCLM = true;
177  }
178 
179  //further check luma conditions
180  if( allowCCLM )
181  {
182  //disallow CCLM if luma 64x64 block uses BT or TT or NS with ISP
183  const Position lumaRefPos( cu.chromaPos().x << getComponentScaleX( COMP_Cb, cu.chromaFormat ), cu.chromaPos().y << getComponentScaleY( COMP_Cb, cu.chromaFormat ) );
184  const CodingUnit* colLumaCu = cu.cs->refCS->getCU( lumaRefPos, CH_L, TREE_D );
185 
186  if( colLumaCu->lwidth() < 64 || colLumaCu->lheight() < 64 ) //further split at 64x64 luma node
187  {
188  const PartSplit cuSplitTypeDepth1Luma = CU::getSplitAtDepth( *colLumaCu, depthFor64x64Node );
189  CHECK( !(cuSplitTypeDepth1Luma >= CU_QUAD_SPLIT && cuSplitTypeDepth1Luma <= CU_TRIV_SPLIT), "split mode shall be BT, TT or QT" );
190  if( cuSplitTypeDepth1Luma != CU_QUAD_SPLIT )
191  {
192  allowCCLM = false;
193  }
194  }
195  else if( colLumaCu->lwidth() == 64 && colLumaCu->lheight() == 64 && colLumaCu->ispMode ) //not split at 64x64 luma node and use ISP mode
196  {
197  allowCCLM = false;
198  }
199  }
200  }
201 
202  return allowCCLM;
203 }

237 bool CU::getRprScaling( const SPS* sps, const PPS* curPPS, Picture* refPic, int& xScale, int& yScale )
238 {
239  const Window& curScalingWindow = curPPS->scalingWindow;
240  int curPicWidth = curPPS->picWidthInLumaSamples - SPS::getWinUnitX( sps->chromaFormatIdc ) * (curScalingWindow.winLeftOffset + curScalingWindow.winRightOffset);
241  int curPicHeight = curPPS->picHeightInLumaSamples - SPS::getWinUnitY( sps->chromaFormatIdc ) * (curScalingWindow.winTopOffset + curScalingWindow.winBottomOffset);
242 
243  const Window& refScalingWindow = refPic->cs->pps->scalingWindow;
244  int refPicWidth = refPic->cs->pps->picWidthInLumaSamples - SPS::getWinUnitX( sps->chromaFormatIdc ) * (refScalingWindow.winLeftOffset + refScalingWindow.winRightOffset);
245  int refPicHeight = refPic->cs->pps->picHeightInLumaSamples - SPS::getWinUnitY( sps->chromaFormatIdc) * (refScalingWindow.winTopOffset + refScalingWindow.winBottomOffset);
246 
247  xScale = ( ( refPicWidth << SCALE_RATIO_BITS ) + ( curPicWidth >> 1 ) ) / curPicWidth;
248  yScale = ( ( refPicHeight << SCALE_RATIO_BITS ) + ( curPicHeight >> 1 ) ) / curPicHeight;
249 
250  int curSeqMaxPicWidthY = sps->maxPicWidthInLumaSamples; // pic_width_max_in_luma_samples
251  int curSeqMaxPicHeightY = sps->maxPicHeightInLumaSamples; // pic_height_max_in_luma_samples
252  int curPicWidthY = curPPS->picWidthInLumaSamples; // pic_width_in_luma_samples
253  int curPicHeightY = curPPS->picHeightInLumaSamples; // pic_height_in_luma_samples
254  int max8MinCbSizeY = std::max((int)8, (1<<sps->log2MinCodingBlockSize)); // Max(8, MinCbSizeY)
255 
256  CHECK((curPicWidth * curSeqMaxPicWidthY) < refPicWidth * (curPicWidthY - max8MinCbSizeY), "(curPicWidth * curSeqMaxPicWidthY) should be greater than or equal to refPicWidth * (curPicWidthY - max8MinCbSizeY))");
257  CHECK((curPicHeight * curSeqMaxPicHeightY) < refPicHeight * (curPicHeightY - max8MinCbSizeY), "(curPicHeight * curSeqMaxPicHeightY) should be greater than or equal to refPicHeight * (curPicHeightY - max8MinCbSizeY))");
258 
259  return refPic->cs->pps->isRefScaled( *curPPS );
260 }

290 int CU::predictQP( const CodingUnit& cu, const int prevQP )
291 {
292  const CodingStructure &cs = *cu.cs;
293 
294  uint32_t ctuRsAddr = getCtuAddr( cu );
295  uint32_t ctuXPosInCtus = ctuRsAddr % cs.pcv->widthInCtus;
296  if ( ctuXPosInCtus == 0 &&
297  !( cu.blocks[cu.chType].x & ( cs.pcv->maxCUSizeMask >> getChannelTypeScaleX( cu.chType, cu.chromaFormat ) ) ) &&
298  !( cu.blocks[cu.chType].y & ( cs.pcv->maxCUSizeMask >> getChannelTypeScaleY( cu.chType, cu.chromaFormat ) ) ) &&
299  ( cs.getCU( cu.blocks[cu.chType].pos().offset( 0, -1 ), cu.chType, cu.treeType) != NULL ) &&
300  CU::isSameSliceAndTile( *cs.getCU( cu.blocks[cu.chType].pos().offset( 0, -1 ), cu.chType, cu.treeType), cu ) )
301  {
302  return ( ( cs.getCU( cu.blocks[cu.chType].pos().offset( 0, -1 ), cu.chType, cu.treeType ) )->qp );
303  }
304  else
305  {
306  const int a = ( cu.blocks[cu.chType].y & ( cs.pcv->maxCUSizeMask >> getChannelTypeScaleY( cu.chType, cu.chromaFormat ) ) ) ? ( cs.getCU(cu.blocks[cu.chType].pos().offset( 0, -1 ), cu.chType, cu.treeType))->qp : prevQP;
307  const int b = ( cu.blocks[cu.chType].x & ( cs.pcv->maxCUSizeMask >> getChannelTypeScaleX( cu.chType, cu.chromaFormat ) ) ) ? ( cs.getCU(cu.blocks[cu.chType].pos().offset( -1, 0 ), cu.chType, cu.treeType))->qp : prevQP;
308 
309  return ( a + b + 1 ) >> 1;
310  }

436 uint32_t CU::getISPSplitDim( const int width, const int height, const PartSplit ispType )
437 {
438  bool divideTuInRows = ispType == TU_1D_HORZ_SPLIT;
439  uint32_t splitDimensionSize, nonSplitDimensionSize, partitionSize, divShift = 2;
440 
441  if( divideTuInRows )
442  {
443  splitDimensionSize = height;
444  nonSplitDimensionSize = width;
445  }
446  else
447  {
448  splitDimensionSize = width;
449  nonSplitDimensionSize = height;
450  }
451 
452  const int minNumberOfSamplesPerCu = 1 << ( ( MIN_TB_LOG2_SIZEY << 1 ) );
453  const int factorToMinSamples = nonSplitDimensionSize < minNumberOfSamplesPerCu ? minNumberOfSamplesPerCu >> Log2(nonSplitDimensionSize) : 1;
454  partitionSize = ( splitDimensionSize >> divShift ) < factorToMinSamples ? factorToMinSamples : ( splitDimensionSize >> divShift );
455 
456  CHECK( Log2(partitionSize) + Log2(nonSplitDimensionSize) < Log2(minNumberOfSamplesPerCu), "A partition has less than the minimum amount of samples!" );
457  return partitionSize;

591  mpm[5] = ((mpm[maxCandModeIdx] + offset) % mod) + 2;
592  }
593  }

595  {
596  mpm[0] = PLANAR_IDX;
597  mpm[1] = (leftIntraDir < aboveIntraDir) ? aboveIntraDir : leftIntraDir;
598  maxCandModeIdx = 1;

600  mpm[3] = ((mpm[maxCandModeIdx] - 1) % mod) + 2;
601  mpm[4] = ((mpm[maxCandModeIdx] + offset - 1) % mod) + 2;
602  mpm[5] = ( mpm[maxCandModeIdx] % mod) + 2;
603  }
604  }
605  }
606  for (int i = 0; i < numMPMs; i++)
607  {
608  CHECK(mpm[i] >= NUM_LUMA_MODE, "Invalid MPM");
609  }
610  CHECK(numCand == 0, "No candidates found");
611  return numCand;
612  }
