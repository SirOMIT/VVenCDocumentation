81 int ReshapeData::calculateChromaAdjVpduNei( const TransformUnit& tu, const CompArea& areaY, const TreeType _treeType )
82 {
83  CodingStructure &cs = *tu.cs;
84  int xPos = areaY.lumaPos().x;
85  int yPos = areaY.lumaPos().y;
86  int numNeighborLog = std::min<unsigned>(6,cs.pcv->maxCUSizeLog2);
87  int numNeighbor = 1<<numNeighborLog;
88  const int mask = 1-numNeighbor-1;
89  xPos &= mask;
90  yPos &= mask;
91 
92  if( !cs.pcv->isEncoder )
93  {
94  if( isVPDUprocessed( xPos, yPos ) )
95  {
96  return m_chromaScale;
97  }
98 
99  setVPDULoc( xPos, yPos );
100  }
101  Position topLeft(xPos, yPos);
102  CodingStructure* pcs = (CS::isDualITree(cs) && cs.slice->sliceType == VVENC_I_SLICE) ? tu.cs->picture->cs : tu.cs;
103  CodingUnit *topLeftLuma = pcs->getCU(topLeft, CH_L, _treeType);
104  const CodingUnit *cuAbove = pcs->getCURestricted( topLeftLuma->lumaPos().offset(0, -1), topLeftLuma->lumaPos(), topLeftLuma->slice->independentSliceIdx, topLeftLuma->tileIdx, CH_L, _treeType );
105  const CodingUnit *cuLeft = pcs->getCURestricted( topLeftLuma->lumaPos().offset(-1, 0), topLeftLuma->lumaPos(), topLeftLuma->slice->independentSliceIdx, topLeftLuma->tileIdx, CH_L, _treeType );
106 
107  xPos = topLeftLuma->lumaPos().x;
108  yPos = topLeftLuma->lumaPos().y;
109 
110  const CPelBuf piRecoY = cs.picture->getRecoBuf(topLeftLuma->Y());
111  const int strideY = piRecoY.stride;
112 
113  const Pel* recSrc0 = piRecoY.bufAt(0, 0);
114  const uint32_t picH = tu.cs->picture->lheight();
115  const uint32_t picW = tu.cs->picture->lwidth();
116  int32_t recLuma = 0;
117 
118  int pelnum = 0;
119  if (cuLeft != nullptr)
120  {
121  pelnum ++;
122  for (int i = 0; i < numNeighbor; i++)
123  {
124  int k = (yPos + i) >= picH ? (picH - yPos - 1) : i;
125  recLuma += recSrc0[-1 + k * strideY];
126  }
127  }
128  if (cuAbove != nullptr)
129  {
130  pelnum++;
131  for (int i = 0; i < numNeighbor; i++)
132  {
133  int k = (xPos + i) >= picW ? (picW - xPos - 1) : i;
134  recLuma += recSrc0[-strideY + k];
135  }
136  }
137 
138  int lumaValue;
139  if( pelnum )
140  {
141  const int shift = numNeighborLog + pelnum - 1;
142  lumaValue = (recLuma + (1 << (shift - 1))) >> shift;
143  }
144  else
145  {
146  lumaValue = 1 << (cs.sps->bitDepths[CH_L] - 1);
147  }
148 
149  int chromaScale = calculateChromaAdj(lumaValue);
150  if( !cs.pcv->isEncoder )
151  {
152  m_chromaScale = chromaScale;
153  }
154  return chromaScale;
155 }

161 void Reshape::createDec(int bitDepth)
162 {
163  m_lumaBD = bitDepth;
164  m_reshapeLUTSize = 1 << m_lumaBD;
165  m_initCW = m_reshapeLUTSize / PIC_CODE_CW_BINS;
166 #if defined( TARGET_SIMD_X86 ) && ENABLE_SIMD_OPT_BUFFER
167  if( m_fwdLUT.empty() )
168  m_fwdLUT.resize( m_reshapeLUTSize + 2, 0 );
169  if( m_invLUT.empty() )
170  m_invLUT.resize( m_reshapeLUTSize + 2, 0 );
171 #else
172  if( m_fwdLUT.empty() )
173  m_fwdLUT.resize( m_reshapeLUTSize, 0 );
174  if( m_invLUT.empty() )
175  m_invLUT.resize( m_reshapeLUTSize, 0 );
176 #endif
177  if (m_binCW.empty())
178  m_binCW.resize(PIC_CODE_CW_BINS, 0);
179  if (m_inputPivot.empty())
180  m_inputPivot.resize(PIC_CODE_CW_BINS + 1, 0);
181  if (m_fwdScaleCoef.empty())
182  m_fwdScaleCoef.resize(PIC_CODE_CW_BINS, 1 << FP_PREC);
183  if (m_invScaleCoef.empty())
184  m_invScaleCoef.resize(PIC_CODE_CW_BINS, 1 << FP_PREC);
185  if (m_reshapePivot.empty())
186  m_reshapePivot.resize(PIC_CODE_CW_BINS + 1, 0);
187  if (m_chromaAdjHelpLUT.empty())
188  m_chromaAdjHelpLUT.resize(PIC_CODE_CW_BINS, 1<<CSCALE_FP_PREC);
189 }

195 void Reshape::constructReshaper()
196 {
197  int pwlFwdLUTsize = PIC_CODE_CW_BINS;
198  int pwlFwdBinLen = m_reshapeLUTSize / PIC_CODE_CW_BINS;
199 
200  for (int i = 0; i < m_sliceReshapeInfo.reshaperModelMinBinIdx; i++)
201  m_binCW[i] = 0;
202  for (int i = m_sliceReshapeInfo.reshaperModelMaxBinIdx + 1; i < PIC_CODE_CW_BINS; i++)
203  m_binCW[i] = 0;
204  for (int i = m_sliceReshapeInfo.reshaperModelMinBinIdx; i <= m_sliceReshapeInfo.reshaperModelMaxBinIdx; i++)
205  m_binCW[i] = (uint16_t)(m_sliceReshapeInfo.reshaperModelBinCWDelta[i] + (int)m_initCW);
206 
207  for (int i = 0; i < pwlFwdLUTsize; i++)
208  {
209  m_reshapePivot[i + 1] = m_reshapePivot[i] + m_binCW[i];
210  m_inputPivot[i + 1] = m_inputPivot[i] + m_initCW;
211  m_fwdScaleCoef[i] = ((int32_t)m_binCW[i] * (1 << FP_PREC) + (1 << (floorLog2(pwlFwdBinLen) - 1))) >> floorLog2(pwlFwdBinLen);
212  if (m_binCW[i] == 0)
213  {
214  m_invScaleCoef[i] = 0;
215  m_chromaAdjHelpLUT[i] = 1 << CSCALE_FP_PREC;
216  }
217  else
218  {
219  m_invScaleCoef[i] = (int32_t)(m_initCW * (1 << FP_PREC) / m_binCW[i]);
220  m_chromaAdjHelpLUT[i] = (int32_t)(m_initCW * (1 << FP_PREC) / ( m_binCW[i] + m_sliceReshapeInfo.chrResScalingOffset ) );
221  }
222  }
223  for (int lumaSample = 0; lumaSample < m_reshapeLUTSize; lumaSample++)
224  {
225  int idxY = lumaSample / m_initCW;
226  int tempVal = m_reshapePivot[idxY] + ((m_fwdScaleCoef[idxY] * (lumaSample - m_inputPivot[idxY]) + (1 << (FP_PREC - 1))) >> FP_PREC);
227  m_fwdLUT[lumaSample] = Clip3((Pel)0, (Pel)((1 << m_lumaBD) - 1), (Pel)(tempVal));
228 
229  int idxYInv = getPWLIdxInv(lumaSample);
230  int invSample = m_inputPivot[idxYInv] + ((m_invScaleCoef[idxYInv] * (lumaSample - m_reshapePivot[idxYInv]) + (1 << (FP_PREC - 1))) >> FP_PREC);
231  m_invLUT[lumaSample] = Clip3((Pel)0, (Pel)((1 << m_lumaBD) - 1), (Pel)(invSample));
232  }
233 }

237 void Reshape::initLumaLevelToWeightTableReshape()
238 {
239  const int lutSize = 1 << m_lumaBD;
240  if (m_reshapeLumaLevelToWeightPLUT.empty())
241  m_reshapeLumaLevelToWeightPLUT.resize(lutSize, 1.0);
242 
243  if (m_lumaLevelToWeightPLUT.empty())
244  m_lumaLevelToWeightPLUT.resize(lutSize, 1.0);
245 
246  if (m_signalType == RESHAPE_SIGNAL_PQ)
247  {
248  for (int i = 0; i < lutSize; i++)
249  {
250  double x = m_lumaBD < 10 ? i << (10 - m_lumaBD) : m_lumaBD > 10 ? i >> (m_lumaBD - 10) : i;
251  double y;
252  y = 0.015*x - 1.5 - 6;
253  y = y < -3 ? -3 : (y > 6 ? 6 : y);
254  m_lumaLevelToWeightPLUT[i] = pow(2.0, y / 3.0);
255  m_reshapeLumaLevelToWeightPLUT[i] = (uint32_t)(m_lumaLevelToWeightPLUT[i]* (double)(1 << 16));
256  }
257  }
258 }

278 void Reshape::updateReshapeLumaLevelToWeightTable(LmcsParam &sliceReshape, Pel* wtTable, double cwt)
279 {
280  if (m_signalType == RESHAPE_SIGNAL_SDR || m_signalType == RESHAPE_SIGNAL_HLG)
281  {
282  if (sliceReshape.sliceReshaperModelPresent )
283  {
284  double wBin = 1.0;
285  double weight = 1.0;
286  int histLens = (1 << m_lumaBD) / PIC_CODE_CW_BINS;
287 
288  for (int i = 0; i < PIC_CODE_CW_BINS; i++)
289  {
290  if ((i < sliceReshape.reshaperModelMinBinIdx) || (i > sliceReshape.reshaperModelMaxBinIdx))
291  weight = 1.0;
292  else
293  {
294  if (sliceReshape.reshaperModelBinCWDelta[i] == 1 || (sliceReshape.reshaperModelBinCWDelta[i] == -1 * histLens))
295  weight = wBin;
296  else
297  {
298  weight = (double)wtTable[i] / (double)histLens;
299  weight = weight*weight;
300  }
301  }
302  for (int j = 0; j < histLens; j++)
303  {
304  int ii = i*histLens + j;
305  m_reshapeLumaLevelToWeightPLUT[ii] = (uint32_t)(weight* (double)(1 << 16));
306  }
307  }
308  m_chromaWeightRS = cwt;
309  }
310  else
311  {
312  THROW("updateReshapeLumaLevelToWeightTable ERROR!!");
313  }
314  }
315  else
316  {
317  THROW("updateReshapeLumaLevelToWeightTable not support other signal types!!");
318  }
319 }
