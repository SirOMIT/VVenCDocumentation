109  class FastGeoCostList
110  {
111  public:
112  FastGeoCostList() { numGeoTemplatesInitialized = 0; };
113  ~FastGeoCostList()
114  {
115  for (int partIdx = 0; partIdx < 2; partIdx++)
116  {
117  for (int splitDir = 0; splitDir < GEO_NUM_PARTITION_MODE; splitDir++)
118  {
119  delete[] singleDistList[partIdx][splitDir];
120  }
121  delete[] singleDistList[partIdx];
122  singleDistList[partIdx] = nullptr;
123  }
124  };
125  SingleGeoMergeEntry **singleDistList[2];
126  void init(int numTemplates, int maxNumGeoCand)
127  {
128  if (numGeoTemplatesInitialized == 0 || numGeoTemplatesInitialized < numTemplates)
129  {
130  for (int partIdx = 0; partIdx < 2; partIdx++)
131  {
132  singleDistList[partIdx] = new SingleGeoMergeEntry *[numTemplates];
133  for (int splitDir = 0; splitDir < numTemplates; splitDir++)
134  {
135  singleDistList[partIdx][splitDir] = new SingleGeoMergeEntry[maxNumGeoCand];
136  }
137  }
138  numGeoTemplatesInitialized = numTemplates;
139  }
140  }
141  void insert(int geoIdx, int partIdx, int mergeIdx, double cost)
142  {
143  assert(geoIdx < numGeoTemplatesInitialized);
144  singleDistList[partIdx][geoIdx][mergeIdx] = SingleGeoMergeEntry(mergeIdx, cost);
145  }
146  int numGeoTemplatesInitialized;
147  };
