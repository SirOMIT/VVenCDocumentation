65 struct SAOStatData //data structure for SAO statistics
66 {
67  int64_t diff[MAX_NUM_SAO_CLASSES];
68  int64_t count[MAX_NUM_SAO_CLASSES];
69 
70  SAOStatData(){}
71  ~SAOStatData(){}
72  void reset()
73  {
74  ::memset(diff, 0, sizeof(int64_t)*MAX_NUM_SAO_CLASSES);
75  ::memset(count, 0, sizeof(int64_t)*MAX_NUM_SAO_CLASSES);
76  }
77  const SAOStatData& operator=(const SAOStatData& src)
78  {
79  ::memcpy(diff, src.diff, sizeof(int64_t)*MAX_NUM_SAO_CLASSES);
80  ::memcpy(count, src.count, sizeof(int64_t)*MAX_NUM_SAO_CLASSES);
81  return *this;
82  }
83  const SAOStatData& operator+= (const SAOStatData& src)
84  {
85  for(int i=0; i< MAX_NUM_SAO_CLASSES; i++)
86  {
87  diff[i] += src.diff[i];
88  count[i] += src.count[i];
89  }
90  return *this;
91  }
92 };
