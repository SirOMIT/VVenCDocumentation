61 template<uint32_t OUTPUT_BITDEPTH_DIV8>
62 static void md5_block(MD5& md5, const Pel* plane, uint32_t n)
63 {
64  /* create a 64 byte buffer for packing Pel's into */
65  uint8_t buf[64/OUTPUT_BITDEPTH_DIV8][OUTPUT_BITDEPTH_DIV8];
66  for (uint32_t i = 0; i < n; i++)
67  {
68  Pel pel = plane[i];
69  /* perform bitdepth and endian conversion */
70  for (uint32_t d = 0; d < OUTPUT_BITDEPTH_DIV8; d++)
71  {
72  buf[i][d] = pel >> (d*8);
73  }
74  }
75  md5.update((uint8_t*)buf, n * OUTPUT_BITDEPTH_DIV8);
76 }

82 template<uint32_t OUTPUT_BITDEPTH_DIV8>
83 static void md5_plane(MD5& md5, const Pel* plane, uint32_t width, uint32_t height, uint32_t stride)
84 {
85  /* N is the number of samples to process per md5 update.
86  * All N samples must fit in buf */
87  uint32_t N = 32;
88  uint32_t width_modN = width % N;
89  uint32_t width_less_modN = width - width_modN;
90 
91  for (uint32_t y = 0; y < height; y++)
92  {
93  /* convert pels into unsigned chars in little endian byte order.
94  * NB, for 8bit data, data is truncated to 8bits. */
95  for (uint32_t x = 0; x < width_less_modN; x += N)
96  {
97  md5_block<OUTPUT_BITDEPTH_DIV8>(md5, &plane[y*stride + x], N);
98  }
99 
100  /* mop up any of the remaining line */
101  md5_block<OUTPUT_BITDEPTH_DIV8>(md5, &plane[y*stride + width_less_modN], width_modN);
102  }
103 }

106 uint32_t compCRC(int bitdepth, const Pel* plane, uint32_t width, uint32_t height, uint32_t stride, PictureHash &digest)
107 {
108  uint32_t crcMsb;
109  uint32_t bitVal;
110  uint32_t crcVal = 0xffff;
111  uint32_t bitIdx;
112  for (uint32_t y = 0; y < height; y++)
113  {
114  for (uint32_t x = 0; x < width; x++)
115  {
116  // take CRC of first pictureData byte
117  for(bitIdx=0; bitIdx<8; bitIdx++)
118  {
119  crcMsb = (crcVal >> 15) & 1;
120  bitVal = (plane[y*stride+x] >> (7 - bitIdx)) & 1;
121  crcVal = (((crcVal << 1) + bitVal) & 0xffff) ^ (crcMsb * 0x1021);
122  }
123  // take CRC of second pictureData byte if bit depth is greater than 8-bits
124  if(bitdepth > 8)
125  {
126  for(bitIdx=0; bitIdx<8; bitIdx++)
127  {
128  crcMsb = (crcVal >> 15) & 1;
129  bitVal = (plane[y*stride+x] >> (15 - bitIdx)) & 1;
130  crcVal = (((crcVal << 1) + bitVal) & 0xffff) ^ (crcMsb * 0x1021);
131  }
132  }
133  }
134  }
135  for(bitIdx=0; bitIdx<16; bitIdx++)
136  {
137  crcMsb = (crcVal >> 15) & 1;
138  crcVal = ((crcVal << 1) & 0xffff) ^ (crcMsb * 0x1021);
139  }
140 
141  digest.hash.push_back((crcVal>>8) & 0xff);
142  digest.hash.push_back( crcVal & 0xff);
143  return 2;
144 }

159 uint32_t compChecksum(int bitdepth, const Pel* plane, uint32_t width, uint32_t height, uint32_t stride, PictureHash &digest, const BitDepths &/*bitDepths*/)
160 {
161  uint32_t checksum = 0;
162  uint8_t xor_mask;
163 
164  for (uint32_t y = 0; y < height; y++)
165  {
166  for (uint32_t x = 0; x < width; x++)
167  {
168  xor_mask = (x & 0xff) ^ (y & 0xff) ^ (x >> 8) ^ (y >> 8);
169  checksum = (checksum + ((plane[y*stride+x] & 0xff) ^ xor_mask)) & 0xffffffff;
170 
171  if(bitdepth > 8)
172  {
173  checksum = (checksum + ((plane[y*stride+x]>>8) ^ xor_mask)) & 0xffffffff;
174  }
175  }
176  }
177 
178  digest.hash.push_back((checksum>>24) & 0xff);
179  digest.hash.push_back((checksum>>16) & 0xff);
180  digest.hash.push_back((checksum>>8) & 0xff);
181  digest.hash.push_back( checksum & 0xff);
182  return 4;
183 }

204 uint32_t calcMD5(const CPelUnitBuf& pic, PictureHash &digest, const BitDepths &bitDepths)
205 {
206  /* choose an md5_plane packing function based on the system bitdepth */
207  typedef void (*MD5PlaneFunc)(MD5&, const Pel*, uint32_t, uint32_t, uint32_t);
208  MD5PlaneFunc md5_plane_func;
209 
210  MD5 md5[MAX_NUM_COMP];
211 
212  digest.hash.clear();
213 
214  for (uint32_t chan = 0; chan< (uint32_t)pic.bufs.size(); chan++)
215  {
216  const ComponentID compID=ComponentID(chan);
217  const CPelBuf area = pic.get(compID);
218  md5_plane_func = bitDepths[toChannelType(compID)] <= 8 ? (MD5PlaneFunc)md5_plane<1> : (MD5PlaneFunc)md5_plane<2>;
219  uint8_t tmp_digest[MD5_DIGEST_STRING_LENGTH];
220  md5_plane_func(md5[compID], area.bufAt(0, 0), area.width, area.height, area.stride );
221  md5[compID].finalize(tmp_digest);
222  for(uint32_t i=0; i<MD5_DIGEST_STRING_LENGTH; i++)
223  {
224  digest.hash.push_back(tmp_digest[i]);
225  }
226  }
227  return 16;
228 }

230 std::string hashToString(const PictureHash &digest, int numChar)
231 {
232  static const char* hex = "0123456789abcdef";
233  std::string result;
234 
235  for(int pos=0; pos<int(digest.hash.size()); pos++)
236  {
237  if ((pos % numChar) == 0 && pos!=0 )
238  {
239  result += ',';
240  }
241  result += hex[digest.hash[pos] >> 4];
242  result += hex[digest.hash[pos] & 0xf];
243  }
244 
245  return result;

248 int calcAndPrintHashStatus(const CPelUnitBuf& pic, const SEIDecodedPictureHash* pictureHashSEI, const BitDepths &bitDepths, const vvencMsgLevel msgl)
249 {
250  /* calculate MD5sum for entire reconstructed picture */
251  PictureHash recon_digest;
252  int numChar=0;
253  const char* hashType = "\0";
254 
255  if (pictureHashSEI)
256  {
257  switch (pictureHashSEI->method)
258  {
259  case VVENC_HASHTYPE_MD5:
260  {
261  hashType = "MD5";
262  numChar = calcMD5(pic, recon_digest, bitDepths);
263  break;
264  }
265  case VVENC_HASHTYPE_CRC:
266  {
267  hashType = "CRC";
268  numChar = calcCRC(pic, recon_digest, bitDepths);
269  break;
270  }
271  case VVENC_HASHTYPE_CHECKSUM:
272  {
273  hashType = "Checksum";
274  numChar = calcChecksum(pic, recon_digest, bitDepths);
275  break;
276  }
277  default:
278  {
279  THROW("Unknown hash type");
280  break;
281  }
282  }
283  }
284 
285  /* compare digest against received version */
286  const char* ok = "(unk)";
287  bool mismatch = false;
288 
289  if (pictureHashSEI)
290  {
291  ok = "(OK)";
292  if (recon_digest != pictureHashSEI->pictureHash)
293  {
294  ok = "(***ERROR***)";
295  mismatch = true;
296  }
297  }
298 
299  msg( msgl, "[%s:%s,%s] ", hashType, hashToString(recon_digest, numChar).c_str(), ok);
300 
301  if (mismatch)
302  {
303  msg( msgl, "[rx%s:%s] ", hashType, hashToString(pictureHashSEI->pictureHash, numChar).c_str());
304  }
305  return mismatch;
306 }
