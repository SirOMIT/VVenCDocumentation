117 inline uint32_t QuantRDOQ::xGetCodedLevel( double& rd64CodedCost,
118  double& rd64CodedCost0,
119  double& rd64CodedCostSig,
120  Intermediate_Int lLevelDouble,
121  uint32_t uiMaxAbsLevel,
122  const BinFracBits* fracBitsSig,
123  const BinFracBits& fracBitsPar,
124  const BinFracBits& fracBitsGt1,
125  const BinFracBits& fracBitsGt2,
126  const int remRegBins,
127  unsigned goRiceZero,
128  uint16_t ui16AbsGoRice,
129  int iQBits,
130  double errorScale,
131  bool bLast,
132  bool useLimitedPrefixLength,
133  const int maxLog2TrDynamicRange
134  ) const
135 {
136  double dCurrCostSig = 0;
137  uint32_t uiBestAbsLevel = 0;
138 
139  if( !bLast && uiMaxAbsLevel < 3 )
140  {
141  rd64CodedCostSig = xGetRateSigCoef( *fracBitsSig, 0 );
142  rd64CodedCost = rd64CodedCost0 + rd64CodedCostSig;
143  if( uiMaxAbsLevel == 0 )
144  {
145  return uiBestAbsLevel;
146  }
147  }
148  else
149  {
150  rd64CodedCost = MAX_DOUBLE;
151  }
152 
153  if( !bLast )
154  {
155  dCurrCostSig = xGetRateSigCoef( *fracBitsSig, 1 );
156  }
157 
158  uint32_t uiMinAbsLevel = ( uiMaxAbsLevel > 1 ? uiMaxAbsLevel - 1 : 1 );
159  for( int uiAbsLevel = uiMaxAbsLevel; uiAbsLevel >= uiMinAbsLevel ; uiAbsLevel-- )
160  {
161  double dErr = double( lLevelDouble - ( Intermediate_Int(uiAbsLevel) << iQBits ) );
162 
163  double dCurrCost = dErr * dErr * errorScale + xGetICost( xGetICRate( uiAbsLevel, fracBitsPar, fracBitsGt1, fracBitsGt2, remRegBins, goRiceZero, ui16AbsGoRice, true, maxLog2TrDynamicRange ) );
164  dCurrCost += dCurrCostSig;
165 
166  if( dCurrCost < rd64CodedCost )
167  {
168  uiBestAbsLevel = uiAbsLevel;
169  rd64CodedCost = dCurrCost;
170  rd64CodedCostSig = dCurrCostSig;
171  }
172  }
173 
174  return uiBestAbsLevel;
175 }

188 inline int QuantRDOQ::xGetICRate( const uint32_t uiAbsLevel,
189  const BinFracBits& fracBitsPar,
190  const BinFracBits& fracBitsGt1,
191  const BinFracBits& fracBitsGt2,
192  const int remRegBins,
193  unsigned goRiceZero,
194  const uint16_t ui16AbsGoRice,
195  const bool useLimitedPrefixLength,
196  const int maxLog2TrDynamicRange ) const
197 {
198  if( remRegBins < 4 )
199  {
200  int iRate = int( xGetIEPRate() ); // cost of sign bit
201  uint32_t symbol = ( uiAbsLevel == 0 ? goRiceZero : uiAbsLevel <= goRiceZero ? uiAbsLevel-1 : uiAbsLevel );
202  uint32_t length;
203  const int threshold = COEF_REMAIN_BIN_REDUCTION;
204  if( symbol < ( threshold << ui16AbsGoRice ) )
205  {
206  length = symbol >> ui16AbsGoRice;
207  iRate += ( length + 1 + ui16AbsGoRice ) << SCALE_BITS;
208  }
209  else if( useLimitedPrefixLength )
210  {
211  const uint32_t maximumPrefixLength = ( 32 - ( COEF_REMAIN_BIN_REDUCTION + maxLog2TrDynamicRange ) );
212 
213  uint32_t prefixLength = 0;
214  uint32_t suffix = ( symbol >> ui16AbsGoRice ) - COEF_REMAIN_BIN_REDUCTION;
215 
216  while( ( prefixLength < maximumPrefixLength ) && ( suffix > ( ( 2 << prefixLength ) - 2 ) ) )
217  {
218  prefixLength++;
219  }
220 
221  const uint32_t suffixLength = ( prefixLength == maximumPrefixLength ) ? ( maxLog2TrDynamicRange - ui16AbsGoRice ) : ( prefixLength + 1/*separator*/ );
222 
223  iRate += ( COEF_REMAIN_BIN_REDUCTION + prefixLength + suffixLength + ui16AbsGoRice ) << SCALE_BITS;
224  }
225  else
226  {
227  length = ui16AbsGoRice;
228  symbol = symbol - ( threshold << ui16AbsGoRice );
229  while( symbol >= ( 1 << length ) )
230  {
231  symbol -= ( 1 << ( length++ ) );
232  }
233  iRate += ( threshold + length + 1 - ui16AbsGoRice + length ) << SCALE_BITS;
234  }
235  return iRate;
236  }
237 
238  int iRate = int( xGetIEPRate() ); // cost of sign bit
239  const uint32_t cthres = 4;
240  if( uiAbsLevel >= cthres )
241  {
242  uint32_t symbol = ( uiAbsLevel - cthres ) >> 1;
243  uint32_t length;
244  const int threshold = COEF_REMAIN_BIN_REDUCTION;
245  if( symbol < ( threshold << ui16AbsGoRice ) )
246  {
247  length = symbol >> ui16AbsGoRice;
248  iRate += ( length + 1 + ui16AbsGoRice ) << SCALE_BITS;
249  }
250  else if( useLimitedPrefixLength )
251  {
252  const uint32_t maximumPrefixLength = ( 32 - ( COEF_REMAIN_BIN_REDUCTION + maxLog2TrDynamicRange ) );
253 
254  uint32_t prefixLength = 0;
255  uint32_t suffix = ( symbol >> ui16AbsGoRice ) - COEF_REMAIN_BIN_REDUCTION;
256 
257  while( ( prefixLength < maximumPrefixLength ) && ( suffix > ( ( 2 << prefixLength ) - 2 ) ) )
258  {
259  prefixLength++;
260  }
261 
262  const uint32_t suffixLength = ( prefixLength == maximumPrefixLength ) ? ( maxLog2TrDynamicRange - ui16AbsGoRice ) : ( prefixLength + 1/*separator*/ );
263 
264  iRate += ( COEF_REMAIN_BIN_REDUCTION + prefixLength + suffixLength + ui16AbsGoRice ) << SCALE_BITS;
265  }
266  else
267  {
268  length = ui16AbsGoRice;
269  symbol = symbol - ( threshold << ui16AbsGoRice );
270  while( symbol >= ( 1 << length ) )
271  {
272  symbol -= ( 1 << ( length++ ) );
273  }
274  iRate += ( threshold + length + 1 - ui16AbsGoRice + length ) << SCALE_BITS;
275  }
276 
277  iRate += fracBitsGt1.intBits[1];
278  iRate += fracBitsPar.intBits[( uiAbsLevel - 2 ) & 1];
279  iRate += fracBitsGt2.intBits[1];
280  }
281  else if( uiAbsLevel == 1 )
282  {
283  iRate += fracBitsGt1.intBits[0];
284  }
285  else if( uiAbsLevel == 2 )
286  {
287  iRate += fracBitsGt1.intBits[1];
288  iRate += fracBitsPar.intBits[0];
289  iRate += fracBitsGt2.intBits[0];
290  }
291  else if( uiAbsLevel == 3 )
292  {
293  iRate += fracBitsGt1.intBits[1];
294  iRate += fracBitsPar.intBits[1];
295  iRate += fracBitsGt2.intBits[0];
296  }
297  else
298  {
299  iRate = 0;
300  }
301  return iRate;
302 }

358 double QuantRDOQ::xGetErrScaleCoeff(const bool needsSqrt2, SizeType width, SizeType height, int qp, const int maxLog2TrDynamicRange, const int channelBitDepth, bool bTransformSkip=false)
359 {
360  const int iTransformShift = bTransformSkip ? 0 : getTransformShift(channelBitDepth, Size(width, height), maxLog2TrDynamicRange);
361  double dErrScale = (double)(1 << SCALE_BITS); // Compensate for scaling of bitcount in Lagrange cost function
362  double dTransShift = (double)iTransformShift + (needsSqrt2 ? -0.5 : 0.0);
363  dErrScale = dErrScale * pow(2.0, (-2.0*dTransShift)); // Compensate for scaling through forward transform
364  const int QStep = g_quantScales[needsSqrt2 ? 1 : 0][qp];
365  double finalErrScale = dErrScale / QStep / QStep / (1 << (DISTORTION_PRECISION_ADJUSTMENT(channelBitDepth) << 1));
366  return finalErrScale;
367 }

378 void QuantRDOQ::xSetErrScaleCoeff( uint32_t list, uint32_t sizeX, uint32_t sizeY, int qp, const int maxLog2TrDynamicRange[MAX_NUM_CH], const BitDepths &bitDepths )
379 {
380  const int width = g_scalingListSizeX[sizeX];
381  const int height = g_scalingListSizeX[sizeY];
382  const ChannelType channelType = ( ( list == 0 ) || ( list == MAX_NUM_COMP ) ) ? CH_L : CH_C;
383  const int channelBitDepth = bitDepths[channelType];
384  const int iTransformShift = getTransformShift( channelBitDepth, Size( g_scalingListSizeX[sizeX], g_scalingListSizeX[sizeY] ), maxLog2TrDynamicRange[channelType] ); // Represents scaling through forward transform
385 
386  double dErrScale = (double)( 1 << SCALE_BITS ); // Compensate for scaling of bitcount in Lagrange cost function
387 
388  const bool needsSqrt2 = ((Log2(width*height)) & 1) == 1;
389  double dTransShift = (double)iTransformShift + ( needsSqrt2 ? -0.5 : 0.0 );
390  dErrScale = dErrScale*pow( 2.0, ( -2.0*dTransShift ) ); // Compensate for scaling through forward transform
391 
392  if( getScalingListEnabled() )
393  {
394  uint32_t i, uiMaxNumCoeff = width * height;
395 
396  int* piQuantcoeff = getQuantCoeff( list, qp, sizeX, sizeY );
397  double* pdErrScale = xGetErrScaleCoeffSL( list, sizeX, sizeY, qp );
398 
399  for( i = 0; i < uiMaxNumCoeff; i++ )
400  {
401  pdErrScale[i] = dErrScale / piQuantcoeff[i] / piQuantcoeff[i] / (1 << (DISTORTION_PRECISION_ADJUSTMENT( bitDepths[channelType] ) << 1));
402  }
403  }
404 
405  int QStep = g_quantScales[needsSqrt2][qp];
406 
407  xGetErrScaleCoeffNoScalingList(list, sizeX, sizeY, qp) =
408  dErrScale / QStep / QStep / (1 << (DISTORTION_PRECISION_ADJUSTMENT(bitDepths[channelType]) << 1));

547  const FracBitsAccess& fracBits = ctx.getFracBitsAcess();

552  const uint32_t uiHeight = rect.height;
553  const ChannelType chType = toChannelType(compID);
554  const int channelBitDepth = sps.bitDepths[ chType ];
555 
556  const bool extendedPrecision = sps.spsRExt.extendedPrecisionProcessing;
557  const int maxLog2TrDynamicRange = sps.getMaxLog2TrDynamicRange(chType);
558 
559  const bool useIntraSubPartitions = tu.cu->ispMode && isLuma(compID);
560  /* for 422 chroma blocks, the effective scaling applied during transformation is not a power of 2, hence it cannot be
561  * implemented as a bit-shift (the quantised result will be sqrt(2) * larger than required). Alternatively, adjust the
562  * uiLog2TrSize applied in iTransformShift, such that the result is 1/sqrt(2) the required result (i.e. smaller)
563  * Then a QP+3 (sqrt(2)) or QP-3 (1/sqrt(2)) method could be used to get the required result
564  */

568 
569  if (tu.mtsIdx[compID]==MTS_SKIP && extendedPrecision)
570  {

587 
588  double *pdCostCoeff = m_pdCostCoeff;

591  int *rateIncUp = m_rateIncUp;
592  int *rateIncDown = m_rateIncDown;
593  int *sigRateDelta = m_sigRateDelta;
594  TCoeff *deltaU = m_deltaU;
595 
596  memset( piDstCoeff, 0, sizeof( TCoeffSig ) * uiMaxNumCoeff );
597  memset( m_pdCostCoeff, 0, sizeof( double ) * uiMaxNumCoeff );
598  memset( m_pdCostSig, 0, sizeof( double ) * uiMaxNumCoeff );
599  memset( m_rateIncUp, 0, sizeof( int ) * uiMaxNumCoeff );
600  memset( m_rateIncDown, 0, sizeof( int ) * uiMaxNumCoeff );
601  memset( m_sigRateDelta, 0, sizeof( int ) * uiMaxNumCoeff );
602  memset( m_deltaU, 0, sizeof( TCoeff ) * uiMaxNumCoeff );
603 
604 
605  const bool needSqrtAdjustment = TU::needsSqrt2Scale( tu, compID );
606  const bool isTransformSkip = tu.mtsIdx[compID]==MTS_SKIP;
607  const double *const pdErrScale = xGetErrScaleCoeffSL(scalingListType, uiLog2BlockWidth, uiLog2BlockHeight, cQP.rem(isTransformSkip));
608  const int *const piQCoef = getQuantCoeff(scalingListType, cQP.rem(isTransformSkip), uiLog2BlockWidth, uiLog2BlockHeight);
609  const bool isLfnstApplied = tu.cu->lfnstIdx > 0 && (CU::isSepTree(*tu.cu) ? true : isLuma(compID));
610  const bool enableScalingLists = getUseScalingList(uiWidth, uiHeight, isTransformSkip, isLfnstApplied);
611  const int defaultQuantisationCoefficient = g_quantScales[ needSqrtAdjustment ?1:0][cQP.rem(isTransformSkip)];
612  const double defaultErrorScale = xGetErrScaleCoeffNoScalingList(scalingListType, uiLog2BlockWidth, uiLog2BlockHeight, cQP.rem(isTransformSkip));
613  const int iQBits = QUANT_SHIFT + cQP.per(isTransformSkip) + iTransformShift + (needSqrtAdjustment?-1:0); // Right shift of non-RDOQ quantizer; level = (coeff*uiQ + offset)>>q_bits
614 
615 
616  const TCoeff entropyCodingMinimum = -(1 << maxLog2TrDynamicRange);
617  const TCoeff entropyCodingMaximum = (1 << maxLog2TrDynamicRange) - 1;
618 
619  CoeffCodingContext cctx(tu, compID, tu.cs->slice->signDataHidingEnabled);
620  const int iCGSizeM1 = (1 << cctx.log2CGSize()) - 1;
621 
622  int iCGLastScanPos = -1;
623  double d64BaseCost = 0;
624  int iLastScanPos = -1;
625 
626  int ctxBinSampleRatio = MAX_TU_LEVEL_CTX_CODED_BIN_CONSTRAINT;
627  int remRegBins = (tu.getTbAreaAfterCoefZeroOut( compID ) * ctxBinSampleRatio) >> 4;
628  uint32_t goRiceParam = 0;
