78 bool tryDecodePicture( Picture* pcEncPic, const int expectedPoc, const std::string& bitstreamFileName, FFwdDecoder& ffwdDecoder, ParameterSetMap<APS>* apsMap, bool bDecodeUntilPocFound /* = false */, int debugPOC /* = -1*/, bool copyToEnc /* = true */ )
79 {
80  PicList* pcListPic = NULL;
81  bool bRet = false;
82 
83  if( pcEncPic )
84  {
85  if( ! ffwdDecoder.pcDecLib )
86  {
87  ffwdDecoder.bitstreamFile = new std::ifstream( bitstreamFileName.c_str(), std::ifstream::in | std::ifstream::binary );
88  ffwdDecoder.bytestream = new InputByteStream( *ffwdDecoder.bitstreamFile );
89 
90  CHECK( !*ffwdDecoder.bitstreamFile, "failed to open bitstream file " << bitstreamFileName.c_str() << " for reading" ) ;
91  // create decoder class
92  ffwdDecoder.pcDecLib = new DecLib;
93  ffwdDecoder.pcDecLib->create();
94 
95  // initialize decoder class
96  ffwdDecoder.pcDecLib->init();
97 
98  ffwdDecoder.pcDecLib->setDecoderInEncoderMode ( true );
99  ffwdDecoder.pcDecLib->setDebugPOC ( debugPOC );
100  ffwdDecoder.pcDecLib->setDecodedPictureHashSEIEnabled( true );
101  if(apsMap) ffwdDecoder.pcDecLib->setAPSMapEnc ( apsMap );
102 
103  msg( VVENC_INFO, "start to decode %s \n", bitstreamFileName.c_str() );
104  }
105 
106  bool goOn = true;
107  DecLib *pcDecLib = ffwdDecoder.pcDecLib;
108  CHECK( pcDecLib == nullptr, "error in setup of decoder lib" );
109 
110  // main decoder loop
111  while( !!*ffwdDecoder.bitstreamFile && goOn )
112  {
113  InputNALUnit nalu;
114  nalu.m_nalUnitType = VVENC_NAL_UNIT_INVALID;
115 
116  // determine if next NAL unit will be the first one from a new picture
117  bool bNewPicture = pcDecLib->isNewPicture( ffwdDecoder.bitstreamFile, ffwdDecoder.bytestream );
118  bool bNewAccessUnit = bNewPicture && pcDecLib->isNewAccessUnit( bNewPicture, ffwdDecoder.bitstreamFile, ffwdDecoder.bytestream );
119  bNewPicture = bNewPicture && bNewAccessUnit;
120 
121  if( !bNewPicture )
122  {
123  AnnexBStats stats = AnnexBStats();
124 
125  byteStreamNALUnit( *ffwdDecoder.bytestream, nalu.getBitstream().getFifo(), stats );
126 
127  // call actual decoding function
128  if( nalu.getBitstream().getFifo().empty() )
129  {
130  /* this can happen if the following occur:
131  * - empty input file
132  * - two back-to-back start_code_prefixes
133  * - start_code_prefix immediately followed by EOF
134  */
135  msg( VVENC_ERROR, "Warning: Attempt to decode an empty NAL unit\n" );
136  }
137  else
138  {
139  read( nalu );
140  int iSkipFrame = 0;
141  pcDecLib->decode( nalu, iSkipFrame, ffwdDecoder.iPOCLastDisplay, 0 );
142  }
143  }
144 
145  if( ( bNewPicture || !*ffwdDecoder.bitstreamFile || nalu.m_nalUnitType == VVENC_NAL_UNIT_EOS ) && !pcDecLib->getFirstSliceInSequence() )
146  {
147  if( ! ffwdDecoder.loopFiltered || *ffwdDecoder.bitstreamFile )
148  {
149  int poc;
150  pcDecLib->finishPictureLight( poc, pcListPic );
151 
152  if( pcListPic )
153  {
154  for( auto & pic : *pcListPic )
155  {
156  if( copyToEnc )
157  {
158  if( pic->poc == poc && (!bDecodeUntilPocFound || expectedPoc == poc ) )
159  {
160  pcEncPic->createTempBuffers( pic->cs->pcv->maxCUSize );
161  pcEncPic->cs->createCoeffs();
162  pcEncPic->cs->createTempBuffers( true );
163  pcEncPic->cs->initStructData( MAX_INT, false, nullptr, true );
164 
165  CHECK( pcEncPic->slices.size() == 0, "at least one slice should be available" );
166 
167  CHECK( expectedPoc != poc, "mismatch in POC - check encoder configuration" );
168 
169  if( poc != debugPOC )
170  {
171  for( int i = 0; i < pic->slices.size(); i++ )
172  {
173  if( pcEncPic->slices.size() <= i )
174  {
175  pcEncPic->slices.push_back( new Slice );
176  pcEncPic->slices.back()->pps = pcEncPic->slices[0]->pps;
177  pcEncPic->slices.back()->sps = pcEncPic->slices[0]->sps;
178  pcEncPic->slices.back()->vps = pcEncPic->slices[0]->vps;
179  pcEncPic->slices.back()->pic = pcEncPic->slices[0]->pic;
180  }
181  pcEncPic->slices[i]->copySliceInfo( pic->slices[i], false );
182  }
183  }
184 
185  pcEncPic->cs->slice = pcEncPic->slices.back();
186 
187  {
188  if ( pic->cs->sps->saoEnabled )
189  {
190  pcEncPic->copySAO( *pic, 0 );
191  }
192 
193  if( pic->cs->sps->alfEnabled )
194  {
195  pcEncPic->resizeAlfCtuBuffers( pic->cs->pcv->sizeInCtus );
196  for( int compIdx = 0; compIdx < MAX_NUM_COMP; compIdx++ )
197  {
198  std::copy( pic->m_alfCtuEnabled[ compIdx ].begin(), pic->m_alfCtuEnabled[ compIdx ].end(), pcEncPic->m_alfCtuEnabled[ compIdx ].begin() );
199  }
200  std::copy( pic->m_alfCtbFilterIndex.begin(), pic->m_alfCtbFilterIndex.end(), pcEncPic->m_alfCtbFilterIndex.begin() );
201 
202  std::copy( pic->m_alfCtuAlternative[COMP_Cb].begin(), pic->m_alfCtuAlternative[COMP_Cb].end(), pcEncPic->m_alfCtuAlternative[COMP_Cb].begin() );
203  std::copy( pic->m_alfCtuAlternative[COMP_Cr].begin(), pic->m_alfCtuAlternative[COMP_Cr].end(), pcEncPic->m_alfCtuAlternative[COMP_Cr].begin() );
204 
205  for( int i = 0; i < pic->slices.size(); i++ )
206  {
207  pcEncPic->slices[i]->tileGroupNumAps = (pic->slices[i]->tileGroupNumAps);
208  pcEncPic->slices[i]->setAlfAPSs(pic->slices[i]->alfAps);
209  pcEncPic->slices[i]-> tileGroupChromaApsId = pic->slices[i]->tileGroupChromaApsId;
210  pcEncPic->slices[i]->tileGroupAlfEnabled[COMP_Y] = pic->slices[i]->tileGroupAlfEnabled[COMP_Y];
211  pcEncPic->slices[i]->tileGroupAlfEnabled[COMP_Cb] = pic->slices[i]->tileGroupAlfEnabled[COMP_Cb];
212  pcEncPic->slices[i]->tileGroupAlfEnabled[COMP_Cr] = pic->slices[i]->tileGroupAlfEnabled[COMP_Cr];
213  pcEncPic->slices[i]->tileGroupCcAlfCbApsId = pic->slices[i]->tileGroupCcAlfCbApsId;
214  pcEncPic->slices[i]->tileGroupCcAlfCbEnabled = pic->slices[i]->tileGroupCcAlfCbEnabled;
215  pcEncPic->slices[i]->tileGroupCcAlfCrApsId = pic->slices[i]->tileGroupCcAlfCrApsId;
216  pcEncPic->slices[i]->tileGroupCcAlfCrEnabled = pic->slices[i]->tileGroupCcAlfCrEnabled;
217  }
218  }
219 
220  pcDecLib->executeLoopFilters();
221  if ( pic->cs->sps->saoEnabled )
222  {
223  pcEncPic->copySAO( *pic, 1 );
224  }
225 
226  pcEncPic->cs->copyStructure( *pic->cs, CH_L, TREE_D, true, true );
227 
228  if( CS::isDualITree( *pcEncPic->cs ) )
229  {
230  pcEncPic->cs->copyStructure( *pic->cs, CH_C, TREE_D, true, true );
231  }
232  }
233 
234  pcEncPic->cs->slice = pcEncPic->slices[ 0 ];
235  pcEncPic->cs->picHeader->copyPicInfo( pic->cs->picHeader, false );
236  for( auto& cu: pcEncPic->cs->cus)
237  {
238  cu->slice = pcEncPic->cs->slice;
239  }
240  goOn = false; // exit the loop return
241  bRet = true;
242  break;
243  }
244  }
245  }
246  }
247  // postpone loop filters
248  if (!bRet)
249  {
250  pcDecLib->executeLoopFilters();
251  }
252 
253  pcDecLib->finishPicture( poc, pcListPic, copyToEnc ? VVENC_DETAILS : VVENC_INFO );
254 
255  // write output
256  if( ! pcListPic->empty())
257  {
258  PicList::iterator iterPic = pcListPic->begin();
259  int numPicsNotYetDisplayed = 0;
260  int dpbFullness = 0;
261  const SPS* activeSPS = (pcListPic->front()->cs->sps);
262  uint32_t maxNrSublayers = activeSPS->maxTLayers;
263  uint32_t numReorderPicsHighestTid = activeSPS->numReorderPics[maxNrSublayers-1];
264  uint32_t maxDecPicBufferingHighestTid = activeSPS->maxDecPicBuffering[maxNrSublayers-1];
265  const VPS* referredVPS = pcListPic->front()->cs->vps;
266 
267  if( referredVPS != nullptr && referredVPS->numLayersInOls[referredVPS->targetOlsIdx] > 1 )
268  {
269  numReorderPicsHighestTid = referredVPS->getNumReorderPics( maxNrSublayers - 1 );
270  maxDecPicBufferingHighestTid = referredVPS->getMaxDecPicBuffering( maxNrSublayers - 1 );
271  }
272 
273  while (iterPic != pcListPic->end())
274  {
275  Picture* pcCurPic = *(iterPic);
276  if(pcCurPic->isNeededForOutput && pcCurPic->getPOC() > ffwdDecoder.iPOCLastDisplay)
277  {
278  numPicsNotYetDisplayed++;
279  dpbFullness++;
280  }
281  else if(pcCurPic->isReferenced)
282  {
283  dpbFullness++;
284  }
285  iterPic++;
286  }
287 
288  iterPic = pcListPic->begin();
289 
290  if (numPicsNotYetDisplayed>2)
291  {
292  iterPic++;
293  }
294 
295  Picture* pcCurPic = *(iterPic);
296  iterPic = pcListPic->begin();
297 
298  while (iterPic != pcListPic->end())
299  {
300  pcCurPic = *(iterPic);
301 
302  if(pcCurPic->isNeededForOutput && pcCurPic->getPOC() > ffwdDecoder.iPOCLastDisplay &&
303  (numPicsNotYetDisplayed > numReorderPicsHighestTid || dpbFullness > maxDecPicBufferingHighestTid))
304  {
305  numPicsNotYetDisplayed--;
306  if( ! pcCurPic->isReferenced )
307  {
308  dpbFullness--;
309  }
310  // update POC of display order
311  ffwdDecoder.iPOCLastDisplay = pcCurPic->getPOC();
312 
313  // erase non-referenced picture in the reference picture list after display
314  if( ! pcCurPic->isReferenced && pcCurPic->isReconstructed )
315  {
316  pcCurPic->isReconstructed = false;
317  }
318  pcCurPic->isNeededForOutput = false;
319  }
320 
321  iterPic++;
322  }
323  }
324 
325 
326  if( ffwdDecoder.bitstreamFile )
327  {
328  pcDecLib->resetAccessUnitNals();
329  pcDecLib->resetAccessUnitApsNals();
330  }
331  }
332  ffwdDecoder.loopFiltered = ( nalu.m_nalUnitType == VVENC_NAL_UNIT_EOS );
333  if( nalu.m_nalUnitType == VVENC_NAL_UNIT_EOS )
334  {
335  pcDecLib->setFirstSliceInSequence( true );
336  }
337 
338  }
339  else if( ( bNewPicture || !*ffwdDecoder.bitstreamFile || nalu.m_nalUnitType == VVENC_NAL_UNIT_EOS ) && pcDecLib->getFirstSliceInSequence() )
340  {
341  pcDecLib->setFirstSliceInPicture( true );
342  }
343  }
344  }
345 
346  if( !bRet )
347  {
348  CHECK( bDecodeUntilPocFound, " decoding failed - check decodeBitstream2 parameter File: " << bitstreamFileName.c_str() );
349  if( ffwdDecoder.pcDecLib )
350  {
351  ffwdDecoder.pcDecLib->destroy();
352  ffwdDecoder.pcDecLib->deletePicBuffer();
353  delete ffwdDecoder.pcDecLib;
354  ffwdDecoder.pcDecLib = nullptr;
355  }
356  ffwdDecoder.loopFiltered = false;
357  ffwdDecoder.iPOCLastDisplay = -MAX_INT;
358 
359  if( ffwdDecoder.bytestream )
360  {
361  delete ffwdDecoder.bytestream;
362  ffwdDecoder.bytestream = nullptr;
363  }
364 
365  if( ffwdDecoder.bitstreamFile )
366  {
367  delete ffwdDecoder.bitstreamFile;
368  ffwdDecoder.bitstreamFile = nullptr;
369  }
370  }
371 
372  return bRet;
373 }

534  pic->isReconstructed = false;
535 
536  return pic;

540 void DecLib::executeLoopFilters()
541 {
542  if( !m_pic )
543  {
544  return; // nothing to deblock
545  }
546  CodingStructure& cs = *m_pic->cs;
547  DTRACE_UPDATE( g_trace_ctx, ( std::make_pair( "poc", cs.slice->poc ) ) );
548  DTRACE(g_trace_ctx, D_CRC, "reconstruction");
549  DTRACE_CRC(g_trace_ctx, D_CRC, cs, cs.getRecoBuf());
550 
551  ReshapeData& reshapeData = m_pic->reshapeData;
552  if (cs.sps->lumaReshapeEnable && reshapeData.getSliceReshaperInfo().sliceReshaperEnabled)
553  {
554  m_pic->getRecoBuf(COMP_Y).rspSignal( reshapeData.getInvLUT());
555 
556  DTRACE(g_trace_ctx, D_CRC, "LMCS");
557  DTRACE_CRC(g_trace_ctx, D_CRC, cs, cs.getRecoBuf());
558  }
559 
560 
561  // deblocking filter
562  m_cLoopFilter.loopFilterPic( cs, false );
563  CS::setRefinedMotionField(cs);
564  DTRACE(g_trace_ctx, D_CRC, "LoopFilter");
565  DTRACE_CRC(g_trace_ctx, D_CRC, cs, cs.getRecoBuf());
566 
567  if( cs.sps->saoEnabled )
568  {
569  m_cSAO.SAOProcess( cs, cs.picture->getSAO() );
570  DTRACE(g_trace_ctx, D_CRC, "SAO");
571  DTRACE_CRC(g_trace_ctx, D_CRC, cs, cs.getRecoBuf());
572  }
573 
574  if( cs.sps->alfEnabled )
575  {
576  m_cALF.m_ccAlfFilterParam = cs.slice->ccAlfFilterParam;
577 
578  // ALF decodes the differentially coded coefficients and stores them in the parameters structure.
579  // Code could be restructured to do directly after parsing. So far we just pass a fresh non-const
580  // copy in case the APS gets used more than once.
581  m_cALF.ALFProcess( cs );
582 
583  DTRACE( g_trace_ctx, D_CRC, "ALF" );
584  DTRACE_CRC( g_trace_ctx, D_CRC, cs, cs.getRecoBuf() );
585  DTRACE_PIC_COMP( D_REC_CB_LUMA_ALF, cs, cs.getRecoBuf(), COMP_Y );
586  DTRACE_PIC_COMP( D_REC_CB_CHROMA_ALF, cs, cs.getRecoBuf(), COMP_Cb );
587  DTRACE_PIC_COMP( D_REC_CB_CHROMA_ALF, cs, cs.getRecoBuf(), COMP_Cr );
588  }
589 
590  // TODO (VG): It looks like a piece of debugging code here, should be removed somehow
591  for (int i = 0; i < cs.pps->numSubPics && m_targetSubPicIdx; i++)
592  {
593  // keep target subpic samples untouched, for other subpics mask their output sample value to 0
594  if (i != m_targetSubPicIdx - 1)
595  {
596  SubPic SubPicNoUse = cs.pps->subPics[i];
