116  void calculateCombinedValues(const ChromaFormat chFmt, double &PSNRyuv, double &MSEyuv, const BitDepths &bitDepths)
117  {
118  MSEyuv = 0;
119  int scale = 0;
120 
121  int maximumBitDepth = bitDepths[CH_L];
122  for (uint32_t ch = 1; ch < MAX_NUM_CH; ch++)
123  {
124  if (bitDepths[ChannelType(ch)] > maximumBitDepth)
125  {
126  maximumBitDepth = bitDepths[ChannelType(ch)];
127  }
128  }
129 
130  const uint32_t maxval = 255 << (maximumBitDepth - 8);
131  const uint32_t numberValidComponents = getNumberValidComponents(chFmt);
132 
133  for (uint32_t comp=0; comp<numberValidComponents; comp++)
134  {
135  const ComponentID compID = ComponentID(comp);
136  const uint32_t csx = getComponentScaleX(compID, chFmt);
137  const uint32_t csy = getComponentScaleY(compID, chFmt);
138  const int scaleChan = (4>>(csx+csy));
139  const uint32_t bitDepthShift = 2 * (maximumBitDepth - bitDepths[toChannelType(compID)]); //*2 because this is a squared number
140 
141  const double channelMSE = (m_MSEyuvframe[compID] * double(1 << bitDepthShift)) / double(getNumPic());
142 
143  scale += scaleChan;
144  MSEyuv += scaleChan * channelMSE;
145  }
146 
147  MSEyuv /= double(scale); // i.e. divide by 6 for 4:2:0, 8 for 4:2:2 etc.
148  PSNRyuv = (MSEyuv == 0) ? 999.99 : 10.0 * log10((maxval * maxval) / MSEyuv);
149  }
