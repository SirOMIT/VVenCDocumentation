118  unsigned sigCtxIdAbs( int scanPos, const TCoeffSig* coeff, const int state )
119  {
120  const uint32_t posY = m_scan[scanPos].y;
121  const uint32_t posX = m_scan[scanPos].x;
122  const TCoeffSig* pData = coeff + posX + posY * m_width;
123  const int diag = posX + posY;
124  int numPos = 0;
125  int sumAbs = 0;
126 #define UPDATE(x) {int a=abs(x);sumAbs+=std::min(4+(a&1),a);numPos+=!!a;}
127  if( posX < m_width-1 )
128  {
129  UPDATE( pData[1] );
130  if( posX < m_width-2 )
131  {
132  UPDATE( pData[2] );
133  }
134  if( posY < m_height-1 )
135  {
136  UPDATE( pData[m_width+1] );
137  }
138  }
139  if( posY < m_height-1 )
140  {
141  UPDATE( pData[m_width] );
142  if( posY < m_height-2 )
143  {
144  UPDATE( pData[m_width<<1] );
145  }
146  }
147 #undef UPDATE
148 
149 
150  int ctxOfs = std::min((sumAbs+1)>>1, 3) + ( diag < 2 ? 4 : 0 );
151 
152  if( m_chType == CH_L )
153  {
154  ctxOfs += diag < 5 ? 4 : 0;
155  }
156 
157  m_tmplCpDiag = diag;
158  m_tmplCpSum1 = sumAbs - numPos;
159  return m_sigFlagCtxSet[std::max( 0, state-1 )]( ctxOfs );
160  }

313  int deriveModCoeff(int rightPixel, int belowPixel, int absCoeff, int bdpcm = 0)
314  {
315  if (absCoeff == 0)
316  return 0;
317 
318  int pred1, absBelow = abs(belowPixel), absRight = abs(rightPixel);
319 
320  int absCoeffMod = absCoeff;
321 
322  if (bdpcm == 0)
323  {
324  pred1 = std::max(absBelow, absRight);
325 
326  if (absCoeff == pred1)
327  {
328  absCoeffMod = 1;
329  }
330  else
331  {
332  absCoeffMod = absCoeff < pred1 ? absCoeff + 1 : absCoeff;
333  }
334  }
335 
336  return(absCoeffMod);

345  pred1 = std::max(absBelow, absRight);

355  absCoeffMod = absCoeff - (absCoeff <= pred1);
