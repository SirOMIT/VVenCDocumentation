89 void DecSlice::decompressSlice( Slice* slice, InputBitstream* bitstream )
90 {
91  const SPS* sps = slice->sps;
92  Picture* pic = slice->pic;
93  CABACReader& cabacReader = *m_CABACDecoder->getCABACReader();
94 
95  // setup coding structure
96  CodingStructure& cs = *pic->cs;
97  cs.slice = slice;
98  cs.sps = sps;
99  cs.pps = slice->pps;
100  memcpy(cs.alfAps, slice->alfAps, sizeof(cs.alfAps));
101 
102  cs.lmcsAps = slice->picHeader->lmcsAps;
103 
104  cs.pcv = slice->pps->pcv;
105  cs.chromaQpAdj = 0;
106 
107  cs.picture->resizeSAO(cs.pcv->sizeInCtus, 0);
108 
109  if (slice->sliceMap.ctuAddrInSlice[0] == 0)
110  {
111  cs.picture->resizeAlfCtuBuffers( cs.pcv->sizeInCtus );
112  }
113 
114  const unsigned numSubstreams = slice->getNumberOfSubstreamSizes() + 1;
115 
116  // init each couple {EntropyDecoder, Substream}
117  // Table of extracted substreams.
118  std::vector<InputBitstream*> ppcSubstreams( numSubstreams );
119  for( unsigned idx = 0; idx < numSubstreams; idx++ )
120  {
121  ppcSubstreams[idx] = bitstream->extractSubstream( idx+1 < numSubstreams ? ( slice->getSubstreamSize(idx) << 3 ) : bitstream->getNumBitsLeft() );
122  }
123 
124  const unsigned widthInCtus = cs.pcv->widthInCtus;
125  const bool wavefrontsEnabled = cs.sps->entropyCodingSyncEnabled;
126  const bool entryPointPresent = cs.sps->entryPointsPresent;
127 
128  cabacReader.initBitstream( ppcSubstreams[0] );
129  cabacReader.initCtxModels( *slice );
130 
131  DTRACE( g_trace_ctx, D_HEADER, "=========== POC: %d ===========\n", slice->poc );
132 
133  // for every CTU in the slice segment...
134  unsigned subStrmId = 0;
135  for( unsigned ctuIdx = 0; ctuIdx < slice->sliceMap.numCtuInSlice; ctuIdx++ )
136  {
137  const unsigned ctuRsAddr = slice->sliceMap.ctuAddrInSlice[ctuIdx];
138  const unsigned ctuXPosInCtus = ctuRsAddr % widthInCtus;
139  const unsigned ctuYPosInCtus = ctuRsAddr / widthInCtus;
140  const unsigned tileXPosInCtus = 0;
141  const unsigned tileYPosInCtus = 0;
142  const unsigned tileColWidth = widthInCtus;
143  const unsigned tileRowHeight = cs.pcv->heightInCtus;
144  const unsigned tileIdx = slice->pps->getTileIdx( Position(ctuXPosInCtus, ctuYPosInCtus));
145  const unsigned maxCUSize = sps->CTUSize;
146  Position pos( ctuXPosInCtus*maxCUSize, ctuYPosInCtus*maxCUSize) ;
147  UnitArea ctuArea(cs.area.chromaFormat, Area( pos.x, pos.y, maxCUSize, maxCUSize ) );
148  const SubPic& curSubPic = slice->pps->getSubPicFromPos(pos);
149  // padding/restore at slice level
150  if (slice->pps->numSubPics>=2 && curSubPic.treatedAsPic && ctuIdx==0)
151  {
152  for (int rlist = REF_PIC_LIST_0; rlist < NUM_REF_PIC_LIST_01; rlist++)
153  {
154  int n = slice->numRefIdx[((RefPicList)rlist)];
155  for (int idx = 0; idx < n; idx++)
156  {
157  Picture *refPic = slice->refPicList[rlist][idx];
158  if (!refPic->isSubPicBorderSaved)
159  {
160  THROW("no support");
161  refPic->isSubPicBorderSaved = (true);
162  }
163  }
164  }
165  }
166 
167  DTRACE_UPDATE( g_trace_ctx, std::make_pair( "ctu", ctuRsAddr ) );
168 
169  cabacReader.initBitstream( ppcSubstreams[subStrmId] );
170 
171  // set up CABAC contexts' state for this CTU
172  if( ctuXPosInCtus == tileXPosInCtus && ctuYPosInCtus == tileYPosInCtus )
173  {
174  if( ctuIdx != 0 ) // if it is the first CTU, then the entropy coder has already been reset
175  {
176  cabacReader.initCtxModels( *slice );
177  }
178  pic->cs->prevQP[0] = pic->cs->prevQP[1] = slice->sliceQp;
179  }
180  else if( ctuXPosInCtus == tileXPosInCtus && wavefrontsEnabled )
181  {
182  // Synchronize cabac probabilities with top CTU if it's available and at the start of a line.
183  if( ctuIdx != 0 ) // if it is the first CTU, then the entropy coder has already been reset
184  {
185  cabacReader.initCtxModels( *slice );
186  }
187  if( cs.getCURestricted( pos.offset(0, -1), pos, slice->independentSliceIdx, tileIdx, CH_L, TREE_D ) )
188  {
189  // Top is available, so use it.
190  cabacReader.getCtx() = m_entropyCodingSyncContextState;
191  }
192  pic->cs->prevQP[0] = pic->cs->prevQP[1] = slice->sliceQp;
193  }
194 
195  if ((cs.slice->sliceType != VVENC_I_SLICE || cs.sps->IBC) && ctuXPosInCtus == tileXPosInCtus)
196  {
197  cs.motionLut.lut.resize(0);
198  cs.motionLut.lutIbc.resize(0);
199  cs.resetIBCBuffer = true;
200  }
201 
202  cabacReader.coding_tree_unit( cs, ctuArea, pic->cs->prevQP, ctuRsAddr );
203 
204  m_pcCuDecoder->decompressCtu( cs, ctuArea );
205 
206  if( ctuXPosInCtus == tileXPosInCtus && wavefrontsEnabled )
207  {
208  m_entropyCodingSyncContextState = cabacReader.getCtx();
209  }
210 
211 
212  if( ctuIdx == slice->sliceMap.numCtuInSlice-1 )
213  {
214  unsigned binVal = cabacReader.terminating_bit();
215  CHECK( !binVal, "Expecting a terminating bit" );
216  cabacReader.remaining_bytes( false );
217  }
218  else if( ( ctuXPosInCtus + 1 == tileXPosInCtus + tileColWidth ) &&
219  ( ctuYPosInCtus + 1 == tileYPosInCtus + tileRowHeight || wavefrontsEnabled ) )
220  {
221  // The sub-stream/stream should be terminated after this CTU.
222  // (end of slice-segment, end of tile, end of wavefront-CTU-row)
223  unsigned binVal = cabacReader.terminating_bit();
224  CHECK( !binVal, "Expecting a terminating bit" );
225  if( entryPointPresent )
226  {
227  cabacReader.remaining_bytes( true );
228  subStrmId++;
229  }
230  }
231  if (slice->pps->numSubPics >= 2 && curSubPic.treatedAsPic && ctuIdx == (slice->sliceMap.numCtuInSlice - 1))
232  // for last Ctu in the slice
233  {
234  for (int rlist = REF_PIC_LIST_0; rlist < NUM_REF_PIC_LIST_01; rlist++)
235  {
236  int n = slice->numRefIdx[((RefPicList)rlist)];
237  for (int idx = 0; idx < n; idx++)
238  {
239  Picture *refPic = slice->refPicList[rlist][idx];
240  if (refPic->isSubPicBorderSaved)
241  {
242  THROW("no support");
243  refPic->isSubPicBorderSaved = (false);
244  }
245  }
246  }
247  }
248  }
249 
250  // deallocate all created substreams, including internal buffers.
251  for( auto substr: ppcSubstreams )
252  {
253  delete substr;
254  }
255 }
