83 void xPredIntraPlanar_Core( PelBuf& pDst, const CPelBuf& pSrc )
84 {
85  const uint32_t width = pDst.width;
86  const uint32_t height = pDst.height;
87  const uint32_t log2W = Log2(width);
88  const uint32_t log2H = Log2(height);
89 
90  int leftColumn[MAX_TB_SIZEY + 1], topRow[MAX_TB_SIZEY + 1], bottomRow[MAX_TB_SIZEY], rightColumn[MAX_TB_SIZEY];
91  const uint32_t offset = 1 << (log2W + log2H);
92 
93  // Get left and above reference column and row
94  for( int k = 0; k < width + 1; k++ )
95  {
96  topRow[k] = pSrc.at( k + 1, 0 );
97  }
98 
99  for( int k = 0; k < height + 1; k++ )
100  {
101  leftColumn[k] = pSrc.at( k + 1, 1 );
102  }
103 
104  // Prepare intermediate variables used in interpolation
105  int bottomLeft = leftColumn[height];
106  int topRight = topRow[width];
107 
108  for( int k = 0; k < width; k++ )
109  {
110  bottomRow[k] = bottomLeft - topRow[k];
111  topRow[k] = topRow[k] << log2H;
112  }
113 
114  // with some optimizations gcc-8 gives spurious "-Wmaybe-uninitialized" warnings here (says leftColumn would be uninitialized here)
115  GCC_WARNING_DISABLE_maybe_uninitialized
116  for( int k = 0; k < height; k++ )
117  {
118  rightColumn[k] = topRight - leftColumn[k];
119  leftColumn[k] = leftColumn[k] << log2W;
120  }
121  GCC_WARNING_RESET
122 
123  const uint32_t finalShift = 1 + log2W + log2H;
124  const uint32_t stride = pDst.stride;
125  Pel* pred = pDst.buf;
126  for( int y = 0; y < height; y++, pred += stride )
127  {
128  int horPred = leftColumn[y];
129 
130  for( int x = 0; x < width; x++ )
131  {
132  horPred += rightColumn[y];
133  topRow[x] += bottomRow[x];
134 
135  int vertPred = topRow[x];
136  pred[x] = ( ( horPred << log2H ) + ( vertPred << log2W ) + offset ) >> finalShift;
137  }
138  }
139 }

141 void IntraPredSampleFilter_Core(PelBuf& dstBuf, const CPelBuf& pSrc)
142 {
143  const int iWidth = dstBuf.width;
144  const int iHeight = dstBuf.height;
145 
146  const int scale = ((Log2(iWidth*iHeight) - 2) >> 2);
147  CHECK(scale < 0 || scale > 31, "PDPC: scale < 0 || scale > 31");
148 
149  for (int y = 0; y < iHeight; y++)
150  {
151  const int wT = 32 >> std::min(31, ((y << 1) >> scale));
152  const Pel left = pSrc.at(y + 1, 1);
153  for (int x = 0; x < iWidth; x++)
154  {
155  const int wL = 32 >> std::min(31, ((x << 1) >> scale));
156  const Pel top = pSrc.at(x + 1, 0);
157  const Pel val = dstBuf.at(x, y);
158  dstBuf.at(x, y) = val + ((wL * (left - val) + wT * (top - val) + 32) >> 6);
159  }
160  }
161 }

296 // Function for calculating DC value of the reference samples used in Intra prediction
297 //NOTE: Bit-Limit - 25-bit source
298 Pel IntraPrediction::xGetPredValDc( const CPelBuf& pSrc, const Size& dstSize )
299 {
300  CHECK( dstSize.width == 0 || dstSize.height == 0, "Empty area provided" );
301 
302  int idx, sum = 0;
303  Pel dcVal;
304  const int width = dstSize.width;
305  const int height = dstSize.height;
306  const auto denom = (width == height) ? (width << 1) : std::max(width,height);
307  const auto divShift = Log2(denom);
308  const auto divOffset = (denom >> 1);
309  const int off = m_ipaParam.multiRefIndex + 1;
310 
311 
312  if ( width >= height )
313  {
314  for( idx = 0; idx < width; idx++ )
315  {
316  sum += pSrc.at( off + idx, 0);
317  }
318  }
319  if ( width <= height )
320  {
321  for( idx = 0; idx < height; idx++ )
322  {
323  sum += pSrc.at( off + idx, 1);
324  }
325  }
326 
327  dcVal = (sum + divOffset) >> divShift;
328  return dcVal;
329 }

331 int IntraPrediction::getWideAngle( int width, int height, int predMode )
332 {
333  if ( predMode > DC_IDX && predMode <= VDIA_IDX )
334  {
335  int modeShift[] = { 0, 6, 10, 12, 14, 15 };
336  int deltaSize = abs(Log2(width) - Log2(height));
337  if (width > height && predMode < 2 + modeShift[deltaSize])
338  {
339  predMode += (VDIA_IDX - 1);
340  }
341  else if (height > width && predMode > VDIA_IDX - modeShift[deltaSize])
342  {
343  predMode -= (VDIA_IDX - 1);
344  }
345  }
346  return predMode;

404 // Function for initialization of intra prediction parameters
405 void IntraPrediction::initPredIntraParams(const CodingUnit& cu, const CompArea area, const SPS& sps)
406 {
407  const ComponentID compId = area.compID;
408  const ChannelType chType = toChannelType(compId);
409 
410  const bool useISP = NOT_INTRA_SUBPARTITIONS != cu.ispMode && isLuma( chType );
411 
412  const Size cuSize = Size( cu.blocks[compId].width, cu.blocks[compId].height );
413  const Size puSize = Size( area.width, area.height );
414  const Size& blockSize = useISP ? cuSize : puSize;
415  const int dirMode = CU::getFinalIntraMode(cu, chType);
416  const int predMode = getWideAngle( blockSize.width, blockSize.height, dirMode );
417 
418  m_ipaParam.isModeVer = predMode >= DIA_IDX;
419  m_ipaParam.multiRefIndex = isLuma (chType) ? cu.multiRefIdx : 0 ;
420  m_ipaParam.refFilterFlag = false;
421  m_ipaParam.interpolationFlag = false;
422  m_ipaParam.applyPDPC = (puSize.width >= MIN_TB_SIZEY && puSize.height >= MIN_TB_SIZEY) && m_ipaParam.multiRefIndex == 0;
423 
424  const int intraPredAngleMode = (m_ipaParam.isModeVer) ? predMode - VER_IDX : -(predMode - HOR_IDX);
425 
426 
427  int absAng = 0;
428  if (dirMode > DC_IDX && dirMode < NUM_LUMA_MODE) // intraPredAngle for directional modes
429  {
430  static const int angTable[32] = { 0, 1, 2, 3, 4, 6, 8, 10, 12, 14, 16, 18, 20, 23, 26, 29, 32, 35, 39, 45, 51, 57, 64, 73, 86, 102, 128, 171, 256, 341, 512, 1024 };
431  static const int invAngTable[32] = {
432  0, 16384, 8192, 5461, 4096, 2731, 2048, 1638, 1365, 1170, 1024, 910, 819, 712, 630, 565,
433  512, 468, 420, 364, 321, 287, 256, 224, 191, 161, 128, 96, 64, 48, 32, 16
434  }; // (512 * 32) / Angle
435 
436  const int absAngMode = abs(intraPredAngleMode);
437  const int signAng = intraPredAngleMode < 0 ? -1 : 1;
438  absAng = angTable [absAngMode];
439 
440  m_ipaParam.absInvAngle = invAngTable[absAngMode];
441  m_ipaParam.intraPredAngle = signAng * absAng;
442  if (intraPredAngleMode < 0)
443  {
444  m_ipaParam.applyPDPC = false;
445  }
446  else if (intraPredAngleMode > 0)
447  {
448  const int sideSize = m_ipaParam.isModeVer ? puSize.height : puSize.width;
449  const int maxScale = 2;
450 
451  m_ipaParam.angularScale = std::min(maxScale, floorLog2(sideSize) - (floorLog2(3 * m_ipaParam.absInvAngle - 2) - 8));
452  m_ipaParam.applyPDPC &= m_ipaParam.angularScale >= 0;
453  }
454  }
455 
456  // high level conditions and DC intra prediction
457  if( sps.spsRExt.intraSmoothingDisabled
458  || !isLuma( chType )
459  || useISP
460  || CU::isMIP( cu, chType ) //th remove this
461  || m_ipaParam.multiRefIndex
462  || DC_IDX == dirMode
463  )
464  {
465  }
466  else if (cu.bdpcmM[chType])
467  {
468  m_ipaParam.refFilterFlag = false;
469  }
470  else if (dirMode == PLANAR_IDX) // Planar intra prediction
471  {
472  m_ipaParam.refFilterFlag = puSize.width * puSize.height > 32 ? true : false;
473  }
474  else if (!useISP)// HOR, VER and angular modes (MDIS)
475  {
476  bool filterFlag = false;
477  {
478  const int diff = std::min<int>( abs( predMode - HOR_IDX ), abs( predMode - VER_IDX ) );
479  const int log2Size = (Log2(puSize.width * puSize.height) >> 1);
480  CHECK( log2Size >= MAX_INTRA_FILTER_DEPTHS, "Size not supported" );
481  filterFlag = (diff > m_aucIntraFilter[log2Size]);
482  }
483 
484  // Selelection of either ([1 2 1] / 4 ) refrence filter OR Gaussian 4-tap interpolation filter
485  if (filterFlag)
486  {
487  const bool isRefFilter = isIntegerSlope(absAng);
488  CHECK( puSize.width * puSize.height <= 32, "DCT-IF interpolation filter is always used for 4x4, 4x8, and 8x4 luma CB" );
489  m_ipaParam.refFilterFlag = isRefFilter;
490  m_ipaParam.interpolationFlag = !isRefFilter;
491  }
492  }
493 }

510 void IntraPrediction::xPredIntraAng( PelBuf& pDst, const CPelBuf& pSrc, const ChannelType channelType, const ClpRng& clpRng)
511 {
512  int width =int(pDst.width);
513  int height=int(pDst.height);
514 
515  const bool bIsModeVer = m_ipaParam.isModeVer;
516  const int multiRefIdx = m_ipaParam.multiRefIndex;
517  const int intraPredAngle = m_ipaParam.intraPredAngle;
518  const int absInvAngle = m_ipaParam.absInvAngle;
519 
520  Pel* refMain;
521  Pel* refSide;
522 
523  Pel refAbove[2 * MAX_CU_SIZE + 3 + 33 * MAX_REF_LINE_IDX];
524  Pel refLeft [2 * MAX_CU_SIZE + 3 + 33 * MAX_REF_LINE_IDX];
525 
526  // Initialize the Main and Left reference array.
527  if (intraPredAngle < 0)
528  {
529  memcpy(&refAbove[height],pSrc.buf,(width + 2 + multiRefIdx)*sizeof(Pel));
530  for (int y = 0; y <= height + 1 + multiRefIdx; y++)
531  {
532  refLeft[y + width] = pSrc.at(y, 1);
533  }
534  refMain = bIsModeVer ? refAbove + height : refLeft + width;
535  refSide = bIsModeVer ? refLeft + width : refAbove + height;
536 
537  // Extend the Main reference to the left.
538  int sizeSide = bIsModeVer ? height : width;
539  for (int k = -sizeSide; k <= -1; k++)
540  {
541  refMain[k] = refSide[std::min((-k * absInvAngle + 256) >> 9, sizeSide)];
542  }
543  }
544  else
545  {
546  memcpy(&refAbove[0], pSrc.buf, ((m_topRefLength)+multiRefIdx + 1) * sizeof(Pel));
547  for (int y = 0; y <= m_leftRefLength + multiRefIdx; y++)
548  {
549  refLeft[y] = pSrc.at(y, 1);
550  }
551 
552  refMain = bIsModeVer ? refAbove : refLeft;
553  refSide = bIsModeVer ? refLeft : refAbove;
554 
555  // Extend main reference to right using replication
556  const int log2Ratio = Log2(width) - Log2(height);
557  const int s = std::max<int>(0, bIsModeVer ? log2Ratio : -log2Ratio);
558  const int maxIndex = (multiRefIdx << s) + 2;
559  const int refLength = bIsModeVer ? m_topRefLength : m_leftRefLength;
560  const Pel val = refMain[refLength + multiRefIdx];
561  for (int z = 1; z <= maxIndex; z++)
562  {
563  refMain[refLength + multiRefIdx + z] = val;
564  }
565  }
566 
567  // swap width/height if we are doing a horizontal mode:
568  if (!bIsModeVer)
569  {
570  std::swap(width, height);
571  }
572  Pel tempArray[MAX_CU_SIZE*MAX_CU_SIZE];
573  const int dstStride = bIsModeVer ? pDst.stride : MAX_CU_SIZE;
574  Pel* pDstBuf = bIsModeVer ? pDst.buf : tempArray;
575 
576  // compensate for line offset in reference line buffers
577  refMain += multiRefIdx;
578  refSide += multiRefIdx;
579 
580  Pel* pDsty = pDstBuf;
581 
582  if( intraPredAngle == 0 ) // pure vertical or pure horizontal
583  {
584  if (m_ipaParam.applyPDPC)
585  {
586  const int scale = (Log2(width * height) - 2) >> 2;
587  IntraHorVerPDPC(pDsty,dstStride,refSide,width,height,scale,refMain,clpRng);
588  }
589  else
590  {
591  for( int y = 0; y < height; y++ )
592  {
593  memcpy(pDsty,&refMain[1],width*sizeof(Pel));
594  pDsty += dstStride;
595  }
596  }
597  }
598  else
599  {
600  if( !isIntegerSlope( abs( intraPredAngle ) ) )
601  {
602  int deltaPos = intraPredAngle * ( 1 + multiRefIdx );
603  if( isLuma( channelType ) )
604  {

615  {
616  const bool useCubicFilter = !m_ipaParam.interpolationFlag;
617

619  TFilterCoeff( 32 - ( deltaFract >> 1 ) ),
620  TFilterCoeff( 16 + ( deltaFract >> 1 ) ),
621  TFilterCoeff( ( deltaFract >> 1 ) ) };

631  p[2] = refMain[deltaInt + x + 2];
632  p[3] = refMain[deltaInt + x + 3];

638  }
639  }
640  }
641  else
642  {
643  IntraPredAngleLuma(pDstBuf, dstStride, refMain, width, height, deltaPos, intraPredAngle, nullptr, !m_ipaParam.interpolationFlag, clpRng);
644  }
645  }
646  else

648  IntraPredAngleChroma(pDstBuf,dstStride,refMain,width,height,deltaPos,intraPredAngle);
649  }
650  }
651  else
652  {
653  for (int y = 0, deltaPos = intraPredAngle * (1 + multiRefIdx); y<height; y++, deltaPos += intraPredAngle, pDsty += dstStride)
654  {
655  const int deltaInt = deltaPos >> 5;
656  // Just copy the integer samples

659  }
660 
661  if (m_ipaParam.applyPDPC)
662  {
663  pDsty = pDstBuf;
664  IntraAnglePDPC(pDsty,dstStride,refSide,width,height,m_ipaParam.angularScale,absInvAngle);

666  } // else
667 
668  // Flip the block if this is the horizontal mode

671  pDst.transposedFrom( CPelBuf( pDstBuf, dstStride, width, height) );

765

767  const int totalLeftUnits = (predHSize + (unitHeight - 1)) / unitHeight;
768  const int totalUnits = totalAboveUnits + totalLeftUnits + 1; //+1 for top-left
769 
770  if( m_lastArea != area || m_lastCh != chType )
771  {
772  m_lastCh = chType;
773  m_lastArea = area;
774  const int numAboveUnits = std::max<int>( tuWidth / unitWidth, 1 );
775  const int numLeftUnits = std::max<int>( tuHeight / unitHeight, 1 );
776  const int numAboveRightUnits = totalAboveUnits - numAboveUnits;
777  const int numLeftBelowUnits = totalLeftUnits - numLeftUnits;
778 
779  CHECK( numAboveUnits <= 0 || numLeftUnits <= 0 || numAboveRightUnits <= 0 || numLeftBelowUnits <= 0, "Size not supported" );
780 
781  // ----- Step 1: analyze neighborhood -----
782  const Position posLT = area;
783  const Position posRT = area.topRight();
784  const Position posLB = area.bottomLeft();
785 
786  m_numIntraNeighbor = 0;
787 
788  memset( m_neighborFlags, 0, totalUnits );

790  m_neighborFlags[totalLeftUnits] = isAboveLeftAvailable( cu, chType, posLT );
791  m_numIntraNeighbor += m_neighborFlags[totalLeftUnits] ? 1 : 0;

793  m_numIntraNeighbor += isAboveRightAvailable( cu, chType, posRT, numAboveRightUnits, unitWidth, (m_neighborFlags + totalLeftUnits + 1 + numAboveUnits) );
794  m_numIntraNeighbor += isLeftAvailable ( cu, chType, posLT, numLeftUnits, unitHeight, (m_neighborFlags + totalLeftUnits - 1) );
795  m_numIntraNeighbor += isBelowLeftAvailable ( cu, chType, posLB, numLeftBelowUnits, unitHeight, (m_neighborFlags + totalLeftUnits - 1 - numLeftUnits) );
796  }
797  // ----- Step 2: fill reference samples (depending on neighborhood) -----
798 
799  const Pel* srcBuf = recoBuf.buf;
800  const int srcStride = recoBuf.stride;
801  Pel* ptrDst = refBufUnfiltered;
802  const Pel* ptrSrc;
803  const Pel valueDC = 1 << (sps.bitDepths[ chType ] - 1);
804 
805 
806  if( m_numIntraNeighbor == 0 )
807  {
808  // Fill border with DC value
809  for (int j = 0; j <= predSize + multiRefIdx; j++) { ptrDst[j] = valueDC; }
810  for (int i = 0; i <= predHSize + multiRefIdx; i++) { ptrDst[i+predStride] = valueDC; }
811  }
812  else if( m_numIntraNeighbor == totalUnits )
813  {
814  // Fill top-left border and top and top right with rec. samples
815  ptrSrc = srcBuf - (1 + multiRefIdx) * srcStride - (1 + multiRefIdx);
816  for (int j = 0; j <= predSize + multiRefIdx; j++) { ptrDst[j] = ptrSrc[j]; }
817  for (int i = 0; i <= predHSize + multiRefIdx; i++)
818  {
819  ptrDst[i + predStride] = ptrSrc[i * srcStride];
820  }
821  }
822  else // reference samples are partially available
823  {
824  // Fill top-left sample(s) if available
825  ptrSrc = srcBuf - (1 + multiRefIdx) * srcStride - (1 + multiRefIdx);
826  ptrDst = refBufUnfiltered;
827  if (m_neighborFlags[totalLeftUnits])
828  {
829  ptrDst[0] = ptrSrc[0];
830  ptrDst[predStride] = ptrSrc[0];
831  for (int i = 1; i <= multiRefIdx; i++)
832  {
833  ptrDst[i] = ptrSrc[i];
834  ptrDst[i + predStride] = ptrSrc[i * srcStride];
835  }
836  }
837 
838  // Fill left & below-left samples if available (downwards)
839  ptrSrc += (1 + multiRefIdx) * srcStride;
840  ptrDst += (1 + multiRefIdx) + predStride;
841  for (int unitIdx = totalLeftUnits - 1; unitIdx > 0; unitIdx--)
842  {
843  if (m_neighborFlags[unitIdx])
844  {
845  for (int i = 0; i < unitHeight; i++)
846  {
847  ptrDst[i] = ptrSrc[i*srcStride];
848  }
849  }
850  ptrSrc += unitHeight * srcStride;
851  ptrDst += unitHeight;
852  }
853  // Fill last below-left sample(s)
854  if (m_neighborFlags[0])
855  {
856  int lastSample = (predHSize % unitHeight == 0) ? unitHeight : predHSize % unitHeight;
857  for (int i = 0; i < lastSample; i++)
858  {
859  ptrDst[i] = ptrSrc[i*srcStride];
860  }
861  }
862 
863  // Fill above & above-right samples if available (left-to-right)
864  ptrSrc = srcBuf - srcStride * (1 + multiRefIdx);
865  ptrDst = refBufUnfiltered + 1 + multiRefIdx;
866  for (int unitIdx = totalLeftUnits + 1; unitIdx < totalUnits - 1; unitIdx++)
867  {
868  if (m_neighborFlags[unitIdx])
869  {
870  memcpy(ptrDst,ptrSrc,unitWidth*sizeof(Pel));
871  }
872  ptrSrc += unitWidth;
873  ptrDst += unitWidth;
874  }
875  // Fill last above-right sample(s)
876  if (m_neighborFlags[totalUnits - 1])
877  {
878  int lastSample = (predSize % unitWidth == 0) ? unitWidth : predSize % unitWidth;
879  memcpy(ptrDst,ptrSrc,lastSample*sizeof(Pel));
880  }
881 
882  // pad from first available down to the last below-left
883  ptrDst = refBufUnfiltered;
884  int lastAvailUnit = 0;
885  if (!m_neighborFlags[0])
886  {
887  int firstAvailUnit = 1;
888  while (firstAvailUnit < totalUnits && !m_neighborFlags[firstAvailUnit])
889  {
890  firstAvailUnit++;
891  }
892 
893  // first available sample
894  int firstAvailRow = -1;
895  int firstAvailCol = 0;
896  if (firstAvailUnit < totalLeftUnits)
897  {
898  firstAvailRow = (totalLeftUnits - firstAvailUnit) * unitHeight + multiRefIdx;
899  }
900  else if (firstAvailUnit == totalLeftUnits)
901  {
902  firstAvailRow = multiRefIdx;
903  }
904  else
905  {
906  firstAvailCol = (firstAvailUnit - totalLeftUnits - 1) * unitWidth + 1 + multiRefIdx;
907  }

909 
910  // last sample below-left (n.a.)
911  int lastRow = predHSize + multiRefIdx;
912 
913  // fill left column
914  for (int i = lastRow; i > firstAvailRow; i--)
915  {
916  ptrDst[i + predStride] = firstAvailSample;
917  }
918  // fill top row
919  if (firstAvailCol > 0)
920  {
921  for (int j = 0; j < firstAvailCol; j++)
922  {
923  ptrDst[j] = firstAvailSample;
924  }
925  }
926  lastAvailUnit = firstAvailUnit;
927  }
928 
929  // pad all other reference samples.
930  int currUnit = lastAvailUnit + 1;
931  while (currUnit < totalUnits)
932  {
933  if (!m_neighborFlags[currUnit]) // samples not available
934  {
935  // last available sample
936  int lastAvailRow = -1;
937  int lastAvailCol = 0;
938  if (lastAvailUnit < totalLeftUnits)
939  {
940  lastAvailRow = (totalLeftUnits - lastAvailUnit - 1) * unitHeight + multiRefIdx + 1;
941  }
942  else if (lastAvailUnit == totalLeftUnits)
943  {
944  lastAvailCol = multiRefIdx;
945  }
946  else
947  {
948  lastAvailCol = (lastAvailUnit - totalLeftUnits) * unitWidth + multiRefIdx;
949  }
950  const Pel lastAvailSample = ptrDst[lastAvailRow < 0 ? lastAvailCol : lastAvailRow + predStride];
951 
952  // fill current unit with last available sample
953  if (currUnit < totalLeftUnits)
954  {
955  for (int i = lastAvailRow - 1; i >= lastAvailRow - unitHeight; i--)
956  {
957  ptrDst[i + predStride] = lastAvailSample;
958  }
959  }
960  else if (currUnit == totalLeftUnits)
961  {
962  for (int i = 0; i < multiRefIdx + 1; i++)
963  {
964  ptrDst[i + predStride] = lastAvailSample;
965  }
966  for (int j = 0; j < multiRefIdx + 1; j++)
967  {
968  ptrDst[j] = lastAvailSample;
969  }
970  }
971  else
972  {
973  int numSamplesInUnit = (currUnit == totalUnits - 1) ? ((predSize % unitWidth == 0) ? unitWidth : predSize % unitWidth) : unitWidth;
974  for (int j = lastAvailCol + 1; j <= lastAvailCol + numSamplesInUnit; j++)
975  {
976  ptrDst[j] = lastAvailSample;
977  }
978  }
979  }
980  lastAvailUnit = currUnit;
981  currUnit++;
982  }
983  }
