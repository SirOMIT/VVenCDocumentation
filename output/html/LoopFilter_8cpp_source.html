117 static inline void xBilinearFilter( Pel* srcP, Pel* srcQ, ptrdiff_t offset, int refMiddle, int refP, int refQ, int numberPSide, int numberQSide, const int* dbCoeffsP, const int* dbCoeffsQ, int tc )
118 {
119  int src;
120  const char tc7[7] = { 6, 5, 4, 3, 2, 1, 1 };
121  const char tc3[3] = { 6, 4, 2 };
122  const char *tcP = ( numberPSide == 3 ) ? tc3 : tc7;
123  const char *tcQ = ( numberQSide == 3 ) ? tc3 : tc7;
124 
125  for( int pos = 0; pos < numberPSide; pos++ )
126  {
127  src = srcP[-offset * pos];
128  int cvalue = ( tc * tcP[pos] ) >> 1;
129  srcP[-offset * pos] = Clip3( src - cvalue, src + cvalue, ( ( refMiddle * dbCoeffsP[pos] + refP * ( 64 - dbCoeffsP[pos] ) + 32 ) >> 6 ) );
130  }
131 
132  for( int pos = 0; pos < numberQSide; pos++ )
133  {
134  src = srcQ[ offset * pos];
135  int cvalue = ( tc * tcQ[pos] ) >> 1;
136  srcQ[ offset * pos] = Clip3( src - cvalue, src + cvalue, ( ( refMiddle * dbCoeffsQ[pos] + refQ * ( 64 - dbCoeffsQ[pos] ) + 32 ) >> 6 ) );

140 void xFilteringPandQCore( Pel* src, ptrdiff_t step, const ptrdiff_t offset, int numberPSide, int numberQSide, int tc )
141 {
142  CHECK( numberPSide <= 3 && numberQSide <= 3, "Short filtering in long filtering function" );
143  CHECK( numberPSide != 3 && numberPSide != 5 && numberPSide != 7, "invalid numberPSide" );
144  CHECK( numberQSide != 3 && numberQSide != 5 && numberQSide != 7, "invalid numberQSide" );
145 
146  const int* dbCoeffsP = numberPSide == 7 ? dbCoeffs7 : ( numberPSide == 5 ) ? dbCoeffs5 : dbCoeffs3;
147  const int* dbCoeffsQ = numberQSide == 7 ? dbCoeffs7 : ( numberQSide == 5 ) ? dbCoeffs5 : dbCoeffs3;
148 
149  for( int i = 0; i < DEBLOCK_SMALLEST_BLOCK / 2; i++ )
150  {
151  Pel* srcP = src + step * i - offset;
152  Pel* srcQ = src + step * i;
153 
154 
155  int refP = 0;
156  int refQ = 0;
157  int refMiddle = 0;
158 
159  switch( numberPSide )
160  {
161  case 7: refP = ( srcP[-6 * offset] + srcP[-7 * offset] + 1 ) >> 1; break;
162  case 3: refP = ( srcP[-2 * offset] + srcP[-3 * offset] + 1 ) >> 1; break;
163  case 5: refP = ( srcP[-4 * offset] + srcP[-5 * offset] + 1 ) >> 1; break;
164  }
165 
166  switch( numberQSide )
167  {
168  case 7: refQ = ( srcQ[6 * offset] + srcQ[7 * offset] + 1 ) >> 1; break;
169  case 3: refQ = ( srcQ[2 * offset] + srcQ[3 * offset] + 1 ) >> 1; break;
170  case 5: refQ = ( srcQ[4 * offset] + srcQ[5 * offset] + 1 ) >> 1; break;
171  }
172 
173  if( numberPSide == numberQSide )
174  {
175  if( numberPSide == 5 )
176  {
177  refMiddle = ( 2 * ( srcP[0] + srcQ[0] + srcP[-offset] + srcQ[offset] + srcP[-2 * offset] + srcQ[2 * offset] ) + srcP[-3 * offset] + srcQ[3 * offset] + srcP[-4 * offset] + srcQ[4 * offset] + 8 ) >> 4;
178  }
179  else
180  {
181  refMiddle = ( 2 * ( srcP[0] + srcQ[0] ) + srcP[-offset] + srcQ[offset] + srcP[-2 * offset] + srcQ[2 * offset] + srcP[-3 * offset] + srcQ[3 * offset] + srcP[-4 * offset] + srcQ[4 * offset] + srcP[-5 * offset] + srcQ[5 * offset] + +srcP[-6 * offset] + srcQ[6 * offset] + 8 ) >> 4;
182  }
183  }
184  else
185  {
186  Pel* srcPt = srcP;
187  Pel* srcQt = srcQ;
188  ptrdiff_t offsetP = -offset;
189  ptrdiff_t offsetQ = offset;
190 
191  int newNumberQSide = numberQSide;
192  int newNumberPSide = numberPSide;
193 
194  if( numberQSide > numberPSide )
195  {
196  std::swap( srcPt, srcQt );
197  std::swap( offsetP, offsetQ );
198  newNumberQSide = numberPSide;
199  newNumberPSide = numberQSide;
200  }
201 
202  if( newNumberPSide == 7 && newNumberQSide == 5 )
203  {
204  refMiddle = ( 2 * ( srcP[0] + srcQ[0] + srcP[-offset] + srcQ[offset] ) + srcP[-2 * offset] + srcQ[2 * offset] + srcP[-3 * offset] + srcQ[3 * offset] + srcP[-4 * offset] + srcQ[4 * offset] + srcP[-5 * offset] + srcQ[5 * offset] + 8 ) >> 4;
205  }
206  else if( newNumberPSide == 7 && newNumberQSide == 3 )
207  {
208  refMiddle = ( 2 * ( srcPt[0] + srcQt[0] ) + srcQt[0] + 2 * ( srcQt[offsetQ] + srcQt[2 * offsetQ] ) + srcPt[offsetP] + srcQt[offsetQ] + srcPt[2 * offsetP] + srcPt[3 * offsetP] + srcPt[4 * offsetP] + srcPt[5 * offsetP] + srcPt[6 * offsetP] + 8 ) >> 4;
209  }
210  else //if (newNumberPSide == 5 && newNumberQSide == 3)
211  {
212  refMiddle = ( srcP[0] + srcQ[0] + srcP[-offset] + srcQ[offset] + srcP[-2 * offset] + srcQ[2 * offset] + srcP[-3 * offset] + srcQ[3 * offset] + 4 ) >> 3;
213  }
214  }
215 
216  xBilinearFilter( srcP, srcQ, offset, refMiddle, refP, refQ, numberPSide, numberQSide, dbCoeffsP, dbCoeffsQ, tc );
217  }

234 void xPelFilterLumaCorePel( Pel* piSrc, const ptrdiff_t iOffset, const int tc, const bool sw,const int iThrCut, const bool bFilterSecondP, const bool bFilterSecondQ, const ClpRng& clpRng )
235 {
236  const Pel m0 = piSrc[-4 * iOffset];
237  const Pel m1 = piSrc[-3 * iOffset];
238  const Pel m2 = piSrc[-2 * iOffset];
239  const Pel m3 = piSrc[-1 * iOffset];
240  const Pel m4 = piSrc[ 0 ];
241  const Pel m5 = piSrc[ 1 * iOffset];
242  const Pel m6 = piSrc[ 2 * iOffset];
243  const Pel m7 = piSrc[ 3 * iOffset];
244 
245  const char tc3[3] = { 3, 2, 1 };
246 
247  if (sw)
248  {
249  piSrc[-3 * iOffset ] = Clip3( m1 - tc3[2] * tc, m1 + tc3[2] * tc, ( 2*m0 + 3*m1 + m2 + m3 + m4 + 4 ) >> 3 );
250  piSrc[-2 * iOffset ] = Clip3( m2 - tc3[1] * tc, m2 + tc3[1] * tc, ( m1 + m2 + m3 + m4 + 2 ) >> 2 );
251  piSrc[-1 * iOffset ] = Clip3( m3 - tc3[0] * tc, m3 + tc3[0] * tc, ( m1 + 2*m2 + 2*m3 + 2*m4 + m5 + 4 ) >> 3 );
252  piSrc[ 0 ] = Clip3( m4 - tc3[0] * tc, m4 + tc3[0] * tc, ( m2 + 2*m3 + 2*m4 + 2*m5 + m6 + 4 ) >> 3 );
253  piSrc[ 1 * iOffset ] = Clip3( m5 - tc3[1] * tc, m5 + tc3[1] * tc, ( m3 + m4 + m5 + m6 + 2 ) >> 2 );
254  piSrc[ 2 * iOffset ] = Clip3( m6 - tc3[2] * tc, m6 + tc3[2] * tc, ( m3 + m4 + m5 + 3*m6 + 2*m7 + 4 ) >> 3 );
255  }
256  else
257  {
258  /* Weak filter */
259  int delta = ( 9 * ( m4 - m3 ) - 3 * ( m5 - m2 ) + 8 ) >> 4;
260 
261  if ( abs(delta) < iThrCut )
262  {
263  delta = Clip3( -tc, tc, delta );
264  const int tc2 = tc >> 1;
265 
266  piSrc[-iOffset * 1] = ClipPel( m3 + delta, clpRng);
267  if( bFilterSecondP )
268  {
269  const int delta1 = Clip3( -tc2, tc2, ( ( ( ( m1 + m3 + 1 ) >> 1 ) - m2 + delta ) >> 1 ) );
270  piSrc[-iOffset * 2] = ClipPel( m2 + delta1, clpRng);
271  }
272 
273  piSrc[0] = ClipPel( m4 - delta, clpRng);
274  if( bFilterSecondQ )
275  {
276  const int delta2 = Clip3( -tc2, tc2, ( ( ( ( m6 + m4 + 1 ) >> 1 ) - m5 - delta ) >> 1 ) );
277  piSrc[iOffset] = ClipPel( m5 + delta2, clpRng);
278  }
279  }
280  }

620 template<DeblockEdgeDir edgeDir>
621 void LoopFilter::xDeblockArea( const CodingStructure& cs, const UnitArea& area, const ChannelType chType, PelUnitBuf& picRecoBuf ) const
622 {
623  if( cs.slice->deblockingFilterDisable )
624  {
625  return;
626  }
627 
628  const PreCalcValues& pcv = *cs.pcv;
629 
630  const Area lumaArea = area.Y().valid() ? static_cast<const Area&>( area.Y() ) : Area( recalcPosition( pcv.chrFormat, CH_C, CH_L, area.chromaPos() ), recalcSize( pcv.chrFormat, CH_C, CH_L, area.chromaSize() ) );
631 
632  bool doLuma = chType == MAX_NUM_CH || isLuma ( chType );
633  bool doChroma = ( chType == MAX_NUM_CH || isChroma( chType ) ) && pcv.chrFormat != CHROMA_400;
634 
635  static constexpr int incx = 4;
636  static constexpr int incy = 4;
637 
638  const int csx = getChannelTypeScaleX( CH_C, pcv.chrFormat );
639  const int csy = getChannelTypeScaleY( CH_C, pcv.chrFormat );
640 
641  const UnitScale scale = cs.getScaling( UnitScale::LF_PARAM_MAP, CH_L );
642 
643  const int lfpPtrLOffset = scale.scaleHor( incx );
644 
645  const LoopFilterParam* lfpPtr = cs.picture->cs->getLFPMapPtr( edgeDir );
646  ptrdiff_t lfpStride = cs.picture->cs->getLFPMapStride();
647  OFFSET( lfpPtr, lfpStride, scale.scaleHor( lumaArea.x ), scale.scaleVer( lumaArea.y ) );
648 
649  for( int dy = 0; dy < lumaArea.height; dy += incy )
650  {
651  LoopFilterParam const* lineLfpPtr = lfpPtr;
652 
653  const int dyInCtu = doChroma ? ( ( area.chromaPos().y + ( dy >> csy ) ) & ( pcv.maxCUSizeMask >> csy ) ) : 0;
654 
655  for( int dx = 0; dx < lumaArea.width; dx += incx )
656  {
657  if( doLuma && BsGet( lineLfpPtr->bs, COMP_Y ) )
658  {
659  xEdgeFilterLuma<edgeDir>( cs, lumaArea.pos().offset( dx, dy ), *lineLfpPtr, picRecoBuf );
660  }
661 
662  const int dxInCtu = doChroma ? ( ( area.chromaPos().x + ( dx >> csx ) ) & ( pcv.maxCUSizeMask >> csx ) ) : 0;
663 
664  if( doChroma
665  && ( ( edgeDir == EDGE_VER && ( dxInCtu & ( DEBLOCK_SMALLEST_BLOCK - 1 ) ) == 0 )
666  || ( edgeDir == EDGE_HOR && ( dyInCtu & ( DEBLOCK_SMALLEST_BLOCK - 1 ) ) == 0 ) )
667  && ( BsGet( lineLfpPtr->bs, COMP_Cb ) | BsGet( lineLfpPtr->bs, COMP_Cr ) ) )
668  {
669  xEdgeFilterChroma<edgeDir>( cs, area.chromaPos().offset( dx >> csx, dy >> csy ), *lineLfpPtr, picRecoBuf );
670  }
671 
672  OFFSETX( lineLfpPtr, lfpStride, lfpPtrLOffset );
673  }
674 
675  OFFSETY( lfpPtr, lfpStride, scale.scaleVer( incy ) );
676  }

770 void LoopFilter::calcFilterStrengths( const CodingUnit& cu, bool clearLF )
771 {
772  if( cu.slice->deblockingFilterDisable )
773  {
774  return;
775  }
776 
777  const PreCalcValues& pcv = *cu.cs->pcv;
778  const Area area = cu.blocks[cu.chType];
779 
780  bool horEdgeFilter = false;
781  bool verEdgeFilter = false;
782  int numHorVirBndry = 0;
783  int numVerVirBndry = 0;
784  int horVirBndryPos[] = { 0, 0, 0 };
785  int verVirBndryPos[] = { 0, 0, 0 };
786 
787  const uint8_t channelScaleX = getChannelTypeScaleX( cu.chType, cu.chromaFormat );
788  const uint8_t channelScaleY = getChannelTypeScaleY( cu.chType, cu.chromaFormat );
789 
790  if( clearLF )
791  {
792  const unsigned uiPelsInPartX = pcv.minCUSize >> channelScaleX;
793  const unsigned uiPelsInPartY = pcv.minCUSize >> channelScaleY;
794  LoopFilterParam* lfpPtrH = cu.cs->picture->cs->getLFPMapPtr( EDGE_HOR );
795  LoopFilterParam* lfpPtrV = cu.cs->picture->cs->getLFPMapPtr( EDGE_VER );
796  ptrdiff_t lfpStride = cu.cs->picture->cs->getLFPMapStride();
797  const Position lfpPos = cu.cs->picture->cs->getScaling( UnitScale::LF_PARAM_MAP, cu.chType ).scale( area.pos() );
798  OFFSET( lfpPtrH, lfpStride, lfpPos.x, lfpPos.y );
799  OFFSET( lfpPtrV, lfpStride, lfpPos.x, lfpPos.y );
800 
801  const size_t bufLen = ( area.width / uiPelsInPartX ) * sizeof( LoopFilterParam );
802  for( int y = 0; y < area.height; y += uiPelsInPartY )
803  {
804  memset( lfpPtrH, 0, bufLen );
805  memset( lfpPtrV, 0, bufLen );
806 
807  INCY( lfpPtrH, lfpStride );
808  INCY( lfpPtrV, lfpStride );
809  }
810  }
811 
812  const bool isCuCrossedByVirtualBoundaries = isCrossedByVirtualBoundaries( cu.cs->sps,
813  Area( area.x << channelScaleX, area.y << channelScaleY, area.width << channelScaleX, area.height << channelScaleY ),
814  numHorVirBndry, numVerVirBndry,
815  horVirBndryPos, verVirBndryPos );
816  if( isCuCrossedByVirtualBoundaries )
817  {
818  CHECK( numHorVirBndry >= (int)( sizeof(horVirBndryPos) / sizeof(horVirBndryPos[0]) ), "Too many virtual boundaries" );
819  CHECK( numHorVirBndry >= (int)( sizeof(verVirBndryPos) / sizeof(verVirBndryPos[0]) ), "Too many virtual boundaries" );
820  }
821 
822 
823  static constexpr int subBlockSize = 8;
824  LFCUParam stLFCUParam { xGetLoopfilterParam( cu ) };
825  const UnitScale scaling = cu.cs->getScaling( UnitScale::LF_PARAM_MAP, cu.chType );
826  // for SUBPU ATMVP and Affine, more PU deblocking needs to be found, for ISP the chroma block will be deferred to the last luma block,
827  // so the processing order is different. For all other cases the boundary strenght can be directly obtained in the TU loop.
828  const bool refineBs = ( cu.mergeFlag && cu.mergeType == MRG_TYPE_SUBPU_ATMVP ) || cu.affine || cu.ispMode;
829 
830  const int maskBlkX = ~( ( 1 << scaling.posx ) - 1 );
831  const int maskBlkY = ~( ( 1 << scaling.posy ) - 1 );
832 
833  for( const TransformUnit* currTU = cu.firstTU; currTU; currTU = currTU->next )
834  {
835  const Area& areaTu = currTU->blocks[currTU->chType];
836 
837  verEdgeFilter = ( areaTu.x & maskBlkX ) == area.x ? stLFCUParam.leftEdge : true;
838  horEdgeFilter = ( areaTu.y & maskBlkY ) == area.y ? stLFCUParam.topEdge : true;
839 
840  if( isCuCrossedByVirtualBoundaries )
841  {
842  xDeriveEdgefilterParam( Position( ( areaTu.x & maskBlkX ) << channelScaleX, ( areaTu.y & maskBlkY ) << channelScaleY ),
843  numVerVirBndry, numHorVirBndry,
844  verVirBndryPos, horVirBndryPos,
845  verEdgeFilter, horEdgeFilter );
846  }
847 
848  xSetMaxFilterLengthPQFromTransformSizes<EDGE_VER>( cu, *currTU, verEdgeFilter, !refineBs );
849  xSetMaxFilterLengthPQFromTransformSizes<EDGE_HOR>( cu, *currTU, horEdgeFilter, !refineBs );
850  }
851 
852  if( !refineBs ) return;
853 
854  if( ( cu.mergeFlag && cu.mergeType == MRG_TYPE_SUBPU_ATMVP ) || cu.affine )
855  {
856  CHECK( cu.chType != CH_L, "This path is only valid for single tree blocks!" );
857 
858  for( int off = subBlockSize; off < area.width; off += subBlockSize )
859  {
860  const Area mvBlockV( cu.Y().x + off, cu.Y().y, subBlockSize, cu.Y().height );
861  verEdgeFilter = true;
862  if( isCuCrossedByVirtualBoundaries )
863  {
864  xDeriveEdgefilterParam( mvBlockV,
865  numVerVirBndry, 0,
866  verVirBndryPos, horVirBndryPos,
867  verEdgeFilter, horEdgeFilter );
868  }
869 
870  xSetEdgeFilterInsidePu<EDGE_VER>( cu, mvBlockV, verEdgeFilter );
871  }
872 
873  xSetMaxFilterLengthPQForCodingSubBlocks<EDGE_VER>( cu );
874 
875  for( int off = subBlockSize; off < area.height; off += subBlockSize )
876  {
877  const Area mvBlockH( cu.Y().x, cu.Y().y + off, cu.Y().width, subBlockSize );
878  horEdgeFilter = true;
879  if( isCuCrossedByVirtualBoundaries )
880  {
881  xDeriveEdgefilterParam( mvBlockH,
882  0, numHorVirBndry,
883  verVirBndryPos, horVirBndryPos,
884  verEdgeFilter, horEdgeFilter );
885  }
886 
887  xSetEdgeFilterInsidePu<EDGE_HOR>( cu, mvBlockH, horEdgeFilter );
888  }
889 
890  xSetMaxFilterLengthPQForCodingSubBlocks<EDGE_HOR>( cu );
891  }
892 
893  const unsigned uiPelsInPartX = pcv.minCUSize >> channelScaleX;
894  const unsigned uiPelsInPartY = pcv.minCUSize >> channelScaleY;
895  const Position lfpPos = scaling.scale( area.pos() );
896 
897  const CodingUnit* cuP = CU::getLeft( cu );
898  const ChannelType chType = cu.chType;
899 
900  {
901  LoopFilterParam* lfpPtrV = cu.cs->picture->cs->getLFPMapPtr( EDGE_VER );
902  ptrdiff_t lfpStride = cu.cs->picture->cs->getLFPMapStride();
903  OFFSET( lfpPtrV, lfpStride, lfpPos.x, lfpPos.y );
904 
905  for( int y = 0; y < area.height; y += uiPelsInPartY )
906  {
907  LoopFilterParam* lineLfpPtrV = lfpPtrV;
908 
909  cuP = !cuP || cuP->blocks[chType].y + cuP->blocks[chType].height > area.y + y ? cuP : cu.cs->getCU( Position{ area.x - 1, area.y + y }, chType, TREE_D );
910 
911  for( int x = 0; x < area.width; x += uiPelsInPartX )
912  {
913  if( lineLfpPtrV->filterEdge( chType ) ) xGetBoundaryStrengthSingle<EDGE_VER>( *lineLfpPtrV, cu, Position{ area.x + x, area.y + y }, x ? cu : *cuP );
914 
915  lineLfpPtrV->bs &= ~BsSet( 3, MAX_NUM_COMP );
916  INCX( lineLfpPtrV, lfpStride );
917  }
918 
919  INCY( lfpPtrV, lfpStride );
920  }
921  }
922 
923  cuP = CU::getAbove( cu );
924 
925  {
926  LoopFilterParam* lfpPtrH = cu.cs->picture->cs->getLFPMapPtr( EDGE_HOR );
927  ptrdiff_t lfpStride = cu.cs->picture->cs->getLFPMapStride();
928  OFFSET( lfpPtrH, lfpStride, lfpPos.x, lfpPos.y );
929 
930  for( int y = 0; y < area.height; y += uiPelsInPartY )
931  {
932  LoopFilterParam* lineLfpPtrH = lfpPtrH;
933 
934  for( int x = 0; x < area.width; x += uiPelsInPartX )
935  {
936  cuP = ( y || ( cuP && cuP->blocks[chType].x + cuP->blocks[chType].width > area.x + x ) ) ? cuP : cu.cs->getCU( Position{ area.x + x, area.y - 1 }, chType, TREE_D );
937 
938  if( lineLfpPtrH->filterEdge( chType ) ) xGetBoundaryStrengthSingle<EDGE_HOR>( *lineLfpPtrH, cu, Position{ area.x + x, area.y + y }, y ? cu : *cuP );
939 
940  lineLfpPtrH->bs &= ~BsSet( 3, MAX_NUM_COMP );
941  INCX( lineLfpPtrH, lfpStride );
942  }
943 
944  INCY( lfpPtrH, lfpStride );
945  }

969
