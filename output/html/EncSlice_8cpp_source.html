265 void EncSlice::xInitSliceLambdaQP( Slice* slice, int gopId )
266 {
267  const vvencGOPEntry* gopList = m_pcEncCfg->m_GOPList;
268 
269  // pre-compute lambda and qp
270  int iQP, adaptedLumaQP = -1;
271  double dQP = xGetQPForPicture( slice, gopId );
272  double dLambda = xCalculateLambda( slice, gopId, slice->depth, dQP, dQP, iQP );
273  int sliceChromaQpOffsetIntraOrPeriodic[ 2 ] = { m_pcEncCfg->m_sliceChromaQpOffsetIntraOrPeriodic[ 0 ], m_pcEncCfg->m_sliceChromaQpOffsetIntraOrPeriodic[ 1 ] };
274 
275  slice->pic->picVisActY = 0.0; // to allow reusing calculated luma visual activity for rate control
276 
277  if (slice->pps->sliceChromaQpFlag && m_pcEncCfg->m_usePerceptQPA &&
278  ((slice->isIntra() && !slice->sps->IBC) || (m_pcEncCfg->m_sliceChromaQpOffsetPeriodicity > 0 && (slice->poc % m_pcEncCfg->m_sliceChromaQpOffsetPeriodicity) == 0)))
279  {
280  adaptedLumaQP = BitAllocation::applyQPAdaptationChroma (slice, m_pcEncCfg, iQP, *m_LineEncRsrc[ 0 ]->m_encCu.getQpPtr(),
281  sliceChromaQpOffsetIntraOrPeriodic, &slice->pic->picVisActY); // adapts sliceChromaQpOffsetIntraOrPeriodic[]
282  }
283  if (m_pcEncCfg->m_usePerceptQPA)
284  {
285  const bool rcIsFirstPassOf2 = (m_pcEncCfg->m_RCTargetBitrate == 0 && slice->pps->useDQP && (m_pcEncCfg->m_usePerceptQPATempFiltISlice == 2) ? m_pcEncCfg->m_RCNumPasses == 2 && !m_pcRateCtrl->rcIsFinalPass : false);
286  uint32_t startCtuTsAddr = slice->sliceMap.ctuAddrInSlice[0];
287  uint32_t boundingCtuTsAddr = slice->pic->cs->pcv->sizeInCtus;
288 
289  if ((m_pcEncCfg->m_RCNumPasses == 2) && m_pcRateCtrl->rcIsFinalPass && slice->pps->useDQP && (m_pcEncCfg->m_usePerceptQPATempFiltISlice == 2) && slice->isIntra() && (boundingCtuTsAddr > startCtuTsAddr))
290  {
291  const int nCtu = int (boundingCtuTsAddr - startCtuTsAddr);
292  const int offs = (slice->poc / m_pcEncCfg->m_IntraPeriod) * ((nCtu + 1) >> 1);
293  std::vector<uint8_t>& ctuQPMem = *m_pcRateCtrl->getIntraPQPAStats(); // unpack pass-1 red. QPs
294  std::vector<int>& ctuPumpRedQP = *m_LineEncRsrc[0]->m_encCu.getQpPtr();
295 
296  if ((ctuPumpRedQP.size() >= nCtu) && (ctuQPMem.size() >= offs + ((nCtu + 1) >> 1)))
297  {
298  for (uint32_t ctuTsAddr = startCtuTsAddr; ctuTsAddr < boundingCtuTsAddr; ctuTsAddr++)
299  {
300  const uint32_t ctuRsAddr = /*tileMap.getCtuBsToRsAddrMap*/ (ctuTsAddr);
301 
302  ctuPumpRedQP[ctuRsAddr] = int ((ctuRsAddr & 1) ? ctuQPMem[offs + (ctuRsAddr >> 1)] >> 4 : ctuQPMem[offs + (ctuRsAddr >> 1)] & 15) - 8;
303  }
304  }
305  }
306 
307  slice->sliceQp = iQP; // start slice QP for reference
308  slice->pic->picInitialQP = iQP;
309 
310  if ((iQP = BitAllocation::applyQPAdaptationLuma (slice, m_pcEncCfg, adaptedLumaQP, dLambda, *m_LineEncRsrc[ 0 ]->m_encCu.getQpPtr(),
311  (rcIsFirstPassOf2 && slice->poc > 0 ? m_pcRateCtrl->getIntraPQPAStats() : nullptr),
312  startCtuTsAddr, boundingCtuTsAddr )) >= 0) // sets pic->ctuAdaptedQP[] & ctuQpaLambda[]
313  {
314  dLambda *= pow (2.0, ((double) iQP - dQP) / 3.0); // adjust lambda based on change of slice QP
315  }
316  else iQP = (int) dQP; // revert to unadapted slice QP
317  }
318 
319  if ( slice->pps->sliceChromaQpFlag )
320  {
321  const bool bUseIntraOrPeriodicOffset = (slice->isIntra() && !slice->sps->IBC) || (m_pcEncCfg->m_sliceChromaQpOffsetPeriodicity > 0 && (slice->poc % m_pcEncCfg->m_sliceChromaQpOffsetPeriodicity) == 0);
322  int cbQP = bUseIntraOrPeriodicOffset ? sliceChromaQpOffsetIntraOrPeriodic[ 0 ] : gopList[ gopId ].m_CbQPoffset;
323  int crQP = bUseIntraOrPeriodicOffset ? sliceChromaQpOffsetIntraOrPeriodic[ 1 ] : gopList[ gopId ].m_CrQPoffset;
324  int cbCrQP = (cbQP + crQP) >> 1; // use floor of average chroma QP offset for joint-Cb/Cr coding
325 
326  cbQP = Clip3( -12, 12, cbQP + slice->pps->chromaQpOffset[COMP_Cb] ) - slice->pps->chromaQpOffset[COMP_Cb];
327  crQP = Clip3( -12, 12, crQP + slice->pps->chromaQpOffset[COMP_Cr] ) - slice->pps->chromaQpOffset[COMP_Cr];
328  slice->sliceChromaQpDelta[COMP_Cb] = Clip3( -12, 12, cbQP);
329  CHECK(!(slice->sliceChromaQpDelta[COMP_Cb]+slice->pps->chromaQpOffset[COMP_Cb]<=12 && slice->sliceChromaQpDelta[COMP_Cb]+slice->pps->chromaQpOffset[COMP_Cb]>=-12), "Unspecified error");
330  slice->sliceChromaQpDelta[COMP_Cr] = Clip3( -12, 12, crQP);
331  CHECK(!(slice->sliceChromaQpDelta[COMP_Cr]+slice->pps->chromaQpOffset[COMP_Cr]<=12 && slice->sliceChromaQpDelta[COMP_Cr]+slice->pps->chromaQpOffset[COMP_Cr]>=-12), "Unspecified error");
332  if (slice->sps->jointCbCr)
333  {
334  cbCrQP = Clip3 (-12, 12, cbCrQP + slice->pps->chromaQpOffset[COMP_JOINT_CbCr]) - slice->pps->chromaQpOffset[COMP_JOINT_CbCr];
335  slice->sliceChromaQpDelta[COMP_JOINT_CbCr] = Clip3 (-12, 12, cbCrQP);
336  }
337  }
338  else
339  {
340  slice->sliceChromaQpDelta[COMP_Cb] = 0;
341  slice->sliceChromaQpDelta[COMP_Cr] = 0;
342  slice->sliceChromaQpDelta[COMP_JOINT_CbCr] = 0;
343  }
344 
345  for( auto& lineRsc : m_LineEncRsrc )
346  {
347  lineRsc->m_encCu.setUpLambda( *slice, dLambda, iQP, true, true );
348  }
349 
350  slice->sliceQp = iQP;
351  slice->chromaQpAdjEnabled = slice->pps->chromaQpOffsetListLen>0;
352 
353  if (slice->pps->sliceChromaQpFlag && CS::isDualITree (*slice->pic->cs) && (!m_pcEncCfg->m_usePerceptQPA) && (m_pcEncCfg->m_sliceChromaQpOffsetPeriodicity == 0))
354  {
355  // overwrite chroma qp offset for dual tree
356  slice->sliceChromaQpDelta[ COMP_Cb ] = m_pcEncCfg->m_chromaCbQpOffsetDualTree;
357  slice->sliceChromaQpDelta[ COMP_Cr ] = m_pcEncCfg->m_chromaCrQpOffsetDualTree;
358  if ( slice->sps->jointCbCr )
359  {
360  slice->sliceChromaQpDelta[ COMP_JOINT_CbCr ] = m_pcEncCfg->m_chromaCbCrQpOffsetDualTree;
361  }
362  for( auto& lineRsc : m_LineEncRsrc )
363  {
364  lineRsc->m_encCu.setUpLambda( *slice, slice->getLambdas()[0], slice->sliceQp, true, false );
365  }
366  }
367 }

388 int EncSlice::xGetQPForPicture( const Slice* slice, unsigned gopId )
389 {
390  const int lumaQpBDOffset = slice->sps->qpBDOffset[ CH_L ];
391  int qp;
392 
393  if ( m_pcEncCfg->m_costMode == VVENC_COST_LOSSLESS_CODING )
394  {
395  qp = LOSSLESS_AND_MIXED_LOSSLESS_RD_COST_TEST_QP;
396  }
397  else
398  {
399  const SliceType sliceType = slice->sliceType;
400  qp = slice->pic->seqBaseQp;
401 
402  if( sliceType == VVENC_I_SLICE )
403  {
404  qp += m_pcEncCfg->m_intraQPOffset;
405  }
406  else
407  {
408  if ( ! ( qp == -lumaQpBDOffset ) )
409  {
410  const vvencGOPEntry &gopEntry = m_pcEncCfg->m_GOPList[ gopId ];
411  // adjust QP according to the QP offset for the GOP entry.
412  qp += gopEntry.m_QPOffset;
413 
414  // adjust QP according to QPOffsetModel for the GOP entry.
415  double dqpOffset = qp * gopEntry.m_QPOffsetModelScale + gopEntry.m_QPOffsetModelOffset + 0.5;
416  int qpOffset = (int)floor( Clip3<double>( 0.0, 3.0, dqpOffset ) );
417  qp += qpOffset ;
418  }
419  }
420  }
421  qp = Clip3( -lumaQpBDOffset, MAX_QP, qp );
422  return qp;
423 }

426 double EncSlice::xCalculateLambda( const Slice* slice,
427  const int GOPid, // entry in the GOP table
428  const int depth, // slice GOP hierarchical depth.
429  const double refQP, // initial slice-level QP
430  const double dQP, // initial double-precision QP
431  int& iQP ) // returned integer QP.
432 {
433  const vvencGOPEntry* gopList = m_pcEncCfg->m_GOPList;
434  const int NumberBFrames = ( m_pcEncCfg->m_GOPSize - 1 );
435  const int SHIFT_QP = 12;
436  const int temporalId = gopList[ GOPid ].m_temporalId;
437  std::vector<double> intraLambdaModifiers;
438  for ( int i = 0; i < VVENC_MAX_TLAYER; i++ )
439  {
440  if( m_pcEncCfg->m_adIntraLambdaModifier[i] != 0.0 ) intraLambdaModifiers.push_back( m_pcEncCfg->m_adIntraLambdaModifier[i] );
441  else break;
442  }
443 
444  int bitdepth_luma_qp_scale = 6
445  * (slice->sps->bitDepths[ CH_L ] - 8
446  - DISTORTION_PRECISION_ADJUSTMENT(slice->sps->bitDepths[ CH_L ]));
447  double qp_temp = dQP + bitdepth_luma_qp_scale - SHIFT_QP;
448  // Case #1: I or P-slices (key-frame)
449  double dQPFactor = gopList[ GOPid ].m_QPFactor;
450  if( slice->sliceType == VVENC_I_SLICE )
451  {
452  if (m_pcEncCfg->m_dIntraQpFactor>=0.0 && gopList[ GOPid ].m_sliceType != VVENC_I_SLICE)
453  {
454  dQPFactor = m_pcEncCfg->m_dIntraQpFactor;
455  }
456  else
457  {
458  dQPFactor = 0.57;
459  if( ! m_pcEncCfg->m_lambdaFromQPEnable)
460  {
461  double dLambda_scale = 1.0 - Clip3( 0.0, 0.5, 0.05 * (double)NumberBFrames );
462  dQPFactor *= dLambda_scale;
463  }
464  }
465  }
466  else if( m_pcEncCfg->m_lambdaFromQPEnable )
467  {
468  dQPFactor=0.57;
469  }
470 
471  double dLambda = dQPFactor*pow( 2.0, qp_temp/3.0 );
472 
473  if( !(m_pcEncCfg->m_lambdaFromQPEnable) && depth>0 )
474  {
475  double qp_temp_ref = refQP + bitdepth_luma_qp_scale - SHIFT_QP;
476  dLambda *= Clip3(2.00, 4.00, (qp_temp_ref / 6.0)); // (j == B_SLICE && p_cur_frm->layer != 0 )
477  }
478 
479  // if hadamard is used in ME process
480  if ( !m_pcEncCfg->m_bUseHADME && slice->sliceType != VVENC_I_SLICE )
481  {
482  dLambda *= 0.95;
483  }
484 
485  double lambdaModifier;
486  if( slice->sliceType != VVENC_I_SLICE || intraLambdaModifiers.empty())
487  {
488  lambdaModifier = m_pcEncCfg->m_adLambdaModifier[ temporalId ];
489  }
490  else
491  {
492  lambdaModifier = intraLambdaModifiers[ (temporalId < intraLambdaModifiers.size()) ? temporalId : (intraLambdaModifiers.size()-1) ];
493  }
494  dLambda *= lambdaModifier;
495 
496  iQP = Clip3( -slice->sps->qpBDOffset[ CH_L ], MAX_QP, (int) floor( dQP + 0.5 ) );
497 
498  if( m_pcEncCfg->m_DepQuantEnabled )
499  {
500  dLambda *= pow( 2.0, 0.25/3.0 ); // slight lambda adjustment for dependent quantization (due to different slope of quantizer)
501  }
502 
503  // NOTE: the lambda modifiers that are sometimes applied later might be best always applied in here.
504  return dLambda;
505 }

553 void setJointCbCrModes( CodingStructure& cs, const Position topLeftLuma, const Size sizeLuma )
554 {
555  bool sgnFlag = true;
556 
557  if( isChromaEnabled( cs.picture->chromaFormat) )
558  {
559  const CompArea cbArea = CompArea( COMP_Cb, cs.picture->chromaFormat, Area(topLeftLuma,sizeLuma), true );
560  const CompArea crArea = CompArea( COMP_Cr, cs.picture->chromaFormat, Area(topLeftLuma,sizeLuma), true );
561 
562  const CPelBuf orgCb = cs.picture->getFilteredOrigBuffer().valid() ? cs.picture->getRspOrigBuf( cbArea ): cs.picture->getOrigBuf( cbArea );
563  const CPelBuf orgCr = cs.picture->getFilteredOrigBuffer().valid() ? cs.picture->getRspOrigBuf( crArea ): cs.picture->getOrigBuf( crArea );
564  const int x0 = ( cbArea.x > 0 ? 0 : 1 );
565  const int y0 = ( cbArea.y > 0 ? 0 : 1 );
566  const int x1 = ( cbArea.x + cbArea.width < cs.picture->Cb().width ? cbArea.width : cbArea.width - 1 );
567  const int y1 = ( cbArea.y + cbArea.height < cs.picture->Cb().height ? cbArea.height : cbArea.height - 1 );
568  const int cbs = orgCb.stride;
569  const int crs = orgCr.stride;
570  const Pel* pCb = orgCb.buf + y0 * cbs;
571  const Pel* pCr = orgCr.buf + y0 * crs;
572  int64_t sumCbCr = 0;
573 
574  // determine inter-chroma transform sign from correlation between high-pass filtered (i.e., zero-mean) Cb and Cr planes
575  for( int y = y0; y < y1; y++, pCb += cbs, pCr += crs )
576  {
577  for( int x = x0; x < x1; x++ )
578  {
579  int cb = ( 12*(int)pCb[x] - 2*((int)pCb[x-1] + (int)pCb[x+1] + (int)pCb[x-cbs] + (int)pCb[x+cbs]) - ((int)pCb[x-1-cbs] + (int)pCb[x+1-cbs] + (int)pCb[x-1+cbs] + (int)pCb[x+1+cbs]) );
580  int cr = ( 12*(int)pCr[x] - 2*((int)pCr[x-1] + (int)pCr[x+1] + (int)pCr[x-crs] + (int)pCr[x+crs]) - ((int)pCr[x-1-crs] + (int)pCr[x+1-crs] + (int)pCr[x-1+crs] + (int)pCr[x+1+crs]) );
581  sumCbCr += cb*cr;
582  }
583  }
584 
585  sgnFlag = ( sumCbCr < 0 );
586  }
587 
588  cs.slice->picHeader->jointCbCrSign = sgnFlag;
589 }

660  {

662  m_ctuAddrMap.resize( pcv.sizeInCtus, 0 );
663  int addr = 0;
664  for( int i = 1; i < pcv.sizeInCtus; i++ )
665  {
666  int x = addr % pcv.widthInCtus;
667  int y = addr / pcv.widthInCtus;

680  addr = y * pcv.widthInCtus + x;
681  m_ctuAddrMap[ i ] = addr;
682  }

765  if( m_pcEncCfg->m_numThreads > 0 )
766  {
767  for( auto& ctuEncParam : ctuEncParams )

796 template<bool checkReadyState>
797 bool EncSlice::xProcessCtuTask( int threadIdx, CtuEncParam* ctuEncParam )
798 {
799  Picture* pic = ctuEncParam->pic;
800  EncSlice* encSlice = ctuEncParam->encSlice;
801  CodingStructure& cs = *pic->cs;
802  Slice& slice = *cs.slice;
803  const PreCalcValues& pcv = *cs.pcv;
804  const int ctuRsAddr = ctuEncParam->ctuRsAddr;
805  const int ctuPosX = ctuEncParam->ctuPosX;
806  const int ctuPosY = ctuEncParam->ctuPosY;
807  const int x = ctuPosX << pcv.maxCUSizeLog2;
808  const int y = ctuPosY << pcv.maxCUSizeLog2;
809  const int width = std::min( pcv.maxCUSize, pcv.lumaWidth - x );
810  const int height = std::min( pcv.maxCUSize, pcv.lumaHeight - y );
811  const int ctuStride = pcv.widthInCtus;
812  ProcessCtuState* processStates = encSlice->m_processStates.data();
813  const UnitArea& ctuArea = ctuEncParam->ctuArea;
814  const bool wppSyncEnabled = cs.sps->entropyCodingSyncEnabled;
815 
816  DTRACE_UPDATE( g_trace_ctx, std::make_pair( "poc", cs.slice->poc ) );
817  DTRACE_UPDATE( g_trace_ctx, std::make_pair( "ctu", ctuRsAddr ) );
818  DTRACE_UPDATE( g_trace_ctx, std::make_pair( "final", processStates[ ctuRsAddr ] == CTU_ENCODE ? 0 : 1 ) );
819 
820  // process ctu's line wise from left to right
821  if( ctuPosX > 0 && processStates[ ctuRsAddr - 1 ] <= processStates[ ctuRsAddr ] && processStates[ ctuRsAddr ] < PROCESS_DONE )
822  return false;
823 
824  switch( processStates[ ctuRsAddr ].load() )
825  {
826  // encode
827  case CTU_ENCODE:
828  {
829  // general wpp conditions, top and top-right ctu have to be encoded
830  if( ctuPosY > 0 && processStates[ ctuRsAddr - ctuStride ] <= CTU_ENCODE )
831  return false;
832  if( ctuPosY > 0 && ctuPosX + 1 < pcv.widthInCtus && processStates[ ctuRsAddr - ctuStride + 1 ] <= CTU_ENCODE && !wppSyncEnabled )
833  return false;
834 
835  if( checkReadyState )
836  return true;
837 
838 #ifdef TRACE_ENABLE_ITT
839  std::stringstream ss;
840  ss << "Encode_" << slice.poc << "_CTU_" << ctuPosY << "_" << ctuPosX;
841  __itt_string_handle* itt_handle_ctuEncode = __itt_string_handle_create( ss.str().c_str() );
842 #endif
843  ITT_TASKSTART( itt_domain_encode, itt_handle_ctuEncode );
844 
845  const int lineIdx = std::min( (int)( encSlice->m_LineEncRsrc.size() ) - 1, ctuPosY );
846  LineEncRsrc* lineEncRsrc = encSlice->m_LineEncRsrc[ lineIdx ];
847  PerThreadRsrc* taskRsrc = encSlice->m_CtuTaskRsrc[ threadIdx ];
848  EncCu& encCu = lineEncRsrc->m_encCu;
849 
850  encCu.setCtuEncRsrc( &lineEncRsrc->m_CABACEstimator, &taskRsrc->m_CtxCache, &lineEncRsrc->m_ReuseUniMv, &lineEncRsrc->m_BlkUniMvInfoBuffer, &lineEncRsrc->m_AffineProfList, &lineEncRsrc->m_CachedBvs );
851  encCu.encodeCtu( pic, lineEncRsrc->m_prevQp, ctuPosX, ctuPosY );
852 
853  // cleanup line memory when last ctu in line done to reduce overall memory consumption
854  if( encSlice->m_pcEncCfg->m_ensureWppBitEqual && ctuPosX == pcv.widthInCtus - 1 )
855  {
856  lineEncRsrc->m_AffineProfList.resetAffineMVList();
857  lineEncRsrc->m_BlkUniMvInfoBuffer.resetUniMvList();
858  lineEncRsrc->m_ReuseUniMv.resetReusedUniMvs();
859  lineEncRsrc->m_CachedBvs.resetIbcBvCand();
860  }
861 
862  DTRACE_UPDATE( g_trace_ctx, std::make_pair( "final", 1 ) );
863  ITT_TASKEND( itt_domain_encode, itt_handle_ctuEncode );
864 
865  processStates[ ctuRsAddr ] = RESHAPE_LF_VER;
866  }
867  break;
868 
869  // reshape + vertical loopfilter
870  case RESHAPE_LF_VER:
871  {
872  // clip check to right picture border
873  const int checkRight = std::min<int>( encSlice->m_ctuEncDelay, (int)pcv.widthInCtus - 1 - ctuPosX );
874 
875  // ensure all surrounding ctu's are encoded (intra pred requires non-reshaped and unfiltered residual, IBC requires unfiltered samples too)
876  // check right with max offset (due to WPP condition above, this implies top-right has been already encoded)
877  if( processStates[ ctuRsAddr + checkRight ] <= CTU_ENCODE )
878  return false;
879  // check bottom right with 1 CTU delay (this is only required for intra pred)
880  // at the right picture border this will check the bottom CTU
881  const int checkBottomRight = std::min<int>( 1, checkRight );
882  if( ctuPosY + 1 < pcv.heightInCtus && processStates[ ctuRsAddr + checkBottomRight + ctuStride ] <= CTU_ENCODE )
883  return false;
884 
885  if( checkReadyState )
886  return true;
887 
888  ITT_TASKSTART( itt_domain_encode, itt_handle_rspLfVer );
889 
890  // reshape
891  if( slice.sps->lumaReshapeEnable && slice.picHeader->lmcsEnabled )
892  {
893  PelBuf reco = pic->getRecoBuf( COMP_Y ).subBuf( x, y, width, height );
894  reco.rspSignal( pic->reshapeData.getInvLUT() );
895  }
896 
897  // loopfilter
898  if( !cs.pps->deblockingFilterControlPresent || !cs.pps->deblockingFilterDisabled || cs.pps->deblockingFilterOverrideEnabled )
899  {
900  // calculate filter strengths
901  encSlice->m_pLoopFilter->calcFilterStrengthsCTU( cs, ctuArea, true );
902 
903  // vertical filter
904  PelUnitBuf reco = cs.picture->getRecoBuf();
905  encSlice->m_pLoopFilter->xDeblockArea<EDGE_VER>( cs, ctuArea, MAX_NUM_CH, reco );
906  }
907 
908  ITT_TASKEND( itt_domain_encode, itt_handle_rspLfVer );
909 
910  processStates[ ctuRsAddr ] = LF_HOR;
911  }
912  break;
913 
914  // horizontal loopfilter
915  case LF_HOR:
916  {
917  // ensure horizontal ordering (from top to bottom)
918  if( ctuPosY > 0 && processStates[ ctuRsAddr - ctuStride ] <= LF_HOR )
919  return false;
920 
921  // ensure vertical loop filter of neighbor ctu's will not modify current residual
922  // check top, top-right and right ctu
923  // (top, top-right checked implicitly due to ordering check above)
924  if( ctuPosX + 1 < pcv.widthInCtus && processStates[ ctuRsAddr + 1 ] <= RESHAPE_LF_VER )
925  return false;
926 
927  if( checkReadyState )
928  return true;
929 
930  ITT_TASKSTART( itt_domain_encode, itt_handle_lfHor );
931 
932  if( !cs.pps->deblockingFilterControlPresent || !cs.pps->deblockingFilterDisabled || cs.pps->deblockingFilterOverrideEnabled )
933  {
934  PelUnitBuf reco = cs.picture->getRecoBuf();
935  encSlice->m_pLoopFilter->xDeblockArea<EDGE_HOR>( cs, ctuArea, MAX_NUM_CH, reco );
936  }
937 
938  ITT_TASKEND( itt_domain_encode, itt_handle_lfHor );
939 
940  processStates[ ctuRsAddr ] = SAO_FILTER;
941  }
942  break;
943 
944  // SAO filter
945  case SAO_FILTER:
946  {
947  // general wpp conditions, top and top-right ctu have to be filtered
948  if( ctuPosY > 0 && processStates[ ctuRsAddr - ctuStride ] <= SAO_FILTER )
949  return false;
950  if( ctuPosY > 0 && ctuPosX + 1 < pcv.widthInCtus && processStates[ ctuRsAddr - ctuStride + 1 ] <= SAO_FILTER )
951  return false;
952 
953  // ensure loop filter of neighbor ctu's will not modify current residual
954  // sao processing dependents on +1 pixel to each side
955  // due to wpp condition above, only right, bottom and bottom-right ctu have to be checked
956  if( ctuPosX + 1 < pcv.widthInCtus && processStates[ ctuRsAddr + 1 ] <= LF_HOR )
957  return false;
958  if( ctuPosY + 1 < pcv.heightInCtus && processStates[ ctuRsAddr + ctuStride ] <= LF_HOR )
959  return false;
960  if( ctuPosX + 1 < pcv.widthInCtus && ctuPosY + 1 < pcv.heightInCtus && processStates[ ctuRsAddr + 1 + ctuStride ] <= LF_HOR )
961  return false;
962 
963  if( checkReadyState )
964  return true;
965 
966  ITT_TASKSTART( itt_domain_encode, itt_handle_sao );
967 
968  // SAO filter
969  if( slice.sps->saoEnabled )
970  {
971  const int lineIdx = std::min( (int)( encSlice->m_LineEncRsrc.size() ) - 1, ctuPosY );
972  LineEncRsrc* lineEncRsrc = encSlice->m_LineEncRsrc[ lineIdx ];
973  PerThreadRsrc* taskRsrc = encSlice->m_CtuTaskRsrc[ threadIdx ];
974  EncSampleAdaptiveOffset& encSao = lineEncRsrc->m_encSao;
975 
976  encSao.setCtuEncRsrc( &lineEncRsrc->m_SaoCABACEstimator, &taskRsrc->m_CtxCache );
977  encSao.storeCtuReco( cs, ctuArea );
978  encSao.getCtuStatistics( cs, encSlice->m_saoStatData, ctuArea, ctuRsAddr );
979  encSao.decideCtuParams( cs, encSlice->m_saoStatData, encSlice->m_saoEnabled, encSlice->m_saoAllDisabled, ctuArea, ctuRsAddr, &encSlice->m_saoReconParams[ 0 ], cs.picture->getSAO() );
980  }
981 
982  // ALF border extension
983  if( slice.sps->alfEnabled )
984  {
985  // we have to do some kind of position aware boundary padding
986  // it's done here because the conditions are readable
