90 bool AdaptiveLoopFilter::isCrossedByVirtualBoundaries( const CodingStructure& cs, const int xPos, const int yPos, const int width, const int height, bool& clipTop, bool& clipBottom, bool& clipLeft, bool& clipRight, int& numHorVirBndry, int& numVerVirBndry, int horVirBndryPos[], int verVirBndryPos[], int& rasterSliceAlfPad )
91 {
92  clipTop = false; clipBottom = false; clipLeft = false; clipRight = false;
93  numHorVirBndry = 0; numVerVirBndry = 0;
94  const PPS* pps = cs.pps;
95  const PicHeader* picHeader = cs.picHeader;
96 
97  if( picHeader->virtualBoundariesEnabled )
98  {
99  for( int i = 0; i < picHeader->numHorVirtualBoundaries; i++ )
100  {
101  if( picHeader->virtualBoundariesPosY[i] == yPos )
102  {
103  clipTop = true;
104  }
105  else if( picHeader->virtualBoundariesPosY[i] == yPos + height )
106  {
107  clipBottom = true;
108  }
109  else if( yPos < picHeader->virtualBoundariesPosY[i] && picHeader->virtualBoundariesPosY[i] < yPos + height )
110  {
111  horVirBndryPos[numHorVirBndry++] = picHeader->virtualBoundariesPosY[i];
112  }
113  }
114  for( int i = 0; i < picHeader->numVerVirtualBoundaries; i++ )
115  {
116  if( picHeader->virtualBoundariesPosX[i] == xPos )
117  {
118  clipLeft = true;
119  }
120  else if( picHeader->virtualBoundariesPosX[i] == xPos + width )
121  {
122  clipRight = true;
123  }
124  else if( xPos < picHeader->virtualBoundariesPosX[i] && picHeader->virtualBoundariesPosX[i] < xPos + width )
125  {
126  verVirBndryPos[numVerVirBndry++] = picHeader->virtualBoundariesPosX[i];
127  }
128  }
129  }
130 
131  const Slice& slice = *(cs.slice);
132  int ctuSize = slice.sps->CTUSize;
133  const Position currCtuPos(xPos, yPos);
134  const CodingUnit *currCtu = cs.getCU(currCtuPos, CH_L, TREE_D);
135  const SubPic& curSubPic = slice.pps->getSubPicFromPos(currCtuPos);
136  bool loopFilterAcrossSubPicEnabled = curSubPic.loopFilterAcrossSubPicEnabled;
137  //top
138  if (yPos >= ctuSize && clipTop == false)
139  {
140  const Position prevCtuPos(xPos, yPos - ctuSize);
141  const CodingUnit *prevCtu = cs.getCU(prevCtuPos, CH_L, TREE_D);
142  if ((!pps->loopFilterAcrossSlicesEnabled && !CU::isSameSlice(*currCtu, *prevCtu)) ||
143  (!pps->loopFilterAcrossTilesEnabled && !CU::isSameTile(*currCtu, *prevCtu))
144  || (!loopFilterAcrossSubPicEnabled && !CU::isSameSubPic(*currCtu, *prevCtu))
145  )
146  {
147  clipTop = true;
148  }
149  }
150 
151  //bottom
152  if (yPos + ctuSize < cs.pcv->lumaHeight && clipBottom == false)
153  {
154  const Position nextCtuPos(xPos, yPos + ctuSize);
155  const CodingUnit *nextCtu = cs.getCU(nextCtuPos, CH_L, TREE_D);
156  if ((!pps->loopFilterAcrossSlicesEnabled && !CU::isSameSlice(*currCtu, *nextCtu)) ||
157  (!pps->loopFilterAcrossTilesEnabled && !CU::isSameTile(*currCtu, *nextCtu))
158  || (!loopFilterAcrossSubPicEnabled && !CU::isSameSubPic(*currCtu, *nextCtu))
159  )
160  {
161  clipBottom = true;
162  }
163  }
164 
165  //left
166  if (xPos >= ctuSize && clipLeft == false)
167  {
168  const Position prevCtuPos(xPos - ctuSize, yPos);
169  const CodingUnit *prevCtu = cs.getCU(prevCtuPos, CH_L, TREE_D);
170  if ((!pps->loopFilterAcrossSlicesEnabled && !CU::isSameSlice(*currCtu, *prevCtu)) ||
171  (!pps->loopFilterAcrossTilesEnabled && !CU::isSameTile(*currCtu, *prevCtu))
172  || (!loopFilterAcrossSubPicEnabled && !CU::isSameSubPic(*currCtu, *prevCtu))
173  )
174  {
175  clipLeft = true;
176  }
177  }
178 
179  //right
180  if (xPos + ctuSize < cs.pcv->lumaWidth && clipRight == false)
181  {
182  const Position nextCtuPos(xPos + ctuSize, yPos);
183  const CodingUnit *nextCtu = cs.getCU(nextCtuPos, CH_L, TREE_D);
184  if ((!pps->loopFilterAcrossSlicesEnabled && !CU::isSameSlice(*currCtu, *nextCtu)) ||
185  (!pps->loopFilterAcrossTilesEnabled && !CU::isSameTile(*currCtu, *nextCtu))
186  || (!loopFilterAcrossSubPicEnabled && !CU::isSameSubPic(*currCtu, *nextCtu))
187  )
188  {
189  clipRight = true;
190  }
191  }
192 
193  rasterSliceAlfPad = 0;
194  if ( !clipTop && !clipLeft )
195  {
196  //top-left CTU
197  if ( xPos >= ctuSize && yPos >= ctuSize )
198  {
199  const Position prevCtuPos( xPos - ctuSize, yPos - ctuSize );
200  const CodingUnit *prevCtu = cs.getCU( prevCtuPos, CH_L, TREE_D );
201  if ( !pps->loopFilterAcrossSlicesEnabled && !CU::isSameSlice( *currCtu, *prevCtu ) )
202  {
203  rasterSliceAlfPad = 1;
204  }
205  }
206  }
207 
208  if ( !clipBottom && !clipRight )
209  {
210  //bottom-right CTU
211  if ( xPos + ctuSize < cs.pcv->lumaWidth && yPos + ctuSize < cs.pcv->lumaHeight )
212  {
213  const Position nextCtuPos( xPos + ctuSize, yPos + ctuSize );
214  const CodingUnit *nextCtu = cs.getCU( nextCtuPos, CH_L, TREE_D );
215  if ( !pps->loopFilterAcrossSlicesEnabled && !CU::isSameSlice( *currCtu, *nextCtu ) )
216  {
217  rasterSliceAlfPad += 2;
218  }
219  }
220  }
221 
222  return numHorVirBndry > 0 || numVerVirBndry > 0 || clipTop || clipBottom || clipLeft || clipRight || rasterSliceAlfPad;
223 }

312 void AdaptiveLoopFilter::applyCcAlfFilter(CodingStructure &cs, ComponentID compID, const PelBuf &dstBuf,
313  const PelUnitBuf &recYuvExt, uint8_t *filterControl,
314  const short filterSet[MAX_NUM_CC_ALF_FILTERS][MAX_NUM_CC_ALF_CHROMA_COEFF],
315  const int selectedFilterIdx)
316 {
317  bool clipTop = false, clipBottom = false, clipLeft = false, clipRight = false;
318  int numHorVirBndry = 0, numVerVirBndry = 0;
319  int horVirBndryPos[] = { 0, 0, 0 };
320  int verVirBndryPos[] = { 0, 0, 0 };
321 
322  const ClpRngs& clpRngs = cs.slice->clpRngs;
323 
324  int ctuIdx = 0;
325  for( int yPos = 0; yPos < m_picHeight; yPos += m_maxCUHeight )
326  {
327  for( int xPos = 0; xPos < m_picWidth; xPos += m_maxCUWidth )
328  {
329  int filterIdx =
330  (filterControl == nullptr)
331  ? selectedFilterIdx
332  : filterControl[(yPos >> cs.pcv->maxCUSizeLog2) * cs.pcv->widthInCtus + (xPos >> cs.pcv->maxCUSizeLog2)];
333  bool skipFiltering = (filterControl != nullptr && filterIdx == 0) ? true : false;
334  if (!skipFiltering)
335  {
336  if (filterControl != nullptr)
337  filterIdx--;
338 
339  const int16_t *filterCoeff = filterSet[filterIdx];
340 
341  const int width = (xPos + m_maxCUWidth > m_picWidth) ? (m_picWidth - xPos) : m_maxCUWidth;
342  const int height = (yPos + m_maxCUHeight > m_picHeight) ? (m_picHeight - yPos) : m_maxCUHeight;
343  const int chromaScaleX = getComponentScaleX(compID, m_chromaFormat);
344  const int chromaScaleY = getComponentScaleY(compID, m_chromaFormat);
345 
346  int rasterSliceAlfPad = 0;
347  if (isCrossedByVirtualBoundaries(cs, xPos, yPos, width, height, clipTop, clipBottom, clipLeft, clipRight,
348  numHorVirBndry, numVerVirBndry, horVirBndryPos, verVirBndryPos,
349  rasterSliceAlfPad))
350  {
351  int yStart = yPos;
352  for (int i = 0; i <= numHorVirBndry; i++)
353  {
354  const int yEnd = i == numHorVirBndry ? yPos + height : horVirBndryPos[i];
355  const int h = yEnd - yStart;
356  const bool clipT = (i == 0 && clipTop) || (i > 0) || (yStart == 0);
357  const bool clipB = (i == numHorVirBndry && clipBottom) || (i < numHorVirBndry) || (yEnd == m_picHeight);
358  int xStart = xPos;
359  for (int j = 0; j <= numVerVirBndry; j++)
360  {
361  const int xEnd = j == numVerVirBndry ? xPos + width : verVirBndryPos[j];
362  const int w = xEnd - xStart;
363  const bool clipL = (j == 0 && clipLeft) || (j > 0) || (xStart == 0);
364  const bool clipR = (j == numVerVirBndry && clipRight) || (j < numVerVirBndry) || (xEnd == m_picWidth);
365  const int wBuf = w + (clipL ? 0 : MAX_ALF_PADDING_SIZE) + (clipR ? 0 : MAX_ALF_PADDING_SIZE);
366  const int hBuf = h + (clipT ? 0 : MAX_ALF_PADDING_SIZE) + (clipB ? 0 : MAX_ALF_PADDING_SIZE);
367  PelUnitBuf buf = m_tempBuf2.subBuf(UnitArea(cs.area.chromaFormat, Area(0, 0, wBuf, hBuf)));
368  buf.copyFrom(recYuvExt.subBuf(
369  UnitArea(cs.area.chromaFormat, Area(xStart - (clipL ? 0 : MAX_ALF_PADDING_SIZE),
370  yStart - (clipT ? 0 : MAX_ALF_PADDING_SIZE), wBuf, hBuf))));
371  // pad top-left unavailable samples for raster slice
372  if (xStart == xPos && yStart == yPos && (rasterSliceAlfPad & 1))
373  {
374  buf.padBorderPel(MAX_ALF_PADDING_SIZE, 1);
375  }
376 
377  // pad bottom-right unavailable samples for raster slice
378  if (xEnd == xPos + width && yEnd == yPos + height && (rasterSliceAlfPad & 2))
379  {
380  buf.padBorderPel(MAX_ALF_PADDING_SIZE, 2);
381  }
382  buf.extendBorderPel(MAX_ALF_PADDING_SIZE);
383  buf = buf.subBuf(UnitArea(
384  cs.area.chromaFormat, Area(clipL ? 0 : MAX_ALF_PADDING_SIZE, clipT ? 0 : MAX_ALF_PADDING_SIZE, w, h)));
385 
386  const Area blkSrc(0, 0, w, h);
387 
388  const Area blkDst(xStart >> chromaScaleX, yStart >> chromaScaleY, w >> chromaScaleX, h >> chromaScaleY);
389  m_filterCcAlf(dstBuf, buf, blkDst, blkSrc, compID, filterCoeff, clpRngs, cs, m_alfVBLumaCTUHeight,
390  m_alfVBLumaPos);
391 
392  xStart = xEnd;
393  }
394 
395  yStart = yEnd;
396  }
397  }
398  else
399  {
400  Area blkDst(xPos >> chromaScaleX, yPos >> chromaScaleY, width >> chromaScaleX, height >> chromaScaleY);
401  Area blkSrc(xPos, yPos, width, height);
402

667 void AdaptiveLoopFilter::reconstructCoeff( AlfParam& alfParam, ChannelType channel, const bool isRdo, const bool isRedo )
668 {
669  const int factor = isRdo ? 0 : (1 << (m_NUM_BITS - 1));
670  const AlfFilterType filterType = isLuma( channel ) ? ALF_FILTER_7 : ALF_FILTER_5;
671  const int numClasses = isLuma( channel ) ? MAX_NUM_ALF_CLASSES : 1;
672  const int numCoeff = filterType == ALF_FILTER_5 ? 7 : 13;
673  const int numCoeffMinus1 = numCoeff - 1;
674  const int numAlts = isLuma( channel ) ? 1 : alfParam.numAlternativesChroma;
675 
676  for( int altIdx = 0; altIdx < numAlts; ++ altIdx )
677  {
678  int numFilters = isLuma( channel ) ? alfParam.numLumaFilters : 1;
679  short* coeff = isLuma( channel ) ? alfParam.lumaCoeff : alfParam.chromaCoeff[altIdx];
680  short* clipp = isLuma( channel ) ? alfParam.lumaClipp : alfParam.chromaClipp[altIdx];
681 
682  for( int filterIdx = 0; filterIdx < numFilters; filterIdx++ )
683  {
684  coeff[filterIdx* MAX_NUM_ALF_LUMA_COEFF + numCoeffMinus1] = factor;
685  }
686 
687  if( isChroma( channel ) )
688  {
689  for( int coeffIdx = 0; coeffIdx < numCoeffMinus1; ++coeffIdx )
690  {
691  m_chromaCoeffFinal[altIdx][coeffIdx] = coeff[coeffIdx];
692  int clipIdx = alfParam.nonLinearFlag[channel] ? clipp[coeffIdx] : 0;
693  m_chromaClippFinal[altIdx][coeffIdx] = isRdo ? clipIdx : m_alfClippingValues[channel][clipIdx];
694  }
695  m_chromaCoeffFinal[altIdx][numCoeffMinus1] = factor;
696  m_chromaClippFinal[altIdx][numCoeffMinus1] = isRdo ? 0 : m_alfClippingValues[channel][0];
697  continue;
698  }
699  for( int classIdx = 0; classIdx < numClasses; classIdx++ )
700  {
701  int filterIdx = alfParam.filterCoeffDeltaIdx[classIdx];
702 
703  CHECK(!(filterIdx >= 0 && filterIdx < alfParam.numLumaFilters), "Bad coeff delta idx in ALF");
704  for (int coeffIdx = 0; coeffIdx < numCoeffMinus1; ++coeffIdx)
705  {
706  m_coeffFinal[classIdx * MAX_NUM_ALF_LUMA_COEFF + coeffIdx] = coeff[filterIdx * MAX_NUM_ALF_LUMA_COEFF + coeffIdx];
707  }
708  m_coeffFinal[classIdx* MAX_NUM_ALF_LUMA_COEFF + numCoeffMinus1] = factor;
709  m_clippFinal[classIdx* MAX_NUM_ALF_LUMA_COEFF + numCoeffMinus1] = isRdo ? 0 : m_alfClippingValues[channel][0];
710  for( int coeffIdx = 0; coeffIdx < numCoeffMinus1; ++coeffIdx )
711  {
712  int clipIdx = alfParam.nonLinearFlag[channel] ? clipp[filterIdx * MAX_NUM_ALF_LUMA_COEFF + coeffIdx] : 0;
713  CHECK(!(clipIdx >= 0 && clipIdx < MaxAlfNumClippingValues), "Bad clip idx in ALF");
714  m_clippFinal[classIdx * MAX_NUM_ALF_LUMA_COEFF + coeffIdx] = isRdo ? clipIdx : m_alfClippingValues[channel][clipIdx];
715  }
716  m_clippFinal[classIdx* MAX_NUM_ALF_LUMA_COEFF + numCoeffMinus1] =
717  isRdo ? 0 :
718  m_alfClippingValues[channel][0];
719  }
720  }

756  for (int i = 1; i < AlfNumClippingValues; ++i)

758  m_alfClippingValues[CH_C][i] = 1 << (7 - 2 * i + shiftChroma);

764  }

770 
771  // Classification

773  {

775  int numVer = (m_numCTUsInHeight * MAX_CU_SIZE)/4;

777  }
778 
779  for (int filterSetIndex = 0; filterSetIndex < NUM_FIXED_FILTER_SETS; filterSetIndex++)
780  {
781  for (int classIdx = 0; classIdx < MAX_NUM_ALF_CLASSES; classIdx++)
782  {
783  int fixedFilterIdx = m_classToFilterMapping[filterSetIndex][classIdx];
784  for (int i = 0; i < MAX_NUM_ALF_LUMA_COEFF - 1; i++)
785  {
786  m_fixedFilterSetCoeffDec[filterSetIndex][classIdx * MAX_NUM_ALF_LUMA_COEFF + i] = m_fixedFilterSetCoeff[fixedFilterIdx][i];
787  }
788  m_fixedFilterSetCoeffDec[filterSetIndex][classIdx * MAX_NUM_ALF_LUMA_COEFF + MAX_NUM_ALF_LUMA_COEFF - 1] = (1 << (m_NUM_BITS - 1));
789  }
790  }
791  for (int i = 0; i < MAX_NUM_ALF_LUMA_COEFF * MAX_NUM_ALF_CLASSES; i++)
792  {
793  m_clipDefault[i] = m_alfClippingValues[CH_L][0];

795 
796  m_ccAlfFilterControl[0] = new uint8_t[m_numCTUsInPic];
797  m_ccAlfFilterControl[1] = new uint8_t[m_numCTUsInPic];
798 
799  m_created = true;

1028  hvd0 = d0;

1030  secondaryDirection = dirTempHV;
1031  }
1032  else
1033  {
1034  hvd1 = hv1;
1035  hvd0 = hv0;
1036  mainDirection = dirTempHV;
1037  secondaryDirection = dirTempD;
1038  }
1039 
1040  int directionStrength = 0;
1041  if( hvd1 > 2 * hvd0 )
1042  {
1043  directionStrength = 1;
1044  }
1045  if( hvd1 * 2 > 9 * hvd0 )
1046  {
1047  directionStrength = 2;
1048  }
1049

1057

1059  }
1060  }
