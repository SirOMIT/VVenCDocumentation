81 void write(std::ostream& out, OutputNALUnit& nalu)
82 {
83  writeNalUnitHeader(out, nalu);
84  /* write out rsbp_byte's, inserting any required
85  * emulation_prevention_three_byte's */
86  /* 7.4.1 ...
87  * emulation_prevention_three_byte is a byte equal to 0x03. When an
88  * emulation_prevention_three_byte is present in the NAL unit, it shall be
89  * discarded by the decoding process.
90  * The last byte of the NAL unit shall not be equal to 0x00.
91  * Within the NAL unit, the following three-byte sequences shall not occur at
92  * any byte-aligned position:
93  * - 0x000000
94  * - 0x000001
95  * - 0x000002
96  * Within the NAL unit, any four-byte sequence that starts with 0x000003
97  * other than the following sequences shall not occur at any byte-aligned
98  * position:
99  * - 0x00000300
100  * - 0x00000301
101  * - 0x00000302
102  * - 0x00000303
103  */
104  std::vector<uint8_t>& rbsp = nalu.m_Bitstream.getFIFO();
105 
106  std::vector<uint8_t> outputBuffer;
107  outputBuffer.resize(rbsp.size()*2+1); //there can never be enough emulation_prevention_three_bytes to require this much space
108  std::size_t outputAmount = 0;
109  int zeroCount = 0;
110  for (std::vector<uint8_t>::iterator it = rbsp.begin(); it != rbsp.end(); it++)
111  {
112  const uint8_t v=(*it);
113  if (zeroCount==2 && v<=3)
114  {
115  outputBuffer[outputAmount++]=emulation_prevention_three_byte;
116  zeroCount=0;
117  }
118 
119  if (v==0)
120  {
121  zeroCount++;
122  }
123  else
124  {
125  zeroCount=0;
126  }
127  outputBuffer[outputAmount++]=v;
128  }
129 
130  /* 7.4.1.1
131  * ... when the last byte of the RBSP data is equal to 0x00 (which can
132  * only occur when the RBSP ends in a cabac_zero_word), a final byte equal
133  * to 0x03 is appended to the end of the data.
134  */
135  if (zeroCount>0)
136  {
137  outputBuffer[outputAmount++]=emulation_prevention_three_byte;
138  }
139  out.write(reinterpret_cast<const char*>(&(*outputBuffer.begin())), outputAmount);
140 }
