106 bool parseCfg( int argc, char* argv[], apputils::VVEncAppCfg& rcVVEncAppCfg )
107 {
108  try
109  {
110  if( ! rcVVEncAppCfg.parseCfg( argc, argv ) )
111  {
112  return false;
113  }
114  }
115  catch( apputils::df::program_options_lite::ParseFailure &e )
116  {
117  msgApp( nullptr, VVENC_ERROR, "Error parsing option \"%s\" with argument \"%s\".\n", e.arg.c_str(), e.val.c_str() );
118  return false;
119  }
120  g_verbosity = rcVVEncAppCfg.m_verbosity;
121 
122  return true;
123 }

125 int main( int argc, char* argv[] )
126 {
127  std::string cAppname = argv[0];
128  std::size_t iPos = (int)cAppname.find_last_of("/");
129  if( std::string::npos != iPos )
130  {
131  cAppname = cAppname.substr(iPos+1 );
132  }
133 
134  int iRet = 0;
135 
136  vvenc_set_logging_callback( nullptr, msgFnc );
137 
138  std::string cInputFile;
139  std::string cOutputfile = "";
140 
141  apputils::VVEncAppCfg vvencappCfg;
142  vvenc_init_default( &vvencappCfg, 1920, 1080, 60, 0, 32, vvencPresetMode::VVENC_MEDIUM );
143 
144  // parse configuration
145  if ( ! parseCfg( argc, argv, vvencappCfg ) )
146  {
147  return 1;
148  }
149  // assign verbosity used for encoder output
150  g_verbosity = vvencappCfg.m_verbosity;
151 
152  if( vvencappCfg.m_showVersion )
153  {
154  std::cout << cAppname << " version " << vvenc_get_version()<< std::endl;
155  return 0;
156  }
157 
158  if( !strcmp( vvencappCfg.m_inputFileName.c_str(), "-" ) )
159  {
160  if( vvencappCfg.m_RCNumPasses > 1 )
161  {
162  std::cout << cAppname << " [error]: 2 pass rate control and reading from stdin is not supported yet" << std::endl;
163  return -1;
164  }
165  else
166  {
167  std::cout << cAppname << " trying to read from stdin" << std::endl;
168  }
169  }
170 
171  if( vvencappCfg.m_bitstreamFileName.empty() )
172  {
173  std::cout << cAppname << " [error]: no output bitstream file given." << std::endl;
174  return -1;
175  }
176 
177  cInputFile = vvencappCfg.m_inputFileName;
178  cOutputfile = vvencappCfg.m_bitstreamFileName;
179 
180  if( vvencappCfg.m_verbosity > VVENC_SILENT && vvencappCfg.m_verbosity < VVENC_NOTICE )
181  {
182  std::cout << "-------------------" << std::endl;
183  std::cout << cAppname << " version " << vvenc_get_version() << std::endl;
184  }
185 
186  vvencEncoder *enc = vvenc_encoder_create();
187  if( nullptr == enc )
188  {
189  return -1;
190  }
191 
192  // initialize the encoder
193  iRet = vvenc_encoder_open( enc, &vvencappCfg );
194  if( 0 != iRet )
195  {
196  printVVEncErrorMsg( cAppname, "cannot create encoder", iRet, vvenc_get_last_error( enc ) );
197  vvenc_encoder_close( enc );
198  return iRet;
199  }
200 
201  if( vvencappCfg.m_verbosity > VVENC_WARNING )
202  {
203  std::cout << cAppname << ": " << vvenc_get_enc_information( enc ) << std::endl;
204  }
205 
206  vvenc_get_config( enc, &vvencappCfg ); // get the adapted config, because changes are needed for the yuv reader (m_MSBExtendedBitDepth)
207 
208  if( vvencappCfg.m_verbosity >= VVENC_INFO )
209  {
210  std::cout << vvencappCfg.getConfigAsString( vvencappCfg.m_verbosity ) << std::endl;
211  }
212 
213  // open output file
214  std::ofstream cOutBitstream;
215  if( !cOutputfile.empty() )
216  {
217  cOutBitstream.open( cOutputfile.c_str(), std::ios::out | std::ios::binary | std::ios::trunc);
218  if( !cOutBitstream.is_open() )
219  {
220  std::cout << cAppname << " [error]: failed to open output file " << cOutputfile << std::endl;
221  return -1;
222  }
223  }
224 
225  // --- allocate memory for output packets
226  vvencAccessUnit AU;
227  vvenc_accessUnit_default( &AU );
228  vvenc_accessUnit_alloc_payload( &AU, vvencappCfg.m_SourceWidth * vvencappCfg.m_SourceHeight );
229 
230  // --- allocate memory for YUV input picture
231  vvencYUVBuffer cYUVInputBuffer;
232  vvenc_YUVBuffer_default( &cYUVInputBuffer );
233  vvenc_YUVBuffer_alloc_buffer( &cYUVInputBuffer, vvencappCfg.m_internChromaFormat, vvencappCfg.m_SourceWidth, vvencappCfg.m_SourceHeight );
234 
235  // --- start timer
236  std::chrono::steady_clock::time_point cTPStartRun = std::chrono::steady_clock::now();
237  if( vvencappCfg.m_verbosity > VVENC_WARNING )
238  {
239  std::time_t startTime2 = std::chrono::system_clock::to_time_t(std::chrono::system_clock::now());
240  std::cout << "started @ " << std::ctime(&startTime2) << std::endl;
241  }
242 
243  // calc temp. rate/scale
244  int temporalRate = vvencappCfg.m_FrameRate;
245  int temporalScale = 1;
246 
247  switch( vvencappCfg.m_FrameRate )
248  {
249  case 23: temporalRate = 24000; temporalScale = 1001; break;
250  case 29: temporalRate = 30000; temporalScale = 1001; break;
251  case 59: temporalRate = 60000; temporalScale = 1001; break;
252  default: break;
253  }
254 
255  unsigned int uiFrames = 0;
256  for( int pass = 0; pass < vvencappCfg.m_RCNumPasses; pass++ )
257  {
258  // initialize the encoder pass
259  iRet = vvenc_init_pass( enc, pass );
260  if( 0 != iRet )
261  {
262  printVVEncErrorMsg( cAppname, "cannot init encoder", iRet, vvenc_get_last_error( enc ) );
263  return iRet;
264  }
265 
266  // open the input file
267  apputils::YuvFileIO cYuvFileInput;
268  if( 0 != cYuvFileInput.open( cInputFile, false, vvencappCfg.m_inputBitDepth[0], vvencappCfg.m_MSBExtendedBitDepth[0], vvencappCfg.m_internalBitDepth[0],
269  vvencappCfg.m_inputFileChromaFormat, vvencappCfg.m_internChromaFormat, vvencappCfg.m_bClipOutputVideoToRec709Range, false ) )
270  {
271  std::cout << cAppname << " [error]: failed to open input file " << cInputFile << std::endl;
272  vvenc_YUVBuffer_free_buffer( &cYUVInputBuffer );
273  vvenc_accessUnit_free_payload( &AU );
274  vvenc_encoder_close( enc );
275  return -1;
276  }
277 
278  const int iFrameSkip = std::max( vvencappCfg.m_FrameSkip - vvenc_get_num_lead_frames(enc), 0 );
279  const int64_t iMaxFrames = vvencappCfg.m_framesToBeEncoded + vvenc_get_num_lead_frames(enc) + vvenc_get_num_trail_frames(enc);
280  int64_t iSeqNumber = 0;
281  bool bEof = false;
282  bool bEncodeDone = false;
283 
284  uiFrames = 0;
285 
286  if( iFrameSkip )
287  {
288  cYuvFileInput.skipYuvFrames(iFrameSkip, vvencappCfg.m_SourceWidth, vvencappCfg.m_SourceHeight);
289  iSeqNumber=iFrameSkip;
290  }
291 
292  while( !bEof || !bEncodeDone )
293  {
294  vvencYUVBuffer* ptrYUVInputBuffer = nullptr;
295  if( !bEof )
296  {
297  if( 0 != cYuvFileInput.readYuvBuf( cYUVInputBuffer, bEof ) )
298  {
299  std::cout << " [error]: read file failed: " << cYuvFileInput.getLastError() << std::endl;
300  vvenc_YUVBuffer_free_buffer( &cYUVInputBuffer );
301  vvenc_accessUnit_free_payload( &AU );
302  vvenc_encoder_close( enc );
303  return -1;
304  }
305  if( ! bEof )
306  {
307  // set sequence number and cts
308  cYUVInputBuffer.sequenceNumber = iSeqNumber;
309  cYUVInputBuffer.cts = iSeqNumber * vvencappCfg.m_TicksPerSecond * temporalScale / temporalRate;
310  cYUVInputBuffer.ctsValid = true;
311  ptrYUVInputBuffer = &cYUVInputBuffer;
312  iSeqNumber++;
313  //std::cout << "process picture " << cYUVInputBuffer.m_uiSequenceNumber << " cts " << cYUVInputBuffer.m_uiCts << std::endl;
314  }
315  else if( vvencappCfg.m_verbosity > VVENC_ERROR && vvencappCfg.m_verbosity < VVENC_NOTICE )
316  {
317  std::cout << "EOF reached" << std::endl;
318  }
319  }
320 
321  // call encode
322  iRet = vvenc_encode( enc, ptrYUVInputBuffer, &AU, &bEncodeDone );
323  if( 0 != iRet )
324  {
325  printVVEncErrorMsg( cAppname, "encoding failed", iRet, vvenc_get_last_error( enc ) );
326  vvenc_YUVBuffer_free_buffer( &cYUVInputBuffer );
327  vvenc_accessUnit_free_payload( &AU );
328  vvenc_encoder_close( enc );
329  return iRet;
330  }
331 
332  if( AU.payloadUsedSize > 0 )
333  {
334  if( cOutBitstream.is_open() )
335  {
336  // write output
337  cOutBitstream.write( (const char*)AU.payload, AU.payloadUsedSize );
338  }
339  uiFrames++;
340  }
341 
342  if( iMaxFrames > 0 && iSeqNumber >= ( iFrameSkip + iMaxFrames ) )
343  {
344  bEof = true;
345  }
346  }
347 
348  cYuvFileInput.close();
349  }
350 
351  std::chrono::steady_clock::time_point cTPEndRun = std::chrono::steady_clock::now();
352  double dTimeSec = (double)std::chrono::duration_cast<std::chrono::milliseconds>((cTPEndRun)-(cTPStartRun)).count() / 1000;
353 
354  if( cOutBitstream.is_open() )
355  {
356  cOutBitstream.close();
357  }
358 
359  vvenc_print_summary(enc);
360 
361  // un-initialize the encoder
362  iRet = vvenc_encoder_close( enc );
363  if( 0 != iRet )
364  {
365  printVVEncErrorMsg( cAppname, "destroy encoder failed", iRet, vvenc_get_last_error( enc ) );
366  return iRet;
367  }
368 
369  vvenc_YUVBuffer_free_buffer( &cYUVInputBuffer );
370  vvenc_accessUnit_free_payload( &AU );
371 
372  if( 0 == uiFrames )
373  {
374  std::cout << "no frames encoded" << std::endl;
375  }
376 
377  if( uiFrames && vvencappCfg.m_verbosity > VVENC_SILENT )
378  {
379  if( vvencappCfg.m_verbosity > VVENC_WARNING )
380  {
381  std::time_t endTime2 = std::chrono::system_clock::to_time_t(std::chrono::system_clock::now());
382  std::cout << "finished @ " << std::ctime(&endTime2) << std::endl;
383  }
384 
385  double dFps = (double)uiFrames / dTimeSec;
386  std::cout << "Total Time: " << dTimeSec << " sec. Fps(avg): " << dFps << " encoded Frames " << uiFrames << std::endl;
387  }
388 
389  return 0;
390 }
