62 static constexpr int PROB_BITS = 15; // Nominal number of bits to represent probabilities
63 static constexpr int PROB_BITS_0 = 10; // Number of bits to represent 1st estimate
64 static constexpr int PROB_BITS_1 = 14; // Number of bits to represent 2nd estimate
65 static constexpr int MASK_0 = ~(~0u << PROB_BITS_0) << (PROB_BITS - PROB_BITS_0);
66 static constexpr int MASK_1 = ~(~0u << PROB_BITS_1) << (PROB_BITS - PROB_BITS_1);
67 static constexpr uint8_t DWS = 8; // 0x47 Default window sizes

91 class BinProbModel : public BinProbModelBase
92 {
93 public:
94  BinProbModel()
95  {
96  uint16_t half = 1 << (PROB_BITS - 1);
97  m_state[0] = half;
98  m_state[1] = half;
99  m_rate = DWS;
100  }
101 
102 public:
103  void init ( int qp, int initId );
104  void update(unsigned bin)
105  {
106  int rate0 = m_rate >> 4;
107  int rate1 = m_rate & 15;
108 
109  m_state[0] -= ( m_state[0] >> rate0 ) & MASK_0;
110  m_state[1] -= ( m_state[1] >> rate1 ) & MASK_1;
111 
112  m_state[0] += ( ( -static_cast< int >( bin ) & 0x7fffu ) >> rate0 ) & MASK_0;
113  m_state[1] += ( ( -static_cast< int >( bin ) & 0x7fffu ) >> rate1 ) & MASK_1;
114  }
115  void setLog2WindowSize(uint8_t log2WindowSize)
116  {
117  int rate0 = 2 + ((log2WindowSize >> 2) & 3);
118  int rate1 = 3 + rate0 + (log2WindowSize& 3);
119  m_rate = 16 * rate0 + rate1;
120  CHECK(rate1 > 9, "Second window size is too large!");
121  }
122  void estFracBitsUpdate(unsigned bin, uint64_t &b)
123  {
124  b += estFracBits(bin);
125  update(bin);
126  }
127  uint32_t estFracBits(unsigned bin) const { return getFracBitsArray().intBits[bin]; }
128  static uint32_t estFracBitsTrm(unsigned bin) { return (bin ? 0x3bfbb : 0x0010c); }
129  const BinFracBits &getFracBitsArray() const { return m_binFracBits[state()]; }
130 public:
131  uint8_t state() const { return (m_state[0] + m_state[1]) >> 8; }
132  uint8_t mps() const { return state() >> 7; }
133  uint8_t getLPS(unsigned range) const
134  {
135  uint16_t q = state();
136  if (q & 0x80)
137  q = q ^ 0xff;
138  return ((q >> 2) * (range >> 5) >> 1) + 4;
139  }
140  static uint8_t getRenormBitsLPS ( unsigned LPS ) { return m_RenormTable_32 [LPS>>3]; }
141  static uint8_t getRenormBitsRange( unsigned range ) { return 1; }
142  uint16_t getState() const { return m_state[0] + m_state[1]; }
143  void setState(uint16_t pState)
144  {
145  m_state[0] = (pState >> 1) & MASK_0;
146  m_state[1] = (pState >> 1) & MASK_1;
147  }
148 public:
149  uint64_t estFracExcessBits(const BinProbModel &r) const
150  {
151  int n = 2 * state() + 1;
152  return ((512 - n) * r.estFracBits(0) + n * r.estFracBits(1) + 256) >> 9;
153  }
154 private:
155  uint16_t m_state[2];
156  uint8_t m_rate;
157 };
