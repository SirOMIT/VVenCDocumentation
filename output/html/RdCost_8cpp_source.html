141 void RdCost::setDistParam( DistParam &rcDP, const CPelBuf& org, const Pel* piRefY, int iRefStride, int bitDepth, ComponentID compID, int subShiftMode, bool useHadamard )
142 {
143  rcDP.bitDepth = bitDepth;
144  rcDP.compID = compID;
145 
146  // set Original & Curr Pointer / Stride
147  rcDP.org = org;
148 
149  rcDP.cur.buf = piRefY;
150  rcDP.cur.stride = iRefStride;
151 
152  // set Block Width / Height
153  rcDP.cur.width = org.width;
154  rcDP.cur.height = org.height;
155  rcDP.maximumDistortionForEarlyExit = MAX_DISTORTION;
156 
157  const int base = (rcDP.bitDepth > 10 || rcDP.applyWeight) ? 1 : 0;
158  if( !useHadamard )
159  {
160  rcDP.distFunc = m_afpDistortFunc[base][ DF_SAD + Log2( org.width ) ];
161  }
162  else
163  {
164  rcDP.distFunc = m_afpDistortFunc[base][ DF_HAD + Log2( org.width ) ];
165  }
166 
167  // initialize
168  rcDP.subShift = 0;
169 
170  if( subShiftMode == 1 )
171  {
172  if( rcDP.org.height > 32 && ( rcDP.org.height & 15 ) == 0 )
173  {
174  rcDP.subShift = 4;
175  }
176  else if( rcDP.org.height > 16 && ( rcDP.org.height & 7 ) == 0 )
177  {
178  rcDP.subShift = 3;
179  }
180  else if( rcDP.org.height > 8 && ( rcDP.org.height & 3 ) == 0 )
181  {
182  rcDP.subShift = 2;
183  }
184  else if( ( rcDP.org.height & 1 ) == 0 )
185  {
186  rcDP.subShift = 1;
187  }
188  }
189  else if( subShiftMode == 2 )
190  {
191  if( rcDP.org.height > 8 && rcDP.org.width <= 64 )
192  {
193  rcDP.subShift = 1;
194  }
195  }
196  else if (subShiftMode == 3)
197  {
198  if (rcDP.org.height > 8)
199  {
200  rcDP.subShift = 1;
201  }
202  }
203 
204 #if ENABLE_MEASURE_SEARCH_SPACE
205  rcDP.xDistFunc = rcDP.distFunc;
206  rcDP.distFunc = xMeasurePredSearchSpaceInterceptor;
207 #endif
208 }

264 Distortion RdCost::getDistPart( const CPelBuf& org, const CPelBuf& cur, int bitDepth, const ComponentID compId, DFunc eDFunc, const CPelBuf* orgLuma )
265 {
266  DistParam dp( org, cur, nullptr, bitDepth, 0, compId );
267 # if ENABLE_MEASURE_SEARCH_SPACE
268  g_searchSpaceAcc.addPrediction( dp.cur.width, dp.cur.height, toChannelType( dp.compID ) );
269 #endif
270  Distortion dist;
271  if( orgLuma )
272  {
273  CHECKD( eDFunc != DF_SSE_WTD, "mismatch func and parameter")
274  dp.orgLuma = orgLuma;
275  dist = RdCost::xGetSSE_WTD( dp );
276  }
277  else
278  {
279  if( ( org.width == 1 ) )
280  {
281  dist = xGetSSE( dp );
282  }
283  else
284  {
285  const int base = (bitDepth > 10) ? 1 : 0;
286  dist = m_afpDistortFunc[base][eDFunc + Log2(org.width)](dp);
287  }
288  }
289  if (isChroma(compId))
290  {
291  return ((Distortion) (m_distortionWeight[ compId ] * dist));
292  }
293  else
294  {
295  return dist;
296  }
297 }

307 Distortion RdCost::xGetSAD( const DistParam& rcDtParam )
308 {
309  if ( rcDtParam.applyWeight )
310  {
311  THROW(" no support");
312  }
313 
314  const Pel* piOrg = rcDtParam.org.buf;
315  const Pel* piCur = rcDtParam.cur.buf;
316  const int iCols = rcDtParam.org.width;
317  int iRows = rcDtParam.org.height;
318  const int iSubShift = rcDtParam.subShift;
319  const int iSubStep = ( 1 << iSubShift );
320  const int iStrideCur = rcDtParam.cur.stride * iSubStep;
321  const int iStrideOrg = rcDtParam.org.stride * iSubStep;
322  const uint32_t distortionShift = DISTORTION_PRECISION_ADJUSTMENT(rcDtParam.bitDepth);
323 
324  Distortion uiSum = 0;
325 
326  for( ; iRows != 0; iRows -= iSubStep )
327  {
328  for (int n = 0; n < iCols; n++ )
329  {
330  uiSum += abs( piOrg[n] - piCur[n] );
331  }
332  if (rcDtParam.maximumDistortionForEarlyExit < ( uiSum >> distortionShift ))
333  {
334  return ( uiSum >> distortionShift );
335  }
336  piOrg += iStrideOrg;
337  piCur += iStrideCur;
338  }
339 
340  uiSum <<= iSubShift;
341  return ( uiSum >> distortionShift );
342 }
