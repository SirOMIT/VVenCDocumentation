70 template<X86_VEXT vext>
71 inline void xPelLumaCore( int64_t m0, int64_t& m1, int64_t& m2, int64_t& m3, int64_t& m4, int64_t& m5, int64_t& m6, int64_t m7, const int tc )
72 {
73  static constexpr int64_t val4 = 4ll | ( 4ll << 32ll ) | ( ( 4ll | ( 4ll << 32ll ) ) << 16ll );
74  static constexpr int64_t val2 = 2ll | ( 2ll << 32ll ) | ( ( 2ll | ( 2ll << 32ll ) ) << 16ll );
75 
76  const int64_t m1234 = m1 + m2 + m3 + m4;
77  const int64_t m3456 = m3 + m4 + m5 + m6;
78 
79  const int64_t r1 = m1234 + val4 + ( ( m0 + m1 ) << 1 );
80  const int64_t r2 = m1234 + val2;
81  const int64_t r3 = ( m1234 << 1 ) + m5 + val4 - m1;
82  const int64_t r4 = ( m3456 << 1 ) + m2 + val4 - m6;
83  const int64_t r5 = m3456 + val2;
84  const int64_t r6 = m3456 + val4 + ( ( m6 + m7 ) << 1 );
85 
86  const char tc3[3] = { 3, 2, 1 };
87 
88  const __m128i vtc0 = _mm_set1_epi16 ( tc3[0] * tc );
89  const __m128i vtc1 = _mm_set1_epi16 ( tc3[1] * tc );
90  const __m128i vtc2 = _mm_set1_epi16 ( tc3[2] * tc );
91  const __m128i vzero = _mm_setzero_si128();
92 
93  __m128i vmax = vtc1;
94  __m128i vmin = _mm_sub_epi16 ( vzero, vmax );
95 
96  __m128i org = _mm_set_epi64x ( m2, m5 );
97  __m128i vec = _mm_set_epi64x ( r2, r5 );
98  vec = _mm_srli_epi16 ( vec, 2 );
99  vec = _mm_sub_epi16 ( vec, org );
100  vec = _mm_min_epi16 ( vmax, _mm_max_epi16( vmin, vec ) );
101  vec = _mm_add_epi16 ( vec, org );
102  m5 = _mm_cvtsi128_si64( vec );
103  m2 = _mm_extract_epi64( vec, 1 );
104 
105  vmax = _mm_blend_epi16 ( vtc0, vtc2, 0xf0 );
106  vmin = _mm_sub_epi16 ( vzero, vmax );
107  org = _mm_set_epi64x ( m1, m3 );
108  vec = _mm_set_epi64x ( r1, r3 );
109  vec = _mm_srli_epi16 ( vec, 3 );
110  vec = _mm_sub_epi16 ( vec, org );
111  vec = _mm_min_epi16 ( vmax, _mm_max_epi16( vmin, vec ) );
112  vec = _mm_add_epi16 ( vec, org );
113  m3 = _mm_cvtsi128_si64( vec );
114  m1 = _mm_extract_epi64( vec, 1 );
115 
116  vmax = _mm_blend_epi16 ( vtc2, vtc0, 0xf0 );
117  vmin = _mm_sub_epi16 ( vzero, vmax );
118  org = _mm_set_epi64x ( m4, m6 );
119  vec = _mm_set_epi64x ( r4, r6 );
120  vec = _mm_srli_epi16 ( vec, 3 );
121  vec = _mm_sub_epi16 ( vec, org );
122  vec = _mm_min_epi16 ( vmax, _mm_max_epi16( vmin, vec ) );
123  vec = _mm_add_epi16 ( vec, org );
124  m6 = _mm_cvtsi128_si64( vec );
125  m4 = _mm_extract_epi64( vec, 1 );
126 }

143 template<X86_VEXT vext>
144 static inline void xPelFilterLumaLoopVer( Pel* piSrc, const ptrdiff_t step, const int tc )
145 {
150 
151  __m128i va01 = _mm_loadu_si128( ( const __m128i* ) &piSrc[-4 + 0 * step] );
152  __m128i va23 = _mm_loadu_si128( ( const __m128i* ) &piSrc[-4 + 1 * step] );
153  __m128i va45 = _mm_loadu_si128( ( const __m128i* ) &piSrc[-4 + 2 * step] );
154  __m128i va67 = _mm_loadu_si128( ( const __m128i* ) &piSrc[-4 + 3 * step] );
155 
156  __m128i va01a23lo = _mm_unpacklo_epi16( va01, va23 );
157  __m128i va01a23hi = _mm_unpackhi_epi16( va01, va23 );
158  __m128i va45a67lo = _mm_unpacklo_epi16( va45, va67 );
159  __m128i va45a67hi = _mm_unpackhi_epi16( va45, va67 );
160 
161  va01 = _mm_unpacklo_epi32( va01a23lo, va45a67lo );
162  va23 = _mm_unpackhi_epi32( va01a23lo, va45a67lo );
163  va45 = _mm_unpacklo_epi32( va01a23hi, va45a67hi );
164  va67 = _mm_unpackhi_epi32( va01a23hi, va45a67hi );
165 
166  int64_t m0 = _mm_extract_epi64( va01, 0 );
167  int64_t m1 = _mm_extract_epi64( va01, 1 );
168  int64_t m2 = _mm_extract_epi64( va23, 0 );
169  int64_t m3 = _mm_extract_epi64( va23, 1 );
170  int64_t m4 = _mm_extract_epi64( va45, 0 );
171  int64_t m5 = _mm_extract_epi64( va45, 1 );
172  int64_t m6 = _mm_extract_epi64( va67, 0 );
173  int64_t m7 = _mm_extract_epi64( va67, 1 );
174 
175  // do the loop filter of the the 4x8 matrix
176  xPelLumaCore<vext>( m0, m1, m2, m3, m4, m5, m6, m7, tc );
177 
178  // Transpose back
179  va01 = _mm_set_epi64x( m4, m0 );
180  va23 = _mm_set_epi64x( m5, m1 );
181  va45 = _mm_set_epi64x( m6, m2 );
182  va67 = _mm_set_epi64x( m7, m3 );
183 
184  va01a23lo = _mm_unpacklo_epi16( va01, va23 );
185  va01a23hi = _mm_unpackhi_epi16( va01, va23 );
186  va45a67lo = _mm_unpacklo_epi16( va45, va67 );
187  va45a67hi = _mm_unpackhi_epi16( va45, va67 );
188 
189  va01 = _mm_unpacklo_epi32( va01a23lo, va45a67lo );
190  va23 = _mm_unpackhi_epi32( va01a23lo, va45a67lo );
191  va45 = _mm_unpacklo_epi32( va01a23hi, va45a67hi );
192  va67 = _mm_unpackhi_epi32( va01a23hi, va45a67hi );
193 
194  __m128i vzr = _mm_setzero_si128();
195 
196  _mm_storel_epi64( ( __m128i* ) &piSrc[-4 + 0 * step], _mm_unpacklo_epi64( va01, vzr ) );
197  _mm_storel_epi64( ( __m128i* ) &piSrc[-4 + 1 * step], _mm_unpackhi_epi64( va01, vzr ) );
198  _mm_storel_epi64( ( __m128i* ) &piSrc[-4 + 2 * step], _mm_unpacklo_epi64( va23, vzr ) );
199  _mm_storel_epi64( ( __m128i* ) &piSrc[-4 + 3 * step], _mm_unpackhi_epi64( va23, vzr ) );
200  _mm_storel_epi64( ( __m128i* ) &piSrc[ 0 + 0 * step], _mm_unpacklo_epi64( va45, vzr ) );
201  _mm_storel_epi64( ( __m128i* ) &piSrc[ 0 + 1 * step], _mm_unpackhi_epi64( va45, vzr ) );
202  _mm_storel_epi64( ( __m128i* ) &piSrc[ 0 + 2 * step], _mm_unpacklo_epi64( va67, vzr ) );
203  _mm_storel_epi64( ( __m128i* ) &piSrc[ 0 + 3 * step], _mm_unpackhi_epi64( va67, vzr ) );
204 }

206 template<X86_VEXT vext>
207 static inline void xPelFilterLumaWeakCore( __m128i &vm1, __m128i &vm2, __m128i &vm3, __m128i &vm4, __m128i &vm5, __m128i &vm6, const int tc, const int iThrCut, const bool bFilterSecondP, const bool bFilterSecondQ, const ClpRng& clpRng )
208 {
209  __m128i vmin = _mm_set1_epi16( clpRng.min );
210  __m128i vmax = _mm_set1_epi16( clpRng.max );
211 
212  __m128i vtmp0 = _mm_sub_epi16( vm4, vm3 );
213  __m128i vtmp1 = _mm_sub_epi16( vm5, vm2 );
214  __m128i vm3s = vm3;
215  __m128i vm4s = vm4;
216 
217  vtmp0 = _mm_unpacklo_epi16( vtmp0, vtmp1 );
218  vtmp1 = _mm_set1_epi32( 0xfffd0009 );
219 
220  vtmp0 = _mm_madd_epi16( vtmp0, vtmp1 );
221  vtmp1 = _mm_add_epi32( vtmp0, _mm_set1_epi32( 8 ) );
222  __m128i vdlt = _mm_srai_epi32( vtmp1, 4 );
223  vdlt = _mm_packs_epi32( vdlt, _mm_setzero_si128() );
224 
225  short deltaV[4] = { 0, 0, 0, 0 };
226  _mm_storel_epi64( ( __m128i* ) deltaV, vdlt );
227 
228  __m128i vmsk = _mm_cmpgt_epi16( _mm_set1_epi16( iThrCut ), _mm_abs_epi16( vdlt ) );
229  __m128i vtc = _mm_set1_epi16( -tc );
230  vdlt = _mm_max_epi16( vdlt, vtc );
231  vtc = _mm_set1_epi16( tc );
232  vdlt = _mm_min_epi16( vdlt, vtc );
233  vtc = _mm_set1_epi16( tc >> 1 );
234  __m128i vtcn = _mm_set1_epi16( -( tc >> 1 ) );
235 
236  vtmp0 = _mm_add_epi16( vm3, vdlt );
237  vtmp1 = _mm_min_epi16( _mm_max_epi16( vmin, vtmp0 ), vmax );
238  vtmp0 = _mm_blendv_epi8( vm3, vtmp1, vmsk );
239  vm3 = _mm_unpacklo_epi64( vtmp0, _mm_setzero_si128() );
240  //_mm_storel_epi64( ( __m128i* ) &m3, vtmp0 );
241 
242  if( bFilterSecondP )
243  {
244  vtmp0 = _mm_srli_epi16( _mm_add_epi16( _mm_add_epi16( vm1, vm3s ), _mm_set1_epi16( 1 ) ), 1 );
245  vtmp1 = _mm_srai_epi16( _mm_add_epi16( _mm_sub_epi16( vtmp0, vm2 ), vdlt ), 1 );
246  vtmp0 = _mm_min_epi16( _mm_max_epi16( vtmp1, vtcn ), vtc );
247 
248  vtmp1 = _mm_add_epi16( vm2, vtmp0 );
249  vtmp0 = _mm_min_epi16( _mm_max_epi16( vmin, vtmp1 ), vmax );
250  vtmp1 = _mm_blendv_epi8( vm2, vtmp0, vmsk );
251  vm2 = _mm_unpacklo_epi64( vtmp1, _mm_setzero_si128() );
252  //_mm_storel_epi64( ( __m128i* ) &m2, vtmp1 );
253  }
254 
255  vtmp0 = _mm_sub_epi16( vm4, vdlt );
256  vtmp1 = _mm_min_epi16( _mm_max_epi16( vmin, vtmp0 ), vmax );
257  vtmp0 = _mm_blendv_epi8( vm4, vtmp1, vmsk );
258  vm4 = _mm_unpacklo_epi64( vtmp0, _mm_setzero_si128() );
259  //_mm_storel_epi64( ( __m128i* ) &m4, vtmp0 );
260 
261  if( bFilterSecondQ )
262  {
263  vtmp0 = _mm_srli_epi16( _mm_add_epi16( _mm_add_epi16( vm6, vm4s ), _mm_set1_epi16( 1 ) ), 1 );
264  vtmp1 = _mm_srai_epi16( _mm_sub_epi16( _mm_sub_epi16( vtmp0, vm5 ), vdlt ), 1 );
265  vtmp0 = _mm_min_epi16( _mm_max_epi16( vtmp1, vtcn ), vtc );
266 
267  vtmp1 = _mm_add_epi16( vm5, vtmp0 );
268  vtmp0 = _mm_min_epi16( _mm_max_epi16( vmin, vtmp1 ), vmax );
269  vtmp1 = _mm_blendv_epi8( vm5, vtmp0, vmsk );
270  vm5 = _mm_unpacklo_epi64( vtmp1, _mm_setzero_si128() );
271  //_mm_storel_epi64( ( __m128i* ) &m5, vtmp1 );
272  }
273 }

368 template<X86_VEXT vext>
369 static inline void xFilteringPandQX86Hor( Pel* src, ptrdiff_t step, const ptrdiff_t offset, int numberPSide, int numberQSide, int tc )
370 {
371  CHECKD( step != 1, "Offset has to be '1' for vertical edge filtering!" );
372 
373  const int* dbCoeffsP = numberPSide == 7 ? dbCoeffs7 : ( numberPSide == 5 ) ? dbCoeffs5 : dbCoeffs3;
374  const int* dbCoeffsQ = numberQSide == 7 ? dbCoeffs7 : ( numberQSide == 5 ) ? dbCoeffs5 : dbCoeffs3;
375  const int* tcP = numberPSide == 3 ? tc3 : tc7;
376  const int* tcQ = numberQSide == 3 ? tc3 : tc7;
377 
378  uint64_t refPx4 = 0, refQx4 = 0, refMiddlex4 = 0;
379 
380  for( int i = 0; i < DEBLOCK_SMALLEST_BLOCK / 2; i++ )
381  {
382  Pel* srcP = src + step * i - offset;
383  Pel* srcQ = src + step * i;
384 
385  Pel refMiddle = 0;
386  const Pel refP = ( srcP[-numberPSide * offset] + srcP[-( numberPSide - 1 ) * offset] + 1 ) >> 1;
387  const Pel refQ = ( srcQ[ numberQSide * offset] + srcQ[ ( numberQSide - 1 ) * offset] + 1 ) >> 1;
388 
389  if( numberPSide == numberQSide )
390  {
391  if( numberPSide == 5 )
392  {
393  refMiddle = ( 2 * ( srcP[0] + srcQ[0] + srcP[-offset] + srcQ[offset] + srcP[-2 * offset] + srcQ[2 * offset] ) + srcP[-3 * offset] + srcQ[3 * offset] + srcP[-4 * offset] + srcQ[4 * offset] + 8 ) >> 4;
394  }
395  else
396  {
397  refMiddle = ( 2 * ( srcP[0] + srcQ[0] ) + srcP[-offset] + srcQ[offset] + srcP[-2 * offset] + srcQ[2 * offset] + srcP[-3 * offset] + srcQ[3 * offset] + srcP[-4 * offset] + srcQ[4 * offset] + srcP[-5 * offset] + srcQ[5 * offset] + +srcP[-6 * offset] + srcQ[6 * offset] + 8 ) >> 4;
398  }
399  }
400  else
401  {
402  Pel* srcPt = srcP;
403  Pel* srcQt = srcQ;
404  ptrdiff_t offsetP = -offset;
405  ptrdiff_t offsetQ = offset;
406 
407  int newNumberQSide = numberQSide;
408  int newNumberPSide = numberPSide;
409 
410  if( numberQSide > numberPSide )
411  {
412  std::swap( srcPt, srcQt );
413  std::swap( offsetP, offsetQ );
414  newNumberQSide = numberPSide;
415  newNumberPSide = numberQSide;
416  }
417 
418  if( newNumberPSide == 7 && newNumberQSide == 5 )
419  {
