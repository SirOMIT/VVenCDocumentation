125 vvencLevel LevelTierFeatures::getLevelForInput( uint32_t width, uint32_t height, bool tier, int temporalRate, int temporalScale, int bitrate )
126 {
127  uint64_t samplesPerSec = ((uint64_t)temporalRate*(uint64_t)width*(uint64_t)height)/temporalScale;
128  uint32_t br = bitrate ? (uint32_t)bitrate/1000 : std::min((uint32_t)320000,(uint32_t)(samplesPerSec/3000)); //assume a min compression factor of 36 in the case of fix qp
129 
130  for (const auto& info: mainLevelTierInfo )
131  {
132  if ( width <= info.getMaxPicWidthInLumaSamples() &&
133  height <= info.getMaxPicHeightInLumaSamples() &&
134  samplesPerSec <= info.maxLumaSr &&
135  br <= info.maxBr[tier?1:0] &&
136  info.level != VVENC_LEVEL_AUTO )
137  {
138  return info.level;
139  }
140  }
141 
142  return VVENC_NUMBER_OF_LEVELS;
143 }

145 void ProfileLevelTierFeatures::extractPTLInformation(const SPS &sps)
146 {
147  const ProfileTierLevel &spsPtl = sps.profileTierLevel;
148 
149  m_pProfile = nullptr;
150  m_pLevelTier = nullptr;
151  m_tier = spsPtl.tierFlag;
152 
153  // Identify the profile from the profile Idc, and possibly other constraints.
154  for(int32_t i=0; validProfiles[i].profile != VVENC_PROFILE_AUTO; i++)
155  {
156  if (spsPtl.profileIdc == validProfiles[i].profile)
157  {
158  m_pProfile = &(validProfiles[i]);
159  break;
160  }
161  }
162 
163  if (m_pProfile != nullptr)
164  {
165  // Now identify the level:
166  const LevelTierFeatures *pLTF = m_pProfile->pLevelTiersListInfo;
167  const vvencLevel spsLevelName = spsPtl.levelIdc;
168  if (spsLevelName!=VVENC_LEVEL15_5 || m_pProfile->canUseLevel15p5)
169  {
170  for(int i=0; pLTF[i].level!=VVENC_LEVEL_AUTO; i++)
171  {
172  if (pLTF[i].level == spsLevelName)
173  {
174  m_pLevelTier = &(pLTF[i]);
175  break;
176  }
177  }
178  }
179  }
180 }

192 uint32_t ProfileLevelTierFeatures::getMaxDpbSize( uint32_t picSizeMaxInSamplesY ) const
193 {
194  const uint32_t maxDpbPicBuf = 8;
195  uint32_t maxDpbSize;
196 
197  if (m_pLevelTier->level == VVENC_LEVEL15_5)
198  {
199  // maxDpbSize is unconstrained in this case
200  maxDpbSize = std::numeric_limits<uint32_t>::max();
201  }
202  else if (2 * picSizeMaxInSamplesY <= m_pLevelTier->maxLumaPs)
203  {
204  maxDpbSize = 2 * maxDpbPicBuf;
205  }
206  else if (3 * picSizeMaxInSamplesY <= 2 * m_pLevelTier->maxLumaPs)
207  {
208  maxDpbSize = 3 * maxDpbPicBuf / 2;
209  }
210  else
211  {
212  maxDpbSize = maxDpbPicBuf;
213  }
214 
215  return maxDpbSize;
216 }
