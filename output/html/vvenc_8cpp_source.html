94 VVENC_DECL void vvenc_YUVBuffer_alloc_buffer( vvencYUVBuffer *yuvBuffer, const vvencChromaFormat chFmt, const int frameWidth, const int frameHeight )
95 {
96  for ( int i = 0; i < 3; i++ )
97  {
98  vvencYUVPlane& yuvPlane = yuvBuffer->planes[ i ];
99  yuvPlane.width = vvenc_get_width_of_component ( chFmt, frameWidth, i );
100  yuvPlane.height = vvenc_get_height_of_component( chFmt, frameHeight, i );
101  yuvPlane.stride = yuvPlane.width;
102  const int size = yuvPlane.stride * yuvPlane.height;
103  yuvPlane.ptr = ( size > 0 ) ? new int16_t[ size ] : nullptr;
104  }
105 }

192 VVENC_DECL int vvenc_encoder_open( vvencEncoder *enc, vvenc_config* config )
193 {
194  if (nullptr == config)
195  {
196  vvenc::msg( VVENC_ERROR, "vvenc_config is null\n" );
197  return VVENC_ERR_PARAMETER;
198  }
199 
200  auto d = (vvenc::VVEncImpl*)enc;
201  if (!d)
202  {
203  return VVENC_ERR_INITIALIZE;
204  }
205 
206  int ret = d->init( *config );
207  if (ret != 0)
208  {
209  // Error initializing the decoder
210  vvenc::msg( VVENC_ERROR, "cannot init the VVenC encoder\n" );
211  return VVENC_ERR_INITIALIZE;
212  }
213 
214  return VVENC_OK;
215 }

400 VVENC_DECL int vvenc_get_width_of_component( const vvencChromaFormat chFmt, const int frameWidth, const int compId )
401 {
402  int w = frameWidth;
403  if ( compId > 0 )
404  {
405  switch ( chFmt )
406  {
407  case VVENC_CHROMA_400: w = 0; break;
408  case VVENC_CHROMA_420:
409  case VVENC_CHROMA_422: w = w >> 1; break;
410  default: break;
411  }
412  }
413  return w;
414 }
