128 int VVEncImpl::init( const vvenc_config& config )
129 {
130  if( m_bInitialized ){ return VVENC_ERR_INITIALIZE; }
131 
132  // Set SIMD extension in case if it hasn't been done before, otherwise it simply reuses the current state
133  std::string curSimd;
134  const char* pSimd = vvenc_set_SIMD_extension( curSimd.c_str() );
135  pSimd == nullptr ? curSimd = "NA" : curSimd = pSimd;
136 
137  m_cVVEncCfgExt = config;
138  m_cVVEncCfg = config;
139 
140  if ( vvenc_init_config_parameter(&m_cVVEncCfg) ) // init auto/dependent options
141  {
142  return VVENC_ERR_INITIALIZE;
143  }
144 
145  std::stringstream cssCap;
146  cssCap << getCompileInfoString() << "[SIMD=" << curSimd <<"]";
147  m_sEncoderCapabilities = cssCap.str();
148 
149  m_cEncoderInfo = "Fraunhofer VVC Encoder ver. " VVENC_VERSION;
150  m_cEncoderInfo += " ";
151  m_cEncoderInfo += m_sEncoderCapabilities;
152 
153  // initialize the encoder
154  m_pEncLib = new EncLib;
155 
156 #if HANDLE_EXCEPTION
157  try
158 #endif
159  {
160  m_pEncLib->initEncoderLib( m_cVVEncCfg );
161  }
162 #if HANDLE_EXCEPTION
163  catch( std::exception& e )
164  {
165  m_cErrorString = e.what();
166  return VVENC_ERR_UNSPECIFIED;
167  }
168 #endif
169 
170  m_bInitialized = true;
171  m_eState = INTERNAL_STATE_INITIALIZED;
172  return VVENC_OK;
173 }

175 int VVEncImpl::initPass( int pass )
176 {
177  if( !m_bInitialized ){ return VVENC_ERR_INITIALIZE; }
178  if( pass > 1 )
179  {
180  std::stringstream css;
181  css << "initPass(" << pass << ") no support for pass " << pass << ". use 0 (first pass) and 1 (second pass)";
182  m_cErrorString = css.str();

184  }
185 
186  if( pass > 1 && m_eState != INTERNAL_STATE_FINALIZED )
187  {
188  std::stringstream css;

190  m_cErrorString = css.str();
191  return VVENC_ERR_INITIALIZE;
192  }
193 
194  if ( m_pEncLib )
195  {
196 #if HANDLE_EXCEPTION
197  try
198 #endif
199  {
200  m_pEncLib->initPass( pass );
201  }
202 #if HANDLE_EXCEPTION
203  catch( std::exception& e )
204  {
205  m_cErrorString = e.what();
206  return VVENC_ERR_UNSPECIFIED;
207  }
208 #endif
209  }
210 
211  m_eState = INTERNAL_STATE_INITIALIZED;
212  return VVENC_OK;

260 int VVEncImpl::encode( vvencYUVBuffer* pcYUVBuffer, vvencAccessUnit* pcAccessUnit, bool* pbEncodeDone )
261 {
262  if( !m_bInitialized ) { return VVENC_ERR_INITIALIZE; }
263  if( m_eState == INTERNAL_STATE_FINALIZED ) { m_cErrorString = "encoder already flushed, please reinit."; return VVENC_ERR_RESTART_REQUIRED; }
264 
265  if( !pcAccessUnit )
266  {
267  m_cErrorString = "vvencAccessUnit is null. AU memory must be allocated before encode call.";
268  return VVENC_NOT_ENOUGH_MEM;
269  }
270  if( pcAccessUnit->payloadSize <= 0 )
271  {
272  m_cErrorString = "vvencAccessUnit has no payload size. AU payload must have a sufficient size to store encoded data.";
273  return VVENC_NOT_ENOUGH_MEM;
274  }
275 
276  int iRet= VVENC_OK;
277 
278  bool bFlush = false;
279  if( pcYUVBuffer )
280  {
281  if( m_eState == INTERNAL_STATE_FLUSHING ) { m_cErrorString = "encoder already received flush indication, please reinit."; return VVENC_ERR_RESTART_REQUIRED; }
282 
283  if( pcYUVBuffer->planes[0].ptr == nullptr )
284  {
285  m_cErrorString = "InputPicture: invalid input buffers";
286  return VVENC_ERR_UNSPECIFIED;
287  }
288 
289  if( m_cVVEncCfg.m_internChromaFormat != VVENC_CHROMA_400 )
290  {
291  if( pcYUVBuffer->planes[1].ptr == nullptr ||
292  pcYUVBuffer->planes[2].ptr == nullptr )
293  {
294  m_cErrorString = "InputPicture: invalid input buffers for chroma";
295  return VVENC_ERR_UNSPECIFIED;
296  }
297  }
298 
299  if( pcYUVBuffer->planes[0].width != m_cVVEncCfg.m_SourceWidth )
300  {
301  m_cErrorString = "InputPicture: unsupported width";
302  return VVENC_ERR_UNSPECIFIED;
303  }
304 
305  if( pcYUVBuffer->planes[0].height != m_cVVEncCfg.m_SourceHeight )
306  {
307  m_cErrorString = "InputPicture: unsupported height";
308  return VVENC_ERR_UNSPECIFIED;
309  }
310 
311  if( pcYUVBuffer->planes[0].width > pcYUVBuffer->planes[0].stride )
312  {
313  m_cErrorString = "InputPicture: unsupported width stride combination";
314  return VVENC_ERR_UNSPECIFIED;
315  }
316 
317  if( m_cVVEncCfg.m_internChromaFormat != VVENC_CHROMA_400 )
318  {
319  if( m_cVVEncCfg.m_internChromaFormat == VVENC_CHROMA_444 )
320  {
321  if( pcYUVBuffer->planes[1].stride && pcYUVBuffer->planes[0].width > pcYUVBuffer->planes[1].stride )
322  {
323  m_cErrorString = "InputPicture: unsupported width cstride combination for 2nd plane";
324  return VVENC_ERR_UNSPECIFIED;
325  }
326 
327  if( pcYUVBuffer->planes[2].stride && pcYUVBuffer->planes[0].width > pcYUVBuffer->planes[2].stride )
328  {
329  m_cErrorString = "InputPicture: unsupported width cstride combination for 3rd plane";
330  return VVENC_ERR_UNSPECIFIED;
331  }
332  }
333  else
334  {
335  if( pcYUVBuffer->planes[1].stride && pcYUVBuffer->planes[0].width/2 > pcYUVBuffer->planes[1].stride )
336  {
337  m_cErrorString = "InputPicture: unsupported width cstride combination for 2nd plane";
338  return VVENC_ERR_UNSPECIFIED;
339  }
340 
341  if( pcYUVBuffer->planes[2].stride && pcYUVBuffer->planes[0].width/2 > pcYUVBuffer->planes[2].stride )
342  {
343  m_cErrorString = "InputPicture: unsupported width cstride combination for 3rd plane";
344  return VVENC_ERR_UNSPECIFIED;
345  }
346  }
347  }
348 
349  if( m_eState == INTERNAL_STATE_INITIALIZED ){ m_eState = INTERNAL_STATE_ENCODING; }
350  }
351  else
352  {
353  if( m_eState == INTERNAL_STATE_ENCODING ){ m_eState = INTERNAL_STATE_FLUSHING; }
354  bFlush = true;
355  }
356 
357  // reset AU data
358  if( pcAccessUnit )
359  {
360  vvenc_accessUnit_reset(pcAccessUnit);
361  }
362  *pbEncodeDone = false;
363 
364  AccessUnitList cAu;
365 #if HANDLE_EXCEPTION
366  try
367 #endif
368  {
369  m_pEncLib->encodePicture( bFlush, pcYUVBuffer, cAu, *pbEncodeDone );
370  }
371 #if HANDLE_EXCEPTION
372  catch( std::exception& e )
373  {
374  m_cErrorString = e.what();
375  return VVENC_ERR_UNSPECIFIED;
376  }
377 #endif
378 
379  if( *pbEncodeDone )
380  {
381  if( m_eState == INTERNAL_STATE_FLUSHING )
382  {
383  m_eState = INTERNAL_STATE_FINALIZED;
384  }
385  else
386  {
387  *pbEncodeDone = false;
388  }
389  }
390 
391  /* copy output AU */
392  if ( !cAu.empty() )
393  {
394  int sizeAu = xGetAccessUnitsSize( cAu );
395  if( pcAccessUnit->payloadSize < sizeAu )
396  {
397  std::stringstream css;
398  css << "vvencAccessUnit payload size is too small to store data. (payload size: " << pcAccessUnit->payloadSize << ", needed " << sizeAu << ")";
399  m_cErrorString =css.str();
400  return VVENC_NOT_ENOUGH_MEM;
401  }
402 
403  iRet = xCopyAu( *pcAccessUnit, cAu );
404  }
405 
406 #if defined( __linux__ )
407  malloc_trim(0); // free unused heap memory
408 #endif
409 
410  return iRet;
411 }

475 int VVEncImpl::xGetAccessUnitsSize( const vvenc::AccessUnitList& rcAuList )
476 {
477  uint32_t sizeSum = 0;
478  if ( ! rcAuList.empty() )
479  {
480  for (vvenc::AccessUnitList::const_iterator it = rcAuList.begin(); it != rcAuList.end(); it++)
481  {
482  const vvenc::NALUnitEBSP& nalu = **it;
483  if (it == rcAuList.begin() ||
484  nalu.m_nalUnitType == VVENC_NAL_UNIT_DCI ||
485  nalu.m_nalUnitType == VVENC_NAL_UNIT_SPS ||
486  nalu.m_nalUnitType == VVENC_NAL_UNIT_VPS ||
487  nalu.m_nalUnitType == VVENC_NAL_UNIT_PPS ||
488  nalu.m_nalUnitType == VVENC_NAL_UNIT_PREFIX_APS ||
489  nalu.m_nalUnitType == VVENC_NAL_UNIT_SUFFIX_APS )
490  {
491  sizeSum += 4;
492  }
493  else
494  {
495  sizeSum += 3;
496  }
497  sizeSum += uint32_t(nalu.m_nalUnitData.str().size());
498  }
499  }
500 
501  return sizeSum;
502 }

505 int VVEncImpl::xCopyAu( vvencAccessUnit& rcAccessUnit, const vvenc::AccessUnitList& rcAuList )
506 {
507  rcAccessUnit.rap = false;
508 
509  std::vector<uint32_t> annexBsizes;
510 
511  /* copy output AU */
512  if ( ! rcAuList.empty() )
513  {
514  uint32_t sizeSum = 0;
515  for (vvenc::AccessUnitList::const_iterator it = rcAuList.begin(); it != rcAuList.end(); it++)
516  {
517  const vvenc::NALUnitEBSP& nalu = **it;
518  uint32_t size = 0; /* size of annexB unit in bytes */
519 
520  if (it == rcAuList.begin() ||
521  nalu.m_nalUnitType == VVENC_NAL_UNIT_DCI ||
522  nalu.m_nalUnitType == VVENC_NAL_UNIT_SPS ||
523  nalu.m_nalUnitType == VVENC_NAL_UNIT_VPS ||
524  nalu.m_nalUnitType == VVENC_NAL_UNIT_PPS ||
525  nalu.m_nalUnitType == VVENC_NAL_UNIT_PREFIX_APS ||
526  nalu.m_nalUnitType == VVENC_NAL_UNIT_SUFFIX_APS )
527  {
528  size += 4;
529  }
530  else
531  {
532  size += 3;
533  }
534  size += uint32_t(nalu.m_nalUnitData.str().size());
535  sizeSum += size;
536  annexBsizes.push_back( size );
537 
538  switch( nalu.m_nalUnitType )
539  {
540  case VVENC_NAL_UNIT_CODED_SLICE_TRAIL:
541  case VVENC_NAL_UNIT_CODED_SLICE_STSA:
542  case VVENC_NAL_UNIT_CODED_SLICE_IDR_W_RADL:
543  case VVENC_NAL_UNIT_CODED_SLICE_IDR_N_LP:
544  case VVENC_NAL_UNIT_CODED_SLICE_CRA:
545  case VVENC_NAL_UNIT_CODED_SLICE_GDR:
546  case VVENC_NAL_UNIT_CODED_SLICE_RADL:
547  case VVENC_NAL_UNIT_CODED_SLICE_RASL:
548  case VVENC_NAL_UNIT_DCI:
549  case VVENC_NAL_UNIT_VPS:
550  case VVENC_NAL_UNIT_SPS:
551  case VVENC_NAL_UNIT_PPS:
552  case VVENC_NAL_UNIT_PREFIX_APS:
553  case VVENC_NAL_UNIT_SUFFIX_APS:
554  rcAccessUnit.essentialBytes += size;
555  break;
556  default:
557  break;
558  }
559  }
560 
561  if( rcAccessUnit.payloadSize < (int)sizeSum )
562  {
563  return -1;
564  }
565 
566  uint32_t iUsedSize = 0;
567  for (vvenc::AccessUnitList::const_iterator it = rcAuList.begin(); it != rcAuList.end(); it++)
568  {
569  const vvenc::NALUnitEBSP& nalu = **it;
570  static const uint8_t start_code_prefix[] = {0,0,0,1};
571  if (it == rcAuList.begin() ||
572  nalu.m_nalUnitType == VVENC_NAL_UNIT_DCI ||
573  nalu.m_nalUnitType == VVENC_NAL_UNIT_SPS ||
574  nalu.m_nalUnitType == VVENC_NAL_UNIT_VPS ||
575  nalu.m_nalUnitType == VVENC_NAL_UNIT_PPS ||
576  nalu.m_nalUnitType == VVENC_NAL_UNIT_PREFIX_APS ||
577  nalu.m_nalUnitType == VVENC_NAL_UNIT_SUFFIX_APS )
578  {
579  /* From AVC, When any of the following conditions are fulfilled, the
580  * zero_byte syntax element shall be present:
581  * - the nal_unit_type within the nal_unit() is equal to 7 (sequence
582  * parameter set) or 8 (picture parameter set),
583  * - the byte stream NAL unit syntax structure contains the first NAL
584  * unit of an access unit in decoding order, as specified by subclause
585  * 7.4.1.2.3.
586  */
587  ::memcpy( rcAccessUnit.payload + iUsedSize, reinterpret_cast<const char*>(start_code_prefix), 4 );
588  iUsedSize += 4;
589  }
590  else
591  {
592  ::memcpy( rcAccessUnit.payload + iUsedSize, reinterpret_cast<const char*>(start_code_prefix+1), 3 );
593  iUsedSize += 3;
594  }
595  uint32_t nalDataSize = uint32_t(nalu.m_nalUnitData.str().size()) ;
596  ::memcpy( rcAccessUnit.payload + iUsedSize, nalu.m_nalUnitData.str().c_str() , nalDataSize );
597  iUsedSize += nalDataSize;
598 
599  if( nalu.m_nalUnitType == VVENC_NAL_UNIT_CODED_SLICE_IDR_W_RADL ||
600  nalu.m_nalUnitType == VVENC_NAL_UNIT_CODED_SLICE_IDR_N_LP ||
601  nalu.m_nalUnitType == VVENC_NAL_UNIT_CODED_SLICE_CRA ||
602  nalu.m_nalUnitType == VVENC_NAL_UNIT_CODED_SLICE_GDR )
603  {
604  rcAccessUnit.rap = true;
605  }
606  //rcAccessUnit.nalUnitTypeVec.push_back( nalu.m_nalUnitType );
607  }
608 
609  rcAccessUnit.payloadUsedSize = iUsedSize;
610  if( iUsedSize != sizeSum )
611  {
612  return VVENC_NOT_ENOUGH_MEM;
613  }
614 
615  //rcAccessUnit.annexBsizeVec = annexBsizes;
616  rcAccessUnit.ctsValid = rcAuList.ctsValid;
617  rcAccessUnit.dtsValid = rcAuList.dtsValid;
618  rcAccessUnit.cts = rcAuList.cts;
619  rcAccessUnit.dts = rcAuList.dts;
620  rcAccessUnit.sliceType = (vvencSliceType)rcAuList.sliceType;
621  rcAccessUnit.refPic = rcAuList.refPic;
622  rcAccessUnit.temporalLayer = rcAuList.temporalLayer;
623  rcAccessUnit.poc = rcAuList.poc;
624  rcAccessUnit.status = rcAuList.status;
625 
626  if ( !rcAuList.InfoString.empty() )
627  {
628  if( rcAuList.InfoString.size() >= VVENC_MAX_STRING_LEN )
629  {
630  rcAuList.InfoString.copy( rcAccessUnit.infoString , VVENC_MAX_STRING_LEN-1 );
631  rcAccessUnit.infoString[VVENC_MAX_STRING_LEN-1] = '\0';
632  }
633  else
634  {
635  rcAuList.InfoString.copy( rcAccessUnit.infoString , rcAuList.InfoString.size()+1 );
636  rcAccessUnit.infoString[rcAuList.InfoString.size()] = '\0';
637  }
638  }
639  else
640  {
641  rcAccessUnit.infoString[0] = '\0';
642  }
643  }
644 
645  return 0;
646 }
