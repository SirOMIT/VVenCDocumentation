116 void CoeffCodingContext::initSubblock( int SubsetId, bool sigGroupFlag )
117 {
118  m_subSetId = SubsetId;
119  m_subSetPos = m_scanCG[m_subSetId].idx;
120  m_subSetPosY = m_subSetPos >> m_log2WidthInGroups;
121  m_subSetPosX = m_subSetPos - ( m_subSetPosY << m_log2WidthInGroups );
122  m_minSubPos = m_subSetId << m_log2CGSize;
123  m_maxSubPos = m_minSubPos + ( 1 << m_log2CGSize ) - 1;
124  if( sigGroupFlag )
125  {
126  m_sigCoeffGroupFlag.set ( m_subSetPos );
127  }
128  unsigned CGPosY = m_subSetPosY;
129  unsigned CGPosX = m_subSetPosX;
130  unsigned sigRight = unsigned( ( CGPosX + 1 ) < m_widthInGroups ? m_sigCoeffGroupFlag[ m_subSetPos + 1 ] : false );
131  unsigned sigLower = unsigned( ( CGPosY + 1 ) < m_heightInGroups ? m_sigCoeffGroupFlag[ m_subSetPos + m_widthInGroups ] : false );
132  m_sigGroupCtxId = Ctx::SigCoeffGroup[m_chType]( sigRight | sigLower );
133  unsigned sigLeft = unsigned( CGPosX > 0 ? m_sigCoeffGroupFlag[m_subSetPos - 1 ] : false );
134  unsigned sigAbove = unsigned( CGPosY > 0 ? m_sigCoeffGroupFlag[m_subSetPos - m_widthInGroups] : false );
135  m_sigGroupCtxIdTS = Ctx::TsSigCoeffGroup( sigLeft + sigAbove );
136 }

150 void DeriveCtx::CtxSplit( const Partitioner& partitioner, unsigned& ctxSpl, unsigned& ctxQt, unsigned& ctxHv, unsigned& ctxHorBt, unsigned& ctxVerBt, const bool canSplit[6] ) const
151 {
152  const ChannelType chType = partitioner.chType;
153  const CodingUnit* cuLeft = cuRestrictedLeft[chType];
154  const CodingUnit* cuAbove = cuRestrictedAbove[chType];
155 
157  // CTX do split (0-8)
159  const unsigned widthCurr = partitioner.currArea().blocks[chType].width;
160  const unsigned heightCurr = partitioner.currArea().blocks[chType].height;
161 
162  ctxSpl = 0;
163 
164  if( cuLeft )
165  {
166  const unsigned heightLeft = cuLeft->blocks[chType].height;
167  ctxSpl += ( heightLeft < heightCurr ? 1 : 0 );
168  }
169  if( cuAbove )
170  {
171  const unsigned widthAbove = cuAbove->blocks[chType].width;
172  ctxSpl += ( widthAbove < widthCurr ? 1 : 0 );
173  }
174 
175  unsigned numSplit = 0;
176  if( canSplit[1] ) numSplit += 2;
177  if( canSplit[2] ) numSplit += 1;
178  if( canSplit[3] ) numSplit += 1;
179  if( canSplit[4] ) numSplit += 1;
180  if( canSplit[5] ) numSplit += 1;
181 
182  if( numSplit > 0 ) numSplit--;
183 
184  ctxSpl += 3 * ( numSplit >> 1 );
185 
187  // CTX is qt split (0-5)
189  ctxQt = ( cuLeft && cuLeft->qtDepth > partitioner.currQtDepth ) ? 1 : 0;
190  ctxQt += ( cuAbove && cuAbove->qtDepth > partitioner.currQtDepth ) ? 1 : 0;
191  ctxQt += partitioner.currQtDepth < 2 ? 0 : 3;
192 
194  // CTX is ver split (0-4)
196  ctxHv = 0;
197 
198  const unsigned numHor = ( canSplit[2] ? 1 : 0 ) + ( canSplit[4] ? 1 : 0 );
199  const unsigned numVer = ( canSplit[3] ? 1 : 0 ) + ( canSplit[5] ? 1 : 0 );
200 
201  if( numVer == numHor )
202  {
203  const Area& area = partitioner.currArea().blocks[chType];
204 
205  const unsigned wAbove = cuAbove ? cuAbove->blocks[chType].width : 1;
206  const unsigned hLeft = cuLeft ? cuLeft ->blocks[chType].height : 1;
207 
208  const unsigned depAbove = area.width / wAbove;
209  const unsigned depLeft = area.height / hLeft;
210 
211  if( depAbove == depLeft || !cuLeft || !cuAbove ) ctxHv = 0;
212  else if( depAbove < depLeft ) ctxHv = 1;
213  else ctxHv = 2;
214  }
215  else if( numVer < numHor )
216  {
217  ctxHv = 3;
218  }
219  else
220  {
221  ctxHv = 4;
222  }
223 
225  // CTX is h/v bt (0-3)
227  ctxHorBt = ( partitioner.currMtDepth <= 1 ? 1 : 0 );
228  ctxVerBt = ( partitioner.currMtDepth <= 1 ? 3 : 2 );
229 }

233 void MergeCtx::setMergeInfo( CodingUnit& cu, int candIdx ) const
234 {
235  CHECK( candIdx >= numValidMergeCand, "Merge candidate does not exist" );
236  cu.regularMergeFlag = !(cu.ciip || cu.geo);
237  cu.mergeFlag = true;
238  cu.mmvdMergeFlag = false;
239  cu.interDir = interDirNeighbours[candIdx];
240  cu.imv = (!cu.geo && useAltHpelIf[candIdx]) ? IMV_HPEL : 0;
241  cu.mergeIdx = candIdx;
242  cu.mergeType = mrgTypeNeighbours[candIdx];
243  cu.mv [REF_PIC_LIST_0][0] = mvFieldNeighbours[(candIdx << 1) + 0].mv;
244  cu.mv [REF_PIC_LIST_1][0] = mvFieldNeighbours[(candIdx << 1) + 1].mv;
245  cu.mvd [REF_PIC_LIST_0][0] = Mv();
246  cu.mvd [REF_PIC_LIST_1][0] = Mv();
247  cu.refIdx [REF_PIC_LIST_0] = mvFieldNeighbours[( candIdx << 1 ) + 0].refIdx;
248  cu.refIdx [REF_PIC_LIST_1] = mvFieldNeighbours[( candIdx << 1 ) + 1].refIdx;
249  cu.mvpIdx [REF_PIC_LIST_0] = NOT_VALID;
250  cu.mvpIdx [REF_PIC_LIST_1] = NOT_VALID;
251  cu.mvpNum [REF_PIC_LIST_0] = NOT_VALID;
252  cu.mvpNum [REF_PIC_LIST_1] = NOT_VALID;
253  if( CU::isIBC( cu ) )
254  {
255  cu.bv = cu.mv[REF_PIC_LIST_0][0];
256  cu.bv .changePrecision( MV_PRECISION_INTERNAL, MV_PRECISION_INT ); // used for only integer resolution
257  cu.imv = cu.imv == IMV_HPEL ? 0 : cu.imv;
258  }
259  cu.BcwIdx = ( interDirNeighbours[candIdx] == 3 ) ? BcwIdx[candIdx] : BCW_DEFAULT;
260 
261  CU::restrictBiPredMergeCandsOne(cu);
262  cu.mcControl = 0;
263 }

265 void MergeCtx::setMmvdMergeCandiInfo(CodingUnit& cu, int candIdx) const
266 {
267  const Slice &slice = *cu.cs->slice;
268  const int mvShift = MV_FRACTIONAL_BITS_DIFF;
269  const int refMvdCands[8] = { 1 << mvShift , 2 << mvShift , 4 << mvShift , 8 << mvShift , 16 << mvShift , 32 << mvShift, 64 << mvShift , 128 << mvShift };
270  int fPosGroup = 0;
271  int fPosBaseIdx = 0;
272  int fPosStep = 0;
273  int tempIdx = 0;
274  int fPosPosition = 0;
275  Mv tempMv[2];
276 
277  tempIdx = candIdx;
278  fPosGroup = tempIdx / (MMVD_BASE_MV_NUM * MMVD_MAX_REFINE_NUM);
279  tempIdx = tempIdx - fPosGroup * (MMVD_BASE_MV_NUM * MMVD_MAX_REFINE_NUM);
280  fPosBaseIdx = tempIdx / MMVD_MAX_REFINE_NUM;
281  tempIdx = tempIdx - fPosBaseIdx * (MMVD_MAX_REFINE_NUM);
282  fPosStep = tempIdx / 4;
283  fPosPosition = tempIdx - fPosStep * (4);
284  int offset = refMvdCands[fPosStep];
285  if ( cu.slice->picHeader->disFracMMVD )
286  {
287  offset <<= 2;
288  }
289  const int refList0 = mmvdBaseMv[fPosBaseIdx][0].refIdx;
290  const int refList1 = mmvdBaseMv[fPosBaseIdx][1].refIdx;
291 
292  if ((refList0 != -1) && (refList1 != -1))
293  {
294  const int poc0 = slice.getRefPOC(REF_PIC_LIST_0, refList0);
295  const int poc1 = slice.getRefPOC(REF_PIC_LIST_1, refList1);
296  const int currPoc = slice.poc;
297  if (fPosPosition == 0)
298  {
299  tempMv[0] = Mv(offset, 0);
300  }
301  else if (fPosPosition == 1)
302  {
303  tempMv[0] = Mv(-offset, 0);
304  }
305  else if (fPosPosition == 2)
306  {
307  tempMv[0] = Mv(0, offset);
308  }
309  else
310  {
311  tempMv[0] = Mv(0, -offset);
312  }
313  if ((poc0 - currPoc) == (poc1 - currPoc))
314  {
315  tempMv[1] = tempMv[0];
316  }
317  else if (abs(poc1 - currPoc) > abs(poc0 - currPoc))
318  {
319  const int scale = CU::getDistScaleFactor(currPoc, poc0, currPoc, poc1);
320  tempMv[1] = tempMv[0];
321  const bool isL0RefLongTerm = slice.getRefPic(REF_PIC_LIST_0, refList0)->isLongTerm;
322  const bool isL1RefLongTerm = slice.getRefPic(REF_PIC_LIST_1, refList1)->isLongTerm;
323  if (isL0RefLongTerm || isL1RefLongTerm)
324  {
325  if ((poc1 - currPoc)*(poc0 - currPoc) > 0)
326  {
327  tempMv[0] = tempMv[1];
328  }
329  else
330  {
331  tempMv[0].set(-1 * tempMv[1].hor, -1 * tempMv[1].ver);
332  }
333  }
334  else
335  tempMv[0] = tempMv[1].scaleMv(scale);
336  }
337  else
338  {
339  const int scale = CU::getDistScaleFactor(currPoc, poc1, currPoc, poc0);
340  const bool isL0RefLongTerm = slice.getRefPic(REF_PIC_LIST_0, refList0)->isLongTerm;
341  const bool isL1RefLongTerm = slice.getRefPic(REF_PIC_LIST_1, refList1)->isLongTerm;
342  if (isL0RefLongTerm || isL1RefLongTerm)
343  {
344  if ((poc1 - currPoc)*(poc0 - currPoc) > 0)
345  {
346  tempMv[1] = tempMv[0];
347  }
348  else
349  {
350  tempMv[1].set(-1 * tempMv[0].hor, -1 * tempMv[0].ver);
351  }
352  }
353  else
354  tempMv[1] = tempMv[0].scaleMv(scale);
355  }
356 
357  cu.interDir = 3;
358  cu.mv[REF_PIC_LIST_0][0] = mmvdBaseMv[fPosBaseIdx][0].mv + tempMv[0];
359  cu.refIdx[REF_PIC_LIST_0] = refList0;
360  cu.mv[REF_PIC_LIST_1][0] = mmvdBaseMv[fPosBaseIdx][1].mv + tempMv[1];
361  cu.refIdx[REF_PIC_LIST_1] = refList1;
362  }
363  else if (refList0 != -1)
364  {
365  if (fPosPosition == 0)
366  {
367  tempMv[0] = Mv(offset, 0);
368  }
369  else if (fPosPosition == 1)
370  {
371  tempMv[0] = Mv(-offset, 0);
372  }
373  else if (fPosPosition == 2)
374  {
375  tempMv[0] = Mv(0, offset);
376  }
377  else
378  {
379  tempMv[0] = Mv(0, -offset);
380  }
381  cu.interDir = 1;
382  cu.mv[REF_PIC_LIST_0][0] = mmvdBaseMv[fPosBaseIdx][0].mv + tempMv[0];
383  cu.refIdx[REF_PIC_LIST_0] = refList0;
384  cu.mv[REF_PIC_LIST_1][0] = Mv(0, 0);
385  cu.refIdx[REF_PIC_LIST_1] = -1;
386  }
387  else if (refList1 != -1)
388  {
389  if (fPosPosition == 0)
390  {
391  tempMv[1] = Mv(offset, 0);
392  }
393  else if (fPosPosition == 1)
394  {
395  tempMv[1] = Mv(-offset, 0);
396  }
397  else if (fPosPosition == 2)
398  {
399  tempMv[1] = Mv(0, offset);
400  }
401  else
402  {
403  tempMv[1] = Mv(0, -offset);
404  }
405  cu.interDir = 2;
406  cu.mv[REF_PIC_LIST_0][0] = Mv(0, 0);
407  cu.refIdx[REF_PIC_LIST_0] = -1;
408  cu.mv[REF_PIC_LIST_1][0] = mmvdBaseMv[fPosBaseIdx][1].mv + tempMv[1];
409  cu.refIdx[REF_PIC_LIST_1] = refList1;
410  }
411 
412  cu.mmvdMergeFlag = true;
413  cu.mmvdMergeIdx = candIdx;
414  cu.mergeFlag = true;
415  cu.regularMergeFlag = true;
416  cu.mergeIdx = candIdx;
417  cu.mergeType = MRG_TYPE_DEFAULT_N;
418  cu.mvd[REF_PIC_LIST_0][0] = Mv();
419  cu.mvd[REF_PIC_LIST_1][0] = Mv();
420  cu.mvpIdx[REF_PIC_LIST_0] = NOT_VALID;
421  cu.mvpIdx[REF_PIC_LIST_1] = NOT_VALID;
422  cu.mvpNum[REF_PIC_LIST_0] = NOT_VALID;
423  cu.mvpNum[REF_PIC_LIST_1] = NOT_VALID;
424  cu.imv = mmvdUseAltHpelIf[fPosBaseIdx] ? IMV_HPEL : 0;
425 
426  cu.BcwIdx = (interDirNeighbours[fPosBaseIdx] == 3) ? BcwIdx[fPosBaseIdx] : BCW_DEFAULT;
427 
428  for (int refList = 0; refList < 2; refList++)
429  {
430  if (cu.refIdx[refList] >= 0)
431  {
432  cu.mv[refList][0].clipToStorageBitDepth();
433  }
434  }
435 
436 
437  CU::restrictBiPredMergeCandsOne(cu);
438 }

440 unsigned DeriveCtx::CtxMipFlag( const CodingUnit& cu ) const
441 {

444  ctxId = (cuLeft && cuLeft->mipFlag) ? 1 : 0;
445

447  ctxId += (cuAbove && cuAbove->mipFlag) ? 1 : 0;
448

450
