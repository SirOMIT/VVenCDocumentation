211 template<typename T, size_t N>
212 uint32_t updateCandList(T uiMode, double uiCost, static_vector<T, N>& candModeList, static_vector<double, N>& candCostList
213  , size_t uiFastCandNum = N, int* iserttPos = nullptr)
214 {
215  CHECK( std::min( uiFastCandNum, candModeList.size() ) != std::min( uiFastCandNum, candCostList.size() ), "Sizes do not match!" );
216  CHECK( uiFastCandNum > candModeList.capacity(), "The vector is to small to hold all the candidates!" );
217 
218  size_t i;
219  size_t shift = 0;
220  size_t currSize = std::min( uiFastCandNum, candCostList.size() );
221 
222  while( shift < uiFastCandNum && shift < currSize && uiCost < candCostList[currSize - 1 - shift] )
223  {
224  shift++;
225  }
226 
227  if( candModeList.size() >= uiFastCandNum && shift != 0 )
228  {
229  for( i = 1; i < shift; i++ )
230  {
231  candModeList[currSize - i] = candModeList[currSize - 1 - i];
232  candCostList[currSize - i] = candCostList[currSize - 1 - i];
233  }
234  candModeList[currSize - shift] = uiMode;
235  candCostList[currSize - shift] = uiCost;
236  if (iserttPos != nullptr)
237  {
238  *iserttPos = int(currSize - shift);
239  }
240  return 1;
241  }
242  else if( currSize < uiFastCandNum )
243  {
244  candModeList.insert( candModeList.end() - shift, uiMode );
245  candCostList.insert( candCostList.end() - shift, uiCost );
246  if (iserttPos != nullptr)
247  {
248  *iserttPos = int(candModeList.size() - shift - 1);
249  }
250  return 1;
251  }
252  if (iserttPos != nullptr)
253  {
254  *iserttPos = -1;
255  }
256  return 0;
257 }
