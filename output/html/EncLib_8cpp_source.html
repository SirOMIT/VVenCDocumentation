219 void EncLib::initPass( int pass )
220 {
221  CHECK( m_numPassInitialized != pass && m_numPassInitialized + 1 != pass, "initialization of passes only in successive order possible" );
222 
223  if( m_numPassInitialized + 1 != pass )
224  {
225  return;
226  }
227 
228  xUninitLib();
229 
230  // set rate control pass
231  m_cRateCtrl.setRCPass( pass, m_cEncCfg.m_RCNumPasses - 1 );
232 
233  // modify encoder config based on rate control pass
234  if( m_cEncCfg.m_RCNumPasses > 1 )
235  {
236  xSetRCEncCfg( pass );
237  }
238 
239  // setup parameter sets
240  const int dciId = m_cEncCfg.m_decodingParameterSetEnabled ? 1 : 0;
241  SPS& sps0 = *( m_spsMap.allocatePS( 0 ) ); // NOTE: implementations that use more than 1 SPS need to be aware of activation issues.
242  PPS& pps0 = *( m_ppsMap.allocatePS( 0 ) );
243 
244  xInitSPS( sps0 );
245  sps0.dciId = m_cDCI.dciId;
246  xInitVPS( m_cVPS );
247  xInitDCI( m_cDCI, sps0, dciId );
248  xInitPPS( pps0, sps0 );
249  xInitRPL( sps0 );
250  xInitHrdParameters( sps0 );
251 
252  // thread pool
253  if( m_cEncCfg.m_numThreads > 0 )
254  {
255  m_threadPool = new NoMallocThreadPool( m_cEncCfg.m_numThreads, "EncSliceThreadPool" );
256  }
257 
258  m_MCTF.init( m_cEncCfg.m_internalBitDepth, m_cEncCfg.m_PadSourceWidth, m_cEncCfg.m_PadSourceHeight, sps0.CTUSize,
259  m_cEncCfg.m_internChromaFormat, m_cEncCfg.m_QP, m_cEncCfg.m_vvencMCTF, m_cEncCfg.m_framesToBeEncoded, m_threadPool );
260 
261  CHECK( m_cGOPEncoder != nullptr, "encoder library already initialised" );
262  m_cGOPEncoder = new EncGOP;
263  m_cGOPEncoder->init( m_cEncCfg, sps0, pps0, m_cRateCtrl, m_cEncHRD, m_threadPool );
264 
265  m_pocToGopId.resize( m_cEncCfg.m_GOPSize, -1 );
266  m_nextPocOffset.resize( m_cEncCfg.m_GOPSize, 0 );
267 
268  int gopPOCadj = m_cEncCfg.m_DecodingRefreshType == 4 ? 1 : 0;
269 
270  for ( int i = 0; i < m_cEncCfg.m_GOPSize; i++ )
271  {
272  const int poc = (m_cEncCfg.m_GOPList[ i ].m_POC-gopPOCadj) % m_cEncCfg.m_GOPSize;
273  CHECK( m_cEncCfg.m_GOPList[ i ].m_POC > m_cEncCfg.m_GOPSize, "error: poc greater than gop size" );
274  CHECK( m_pocToGopId[ poc ] != -1, "error: multiple entries in gop list map to same poc modulo gop size" );
275  m_pocToGopId[ poc ] = i;
276  const int nextGopNum = ( i + 1 ) / m_cEncCfg.m_GOPSize;
277  const int nextGopId = ( i + 1 ) % m_cEncCfg.m_GOPSize;
278  const int nextPoc = nextGopNum * m_cEncCfg.m_GOPSize + m_cEncCfg.m_GOPList[ nextGopId ].m_POC-gopPOCadj;
279  m_nextPocOffset[ poc ] = nextPoc - (m_cEncCfg.m_GOPList[ i ].m_POC-gopPOCadj);
280  }
281  for ( int i = 0; i < m_cEncCfg.m_GOPSize; i++ )
282  {
283  CHECK( m_pocToGopId[ i ] < 0 || m_nextPocOffset[ i ] == 0, "error: poc not found in gop list" );
284  }
285 
286  if ( m_cEncCfg.m_RCTargetBitrate > 0 )
287  {
288  m_cRateCtrl.init( m_cEncCfg.m_framesToBeEncoded, m_cEncCfg.m_RCTargetBitrate, (int)( (double)m_cEncCfg.m_FrameRate / m_cEncCfg.m_temporalSubsampleRatio + 0.5 ), m_cEncCfg.m_IntraPeriod, m_cEncCfg.m_GOPSize, m_cEncCfg.m_PadSourceWidth, m_cEncCfg.m_PadSourceHeight,
289  m_cEncCfg.m_CTUSize, m_cEncCfg.m_CTUSize, m_cEncCfg.m_internalBitDepth[ CH_L ], m_cEncCfg.m_GOPList, m_cEncCfg.m_maxParallelFrames );
290 
291  if ( pass == 1 )
292  {
293  m_cRateCtrl.processFirstPassData (m_cEncCfg.m_QP);
294  // update first-pass data
295  m_cRateCtrl.encRCSeq->firstPassData = m_cRateCtrl.getFirstPassStats();
296  }
297  }
298 
299  int iOffset = -1;
300  while((1<<(++iOffset)) < m_cEncCfg.m_GOPSize);
301  m_GOPSizeLog2 = iOffset;
302 
303  if( m_cEncCfg.m_FrameRate )
304  {
305  int iTempRate = m_cEncCfg.m_FrameRate;
306  int iTempScale = 1;
307  switch( m_cEncCfg.m_FrameRate )
308  {
309  case 23: iTempRate = 24000; iTempScale = 1001; break;
310  case 29: iTempRate = 30000; iTempScale = 1001; break;
311  case 59: iTempRate = 60000; iTempScale = 1001; break;
312  default: break;
313  }
314  m_TicksPerFrameMul4 = (int)((int64_t)4 *(int64_t)m_cEncCfg.m_TicksPerSecond * (int64_t)iTempScale/(int64_t)iTempRate);
315  }
316 
317  m_numPassInitialized = pass;
318 }

359 void EncLib::xSetRCEncCfg( int pass )
360 {
361  // restore encoder configuration for second rate control passes
362  const_cast<VVEncCfg&>(m_cEncCfg) = m_cBckCfg;
363 
364  // set encoder config for rate control first pass
365  if( ! m_cRateCtrl.rcIsFinalPass )
366  {
367  const double d = (3840.0 * 2160.0) / double (m_cEncCfg.m_SourceWidth * m_cEncCfg.m_SourceHeight);
368 
369  // preserve the CTU size, MCTF and IBC settings
370  const unsigned cs = m_cBckCfg.m_CTUSize;
371  const int mctf = m_cBckCfg.m_vvencMCTF.MCTF;
372  const int ibcMode = m_cBckCfg.m_IBCMode;
373 
374  vvenc_init_preset( &m_cBckCfg, vvencPresetMode::VVENC_FIRSTPASS );
375 
376  // fixed-QP encoding in first rate control pass
377  m_cBckCfg.m_RCTargetBitrate = 0;
378  m_cBckCfg.m_QP /*base QP*/ = (m_cEncCfg.m_RCInitialQP > 0 ? Clip3 (17, MAX_QP, m_cEncCfg.m_RCInitialQP) : std::max (17, MAX_QP_PERCEPT_QPA - 2 - int (0.5 + sqrt ((d * m_cEncCfg.m_RCTargetBitrate) / 500000.0))));
379 
380  // restore the settings
381  if (m_cBckCfg.m_usePerceptQPA && (m_cBckCfg.m_QP <= MAX_QP_PERCEPT_QPA))
382  {
383  m_cBckCfg.m_CTUSize = cs;
384  }
385  m_cBckCfg.m_vvencMCTF.MCTF = mctf;
386  m_cBckCfg.m_IBCMode = ibcMode;
387 
388  // clear MaxCuDQPSubdiv
389  if( m_cBckCfg.m_CTUSize < 128 )
390  {
391  m_cBckCfg.m_cuQpDeltaSubdiv = 0;
392  }
393 
394  std::swap( const_cast<VVEncCfg&>(m_cEncCfg), m_cBckCfg );
395  }
396  else // estimate near-optimal base QP for PPS in second RC pass
397  {
398  const unsigned fps = m_cEncCfg.m_FrameRate;
399  uint64_t sumFrBits = 0, sumVisAct = 0; // for first-pass data
400  std::list<TRCPassStats>& firstPassData = m_cRateCtrl.getFirstPassStats();
401  std::list<TRCPassStats>::iterator it;
402 
403  for (it = firstPassData.begin(); it != firstPassData.end(); it++)
404  {
405  sumFrBits += it->numBits;
406  sumVisAct += it->visActY;
407  }
408  if ((firstPassData.size() > 0) && (fps > 0))
409  {
410  double d = (3840.0 * 2160.0) / double (m_cEncCfg.m_SourceWidth * m_cEncCfg.m_SourceHeight);
411  const int firstPassBaseQP = std::max (17, MAX_QP_PERCEPT_QPA - 2 - int (0.5 + sqrt ((d * m_cEncCfg.m_RCTargetBitrate) / 500000.0)));
412  const int log2HeightMinus7 = int (0.5 + log ((double) std::max (128, m_cEncCfg.m_SourceHeight)) / log (2.0)) - 7;
413 
414  d = (double) m_cEncCfg.m_RCTargetBitrate * (double) firstPassData.size() / double (fps * sumFrBits);
415  d = firstPassBaseQP - (105.0 / 128.0) * sqrt ((double) std::max (1, firstPassBaseQP)) * log (d) / log (2.0);
416  const_cast<VVEncCfg&>(m_cEncCfg).m_QP = int (0.5 + d + 0.125 * log2HeightMinus7 * std::max (0.0, 24.0 + 0.001/*log2HeightMinus7*/ * (log ((double) sumVisAct / firstPassData.size()) / log (2.0) - 0.5 * m_cEncCfg.m_internalBitDepth[CH_L] - 3.0) - d));
417  const_cast<VVEncCfg&>(m_cEncCfg).m_QP = Clip3 (17, MAX_QP, m_cEncCfg.m_QP);
418  }
419  }

426 void EncLib::encodePicture( bool flush, const vvencYUVBuffer* yuvInBuf, AccessUnitList& au, bool& isQueueEmpty )
427 {
428  PROFILER_ACCUM_AND_START_NEW_SET( 1, g_timeProfiler, P_PIC_LEVEL );
429 
430  // clear output access unit
431  au.clearAu();
432 
433  // setup picture and store original yuv
434  Picture* pic = nullptr;
435  if ( ! flush )
436  {
437  CHECK( m_ppsMap.getFirstPS() == nullptr || m_spsMap.getPS( m_ppsMap.getFirstPS()->spsId ) == nullptr, "picture set not initialised" );
438  CHECK( yuvInBuf == nullptr, "no input picture given" );
439 
440  if ( m_cEncCfg.m_vvencMCTF.MCTF && m_numPicsRcvd <= 0 && m_MCTF.getNumLeadFrames() < m_cEncCfg.m_vvencMCTF.MCTFNumLeadFrames )
441  {
442  m_MCTF.addLeadFrame( *yuvInBuf );
443  }
444  else if ( m_cEncCfg.m_vvencMCTF.MCTF && m_cEncCfg.m_framesToBeEncoded > 0 && m_numPicsRcvd >= m_cEncCfg.m_framesToBeEncoded )
445  {
446  m_MCTF.addTrailFrame( *yuvInBuf );
447  }
448  else
449  {
450  PPS& pps = *(m_ppsMap.getFirstPS());
451  const SPS& sps = *(m_spsMap.getPS( pps.spsId ));
452 
453  pic = xGetNewPicBuffer( pps, sps );
454 
455  copyPadToPelUnitBuf( pic->getOrigBuf(), *yuvInBuf, m_cEncCfg.m_internChromaFormat );
456 
457  if( yuvInBuf->ctsValid )
458  {
459  pic->cts = yuvInBuf->cts;
460  pic->ctsValid = true;
461  }
462 
463  xInitPicture( *pic, m_numPicsRcvd, pps, sps, m_cVPS, m_cDCI );
464  xDetectScreenC(*pic, pic->getOrigBuf());
465  m_numPicsRcvd += 1;
466  m_numPicsInQueue += 1;
467  }
468  }
469  else
470  {
471  CHECK( 0 == m_numPicsRcvd, "invalid call, non-empty input buffer expected" );
472  }
473 
474  // MCTF process
475  if (m_cEncCfg.m_usePerceptQPA || (m_cEncCfg.m_RCNumPasses == 2))
476  {
477  m_MCTF.assignQpaBufs( pic );
478  }
479  if ( m_cEncCfg.m_vvencMCTF.MCTF )
480  {
481  do
482  {
483  m_MCTF.filter( pic );
484  } while ( flush && m_MCTF.getCurDelay() > 0 );
485  }
486 
487  // encode picture
488  if ( m_numPicsInQueue >= m_cEncCfg.m_InputQueueSize
489  || ( m_numPicsInQueue > 0 && flush ) )
490  {
491  if ( m_cEncCfg.m_RCTargetBitrate > 0 )
492  {
493  if ( m_numPicsRcvd == m_numPicsInQueue )
494  {
495  m_cRateCtrl.initRCGOP( 1 );
496  }
497  else if ( 1 == ( m_numPicsRcvd - m_numPicsInQueue ) % m_cEncCfg.m_GOPSize )
498  {
499  m_cRateCtrl.destroyRCGOP();
500  m_cRateCtrl.initRCGOP( m_numPicsInQueue >= m_cEncCfg.m_GOPSize ? m_cEncCfg.m_GOPSize : m_numPicsInQueue );
501  }
502  }
503 
504  // update current poc
505  m_pocEncode = xGetNextPocICO( m_pocEncode, flush, m_numPicsRcvd, m_cEncCfg.m_DecodingRefreshType == 4 );
506 
507  if ((m_cEncCfg.m_RCNumPasses == 2) && (m_cRateCtrl.flushPOC < 0) && flush) m_cRateCtrl.flushPOC = m_pocEncode;
508 
509  std::vector<Picture*> encList;
510  xCreateCodingOrder( m_pocEncode, m_numPicsRcvd, m_numPicsInQueue, flush, encList, m_cEncCfg.m_DecodingRefreshType == 4 );
511 
512  // create cts / dts
513  if( !encList.empty() && encList[0]->ctsValid )
514  {
