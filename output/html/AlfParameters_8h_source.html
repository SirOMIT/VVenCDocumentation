70 struct AlfFilterShape
71 {
72  AlfFilterShape( int size )
73  : filterLength( size ),
74  numCoeff( size * size / 4 + 1 ),
75  filterSize( size * size / 2 + 1 )
76  {
77  if( size == 5 )
78  {
79  pattern = {
80  0,
81  1, 2, 3,
82  4, 5, 6, 5, 4,
83  3, 2, 1,
84  0
85  };
86 
87  weights = {
88  2,
89  2, 2, 2,
90  2, 2, 1, 1
91  };
92 
93  filterType = ALF_FILTER_5;
94  }
95  else if( size == 7 )
96  {
97  pattern = {
98  0,
99  1, 2, 3,
100  4, 5, 6, 7, 8,
101  9, 10, 11, 12, 11, 10, 9,
102  8, 7, 6, 5, 4,
103  3, 2, 1,
104  0
105  };
106 
107  weights = {
108  2,
109  2, 2, 2,
110  2, 2, 2, 2, 2,
111  2, 2, 2, 1, 1
112  };
113 
114  filterType = ALF_FILTER_7;
115  }
116  else if (size == size_CC_ALF)
117  {
118  size = 4;
119  filterLength = 8;
120  numCoeff = 8;
121  filterSize = 8;
122  filterType = CC_ALF;
123  }
124  else
125  {
126  filterType = ALF_NUM_OF_FILTER_TYPES;
127  CHECK( 0, "Wrong ALF filter shape" );
128  }
129  }
130 
131  AlfFilterType filterType;
132  int filterLength;
133  int numCoeff; //TO DO: check whether we need both numCoeff and filterSize
134  int filterSize;
135  std::vector<int> pattern;
136  std::vector<int> weights;
137 };
