120 void BinEncoderBase::finish()
121 {
122  if( m_Low >> ( 32 - m_bitsLeft ) )
123  {
124  m_Bitstream->write( m_bufferedByte + 1, 8 );
125  while( m_numBufferedBytes > 1 )
126  {
127  m_Bitstream->write( 0x00, 8 );
128  m_numBufferedBytes--;
129  }
130  m_Low -= 1 << ( 32 - m_bitsLeft );
131  }
132  else
133  {
134  if( m_numBufferedBytes > 0 )
135  {
136  m_Bitstream->write( m_bufferedByte, 8 );
137  }
138  while( m_numBufferedBytes > 1 )
139  {
140  m_Bitstream->write( 0xff, 8 );
141  m_numBufferedBytes--;
142  }
143  }
144  m_Bitstream->write( m_Low >> 8, 24 - m_bitsLeft );

188 void BinEncoderBase::encodeBinsEP( unsigned bins, unsigned numBins )
189 {
190  for(int i = 0; i < numBins; i++)
191  {
192  DTRACE( g_trace_ctx, D_CABAC, "%d" " " "%d" " EP=%d \n", DTRACE_GET_COUNTER( g_trace_ctx, D_CABAC ), m_Range, ( bins >> ( numBins - 1 - i ) ) & 1 );
193  }
194 
195  BinCounter::addEP( numBins );
196  if( m_Range == 256 )
197  {
198  encodeAlignedBinsEP( bins, numBins );
199  return;
200  }
201  while( numBins > 8 )
202  {
203  numBins -= 8;
204  unsigned pattern = bins >> numBins;
205  m_Low <<= 8;
206  m_Low += m_Range * pattern;
207  bins -= pattern << numBins;
208  m_bitsLeft -= 8;
209  if( m_bitsLeft < 12 )
210  {
211  writeOut();
212  }
213  }
214  m_Low <<= numBins;
215  m_Low += m_Range * bins;
216  m_bitsLeft -= numBins;
217  if( m_bitsLeft < 12 )
218  {
219  writeOut();
220  }
221 }

223 void BinEncoderBase::encodeRemAbsEP(unsigned bins, unsigned goRicePar, unsigned cutoff, int maxLog2TrDynamicRange)
224 {
225  const unsigned threshold = cutoff << goRicePar;
226  if (bins < threshold)
227  {
228  const unsigned bitMask = (1 << goRicePar) - 1;
229  const unsigned length = (bins >> goRicePar) + 1;
230  encodeBinsEP((1 << length) - 2, length);
231  encodeBinsEP(bins & bitMask, goRicePar);
232  }
233  else
234  {
235  const unsigned maxPrefixLength = 32 - cutoff - maxLog2TrDynamicRange;
236  unsigned prefixLength = 0;
237  unsigned codeValue = (bins >> goRicePar) - cutoff;
238  unsigned suffixLength;
239  if (codeValue >= ((1 << maxPrefixLength) - 1))
240  {
241  prefixLength = maxPrefixLength;
242  suffixLength = maxLog2TrDynamicRange;
243  }
244  else
245  {
246  while (codeValue > ((2 << prefixLength) - 2))
247  {
248  prefixLength++;
249  }
250  suffixLength = prefixLength + goRicePar + 1; //+1 for the separator bit
251  }
252  const unsigned totalPrefixLength = prefixLength + cutoff;
253  const unsigned bitMask = (1 << goRicePar) - 1;
254  const unsigned prefix = (1 << totalPrefixLength) - 1;
255  const unsigned suffix = ((codeValue - ((1 << prefixLength) - 1)) << goRicePar) | (bins & bitMask);
256  encodeBinsEP(prefix, totalPrefixLength); //prefix
257  encodeBinsEP(suffix, suffixLength); //separator, suffix, and rParam bits
258  }
259 }

261 void BinEncoderBase::encodeBinTrm( unsigned bin )
262 {
263  BinCounter::addTrm();

265  if( bin )
266  {
267  m_Low += m_Range;
268  m_Low <<= 7;
269  m_Range = 2 << 7;
270  m_bitsLeft -= 7;
271  }
272  else if( m_Range >= 256 )
273  {
274  return;
275  }
276  else
277  {
278  m_Low <<= 1;
279  m_Range <<= 1;
280  m_bitsLeft--;
281  }
282  if( m_bitsLeft < 12 )
283  {
284  writeOut();
285  }

295 void BinEncoderBase::encodeAlignedBinsEP( unsigned bins, unsigned numBins )
296 {
297  unsigned remBins = numBins;
298  while( remBins > 0 )
299  {
300  //The process of encoding an EP bin is the same as that of coding a normal
301  //bin where the symbol ranges for 1 and 0 are both half the range:
302  //
303  // low = (low + range/2) << 1 (to encode a 1)
304  // low = low << 1 (to encode a 0)
305  //
306  // i.e.
307  // low = (low + (bin * range/2)) << 1
308  //
309  // which is equivalent to:
310  //
311  // low = (low << 1) + (bin * range)
312  //
313  // this can be generalised for multiple bins, producing the following expression:
314  //
315  unsigned binsToCode = std::min<unsigned>( remBins, 8); //code bytes if able to take advantage of the system's byte-write function
316  unsigned binMask = ( 1 << binsToCode ) - 1;
317  unsigned newBins = ( bins >> ( remBins - binsToCode ) ) & binMask;
318  m_Low = ( m_Low << binsToCode ) + ( newBins << 8 ); //range is known to be 256
319  remBins -= binsToCode;
320  m_bitsLeft -= binsToCode;
321  if( m_bitsLeft < 12 )
322  {
323  writeOut();
324  }
325  }
326 }

328 void BinEncoderBase::writeOut()
329 {
330  unsigned leadByte = m_Low >> ( 24 - m_bitsLeft );
331  m_bitsLeft += 8;
332  m_Low &= 0xffffffffu >> m_bitsLeft;
333  if( leadByte == 0xff )
334  {
335  m_numBufferedBytes++;
336  }
337  else
338  {
339  if( m_numBufferedBytes > 0 )
340  {
341  unsigned carry = leadByte >> 8;
342  unsigned byte = m_bufferedByte + carry;
343  m_bufferedByte = leadByte & 0xff;
344  m_Bitstream->write( byte, 8 );
345  byte = ( 0xff + carry ) & 0xff;
346  while( m_numBufferedBytes > 1 )
347  {
348  m_Bitstream->write( byte, 8 );
349  m_numBufferedBytes--;
350  }
351  }
352  else
353  {
354  m_numBufferedBytes = 1;
355  m_bufferedByte = leadByte;
356  }
357  }
358 }

367 void BinEncoder::encodeBin( unsigned bin, unsigned ctxId )
368 {
369  BinCounter::addCtx( ctxId );
370  BinProbModel& rcProbModel = m_Ctx[ctxId];
371  uint32_t LPS = rcProbModel.getLPS( m_Range );
372 
373 // DTRACE( g_trace_ctx, D_CABAC, "%d" " %d " "%d" " " "[%d:%d]" " " "%2d(MPS=%d)" " " " - " "%d" "\n", DTRACE_GET_COUNTER( g_trace_ctx, D_CABAC ), ctxId, m_Range, m_Range - LPS, LPS, ( unsigned int ) ( rcProbModel.state() ), bin == rcProbModel.mps(), bin );
374  DTRACE( g_trace_ctx, D_CABAC, " %d " "%d" " " "[%d:%d]" " " "%2d(MPS=%d)" " " " - " "%d" "\n", DTRACE_GET_COUNTER( g_trace_ctx, D_CABAC ), m_Range, m_Range - LPS, LPS, (unsigned int)( rcProbModel.state() ), bin == rcProbModel.mps(), bin );
375 
376  m_Range -= LPS;
377  if( bin != rcProbModel.mps() )
378  {
379  int numBits = rcProbModel.getRenormBitsLPS( LPS );
380  m_bitsLeft -= numBits;
381  m_Low += m_Range;
382  m_Low = m_Low << numBits;
383  m_Range = LPS << numBits;
384  if( m_bitsLeft < 12 )
385  {
386  writeOut();
387  }
388  }
389  else
390  {
391  if( m_Range < 256 )
392  {
393  int numBits = rcProbModel.getRenormBitsRange( m_Range );
394  m_bitsLeft -= numBits;
395  m_Low <<= numBits;
396  m_Range <<= numBits;
397  if( m_bitsLeft < 12 )
398  {
399  writeOut();
400  }
401  }
402  }
403  rcProbModel.update( bin );
404  BinEncoderBase::m_BinStore.addBin( bin, ctxId );
405 }
