174  int getNumIntraCiip ( const CodingUnit& cu )
175  {
176  const Position posBL = cu.Y().bottomLeft();
177  const Position posTR = cu.Y().topRight();
178  const CodingUnit *neigh0 = cu.cs->getCURestricted(posBL.offset(-1, 0), cu, CH_L);
179  const CodingUnit *neigh1 = cu.cs->getCURestricted(posTR.offset(0, -1), cu, CH_L);
180 
181  int numIntra = 0;
182  numIntra += (neigh0 && (neigh0->predMode == MODE_INTRA))?1:0;
183  numIntra += (neigh1 && (neigh1->predMode == MODE_INTRA))?1:0;
184 
185  return numIntra;
186  }
