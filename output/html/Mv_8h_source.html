64 enum MvPrecision
65 {
66  MV_PRECISION_4PEL = 0, // 4-pel
67  MV_PRECISION_INT = 2, // 1-pel, shift 2 bits from 4-pel
68  MV_PRECISION_HALF = 3, // 1/2-pel
69  MV_PRECISION_QUARTER = 4, // 1/4-pel (the precision of regular MV difference signaling), shift 4 bits from 4-pel
70  MV_PRECISION_SIXTEENTH = 6, // 1/16-pel (the precision of internal MV), shift 6 bits from 4-pel
71  MV_PRECISION_INTERNAL = 2 + MV_FRACTIONAL_BITS_INTERNAL,
72 };

193  void changePrecision(const MvPrecision& src, const MvPrecision& dst)
194  {
195  const int shift = (int)dst - (int)src;
196  if (shift >= 0)
197  {
198  *this <<= shift;
199  }
200  else
201  {
202  const int rightShift = -shift;
203  const int nOffset = 1 << (rightShift - 1);
204  hor = hor >= 0 ? (hor + nOffset - 1) >> rightShift : (hor + nOffset) >> rightShift;
205  ver = ver >= 0 ? (ver + nOffset - 1) >> rightShift : (ver + nOffset) >> rightShift;
206  }
207  }

274  void mvCliptoStorageBitDepth() // periodic clipping
275  {
276  hor = (hor + mvClipPeriod) & (mvClipPeriod - 1);
277  hor = (hor >= halMvClipPeriod) ? (hor - mvClipPeriod) : hor;
278  ver = (ver + mvClipPeriod) & (mvClipPeriod - 1);
279  ver = (ver >= halMvClipPeriod) ? (ver - mvClipPeriod) : ver;
